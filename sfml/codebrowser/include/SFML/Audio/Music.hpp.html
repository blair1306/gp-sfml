52 ////////////////////////////////////////////////////////////
53 /// \brief Structure defining a time range using the template type
54 ///
55 ////////////////////////////////////////////////////////////
56 template <typename T>
57 struct Span
58 {
59 ////////////////////////////////////////////////////////////
60 /// \brief Default constructor
61 ///
62 ////////////////////////////////////////////////////////////
63 Span()
64 {
65
66 }
67
68 ////////////////////////////////////////////////////////////
69 /// \brief Initialization constructor
70 ///
71 /// \param off Initial Offset
72 /// \param len Initial Length
73 ///
74 ////////////////////////////////////////////////////////////
75 Span(T off, T len):
76 offset(off),
77 length(len)
78 {
79
80 }
81
82 T offset; ///< The beginning offset of the time range
83 T length; ///< The length of the time range
84 };
85
86 // Define the relevant Span types
87 typedef Span<Time> TimeSpan;
88
89 ////////////////////////////////////////////////////////////
90 /// \brief Default constructor
91 ///
92 ////////////////////////////////////////////////////////////
93 Music();

95 ////////////////////////////////////////////////////////////
96 /// \brief Destructor
97 ///
98 ////////////////////////////////////////////////////////////
99 ~Music();

101 ////////////////////////////////////////////////////////////
102 /// \brief Open a music from an audio file
103 ///
104 /// This function doesn't start playing the music (call play()
105 /// to do so).
106 /// See the documentation of sf::InputSoundFile for the list
107 /// of supported formats.
108 ///
109 /// \warning Since the music is not loaded at once but rather
110 /// streamed continuously, the file must remain accessible until
111 /// the sf::Music object loads a new music or is destroyed.
112 ///
113 /// \param filename Path of the music file to open
114 ///
115 /// \return True if loading succeeded, false if it failed
116 ///
117 /// \see openFromMemory, openFromStream
118 ///
119 ////////////////////////////////////////////////////////////
120 bool openFromFile(const std::string& filename);
121
122 ////////////////////////////////////////////////////////////
123 /// \brief Open a music from an audio file in memory
124 ///
125 /// This function doesn't start playing the music (call play()
126 /// to do so).
127 /// See the documentation of sf::InputSoundFile for the list
128 /// of supported formats.
129 ///
130 /// \warning Since the music is not loaded at once but rather streamed
131 /// continuously, the \a data buffer must remain accessible until
132 /// the sf::Music object loads a new music or is destroyed. That is,
133 /// you can't deallocate the buffer right after calling this function.
134 ///
135 /// \param data Pointer to the file data in memory
136 /// \param sizeInBytes Size of the data to load, in bytes
137 ///
138 /// \return True if loading succeeded, false if it failed
139 ///
140 /// \see openFromFile, openFromStream
141 ///
142 ////////////////////////////////////////////////////////////
143 bool openFromMemory(const void* data, std::size_t sizeInBytes);
144
145 ////////////////////////////////////////////////////////////
146 /// \brief Open a music from an audio file in a custom stream
147 ///
148 /// This function doesn't start playing the music (call play()
149 /// to do so).
150 /// See the documentation of sf::InputSoundFile for the list
151 /// of supported formats.
152 ///
153 /// \warning Since the music is not loaded at once but rather
154 /// streamed continuously, the \a stream must remain accessible
155 /// until the sf::Music object loads a new music or is destroyed.
156 ///
157 /// \param stream Source stream to read from
158 ///
159 /// \return True if loading succeeded, false if it failed
160 ///
161 /// \see openFromFile, openFromMemory
162 ///
163 ////////////////////////////////////////////////////////////
164 bool openFromStream(InputStream& stream);
165
166 ////////////////////////////////////////////////////////////
167 /// \brief Get the total duration of the music
168 ///
169 /// \return Music duration
170 ///
171 ////////////////////////////////////////////////////////////
172 Time getDuration() const;
173
174 ////////////////////////////////////////////////////////////
175 /// \brief Get the positions of the of the sound's looping sequence
176 ///
177 /// \return Loop Time position class.
178 ///
179 /// \warning Since setLoopPoints() performs some adjustments on the
180 /// provided values and rounds them to internal samples, a call to
181 /// getLoopPoints() is not guaranteed to return the same times passed
182 /// into a previous call to setLoopPoints(). However, it is guaranteed
183 /// to return times that will map to the valid internal samples of
184 /// this Music if they are later passed to setLoopPoints().
185 ///
186 /// \see setLoopPoints
187 ///
188 ////////////////////////////////////////////////////////////
189 TimeSpan getLoopPoints() const;
190
191 ////////////////////////////////////////////////////////////
192 /// \brief Sets the beginning and end of the sound's looping sequence using sf::Time
193 ///
194 /// Loop points allow one to specify a pair of positions such that, when the music
195 /// is enabled for looping, it will seamlessly seek to the beginning whenever it
196 /// encounters the end. Valid ranges for timePoints.offset and timePoints.length are
197 /// [0, Dur) and (0, Dur-offset] respectively, where Dur is the value returned by getDuration().
198 /// Note that the EOF "loop point" from the end to the beginning of the stream is still honored,
199 /// in case the caller seeks to a point after the end of the loop range. This function can be
200 /// safely called at any point after a stream is opened, and will be applied to a playing sound
201 /// without affecting the current playing offset.
202 ///
203 /// \warning Setting the loop points while the stream's status is Paused
204 /// will set its status to Stopped. The playing offset will be unaffected.
205 ///
206 /// \param timePoints The definition of the loop. Can be any time points within the sound's length
207 ///
208 /// \see getLoopPoints
209 ///
210 ////////////////////////////////////////////////////////////
211 void setLoopPoints(TimeSpan timePoints);
212
213protected:
214
215 ////////////////////////////////////////////////////////////
216 /// \brief Request a new chunk of audio samples from the stream source
217 ///
218 /// This function fills the chunk from the next samples
219 /// to read from the audio file.
220 ///
221 /// \param data Chunk of data to fill
222 ///
223 /// \return True to continue playback, false to stop
224 ///
225 ////////////////////////////////////////////////////////////
226 virtual bool onGetData(Chunk& data);
227
228 ////////////////////////////////////////////////////////////
229 /// \brief Change the current playing position in the stream source
230 ///
231 /// \param timeOffset New playing position, from the beginning of the music
232 ///
233 ////////////////////////////////////////////////////////////
234 virtual void onSeek(Time timeOffset);
235
236 ////////////////////////////////////////////////////////////
237 /// \brief Change the current playing position in the stream source to the loop offset
238 ///
239 /// This is called by the underlying SoundStream whenever it needs us to reset
240 /// the seek position for a loop. We then determine whether we are looping on a
241 /// loop point or the end-of-file, perform the seek, and return the new position.
242 ///
243 /// \return The seek position after looping (or -1 if there's no loop)
244 ///
245 ////////////////////////////////////////////////////////////
246 virtual Int64 onLoop();
247
248private:
249
250 ////////////////////////////////////////////////////////////
251 /// \brief Initialize the internal state after loading a new music
252 ///
253 ////////////////////////////////////////////////////////////
254 void initialize();
255
256 ////////////////////////////////////////////////////////////
257 /// \brief Helper to convert an sf::Time to a sample position
258 ///
259 /// \param position Time to convert to samples
260 ///
261 /// \return The number of samples elapsed at the given time
262 ///
263 ////////////////////////////////////////////////////////////

266 ////////////////////////////////////////////////////////////
267 /// \brief Helper to convert a sample position to an sf::Time
268 ///
269 /// \param position Sample count to convert to Time

271 /// \return The Time position of the given sample
272 ///
273 ////////////////////////////////////////////////////////////
274 Time samplesToTime(Uint64 samples) const;

291////////////////////////////////////////////////////////////
292/// \class sf::Music
293/// \ingroup audio
294///
295/// Musics are sounds that are streamed rather than completely
296/// loaded in memory. This is especially useful for compressed
297/// musics that usually take hundreds of MB when they are
298/// uncompressed: by streaming it instead of loading it entirely,
299/// you avoid saturating the memory and have almost no loading delay.
300/// This implies that the underlying resource (file, stream or
301/// memory buffer) must remain valid for the lifetime of the
302/// sf::Music object.
303///
304/// Apart from that, a sf::Music has almost the same features as
305/// the sf::SoundBuffer / sf::Sound pair: you can play/pause/stop
306/// it, request its parameters (channels, sample rate), change
307/// the way it is played (pitch, volume, 3D position, ...), etc.
308///
309/// As a sound stream, a music is played in its own thread in order
310/// not to block the rest of the program. This means that you can
311/// leave the music alone after calling play(), it will manage itself
312/// very well.
313///
314/// Usage example:
315/// \code
316/// // Declare a new music
317/// sf::Music music;
318///
319/// // Open it from an audio file
320/// if (!music.openFromFile("music.ogg"))
321/// {
322/// // error...
323/// }
324///
325/// // Change some parameters
326/// music.setPosition(0, 1, 10); // change its 3D position
327/// music.setPitch(2); // increase the pitch
328/// music.setVolume(50); // reduce the volume
329/// music.setLoop(true); // make it loop
330///
331/// // Play it
332/// music.play();
333/// \endcode
334///
335/// \see sf::Sound, sf::SoundStream
336///
