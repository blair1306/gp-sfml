38/// \brief The audio listener is the point in the scene
39/// from where all the sounds are heard
40///
41////////////////////////////////////////////////////////////
42class SFML_AUDIO_API Listener
43{
44public:
45
46 ////////////////////////////////////////////////////////////
47 /// \brief Change the global volume of all the sounds and musics
48 ///
49 /// The volume is a number between 0 and 100; it is combined with
50 /// the individual volume of each sound / music.
51 /// The default value for the volume is 100 (maximum).
52 ///
53 /// \param volume New global volume, in the range [0, 100]
54 ///
55 /// \see getGlobalVolume
56 ///
57 ////////////////////////////////////////////////////////////
58 static void setGlobalVolume(float volume);
59
60 ////////////////////////////////////////////////////////////
61 /// \brief Get the current value of the global volume
62 ///
63 /// \return Current global volume, in the range [0, 100]
64 ///
65 /// \see setGlobalVolume
66 ///
67 ////////////////////////////////////////////////////////////
68 static float getGlobalVolume();
69
70 ////////////////////////////////////////////////////////////
71 /// \brief Set the position of the listener in the scene
72 ///
73 /// The default listener's position is (0, 0, 0).
74 ///
75 /// \param x X coordinate of the listener's position
76 /// \param y Y coordinate of the listener's position
77 /// \param z Z coordinate of the listener's position
78 ///
79 /// \see getPosition, setDirection
80 ///
81 ////////////////////////////////////////////////////////////
82 static void setPosition(float x, float y, float z);
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Set the position of the listener in the scene
86 ///
87 /// The default listener's position is (0, 0, 0).
88 ///
89 /// \param position New listener's position
90 ///
91 /// \see getPosition, setDirection
92 ///
93 ////////////////////////////////////////////////////////////
94 static void setPosition(const Vector3f& position);
95
96 ////////////////////////////////////////////////////////////
97 /// \brief Get the current position of the listener in the scene
98 ///
99 /// \return Listener's position
100 ///
101 /// \see setPosition
102 ///
103 ////////////////////////////////////////////////////////////
104 static Vector3f getPosition();
105
106 ////////////////////////////////////////////////////////////
107 /// \brief Set the forward vector of the listener in the scene
108 ///
109 /// The direction (also called "at vector") is the vector
110 /// pointing forward from the listener's perspective. Together
111 /// with the up vector, it defines the 3D orientation of the
112 /// listener in the scene. The direction vector doesn't
113 /// have to be normalized.
114 /// The default listener's direction is (0, 0, -1).
115 ///
116 /// \param x X coordinate of the listener's direction
117 /// \param y Y coordinate of the listener's direction
118 /// \param z Z coordinate of the listener's direction
119 ///
120 /// \see getDirection, setUpVector, setPosition
121 ///
122 ////////////////////////////////////////////////////////////
123 static void setDirection(float x, float y, float z);
124
125 ////////////////////////////////////////////////////////////
126 /// \brief Set the forward vector of the listener in the scene
127 ///
128 /// The direction (also called "at vector") is the vector
129 /// pointing forward from the listener's perspective. Together
130 /// with the up vector, it defines the 3D orientation of the
131 /// listener in the scene. The direction vector doesn't
132 /// have to be normalized.
133 /// The default listener's direction is (0, 0, -1).
134 ///
135 /// \param direction New listener's direction
136 ///
137 /// \see getDirection, setUpVector, setPosition
138 ///
139 ////////////////////////////////////////////////////////////
140 static void setDirection(const Vector3f& direction);
141
142 ////////////////////////////////////////////////////////////
143 /// \brief Get the current forward vector of the listener in the scene
144 ///
145 /// \return Listener's forward vector (not normalized)
146 ///
147 /// \see setDirection
148 ///
149 ////////////////////////////////////////////////////////////
150 static Vector3f getDirection();
151
152 ////////////////////////////////////////////////////////////
153 /// \brief Set the upward vector of the listener in the scene
154 ///
155 /// The up vector is the vector that points upward from the
156 /// listener's perspective. Together with the direction, it
157 /// defines the 3D orientation of the listener in the scene.
158 /// The up vector doesn't have to be normalized.
159 /// The default listener's up vector is (0, 1, 0). It is usually
160 /// not necessary to change it, especially in 2D scenarios.
161 ///
162 /// \param x X coordinate of the listener's up vector
163 /// \param y Y coordinate of the listener's up vector
164 /// \param z Z coordinate of the listener's up vector
165 ///
166 /// \see getUpVector, setDirection, setPosition
167 ///
168 ////////////////////////////////////////////////////////////
169 static void setUpVector(float x, float y, float z);
170
171 ////////////////////////////////////////////////////////////
172 /// \brief Set the upward vector of the listener in the scene
173 ///
174 /// The up vector is the vector that points upward from the
175 /// listener's perspective. Together with the direction, it
176 /// defines the 3D orientation of the listener in the scene.
177 /// The up vector doesn't have to be normalized.
178 /// The default listener's up vector is (0, 1, 0). It is usually
179 /// not necessary to change it, especially in 2D scenarios.
180 ///
181 /// \param upVector New listener's up vector
182 ///
183 /// \see getUpVector, setDirection, setPosition
184 ///
185 ////////////////////////////////////////////////////////////
186 static void setUpVector(const Vector3f& upVector);
187
188 ////////////////////////////////////////////////////////////
189 /// \brief Get the current upward vector of the listener in the scene
190 ///
191 /// \return Listener's upward vector (not normalized)
192 ///
193 /// \see setUpVector
194 ///
195 ////////////////////////////////////////////////////////////
196 static Vector3f getUpVector();
197};

206/// \class sf::Listener
207/// \ingroup audio
208///
209/// The audio listener defines the global properties of the
210/// audio environment, it defines where and how sounds and musics
211/// are heard. If sf::View is the eyes of the user, then sf::Listener
212/// is his ears (by the way, they are often linked together --
213/// same position, orientation, etc.).
214///
215/// sf::Listener is a simple interface, which allows to setup the
216/// listener in the 3D audio environment (position, direction and
217/// up vector), and to adjust the global volume.
218///
219/// Because the listener is unique in the scene, sf::Listener only
220/// contains static functions and doesn't have to be instantiated.
221///
222/// Usage example:
223/// \code
224/// // Move the listener to the position (1, 0, -5)
225/// sf::Listener::setPosition(1, 0, -5);
226///
227/// // Make it face the right axis (1, 0, 0)
228/// sf::Listener::setDirection(1, 0, 0);
229///
230/// // Reduce the global volume
231/// sf::Listener::setGlobalVolume(50);
232/// \endcode
233///
