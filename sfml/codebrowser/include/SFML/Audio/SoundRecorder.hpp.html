41////////////////////////////////////////////////////////////
42/// \brief Abstract base class for capturing sound data
43///
44////////////////////////////////////////////////////////////
45class SFML_AUDIO_API SoundRecorder : AlResource
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 /// \brief destructor
51 ///
52 ////////////////////////////////////////////////////////////
53 virtual ~SoundRecorder();
54
55 ////////////////////////////////////////////////////////////
56 /// \brief Start the capture
57 ///
58 /// The \a sampleRate parameter defines the number of audio samples
59 /// captured per second. The higher, the better the quality
60 /// (for example, 44100 samples/sec is CD quality).
61 /// This function uses its own thread so that it doesn't block
62 /// the rest of the program while the capture runs.
63 /// Please note that only one capture can happen at the same time.
64 /// You can select which capture device will be used, by passing
65 /// the name to the setDevice() method. If none was selected
66 /// before, the default capture device will be used. You can get a
67 /// list of the names of all available capture devices by calling
68 /// getAvailableDevices().
69 ///
70 /// \param sampleRate Desired capture rate, in number of samples per second
71 ///
72 /// \return True, if start of capture was successful
73 ///
74 /// \see stop, getAvailableDevices
75 ///
76 ////////////////////////////////////////////////////////////
77 bool start(unsigned int sampleRate = 44100);
78
79 ////////////////////////////////////////////////////////////
80 /// \brief Stop the capture
81 ///
82 /// \see start
83 ///
84 ////////////////////////////////////////////////////////////
85 void stop();
86
87 ////////////////////////////////////////////////////////////
88 /// \brief Get the sample rate
89 ///
90 /// The sample rate defines the number of audio samples
91 /// captured per second. The higher, the better the quality
92 /// (for example, 44100 samples/sec is CD quality).
93 ///
94 /// \return Sample rate, in samples per second
95 ///
96 ////////////////////////////////////////////////////////////
97 unsigned int getSampleRate() const;
98
99 ////////////////////////////////////////////////////////////
100 /// \brief Get a list of the names of all available audio capture devices
101 ///
102 /// This function returns a vector of strings, containing
103 /// the names of all available audio capture devices.
104 ///
105 /// \return A vector of strings containing the names
106 ///
107 ////////////////////////////////////////////////////////////
108 static std::vector<std::string> getAvailableDevices();
109
110 ////////////////////////////////////////////////////////////
111 /// \brief Get the name of the default audio capture device
112 ///
113 /// This function returns the name of the default audio
114 /// capture device. If none is available, an empty string
115 /// is returned.
116 ///
117 /// \return The name of the default audio capture device
118 ///
119 ////////////////////////////////////////////////////////////
120 static std::string getDefaultDevice();
121
122 ////////////////////////////////////////////////////////////
123 /// \brief Set the audio capture device
124 ///
125 /// This function sets the audio capture device to the device
126 /// with the given \a name. It can be called on the fly (i.e:
127 /// while recording). If you do so while recording and
128 /// opening the device fails, it stops the recording.
129 ///
130 /// \param name The name of the audio capture device
131 ///
132 /// \return True, if it was able to set the requested device
133 ///
134 /// \see getAvailableDevices, getDefaultDevice
135 ///
136 ////////////////////////////////////////////////////////////
137 bool setDevice(const std::string& name);
138
139 ////////////////////////////////////////////////////////////
140 /// \brief Get the name of the current audio capture device
141 ///
142 /// \return The name of the current audio capture device
143 ///
144 ////////////////////////////////////////////////////////////
145 const std::string& getDevice() const;
146
147 ////////////////////////////////////////////////////////////
148 /// \brief Set the channel count of the audio capture device
149 ///
150 /// This method allows you to specify the number of channels
151 /// used for recording. Currently only 16-bit mono and
152 /// 16-bit stereo are supported.
153 ///
154 /// \param channelCount Number of channels. Currently only
155 /// mono (1) and stereo (2) are supported.
156 ///
157 /// \see getChannelCount
158 ///
159 ////////////////////////////////////////////////////////////
160 void setChannelCount(unsigned int channelCount);
161
162 ////////////////////////////////////////////////////////////
163 /// \brief Get the number of channels used by this recorder
164 ///
165 /// Currently only mono and stereo are supported, so the
166 /// value is either 1 (for mono) or 2 (for stereo).
167 ///
168 /// \return Number of channels
169 ///
170 /// \see setChannelCount
171 ///
172 ////////////////////////////////////////////////////////////
173 unsigned int getChannelCount() const;
174
175 ////////////////////////////////////////////////////////////
176 /// \brief Check if the system supports audio capture
177 ///
178 /// This function should always be called before using
179 /// the audio capture features. If it returns false, then
180 /// any attempt to use sf::SoundRecorder or one of its derived
181 /// classes will fail.
182 ///
183 /// \return True if audio capture is supported, false otherwise
184 ///
185 ////////////////////////////////////////////////////////////
186 static bool isAvailable();
187
188protected:
189
190 ////////////////////////////////////////////////////////////
191 /// \brief Default constructor
192 ///
193 /// This constructor is only meant to be called by derived classes.
194 ///
195 ////////////////////////////////////////////////////////////
196 SoundRecorder();
197
198 ////////////////////////////////////////////////////////////
199 /// \brief Set the processing interval
200 ///
201 /// The processing interval controls the period
202 /// between calls to the onProcessSamples function. You may
203 /// want to use a small interval if you want to process the
204 /// recorded data in real time, for example.
205 ///
206 /// Note: this is only a hint, the actual period may vary.
207 /// So don't rely on this parameter to implement precise timing.
208 ///
209 /// The default processing interval is 100 ms.
210 ///
211 /// \param interval Processing interval
212 ///
213 ////////////////////////////////////////////////////////////
214 void setProcessingInterval(Time interval);
215
216 ////////////////////////////////////////////////////////////
217 /// \brief Start capturing audio data
218 ///
219 /// This virtual function may be overridden by a derived class
220 /// if something has to be done every time a new capture
221 /// starts. If not, this function can be ignored; the default
222 /// implementation does nothing.
223 ///
224 /// \return True to start the capture, or false to abort it
225 ///
226 ////////////////////////////////////////////////////////////
227 virtual bool onStart();
228
229 ////////////////////////////////////////////////////////////
230 /// \brief Process a new chunk of recorded samples
231 ///
232 /// This virtual function is called every time a new chunk of
233 /// recorded data is available. The derived class can then do
234 /// whatever it wants with it (storing it, playing it, sending
235 /// it over the network, etc.).
236 ///
237 /// \param samples Pointer to the new chunk of recorded samples
238 /// \param sampleCount Number of samples pointed by \a samples
239 ///
240 /// \return True to continue the capture, or false to stop it
241 ///
242 ////////////////////////////////////////////////////////////
243 virtual bool onProcessSamples(const Int16* samples, std::size_t sampleCount) = 0;
244
245 ////////////////////////////////////////////////////////////
246 /// \brief Stop capturing audio data
247 ///
248 /// This virtual function may be overridden by a derived class
249 /// if something has to be done every time the capture
250 /// ends. If not, this function can be ignored; the default
251 /// implementation does nothing.
252 ///
253 ////////////////////////////////////////////////////////////
254 virtual void onStop();
255
256private:
257
258 ////////////////////////////////////////////////////////////
259 /// \brief Function called as the entry point of the thread
260 ///
261 /// This function starts the recording loop, and returns
262 /// only when the capture is stopped.
263 ///
264 ////////////////////////////////////////////////////////////
265 void record();
266
267 ////////////////////////////////////////////////////////////
268 /// \brief Get the new available audio samples and process them
269 ///
270 /// This function is called continuously during the
271 /// capture loop. It retrieves the captured samples and
272 /// forwards them to the derived class.
273 ///
274 ////////////////////////////////////////////////////////////
275 void processCapturedSamples();
276
277 ////////////////////////////////////////////////////////////
278 /// \brief Clean up the recorder's internal resources
279 ///
280 /// This function is called when the capture stops.
281 ///
282 ////////////////////////////////////////////////////////////
283 void cleanup();
284
285 ////////////////////////////////////////////////////////////
286 // Member data
287 ////////////////////////////////////////////////////////////
288 Thread m_thread; ///< Thread running the background recording task
289 std::vector<Int16> m_samples; ///< Buffer to store captured samples
290 unsigned int m_sampleRate; ///< Sample rate
291 Time m_processingInterval; ///< Time period between calls to onProcessSamples
292 bool m_isCapturing; ///< Capturing state
293 std::string m_deviceName; ///< Name of the audio capture device
294 unsigned int m_channelCount; ///< Number of recording channels
295};
296
297} // namespace sf
298
299
300#endif // SFML_SOUNDRECORDER_HPP

303////////////////////////////////////////////////////////////
304/// \class sf::SoundRecorder
305/// \ingroup audio
306///
307/// sf::SoundBuffer provides a simple interface to access
308/// the audio recording capabilities of the computer
309/// (the microphone). As an abstract base class, it only cares
310/// about capturing sound samples, the task of making something
311/// useful with them is left to the derived class. Note that
312/// SFML provides a built-in specialization for saving the
313/// captured data to a sound buffer (see sf::SoundBufferRecorder).
314///
315/// A derived class has only one virtual function to override:
316/// \li onProcessSamples provides the new chunks of audio samples while the capture happens
317///
318/// Moreover, two additional virtual functions can be overridden
319/// as well if necessary:
320/// \li onStart is called before the capture happens, to perform custom initializations
321/// \li onStop is called after the capture ends, to perform custom cleanup
322///
323/// A derived class can also control the frequency of the onProcessSamples
324/// calls, with the setProcessingInterval protected function. The default
325/// interval is chosen so that recording thread doesn't consume too much
326/// CPU, but it can be changed to a smaller value if you need to process
327/// the recorded data in real time, for example.
328///
329/// The audio capture feature may not be supported or activated
330/// on every platform, thus it is recommended to check its
331/// availability with the isAvailable() function. If it returns
332/// false, then any attempt to use an audio recorder will fail.
333///
334/// If you have multiple sound input devices connected to your
335/// computer (for example: microphone, external soundcard, webcam mic, ...)
336/// you can get a list of all available devices through the
337/// getAvailableDevices() function. You can then select a device
338/// by calling setDevice() with the appropriate device. Otherwise
339/// the default capturing device will be used.
340///
341/// By default the recording is in 16-bit mono. Using the
342/// setChannelCount method you can change the number of channels
343/// used by the audio capture device to record. Note that you
344/// have to decide whether you want to record in mono or stereo
345/// before starting the recording.
346///
347/// It is important to note that the audio capture happens in a
348/// separate thread, so that it doesn't block the rest of the
349/// program. In particular, the onProcessSamples virtual function
350/// (but not onStart and not onStop) will be called
351/// from this separate thread. It is important to keep this in
352/// mind, because you may have to take care of synchronization
353/// issues if you share data between threads.
354/// Another thing to bear in mind is that you must call stop()
355/// in the destructor of your derived class, so that the recording
356/// thread finishes before your object is destroyed.
357///
358/// Usage example:
359/// \code
360/// class CustomRecorder : public sf::SoundRecorder
361/// {
362/// ~CustomRecorder()
363/// {
364/// // Make sure to stop the recording thread
365/// stop();
366/// }
367///
368/// virtual bool onStart() // optional
369/// {
370/// // Initialize whatever has to be done before the capture starts
371/// ...
372///
373/// // Return true to start playing
374/// return true;
375/// }
376///
377/// virtual bool onProcessSamples(const Int16* samples, std::size_t sampleCount)
378/// {
379/// // Do something with the new chunk of samples (store them, send them, ...)
380/// ...
381///
382/// // Return true to continue playing
383/// return true;
384/// }
385///
386/// virtual void onStop() // optional
387/// {
388/// // Clean up whatever has to be done after the capture ends
389/// ...
390/// }
391/// }
392///
393/// // Usage
394/// if (CustomRecorder::isAvailable())
395/// {
396/// CustomRecorder recorder;
397///
398/// if (!recorder.start())
399/// return -1;
400///
401/// ...
402/// recorder.stop();
403/// }
404/// \endcode
405///
406/// \see sf::SoundBufferRecorder
407///
