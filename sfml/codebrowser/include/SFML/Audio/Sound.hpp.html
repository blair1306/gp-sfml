41////////////////////////////////////////////////////////////
42/// \brief Regular sound that can be played in the audio environment
43///
44////////////////////////////////////////////////////////////
45class SFML_AUDIO_API Sound : public SoundSource
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 /// \brief Default constructor
51 ///
52 ////////////////////////////////////////////////////////////
53 Sound();
54
55 ////////////////////////////////////////////////////////////
56 /// \brief Construct the sound with a buffer
57 ///
58 /// \param buffer Sound buffer containing the audio data to play with the sound
59 ///
60 ////////////////////////////////////////////////////////////
61 explicit Sound(const SoundBuffer& buffer);
62
63 ////////////////////////////////////////////////////////////
64 /// \brief Copy constructor
65 ///
66 /// \param copy Instance to copy
67 ///
68 ////////////////////////////////////////////////////////////
69 Sound(const Sound& copy);
70
71 ////////////////////////////////////////////////////////////
72 /// \brief Destructor
73 ///
74 ////////////////////////////////////////////////////////////
75 ~Sound();
76
77 ////////////////////////////////////////////////////////////
78 /// \brief Start or resume playing the sound
79 ///
80 /// This function starts the stream if it was stopped, resumes
81 /// it if it was paused, and restarts it from beginning if it
82 /// was it already playing.
83 /// This function uses its own thread so that it doesn't block
84 /// the rest of the program while the sound is played.
85 ///
86 /// \see pause, stop
87 ///
88 ////////////////////////////////////////////////////////////
89 void play();
90
91 ////////////////////////////////////////////////////////////
92 /// \brief Pause the sound
93 ///
94 /// This function pauses the sound if it was playing,
95 /// otherwise (sound already paused or stopped) it has no effect.
96 ///
97 /// \see play, stop
98 ///
99 ////////////////////////////////////////////////////////////
100 void pause();
101
102 ////////////////////////////////////////////////////////////
103 /// \brief stop playing the sound
104 ///
105 /// This function stops the sound if it was playing or paused,
106 /// and does nothing if it was already stopped.
107 /// It also resets the playing position (unlike pause()).
108 ///
109 /// \see play, pause
110 ///
111 ////////////////////////////////////////////////////////////
112 void stop();
113
114 ////////////////////////////////////////////////////////////
115 /// \brief Set the source buffer containing the audio data to play
116 ///
117 /// It is important to note that the sound buffer is not copied,
118 /// thus the sf::SoundBuffer instance must remain alive as long
119 /// as it is attached to the sound.
120 ///
121 /// \param buffer Sound buffer to attach to the sound
122 ///
123 /// \see getBuffer
124 ///
125 ////////////////////////////////////////////////////////////
126 void setBuffer(const SoundBuffer& buffer);
127
128 ////////////////////////////////////////////////////////////
129 /// \brief Set whether or not the sound should loop after reaching the end
130 ///
131 /// If set, the sound will restart from beginning after
132 /// reaching the end and so on, until it is stopped or
133 /// setLoop(false) is called.
134 /// The default looping state for sound is false.
135 ///
136 /// \param loop True to play in loop, false to play once
137 ///
138 /// \see getLoop
139 ///
140 ////////////////////////////////////////////////////////////
141 void setLoop(bool loop);
142
143 ////////////////////////////////////////////////////////////
144 /// \brief Change the current playing position of the sound
145 ///
146 /// The playing position can be changed when the sound is
147 /// either paused or playing. Changing the playing position
148 /// when the sound is stopped has no effect, since playing
149 /// the sound will reset its position.
150 ///
151 /// \param timeOffset New playing position, from the beginning of the sound
152 ///
153 /// \see getPlayingOffset
154 ///
155 ////////////////////////////////////////////////////////////
156 void setPlayingOffset(Time timeOffset);
157
158 ////////////////////////////////////////////////////////////
159 /// \brief Get the audio buffer attached to the sound
160 ///
161 /// \return Sound buffer attached to the sound (can be NULL)
162 ///
163 ////////////////////////////////////////////////////////////
164 const SoundBuffer* getBuffer() const;
165
166 ////////////////////////////////////////////////////////////
167 /// \brief Tell whether or not the sound is in loop mode
168 ///
169 /// \return True if the sound is looping, false otherwise
170 ///
171 /// \see setLoop
172 ///
173 ////////////////////////////////////////////////////////////
174 bool getLoop() const;
175
176 ////////////////////////////////////////////////////////////
177 /// \brief Get the current playing position of the sound
178 ///
179 /// \return Current playing position, from the beginning of the sound
180 ///
181 /// \see setPlayingOffset
182 ///
183 ////////////////////////////////////////////////////////////
184 Time getPlayingOffset() const;
185
186 ////////////////////////////////////////////////////////////
187 /// \brief Get the current status of the sound (stopped, paused, playing)
188 ///
189 /// \return Current status of the sound
190 ///
191 ////////////////////////////////////////////////////////////
192 Status getStatus() const;
193
194 ////////////////////////////////////////////////////////////
195 /// \brief Overload of assignment operator
196 ///
197 /// \param right Instance to assign
198 ///
199 /// \return Reference to self
200 ///
201 ////////////////////////////////////////////////////////////
202 Sound& operator =(const Sound& right);
203
204 ////////////////////////////////////////////////////////////
205 /// \brief Reset the internal buffer of the sound
206 ///
207 /// This function is for internal use only, you don't have
208 /// to use it. It is called by the sf::SoundBuffer that
209 /// this sound uses, when it is destroyed in order to prevent
210 /// the sound from using a dead buffer.
211 ///
212 ////////////////////////////////////////////////////////////
213 void resetBuffer();
214
215private:
216
217 ////////////////////////////////////////////////////////////
218 // Member data
219 ////////////////////////////////////////////////////////////
220 const SoundBuffer* m_buffer; ///< Sound buffer bound to the source
221};
222

229////////////////////////////////////////////////////////////
230/// \class sf::Sound
231/// \ingroup audio
232///
233/// sf::Sound is the class to use to play sounds.
234/// It provides:
235/// \li Control (play, pause, stop)
236/// \li Ability to modify output parameters in real-time (pitch, volume, ...)
237/// \li 3D spatial features (position, attenuation, ...).
238///
239/// sf::Sound is perfect for playing short sounds that can
240/// fit in memory and require no latency, like foot steps or
241/// gun shots. For longer sounds, like background musics
242/// or long speeches, rather see sf::Music (which is based
243/// on streaming).
244///
245/// In order to work, a sound must be given a buffer of audio
246/// data to play. Audio data (samples) is stored in sf::SoundBuffer,
247/// and attached to a sound with the setBuffer() function.
248/// The buffer object attached to a sound must remain alive
249/// as long as the sound uses it. Note that multiple sounds
250/// can use the same sound buffer at the same time.
251///
252/// Usage example:
253/// \code
254/// sf::SoundBuffer buffer;
255/// buffer.loadFromFile("sound.wav");
256///
257/// sf::Sound sound;
258/// sound.setBuffer(buffer);
259/// sound.play();
260/// \endcode
261///
262/// \see sf::SoundBuffer, sf::Music
263///
