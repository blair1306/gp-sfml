41////////////////////////////////////////////////////////////
42/// \brief Abstract base class for streamed audio sources
43///
44////////////////////////////////////////////////////////////
45class SFML_AUDIO_API SoundStream : public SoundSource
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 /// \brief Structure defining a chunk of audio data to stream
51 ///
52 ////////////////////////////////////////////////////////////
53 struct Chunk
54 {
55 const Int16* samples; ///< Pointer to the audio samples
56 std::size_t sampleCount; ///< Number of samples pointed by Samples
57 };
58
59 ////////////////////////////////////////////////////////////
60 /// \brief Destructor
61 ///
62 ////////////////////////////////////////////////////////////
63 virtual ~SoundStream();
64
65 ////////////////////////////////////////////////////////////
66 /// \brief Start or resume playing the audio stream
67 ///
68 /// This function starts the stream if it was stopped, resumes
69 /// it if it was paused, and restarts it from the beginning if
70 /// it was already playing.
71 /// This function uses its own thread so that it doesn't block
72 /// the rest of the program while the stream is played.
73 ///
74 /// \see pause, stop
75 ///
76 ////////////////////////////////////////////////////////////
77 void play();
78
79 ////////////////////////////////////////////////////////////
80 /// \brief Pause the audio stream
81 ///
82 /// This function pauses the stream if it was playing,
83 /// otherwise (stream already paused or stopped) it has no effect.
84 ///
85 /// \see play, stop
86 ///
87 ////////////////////////////////////////////////////////////
88 void pause();
89
90 ////////////////////////////////////////////////////////////
91 /// \brief Stop playing the audio stream
92 ///
93 /// This function stops the stream if it was playing or paused,
94 /// and does nothing if it was already stopped.
95 /// It also resets the playing position (unlike pause()).
96 ///
97 /// \see play, pause
98 ///
99 ////////////////////////////////////////////////////////////
100 void stop();
101
102 ////////////////////////////////////////////////////////////
103 /// \brief Return the number of channels of the stream
104 ///
105 /// 1 channel means a mono sound, 2 means stereo, etc.
106 ///
107 /// \return Number of channels
108 ///
109 ////////////////////////////////////////////////////////////
110 unsigned int getChannelCount() const;
111
112 ////////////////////////////////////////////////////////////
113 /// \brief Get the stream sample rate of the stream
114 ///
115 /// The sample rate is the number of audio samples played per
116 /// second. The higher, the better the quality.
117 ///
118 /// \return Sample rate, in number of samples per second
119 ///
120 ////////////////////////////////////////////////////////////
121 unsigned int getSampleRate() const;
122
123 ////////////////////////////////////////////////////////////
124 /// \brief Get the current status of the stream (stopped, paused, playing)
125 ///
126 /// \return Current status
127 ///
128 ////////////////////////////////////////////////////////////
129 Status getStatus() const;
130
131 ////////////////////////////////////////////////////////////
132 /// \brief Change the current playing position of the stream
133 ///
134 /// The playing position can be changed when the stream is
135 /// either paused or playing. Changing the playing position
136 /// when the stream is stopped has no effect, since playing
137 /// the stream would reset its position.
138 ///
139 /// \param timeOffset New playing position, from the beginning of the stream
140 ///
141 /// \see getPlayingOffset
142 ///
143 ////////////////////////////////////////////////////////////
144 void setPlayingOffset(Time timeOffset);
145
146 ////////////////////////////////////////////////////////////
147 /// \brief Get the current playing position of the stream
148 ///
149 /// \return Current playing position, from the beginning of the stream
150 ///
151 /// \see setPlayingOffset
152 ///
153 ////////////////////////////////////////////////////////////
154 Time getPlayingOffset() const;
155
156 ////////////////////////////////////////////////////////////
157 /// \brief Set whether or not the stream should loop after reaching the end
158 ///
159 /// If set, the stream will restart from beginning after
160 /// reaching the end and so on, until it is stopped or
161 /// setLoop(false) is called.
162 /// The default looping state for streams is false.
163 ///
164 /// \param loop True to play in loop, false to play once
165 ///
166 /// \see getLoop
167 ///
168 ////////////////////////////////////////////////////////////
169 void setLoop(bool loop);
170
171 ////////////////////////////////////////////////////////////
172 /// \brief Tell whether or not the stream is in loop mode
173 ///
174 /// \return True if the stream is looping, false otherwise
175 ///
176 /// \see setLoop
177 ///
178 ////////////////////////////////////////////////////////////
179 bool getLoop() const;
180
181protected:
182
183 enum
184 {
185 NoLoop = -1 ///< "Invalid" endSeeks value, telling us to continue uninterrupted
186 };
187
188 ////////////////////////////////////////////////////////////
189 /// \brief Default constructor
190 ///
191 /// This constructor is only meant to be called by derived classes.
192 ///
193 ////////////////////////////////////////////////////////////
194 SoundStream();
195
196 ////////////////////////////////////////////////////////////
197 /// \brief Define the audio stream parameters
198 ///
199 /// This function must be called by derived classes as soon
200 /// as they know the audio settings of the stream to play.
201 /// Any attempt to manipulate the stream (play(), ...) before
202 /// calling this function will fail.
203 /// It can be called multiple times if the settings of the
204 /// audio stream change, but only when the stream is stopped.
205 ///
206 /// \param channelCount Number of channels of the stream
207 /// \param sampleRate Sample rate, in samples per second
208 ///
209 ////////////////////////////////////////////////////////////
210 void initialize(unsigned int channelCount, unsigned int sampleRate);
211
212 ////////////////////////////////////////////////////////////
213 /// \brief Request a new chunk of audio samples from the stream source
214 ///
215 /// This function must be overridden by derived classes to provide
216 /// the audio samples to play. It is called continuously by the
217 /// streaming loop, in a separate thread.
218 /// The source can choose to stop the streaming loop at any time, by
219 /// returning false to the caller.
220 /// If you return true (i.e. continue streaming) it is important that
221 /// the returned array of samples is not empty; this would stop the stream
222 /// due to an internal limitation.
223 ///
224 /// \param data Chunk of data to fill
225 ///
226 /// \return True to continue playback, false to stop
227 ///
228 ////////////////////////////////////////////////////////////
229 virtual bool onGetData(Chunk& data) = 0;
230
231 ////////////////////////////////////////////////////////////
232 /// \brief Change the current playing position in the stream source
233 ///
234 /// This function must be overridden by derived classes to
235 /// allow random seeking into the stream source.
236 ///
237 /// \param timeOffset New playing position, relative to the beginning of the stream
238 ///
239 ////////////////////////////////////////////////////////////
240 virtual void onSeek(Time timeOffset) = 0;
241
242 ////////////////////////////////////////////////////////////
243 /// \brief Change the current playing position in the stream source to the beginning of the loop
244 ///
245 /// This function can be overridden by derived classes to
246 /// allow implementation of custom loop points. Otherwise,
247 /// it just calls onSeek(Time::Zero) and returns 0.
248 ///
249 /// \return The seek position after looping (or -1 if there's no loop)
250 ///
251 ////////////////////////////////////////////////////////////
252 virtual Int64 onLoop();
253
254private:
255
256 ////////////////////////////////////////////////////////////
257 /// \brief Function called as the entry point of the thread
258 ///
259 /// This function starts the streaming loop, and returns
260 /// only when the sound is stopped.
261 ///
262 ////////////////////////////////////////////////////////////
263 void streamData();
264
265 ////////////////////////////////////////////////////////////
266 /// \brief Fill a new buffer with audio samples, and append
267 /// it to the playing queue
268 ///
269 /// This function is called as soon as a buffer has been fully
270 /// consumed; it fills it again and inserts it back into the
271 /// playing queue.
272 ///
273 /// \param bufferNum Number of the buffer to fill (in [0, BufferCount])
274 /// \param immediateLoop Treat empty buffers as spent, and act on loops immediately
275 ///
276 /// \return True if the stream source has requested to stop, false otherwise
277 ///
278 ////////////////////////////////////////////////////////////
279 bool fillAndPushBuffer(unsigned int bufferNum, bool immediateLoop = false);
280
281 ////////////////////////////////////////////////////////////
282 /// \brief Fill the audio buffers and put them all into the playing queue
283 ///
284 /// This function is called when playing starts and the
285 /// playing queue is empty.
286 ///
287 /// \return True if the derived class has requested to stop, false otherwise
288 ///
289 ////////////////////////////////////////////////////////////
290 bool fillQueue();
291
292 ////////////////////////////////////////////////////////////
293 /// \brief Clear all the audio buffers and empty the playing queue
294 ///
295 /// This function is called when the stream is stopped.
296 ///
297 ////////////////////////////////////////////////////////////
298 void clearQueue();
299
300 enum
301 {
302 BufferCount = 3, ///< Number of audio buffers used by the streaming loop
303 BufferRetries = 2 ///< Number of retries (excluding initial try) for onGetData()
304 };
305
306 ////////////////////////////////////////////////////////////
307 // Member data
308 ////////////////////////////////////////////////////////////
309 Thread m_thread; ///< Thread running the background tasks
310 mutable Mutex m_threadMutex; ///< Thread mutex
311 Status m_threadStartState; ///< State the thread starts in (Playing, Paused, Stopped)
312 bool m_isStreaming; ///< Streaming state (true = playing, false = stopped)
313 unsigned int m_buffers[BufferCount]; ///< Sound buffers used to store temporary audio data
314 unsigned int m_channelCount; ///< Number of channels (1 = mono, 2 = stereo, ...)
315 unsigned int m_sampleRate; ///< Frequency (samples / second)
316 Uint32 m_format; ///< Format of the internal sound buffers
317 bool m_loop; ///< Loop flag (true to loop, false to play once)
318 Uint64 m_samplesProcessed; ///< Number of buffers processed since beginning of the stream
319 Int64 m_bufferSeeks[BufferCount]; ///< If buffer is an "end buffer", holds next seek position, else NoLoop. For play offset calculation.
320};
321
322} // namespace sf

329/// \class sf::SoundStream
330/// \ingroup audio
331///
332/// Unlike audio buffers (see sf::SoundBuffer), audio streams
333/// are never completely loaded in memory. Instead, the audio
334/// data is acquired continuously while the stream is playing.
335/// This behavior allows to play a sound with no loading delay,
336/// and keeps the memory consumption very low.
337///
338/// Sound sources that need to be streamed are usually big files
339/// (compressed audio musics that would eat hundreds of MB in memory)
340/// or files that would take a lot of time to be received
341/// (sounds played over the network).
342///
343/// sf::SoundStream is a base class that doesn't care about the
344/// stream source, which is left to the derived class. SFML provides
345/// a built-in specialization for big files (see sf::Music).
346/// No network stream source is provided, but you can write your own
347/// by combining this class with the network module.
348///
349/// A derived class has to override two virtual functions:
350/// \li onGetData fills a new chunk of audio data to be played
351/// \li onSeek changes the current playing position in the source
352///
353/// It is important to note that each SoundStream is played in its
354/// own separate thread, so that the streaming loop doesn't block the
355/// rest of the program. In particular, the OnGetData and OnSeek
356/// virtual functions may sometimes be called from this separate thread.
357/// It is important to keep this in mind, because you may have to take
358/// care of synchronization issues if you share data between threads.
359///
360/// Usage example:
361/// \code
362/// class CustomStream : public sf::SoundStream
363/// {
364/// public:
365///
366/// bool open(const std::string& location)
367/// {
368/// // Open the source and get audio settings
369/// ...
370/// unsigned int channelCount = ...;
371/// unsigned int sampleRate = ...;
372///
373/// // Initialize the stream -- important!
374/// initialize(channelCount, sampleRate);
375/// }
376///
377/// private:
378///
379/// virtual bool onGetData(Chunk& data)
380/// {
381/// // Fill the chunk with audio data from the stream source
382/// // (note: must not be empty if you want to continue playing)
383/// data.samples = ...;
384/// data.sampleCount = ...;
385///
386/// // Return true to continue playing
387/// return true;
388/// }
389///
390/// virtual void onSeek(Uint32 timeOffset)
391/// {
392/// // Change the current position in the stream source
393/// ...
394/// }
395/// }
396///
397/// // Usage
398/// CustomStream stream;
399/// stream.open("path/to/stream");
400/// stream.play();
401/// \endcode
402///
403/// \see sf::Music
404///
