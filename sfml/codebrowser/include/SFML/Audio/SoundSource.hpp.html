25#ifndef SFML_SOUNDSOURCE_HPP
26#define SFML_SOUNDSOURCE_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Audio/Export.hpp>
32#include <SFML/Audio/AlResource.hpp>
33#include <SFML/System/Vector3.hpp>
34
35
36namespace sf
37{
38////////////////////////////////////////////////////////////
39/// \brief Base class defining a sound's properties
40///
41////////////////////////////////////////////////////////////
42class SFML_AUDIO_API SoundSource : AlResource
43{
44public:
45
46 ////////////////////////////////////////////////////////////
47 /// \brief Enumeration of the sound source states
48 ///
49 ////////////////////////////////////////////////////////////
50 enum Status
51 {
52 Stopped, ///< Sound is not playing
53 Paused, ///< Sound is paused
54 Playing ///< Sound is playing
55 };
56
57 ////////////////////////////////////////////////////////////
58 /// \brief Copy constructor
59 ///
60 /// \param copy Instance to copy
61 ///
62 ////////////////////////////////////////////////////////////
63 SoundSource(const SoundSource& copy);
64
65 ////////////////////////////////////////////////////////////
66 /// \brief Destructor
67 ///
68 ////////////////////////////////////////////////////////////
69 virtual ~SoundSource();
70
71 ////////////////////////////////////////////////////////////
72 /// \brief Set the pitch of the sound
73 ///
74 /// The pitch represents the perceived fundamental frequency
75 /// of a sound; thus you can make a sound more acute or grave
76 /// by changing its pitch. A side effect of changing the pitch
77 /// is to modify the playing speed of the sound as well.
78 /// The default value for the pitch is 1.
79 ///
80 /// \param pitch New pitch to apply to the sound
81 ///
82 /// \see getPitch
83 ///
84 ////////////////////////////////////////////////////////////
85 void setPitch(float pitch);
86
87 ////////////////////////////////////////////////////////////
88 /// \brief Set the volume of the sound
89 ///
90 /// The volume is a value between 0 (mute) and 100 (full volume).
91 /// The default value for the volume is 100.
92 ///
93 /// \param volume Volume of the sound
94 ///
95 /// \see getVolume
96 ///
97 ////////////////////////////////////////////////////////////
98 void setVolume(float volume);
99
100 ////////////////////////////////////////////////////////////
101 /// \brief Set the 3D position of the sound in the audio scene
102 ///
103 /// Only sounds with one channel (mono sounds) can be
104 /// spatialized.
105 /// The default position of a sound is (0, 0, 0).
106 ///
107 /// \param x X coordinate of the position of the sound in the scene
108 /// \param y Y coordinate of the position of the sound in the scene
109 /// \param z Z coordinate of the position of the sound in the scene
110 ///
111 /// \see getPosition
112 ///
113 ////////////////////////////////////////////////////////////
114 void setPosition(float x, float y, float z);
115
116 ////////////////////////////////////////////////////////////
117 /// \brief Set the 3D position of the sound in the audio scene
118 ///
119 /// Only sounds with one channel (mono sounds) can be
120 /// spatialized.
121 /// The default position of a sound is (0, 0, 0).
122 ///
123 /// \param position Position of the sound in the scene
124 ///
125 /// \see getPosition
126 ///
127 ////////////////////////////////////////////////////////////
128 void setPosition(const Vector3f& position);
129
130 ////////////////////////////////////////////////////////////
131 /// \brief Make the sound's position relative to the listener or absolute
132 ///
133 /// Making a sound relative to the listener will ensure that it will always
134 /// be played the same way regardless of the position of the listener.
135 /// This can be useful for non-spatialized sounds, sounds that are
136 /// produced by the listener, or sounds attached to it.
137 /// The default value is false (position is absolute).
138 ///
139 /// \param relative True to set the position relative, false to set it absolute
140 ///
141 /// \see isRelativeToListener
142 ///
143 ////////////////////////////////////////////////////////////
144 void setRelativeToListener(bool relative);
145
146 ////////////////////////////////////////////////////////////
147 /// \brief Set the minimum distance of the sound
148 ///
149 /// The "minimum distance" of a sound is the maximum
150 /// distance at which it is heard at its maximum volume. Further
151 /// than the minimum distance, it will start to fade out according
152 /// to its attenuation factor. A value of 0 ("inside the head
153 /// of the listener") is an invalid value and is forbidden.
154 /// The default value of the minimum distance is 1.
155 ///
156 /// \param distance New minimum distance of the sound
157 ///
158 /// \see getMinDistance, setAttenuation
159 ///
160 ////////////////////////////////////////////////////////////
161 void setMinDistance(float distance);
162
163 ////////////////////////////////////////////////////////////
164 /// \brief Set the attenuation factor of the sound
165 ///
166 /// The attenuation is a multiplicative factor which makes
167 /// the sound more or less loud according to its distance
168 /// from the listener. An attenuation of 0 will produce a
169 /// non-attenuated sound, i.e. its volume will always be the same
170 /// whether it is heard from near or from far. On the other hand,
171 /// an attenuation value such as 100 will make the sound fade out
172 /// very quickly as it gets further from the listener.
173 /// The default value of the attenuation is 1.
174 ///
175 /// \param attenuation New attenuation factor of the sound
176 ///
177 /// \see getAttenuation, setMinDistance
178 ///
179 ////////////////////////////////////////////////////////////
180 void setAttenuation(float attenuation);
181
182 ////////////////////////////////////////////////////////////
183 /// \brief Get the pitch of the sound
184 ///
185 /// \return Pitch of the sound
186 ///
187 /// \see setPitch
188 ///
189 ////////////////////////////////////////////////////////////
190 float getPitch() const;
191
192 ////////////////////////////////////////////////////////////
193 /// \brief Get the volume of the sound
194 ///
195 /// \return Volume of the sound, in the range [0, 100]
196 ///
197 /// \see setVolume
198 ///
199 ////////////////////////////////////////////////////////////
200 float getVolume() const;
201
202 ////////////////////////////////////////////////////////////
203 /// \brief Get the 3D position of the sound in the audio scene
204 ///
205 /// \return Position of the sound
206 ///
207 /// \see setPosition
208 ///
209 ////////////////////////////////////////////////////////////
210 Vector3f getPosition() const;
211
212 ////////////////////////////////////////////////////////////
213 /// \brief Tell whether the sound's position is relative to the
214 /// listener or is absolute
215 ///
216 /// \return True if the position is relative, false if it's absolute
217 ///
218 /// \see setRelativeToListener
219 ///
220 ////////////////////////////////////////////////////////////
221 bool isRelativeToListener() const;
222
223 ////////////////////////////////////////////////////////////
224 /// \brief Get the minimum distance of the sound
225 ///
226 /// \return Minimum distance of the sound
227 ///
228 /// \see setMinDistance, getAttenuation
229 ///
230 ////////////////////////////////////////////////////////////
231 float getMinDistance() const;
232
233 ////////////////////////////////////////////////////////////
234 /// \brief Get the attenuation factor of the sound
235 ///
236 /// \return Attenuation factor of the sound
237 ///
238 /// \see setAttenuation, getMinDistance
239 ///
240 ////////////////////////////////////////////////////////////
241 float getAttenuation() const;
242
243 ////////////////////////////////////////////////////////////
244 /// \brief Overload of assignment operator
245 ///
246 /// \param right Instance to assign
247 ///
248 /// \return Reference to self
249 ///
250 ////////////////////////////////////////////////////////////
251 SoundSource& operator =(const SoundSource& right);
252
253protected:
254
255 ////////////////////////////////////////////////////////////
256 /// \brief Default constructor
257 ///
258 /// This constructor is meant to be called by derived classes only.
259 ///
260 ////////////////////////////////////////////////////////////
261 SoundSource();
262
263 ////////////////////////////////////////////////////////////
264 /// \brief Get the current status of the sound (stopped, paused, playing)
265 ///
266 /// \return Current status of the sound
267 ///
268 ////////////////////////////////////////////////////////////
269 Status getStatus() const;
270
271 ////////////////////////////////////////////////////////////
272 // Member data
273 ////////////////////////////////////////////////////////////
274 unsigned int m_source; ///< OpenAL source identifier
275};
276
277} // namespace sf
278
279
280#endif // SFML_SOUNDSOURCE_HPP

283////////////////////////////////////////////////////////////
284/// \class sf::SoundSource
285/// \ingroup audio
286///
287/// sf::SoundSource is not meant to be used directly, it
288/// only serves as a common base for all audio objects
289/// that can live in the audio environment.
290///
291/// It defines several properties for the sound: pitch,
292/// volume, position, attenuation, etc. All of them can be
293/// changed at any time with no impact on performances.
294///
295/// \see sf::Sound, sf::SoundStream
296///
