36namespace sf
37{
38class InputStream;
39class SoundFileReader;
40class SoundFileWriter;

42////////////////////////////////////////////////////////////
43/// \brief Manages and instantiates sound file readers and writers
44///
45////////////////////////////////////////////////////////////
46class SFML_AUDIO_API SoundFileFactory
47{
48public:
49
50 ////////////////////////////////////////////////////////////
51 /// \brief Register a new reader
52 ///
53 /// \see unregisterReader
54 ///
55 ////////////////////////////////////////////////////////////
56 template <typename T>
57 static void registerReader();
58
59 ////////////////////////////////////////////////////////////
60 /// \brief Unregister a reader
61 ///
62 /// \see registerReader
63 ///
64 ////////////////////////////////////////////////////////////
65 template <typename T>
66 static void unregisterReader();
67
68 ////////////////////////////////////////////////////////////
69 /// \brief Register a new writer
70 ///
71 /// \see unregisterWriter
72 ///
73 ////////////////////////////////////////////////////////////
74 template <typename T>
75 static void registerWriter();
76
77 ////////////////////////////////////////////////////////////
78 /// \brief Unregister a writer
79 ///
80 /// \see registerWriter
81 ///
82 ////////////////////////////////////////////////////////////
83 template <typename T>
84 static void unregisterWriter();
85
86 ////////////////////////////////////////////////////////////
87 /// \brief Instantiate the right reader for the given file on disk
88 ///
89 /// It's up to the caller to release the returned reader
90 ///
91 /// \param filename Path of the sound file
92 ///
93 /// \return A new sound file reader that can read the given file, or null if no reader can handle it
94 ///
95 /// \see createReaderFromMemory, createReaderFromStream
96 ///
97 ////////////////////////////////////////////////////////////
98 static SoundFileReader* createReaderFromFilename(const std::string& filename);
99
100 ////////////////////////////////////////////////////////////
101 /// \brief Instantiate the right codec for the given file in memory
102 ///
103 /// It's up to the caller to release the returned reader
104 ///
105 /// \param data Pointer to the file data in memory
106 /// \param sizeInBytes Total size of the file data, in bytes
107 ///
108 /// \return A new sound file codec that can read the given file, or null if no codec can handle it
109 ///
110 /// \see createReaderFromFilename, createReaderFromStream
111 ///
112 ////////////////////////////////////////////////////////////
113 static SoundFileReader* createReaderFromMemory(const void* data, std::size_t sizeInBytes);
114
115 ////////////////////////////////////////////////////////////
116 /// \brief Instantiate the right codec for the given file in stream
117 ///
118 /// It's up to the caller to release the returned reader
119 ///
120 /// \param stream Source stream to read from
121 ///
122 /// \return A new sound file codec that can read the given file, or null if no codec can handle it
123 ///
124 /// \see createReaderFromFilename, createReaderFromMemory
125 ///
126 ////////////////////////////////////////////////////////////
127 static SoundFileReader* createReaderFromStream(InputStream& stream);
128
129 ////////////////////////////////////////////////////////////
130 /// \brief Instantiate the right writer for the given file on disk
131 ///
132 /// It's up to the caller to release the returned writer
133 ///
134 /// \param filename Path of the sound file
135 ///
136 /// \return A new sound file writer that can write given file, or null if no writer can handle it
137 ///
138 ////////////////////////////////////////////////////////////
139 static SoundFileWriter* createWriterFromFilename(const std::string& filename);
140
141private:
142
143 ////////////////////////////////////////////////////////////
144 // Types
145 ////////////////////////////////////////////////////////////
146 struct ReaderFactory
147 {
148 bool (*check)(InputStream&);
149 SoundFileReader* (*create)();
150 };
151 typedef std::vector<ReaderFactory> ReaderFactoryArray;
152
153 struct WriterFactory
154 {
155 bool (*check)(const std::string&);
156 SoundFileWriter* (*create)();
157 };
158 typedef std::vector<WriterFactory> WriterFactoryArray;
159
160 ////////////////////////////////////////////////////////////
161 // Static member data
162 ////////////////////////////////////////////////////////////
163 static ReaderFactoryArray s_readers; ///< List of all registered readers
164 static WriterFactoryArray s_writers; ///< List of all registered writers
165};
166
167} // namespace sf

174////////////////////////////////////////////////////////////
175/// \class sf::SoundFileFactory
176/// \ingroup audio
177///
178/// This class is where all the sound file readers and writers are
179/// registered. You should normally only need to use its registration
180/// and unregistration functions; readers/writers creation and manipulation
181/// are wrapped into the higher-level classes sf::InputSoundFile and
182/// sf::OutputSoundFile.
183///
184/// To register a new reader (writer) use the sf::SoundFileFactory::registerReader
185/// (registerWriter) static function. You don't have to call the unregisterReader
186/// (unregisterWriter) function, unless you want to unregister a format before your
187/// application ends (typically, when a plugin is unloaded).
188///
189/// Usage example:
190/// \code
191/// sf::SoundFileFactory::registerReader<MySoundFileReader>();
192/// sf::SoundFileFactory::registerWriter<MySoundFileWriter>();
193/// \endcode
194///
195/// \see sf::InputSoundFile, sf::OutputSoundFile, sf::SoundFileReader, sf::SoundFileWriter
196///
