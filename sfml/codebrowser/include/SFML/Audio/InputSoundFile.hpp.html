31#include <SFML/Audio/Export.hpp>

33#include <SFML/System/Time.hpp>

35#include <algorithm>

38namespace sf
39{
40class InputStream;
41class SoundFileReader;
42
43////////////////////////////////////////////////////////////
44/// \brief Provide read access to sound files
45///
46////////////////////////////////////////////////////////////
47class SFML_AUDIO_API InputSoundFile : NonCopyable
48{
49public:
50
51 ////////////////////////////////////////////////////////////
52 /// \brief Default constructor
53 ///
54 ////////////////////////////////////////////////////////////
55 InputSoundFile();
56
57 ////////////////////////////////////////////////////////////
58 /// \brief Destructor
59 ///
60 ////////////////////////////////////////////////////////////
61 ~InputSoundFile();
62
63 ////////////////////////////////////////////////////////////
64 /// \brief Open a sound file from the disk for reading
65 ///
66 /// The supported audio formats are: WAV (PCM only), OGG/Vorbis, FLAC.
67 /// The supported sample sizes for FLAC and WAV are 8, 16, 24 and 32 bit.
68 ///
69 /// \param filename Path of the sound file to load
70 ///
71 /// \return True if the file was successfully opened
72 ///
73 ////////////////////////////////////////////////////////////
74 bool openFromFile(const std::string& filename);
75
76 ////////////////////////////////////////////////////////////
77 /// \brief Open a sound file in memory for reading
78 ///
79 /// The supported audio formats are: WAV (PCM only), OGG/Vorbis, FLAC.
80 /// The supported sample sizes for FLAC and WAV are 8, 16, 24 and 32 bit.
81 ///
82 /// \param data Pointer to the file data in memory
83 /// \param sizeInBytes Size of the data to load, in bytes
84 ///
85 /// \return True if the file was successfully opened
86 ///
87 ////////////////////////////////////////////////////////////
88 bool openFromMemory(const void* data, std::size_t sizeInBytes);
89
90 ////////////////////////////////////////////////////////////
91 /// \brief Open a sound file from a custom stream for reading
92 ///
93 /// The supported audio formats are: WAV (PCM only), OGG/Vorbis, FLAC.
94 /// The supported sample sizes for FLAC and WAV are 8, 16, 24 and 32 bit.
95 ///
96 /// \param stream Source stream to read from
97 ///
98 /// \return True if the file was successfully opened
99 ///
100 ////////////////////////////////////////////////////////////
101 bool openFromStream(InputStream& stream);
102
103 ////////////////////////////////////////////////////////////
104 /// \brief Get the total number of audio samples in the file
105 ///
106 /// \return Number of samples
107 ///
108 ////////////////////////////////////////////////////////////
109 Uint64 getSampleCount() const;
110
111 ////////////////////////////////////////////////////////////
112 /// \brief Get the number of channels used by the sound
113 ///
114 /// \return Number of channels (1 = mono, 2 = stereo)
115 ///
116 ////////////////////////////////////////////////////////////
117 unsigned int getChannelCount() const;
118
119 ////////////////////////////////////////////////////////////
120 /// \brief Get the sample rate of the sound
121 ///
122 /// \return Sample rate, in samples per second
123 ///
124 ////////////////////////////////////////////////////////////
125 unsigned int getSampleRate() const;
126
127 ////////////////////////////////////////////////////////////
128 /// \brief Get the total duration of the sound file
129 ///
130 /// This function is provided for convenience, the duration is
131 /// deduced from the other sound file attributes.
132 ///
133 /// \return Duration of the sound file
134 ///
135 ////////////////////////////////////////////////////////////
136 Time getDuration() const;
137
138 ////////////////////////////////////////////////////////////
139 /// \brief Get the read offset of the file in time
140 ///
141 /// \return Time position
142 ///
143 ////////////////////////////////////////////////////////////
144 Time getTimeOffset() const;
145
146 ////////////////////////////////////////////////////////////
147 /// \brief Get the read offset of the file in samples
148 ///
149 /// \return Sample position
150 ///
151 ////////////////////////////////////////////////////////////
152 Uint64 getSampleOffset() const;
153
154 ////////////////////////////////////////////////////////////
155 /// \brief Change the current read position to the given sample offset
156 ///
157 /// This function takes a sample offset to provide maximum
158 /// precision. If you need to jump to a given time, use the
159 /// other overload.
160 ///
161 /// The sample offset takes the channels into account.
162 /// Offsets can be calculated like this:
163 /// `sampleNumber * sampleRate * channelCount`
164 /// If the given offset exceeds to total number of samples,
165 /// this function jumps to the end of the sound file.
166 ///
167 /// \param sampleOffset Index of the sample to jump to, relative to the beginning
168 ///
169 ////////////////////////////////////////////////////////////
170 void seek(Uint64 sampleOffset);
171
172 ////////////////////////////////////////////////////////////
173 /// \brief Change the current read position to the given time offset
174 ///
175 /// Using a time offset is handy but imprecise. If you need an accurate
176 /// result, consider using the overload which takes a sample offset.
177 ///
178 /// If the given time exceeds to total duration, this function jumps
179 /// to the end of the sound file.
180 ///
181 /// \param timeOffset Time to jump to, relative to the beginning
182 ///
183 ////////////////////////////////////////////////////////////
184 void seek(Time timeOffset);
185
186 ////////////////////////////////////////////////////////////
187 /// \brief Read audio samples from the open file
188 ///
189 /// \param samples Pointer to the sample array to fill
190 /// \param maxCount Maximum number of samples to read
191 ///
192 /// \return Number of samples actually read (may be less than \a maxCount)
193 ///
194 ////////////////////////////////////////////////////////////
195 Uint64 read(Int16* samples, Uint64 maxCount);
196
197private:
198
199 ////////////////////////////////////////////////////////////
200 /// \brief Close the current file
201 ///
202 ////////////////////////////////////////////////////////////
203 void close();
204
205 ////////////////////////////////////////////////////////////
206 // Member data
207 ////////////////////////////////////////////////////////////
208 SoundFileReader* m_reader; ///< Reader that handles I/O on the file's format
209 InputStream* m_stream; ///< Input stream used to access the file's data
210 bool m_streamOwned; ///< Is the stream internal or external?
211 Uint64 m_sampleOffset; ///< Sample Read Position
212 Uint64 m_sampleCount; ///< Total number of samples in the file
213 unsigned int m_channelCount; ///< Number of channels of the sound
214 unsigned int m_sampleRate; ///< Number of samples per second
215};
216
217} // namespace sf

220#endif // SFML_INPUTSOUNDFILE_HPP

223////////////////////////////////////////////////////////////
224/// \class sf::InputSoundFile
225/// \ingroup audio
226///
227/// This class decodes audio samples from a sound file. It is
228/// used internally by higher-level classes such as sf::SoundBuffer
229/// and sf::Music, but can also be useful if you want to process
230/// or analyze audio files without playing them, or if you want to
231/// implement your own version of sf::Music with more specific
232/// features.
233///
234/// Usage example:
235/// \code
236/// // Open a sound file
237/// sf::InputSoundFile file;
238/// if (!file.openFromFile("music.ogg"))
239/// /* error */;
240///
241/// // Print the sound attributes
242/// std::cout << "duration: " << file.getDuration().asSeconds() << std::endl;
243/// std::cout << "channels: " << file.getChannelCount() << std::endl;
244/// std::cout << "sample rate: " << file.getSampleRate() << std::endl;
245/// std::cout << "sample count: " << file.getSampleCount() << std::endl;
246///
247/// // Read and process batches of samples until the end of file is reached
248/// sf::Int16 samples[1024];
249/// sf::Uint64 count;
250/// do
251/// {
252/// count = file.read(samples, 1024);
253///
254/// // process, analyze, play, convert, or whatever
255/// // you want to do with the samples...
256/// }
257/// while (count > 0);
258/// \endcode
259///
260/// \see sf::SoundFileReader, sf::OutputSoundFile
261///
