39////////////////////////////////////////////////////////////
40/// \brief Abstract base class for sound file decoding
41///
42////////////////////////////////////////////////////////////
43class SFML_AUDIO_API SoundFileReader
44{
45public:
46
47 ////////////////////////////////////////////////////////////
48 /// \brief Structure holding the audio properties of a sound file
49 ///
50 ////////////////////////////////////////////////////////////
51 struct Info
52 {
53 Uint64 sampleCount; ///< Total number of samples in the file
54 unsigned int channelCount; ///< Number of channels of the sound
55 unsigned int sampleRate; ///< Samples rate of the sound, in samples per second
56 };
57
58 ////////////////////////////////////////////////////////////
59 /// \brief Virtual destructor
60 ///
61 ////////////////////////////////////////////////////////////
62 virtual ~SoundFileReader() {}
63
64 ////////////////////////////////////////////////////////////
65 /// \brief Open a sound file for reading
66 ///
67 /// The provided stream reference is valid as long as the
68 /// SoundFileReader is alive, so it is safe to use/store it
69 /// during the whole lifetime of the reader.
70 ///
71 /// \param stream Source stream to read from
72 /// \param info Structure to fill with the properties of the loaded sound
73 ///
74 /// \return True if the file was successfully opened
75 ///
76 ////////////////////////////////////////////////////////////
77 virtual bool open(InputStream& stream, Info& info) = 0;
78
79 ////////////////////////////////////////////////////////////
80 /// \brief Change the current read position to the given sample offset
81 ///
82 /// The sample offset takes the channels into account.
83 /// Offsets can be calculated like this:
84 /// `sampleNumber * sampleRate * channelCount`
85 /// If the given offset exceeds to total number of samples,
86 /// this function must jump to the end of the file.
87 ///
88 /// \param sampleOffset Index of the sample to jump to, relative to the beginning
89 ///
90 ////////////////////////////////////////////////////////////
91 virtual void seek(Uint64 sampleOffset) = 0;
92
93 ////////////////////////////////////////////////////////////
94 /// \brief Read audio samples from the open file
95 ///
96 /// \param samples Pointer to the sample array to fill
97 /// \param maxCount Maximum number of samples to read
98 ///
99 /// \return Number of samples actually read (may be less than \a maxCount)
100 ///
101 ////////////////////////////////////////////////////////////
102 virtual Uint64 read(Int16* samples, Uint64 maxCount) = 0;
103};
104
105} // namespace sf

111////////////////////////////////////////////////////////////
112/// \class sf::SoundFileReader
113/// \ingroup audio
114///
115/// This class allows users to read audio file formats not natively
116/// supported by SFML, and thus extend the set of supported readable
117/// audio formats.
118///
119/// A valid sound file reader must override the open, seek and write functions,
120/// as well as providing a static check function; the latter is used by
121/// SFML to find a suitable writer for a given input file.
122///
123/// To register a new reader, use the sf::SoundFileFactory::registerReader
124/// template function.
125///
126/// Usage example:
127/// \code
128/// class MySoundFileReader : public sf::SoundFileReader
129/// {
130/// public:
131///
132/// static bool check(sf::InputStream& stream)
133/// {
134/// // typically, read the first few header bytes and check fields that identify the format
135/// // return true if the reader can handle the format
136/// }
137///
138/// virtual bool open(sf::InputStream& stream, Info& info)
139/// {
140/// // read the sound file header and fill the sound attributes
141/// // (channel count, sample count and sample rate)
142/// // return true on success
143/// }
144///
145/// virtual void seek(sf::Uint64 sampleOffset)
146/// {
147/// // advance to the sampleOffset-th sample from the beginning of the sound
148/// }
149///
150/// virtual sf::Uint64 read(sf::Int16* samples, sf::Uint64 maxCount)
151/// {
152/// // read up to 'maxCount' samples into the 'samples' array,
153/// // convert them (for example from normalized float) if they are not stored
154/// // as 16-bits signed integers in the file
155/// // return the actual number of samples read
156/// }
157/// };
158///
159/// sf::SoundFileFactory::registerReader<MySoundFileReader>();
160/// \endcode
161///
162/// \see sf::InputSoundFile, sf::SoundFileFactory, sf::SoundFileWriter
163///
