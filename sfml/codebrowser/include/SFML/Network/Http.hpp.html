50 ////////////////////////////////////////////////////////////
51 /// \brief Define a HTTP request
52 ///
53 ////////////////////////////////////////////////////////////
54 class SFML_NETWORK_API Request
55 {
56 public:
57
58 ////////////////////////////////////////////////////////////
59 /// \brief Enumerate the available HTTP methods for a request
60 ///
61 ////////////////////////////////////////////////////////////
62 enum Method
63 {
64 Get, ///< Request in get mode, standard method to retrieve a page
65 Post, ///< Request in post mode, usually to send data to a page
66 Head, ///< Request a page's header only
67 Put, ///< Request in put mode, useful for a REST API
68 Delete ///< Request in delete mode, useful for a REST API
69 };
70
71 ////////////////////////////////////////////////////////////
72 /// \brief Default constructor
73 ///
74 /// This constructor creates a GET request, with the root
75 /// URI ("/") and an empty body.
76 ///
77 /// \param uri Target URI
78 /// \param method Method to use for the request
79 /// \param body Content of the request's body
80 ///
81 ////////////////////////////////////////////////////////////
82 Request(const std::string& uri = "/", Method method = Get, const std::string& body = "");
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Set the value of a field
86 ///
87 /// The field is created if it doesn't exist. The name of
88 /// the field is case-insensitive.
89 /// By default, a request doesn't contain any field (but the
90 /// mandatory fields are added later by the HTTP client when
91 /// sending the request).
92 ///
93 /// \param field Name of the field to set
94 /// \param value Value of the field
95 ///
96 ////////////////////////////////////////////////////////////
97 void setField(const std::string& field, const std::string& value);
98
99 ////////////////////////////////////////////////////////////
100 /// \brief Set the request method
101 ///
102 /// See the Method enumeration for a complete list of all
103 /// the availale methods.
104 /// The method is Http::Request::Get by default.
105 ///
106 /// \param method Method to use for the request
107 ///
108 ////////////////////////////////////////////////////////////
109 void setMethod(Method method);
110
111 ////////////////////////////////////////////////////////////
112 /// \brief Set the requested URI
113 ///
114 /// The URI is the resource (usually a web page or a file)
115 /// that you want to get or post.
116 /// The URI is "/" (the root page) by default.
117 ///
118 /// \param uri URI to request, relative to the host
119 ///
120 ////////////////////////////////////////////////////////////
121 void setUri(const std::string& uri);
122
123 ////////////////////////////////////////////////////////////
124 /// \brief Set the HTTP version for the request
125 ///
126 /// The HTTP version is 1.0 by default.
127 ///
128 /// \param major Major HTTP version number
129 /// \param minor Minor HTTP version number
130 ///
131 ////////////////////////////////////////////////////////////
132 void setHttpVersion(unsigned int major, unsigned int minor);
133
134 ////////////////////////////////////////////////////////////
135 /// \brief Set the body of the request
136 ///
137 /// The body of a request is optional and only makes sense
138 /// for POST requests. It is ignored for all other methods.
139 /// The body is empty by default.
140 ///
141 /// \param body Content of the body
142 ///
143 ////////////////////////////////////////////////////////////
144 void setBody(const std::string& body);
145
146 private:
147
148 friend class Http;
149
150 ////////////////////////////////////////////////////////////
151 /// \brief Prepare the final request to send to the server
152 ///
153 /// This is used internally by Http before sending the
154 /// request to the web server.
155 ///
156 /// \return String containing the request, ready to be sent
157 ///
158 ////////////////////////////////////////////////////////////
159 std::string prepare() const;
160
161 ////////////////////////////////////////////////////////////
162 /// \brief Check if the request defines a field
163 ///
164 /// This function uses case-insensitive comparisons.
165 ///
166 /// \param field Name of the field to test
167 ///
168 /// \return True if the field exists, false otherwise
169 ///
170 ////////////////////////////////////////////////////////////
171 bool hasField(const std::string& field) const;
172
173 ////////////////////////////////////////////////////////////
174 // Types
175 ////////////////////////////////////////////////////////////
176 typedef std::map<std::string, std::string> FieldTable;
177
178 ////////////////////////////////////////////////////////////
179 // Member data
180 ////////////////////////////////////////////////////////////
181 FieldTable m_fields; ///< Fields of the header associated to their value
182 Method m_method; ///< Method to use for the request
183 std::string m_uri; ///< Target URI of the request
184 unsigned int m_majorVersion; ///< Major HTTP version
185 unsigned int m_minorVersion; ///< Minor HTTP version
186 std::string m_body; ///< Body of the request
187 };

189 ////////////////////////////////////////////////////////////
190 /// \brief Define a HTTP response
191 ///
192 ////////////////////////////////////////////////////////////
193 class SFML_NETWORK_API Response
194 {
195 public:
196
197 ////////////////////////////////////////////////////////////
198 /// \brief Enumerate all the valid status codes for a response
199 ///
200 ////////////////////////////////////////////////////////////
201 enum Status
202 {
203 // 2xx: success
204 Ok = 200, ///< Most common code returned when operation was successful
205 Created = 201, ///< The resource has successfully been created
206 Accepted = 202, ///< The request has been accepted, but will be processed later by the server
207 NoContent = 204, ///< The server didn't send any data in return
208 ResetContent = 205, ///< The server informs the client that it should clear the view (form) that caused the request to be sent
209 PartialContent = 206, ///< The server has sent a part of the resource, as a response to a partial GET request
210
211 // 3xx: redirection
212 MultipleChoices = 300, ///< The requested page can be accessed from several locations
213 MovedPermanently = 301, ///< The requested page has permanently moved to a new location
214 MovedTemporarily = 302, ///< The requested page has temporarily moved to a new location
215 NotModified = 304, ///< For conditional requests, means the requested page hasn't changed and doesn't need to be refreshed
216
217 // 4xx: client error
218 BadRequest = 400, ///< The server couldn't understand the request (syntax error)
219 Unauthorized = 401, ///< The requested page needs an authentication to be accessed
220 Forbidden = 403, ///< The requested page cannot be accessed at all, even with authentication
221 NotFound = 404, ///< The requested page doesn't exist
222 RangeNotSatisfiable = 407, ///< The server can't satisfy the partial GET request (with a "Range" header field)
223
224 // 5xx: server error
225 InternalServerError = 500, ///< The server encountered an unexpected error
226 NotImplemented = 501, ///< The server doesn't implement a requested feature
227 BadGateway = 502, ///< The gateway server has received an error from the source server
228 ServiceNotAvailable = 503, ///< The server is temporarily unavailable (overloaded, in maintenance, ...)
229 GatewayTimeout = 504, ///< The gateway server couldn't receive a response from the source server
230 VersionNotSupported = 505, ///< The server doesn't support the requested HTTP version
231
232 // 10xx: SFML custom codes
233 InvalidResponse = 1000, ///< Response is not a valid HTTP one
234 ConnectionFailed = 1001 ///< Connection with server failed
235 };
236
237 ////////////////////////////////////////////////////////////
238 /// \brief Default constructor
239 ///
240 /// Constructs an empty response.
241 ///
242 ////////////////////////////////////////////////////////////
243 Response();
244
245 ////////////////////////////////////////////////////////////
246 /// \brief Get the value of a field
247 ///
248 /// If the field \a field is not found in the response header,
249 /// the empty string is returned. This function uses
250 /// case-insensitive comparisons.
251 ///
252 /// \param field Name of the field to get
253 ///
254 /// \return Value of the field, or empty string if not found
255 ///
256 ////////////////////////////////////////////////////////////
257 const std::string& getField(const std::string& field) const;
258
259 ////////////////////////////////////////////////////////////
260 /// \brief Get the response status code
261 ///
262 /// The status code should be the first thing to be checked
263 /// after receiving a response, it defines whether it is a
264 /// success, a failure or anything else (see the Status
265 /// enumeration).
266 ///
267 /// \return Status code of the response
268 ///
269 ////////////////////////////////////////////////////////////
270 Status getStatus() const;
271
272 ////////////////////////////////////////////////////////////
273 /// \brief Get the major HTTP version number of the response
274 ///
275 /// \return Major HTTP version number
276 ///
277 /// \see getMinorHttpVersion
278 ///
279 ////////////////////////////////////////////////////////////
280 unsigned int getMajorHttpVersion() const;
281
282 ////////////////////////////////////////////////////////////
283 /// \brief Get the minor HTTP version number of the response
284 ///
285 /// \return Minor HTTP version number
286 ///
287 /// \see getMajorHttpVersion
288 ///
289 ////////////////////////////////////////////////////////////
290 unsigned int getMinorHttpVersion() const;
291
292 ////////////////////////////////////////////////////////////
293 /// \brief Get the body of the response
294 ///
295 /// The body of a response may contain:
296 /// \li the requested page (for GET requests)
297 /// \li a response from the server (for POST requests)
298 /// \li nothing (for HEAD requests)
299 /// \li an error message (in case of an error)
300 ///
301 /// \return The response body
302 ///
303 ////////////////////////////////////////////////////////////
304 const std::string& getBody() const;
305
306 private:
307
308 friend class Http;
309
310 ////////////////////////////////////////////////////////////
311 /// \brief Construct the header from a response string
312 ///
313 /// This function is used by Http to build the response
314 /// of a request.
315 ///
316 /// \param data Content of the response to parse
317 ///
318 ////////////////////////////////////////////////////////////
319 void parse(const std::string& data);
320
321
322 ////////////////////////////////////////////////////////////
323 /// \brief Read values passed in the answer header
324 ///
325 /// This function is used by Http to extract values passed
326 /// in the response.
327 ///
328 /// \param in String stream containing the header values
329 ///
330 ////////////////////////////////////////////////////////////
331 void parseFields(std::istream &in);
332
333 ////////////////////////////////////////////////////////////
334 // Types
335 ////////////////////////////////////////////////////////////
336 typedef std::map<std::string, std::string> FieldTable;
337
338 ////////////////////////////////////////////////////////////
339 // Member data
340 ////////////////////////////////////////////////////////////
341 FieldTable m_fields; ///< Fields of the header
342 Status m_status; ///< Status code
343 unsigned int m_majorVersion; ///< Major HTTP version
344 unsigned int m_minorVersion; ///< Minor HTTP version
345 std::string m_body; ///< Body of the response
346 };
347
348 ////////////////////////////////////////////////////////////
349 /// \brief Default constructor
350 ///
351 ////////////////////////////////////////////////////////////
352 Http();
353
354 ////////////////////////////////////////////////////////////
355 /// \brief Construct the HTTP client with the target host
356 ///
357 /// This is equivalent to calling setHost(host, port).
358 /// The port has a default value of 0, which means that the
359 /// HTTP client will use the right port according to the
360 /// protocol used (80 for HTTP). You should leave it like
361 /// this unless you really need a port other than the
362 /// standard one, or use an unknown protocol.
363 ///
364 /// \param host Web server to connect to
365 /// \param port Port to use for connection
366 ///
367 ////////////////////////////////////////////////////////////
368 Http(const std::string& host, unsigned short port = 0);
369
370 ////////////////////////////////////////////////////////////
371 /// \brief Set the target host
372 ///
373 /// This function just stores the host address and port, it
374 /// doesn't actually connect to it until you send a request.
375 /// The port has a default value of 0, which means that the
376 /// HTTP client will use the right port according to the
377 /// protocol used (80 for HTTP). You should leave it like
378 /// this unless you really need a port other than the
379 /// standard one, or use an unknown protocol.
380 ///
381 /// \param host Web server to connect to
382 /// \param port Port to use for connection
383 ///
384 ////////////////////////////////////////////////////////////
385 void setHost(const std::string& host, unsigned short port = 0);
386
387 ////////////////////////////////////////////////////////////
388 /// \brief Send a HTTP request and return the server's response.
389 ///
390 /// You must have a valid host before sending a request (see setHost).
391 /// Any missing mandatory header field in the request will be added
392 /// with an appropriate value.
393 /// Warning: this function waits for the server's response and may
394 /// not return instantly; use a thread if you don't want to block your
395 /// application, or use a timeout to limit the time to wait. A value
396 /// of Time::Zero means that the client will use the system default timeout
397 /// (which is usually pretty long).
398 ///
399 /// \param request Request to send
400 /// \param timeout Maximum time to wait
401 ///
402 /// \return Server's response
403 ///
404 ////////////////////////////////////////////////////////////
405 Response sendRequest(const Request& request, Time timeout = Time::Zero);
406
407private:
408
409 ////////////////////////////////////////////////////////////
410 // Member data
411 ////////////////////////////////////////////////////////////
412 TcpSocket m_connection; ///< Connection to the host
413 IpAddress m_host; ///< Web host address
414 std::string m_hostName; ///< Web host name
415 unsigned short m_port; ///< Port used for connection with host
416};

425/// \class sf::Http
426/// \ingroup network
427///
428/// sf::Http is a very simple HTTP client that allows you
429/// to communicate with a web server. You can retrieve
430/// web pages, send data to an interactive resource,
431/// download a remote file, etc. The HTTPS protocol is
432/// not supported.
433///
434/// The HTTP client is split into 3 classes:
435/// \li sf::Http::Request
436/// \li sf::Http::Response
437/// \li sf::Http
438///
439/// sf::Http::Request builds the request that will be
440/// sent to the server. A request is made of:
441/// \li a method (what you want to do)
442/// \li a target URI (usually the name of the web page or file)
443/// \li one or more header fields (options that you can pass to the server)
444/// \li an optional body (for POST requests)
445///
446/// sf::Http::Response parse the response from the web server
447/// and provides getters to read them. The response contains:
448/// \li a status code
449/// \li header fields (that may be answers to the ones that you requested)
450/// \li a body, which contains the contents of the requested resource
451///
452/// sf::Http provides a simple function, SendRequest, to send a
453/// sf::Http::Request and return the corresponding sf::Http::Response
454/// from the server.
455///
456/// Usage example:
457/// \code
458/// // Create a new HTTP client
459/// sf::Http http;
460///
461/// // We'll work on http://www.sfml-dev.org
462/// http.setHost("http://www.sfml-dev.org");
463///
464/// // Prepare a request to get the 'features.php' page
465/// sf::Http::Request request("features.php");
466///
467/// // Send the request
468/// sf::Http::Response response = http.sendRequest(request);
469///
470/// // Check the status code and display the result
471/// sf::Http::Response::Status status = response.getStatus();
472/// if (status == sf::Http::Response::Ok)
473/// {
474/// std::cout << response.getBody() << std::endl;
475/// }
476/// else
477/// {
478/// std::cout << "Error " << status << std::endl;
479/// }
480/// \endcode
481///
