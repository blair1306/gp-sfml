42////////////////////////////////////////////////////////////
43/// \brief Specialized socket using the TCP protocol
44///
45////////////////////////////////////////////////////////////
46class SFML_NETWORK_API TcpSocket : public Socket
47{
48public:
49
50 ////////////////////////////////////////////////////////////
51 /// \brief Default constructor
52 ///
53 ////////////////////////////////////////////////////////////
54 TcpSocket();
55
56 ////////////////////////////////////////////////////////////
57 /// \brief Get the port to which the socket is bound locally
58 ///
59 /// If the socket is not connected, this function returns 0.
60 ///
61 /// \return Port to which the socket is bound
62 ///
63 /// \see connect, getRemotePort
64 ///
65 ////////////////////////////////////////////////////////////
66 unsigned short getLocalPort() const;
67
68 ////////////////////////////////////////////////////////////
69 /// \brief Get the address of the connected peer
70 ///
71 /// It the socket is not connected, this function returns
72 /// sf::IpAddress::None.
73 ///
74 /// \return Address of the remote peer
75 ///
76 /// \see getRemotePort
77 ///
78 ////////////////////////////////////////////////////////////
79 IpAddress getRemoteAddress() const;
80
81 ////////////////////////////////////////////////////////////
82 /// \brief Get the port of the connected peer to which
83 /// the socket is connected
84 ///
85 /// If the socket is not connected, this function returns 0.
86 ///
87 /// \return Remote port to which the socket is connected
88 ///
89 /// \see getRemoteAddress
90 ///
91 ////////////////////////////////////////////////////////////
92 unsigned short getRemotePort() const;
93
94 ////////////////////////////////////////////////////////////
95 /// \brief Connect the socket to a remote peer
96 ///
97 /// In blocking mode, this function may take a while, especially
98 /// if the remote peer is not reachable. The last parameter allows
99 /// you to stop trying to connect after a given timeout.
100 /// If the socket was previously connected, it is first disconnected.
101 ///
102 /// \param remoteAddress Address of the remote peer
103 /// \param remotePort Port of the remote peer
104 /// \param timeout Optional maximum time to wait
105 ///
106 /// \return Status code
107 ///
108 /// \see disconnect
109 ///
110 ////////////////////////////////////////////////////////////
111 Status connect(const IpAddress& remoteAddress, unsigned short remotePort, Time timeout = Time::Zero);
112
113 ////////////////////////////////////////////////////////////
114 /// \brief Disconnect the socket from its remote peer
115 ///
116 /// This function gracefully closes the connection. If the
117 /// socket is not connected, this function has no effect.
118 ///
119 /// \see connect
120 ///
121 ////////////////////////////////////////////////////////////
122 void disconnect();
123
124 ////////////////////////////////////////////////////////////
125 /// \brief Send raw data to the remote peer
126 ///
127 /// To be able to handle partial sends over non-blocking
128 /// sockets, use the send(const void*, std::size_t, std::size_t&)
129 /// overload instead.
130 /// This function will fail if the socket is not connected.
131 ///
132 /// \param data Pointer to the sequence of bytes to send
133 /// \param size Number of bytes to send
134 ///
135 /// \return Status code
136 ///
137 /// \see receive
138 ///
139 ////////////////////////////////////////////////////////////
140 Status send(const void* data, std::size_t size);
141
142 ////////////////////////////////////////////////////////////
143 /// \brief Send raw data to the remote peer
144 ///
145 /// This function will fail if the socket is not connected.
146 ///
147 /// \param data Pointer to the sequence of bytes to send
148 /// \param size Number of bytes to send
149 /// \param sent The number of bytes sent will be written here
150 ///
151 /// \return Status code
152 ///
153 /// \see receive
154 ///
155 ////////////////////////////////////////////////////////////
156 Status send(const void* data, std::size_t size, std::size_t& sent);
157
158 ////////////////////////////////////////////////////////////
159 /// \brief Receive raw data from the remote peer
160 ///
161 /// In blocking mode, this function will wait until some
162 /// bytes are actually received.
163 /// This function will fail if the socket is not connected.
164 ///
165 /// \param data Pointer to the array to fill with the received bytes
166 /// \param size Maximum number of bytes that can be received
167 /// \param received This variable is filled with the actual number of bytes received
168 ///
169 /// \return Status code
170 ///
171 /// \see send
172 ///
173 ////////////////////////////////////////////////////////////
174 Status receive(void* data, std::size_t size, std::size_t& received);
175
176 ////////////////////////////////////////////////////////////
177 /// \brief Send a formatted packet of data to the remote peer
178 ///
179 /// In non-blocking mode, if this function returns sf::Socket::Partial,
180 /// you \em must retry sending the same unmodified packet before sending
181 /// anything else in order to guarantee the packet arrives at the remote
182 /// peer uncorrupted.
183 /// This function will fail if the socket is not connected.
184 ///
185 /// \param packet Packet to send
186 ///
187 /// \return Status code
188 ///
189 /// \see receive
190 ///
191 ////////////////////////////////////////////////////////////
192 Status send(Packet& packet);
193
194 ////////////////////////////////////////////////////////////
195 /// \brief Receive a formatted packet of data from the remote peer
196 ///
197 /// In blocking mode, this function will wait until the whole packet
198 /// has been received.
199 /// This function will fail if the socket is not connected.
200 ///
201 /// \param packet Packet to fill with the received data
202 ///
203 /// \return Status code
204 ///
205 /// \see send
206 ///
207 ////////////////////////////////////////////////////////////
208 Status receive(Packet& packet);
209
210private:
211
212 friend class TcpListener;
213
214 ////////////////////////////////////////////////////////////
215 /// \brief Structure holding the data of a pending packet
216 ///
217 ////////////////////////////////////////////////////////////
218 struct PendingPacket
219 {
220 PendingPacket();
221
222 Uint32 Size; ///< Data of packet size
223 std::size_t SizeReceived; ///< Number of size bytes received so far
224 std::vector<char> Data; ///< Data of the packet
225 };
226
227 ////////////////////////////////////////////////////////////
228 // Member data
229 ////////////////////////////////////////////////////////////
230 PendingPacket m_pendingPacket; ///< Temporary data of the packet currently being received
231};

240/// \class sf::TcpSocket
241/// \ingroup network
242///
243/// TCP is a connected protocol, which means that a TCP
244/// socket can only communicate with the host it is connected
245/// to. It can't send or receive anything if it is not connected.
246///
247/// The TCP protocol is reliable but adds a slight overhead.
248/// It ensures that your data will always be received in order
249/// and without errors (no data corrupted, lost or duplicated).
250///
251/// When a socket is connected to a remote host, you can
252/// retrieve informations about this host with the
253/// getRemoteAddress and getRemotePort functions. You can
254/// also get the local port to which the socket is bound
255/// (which is automatically chosen when the socket is connected),
256/// with the getLocalPort function.
257///
258/// Sending and receiving data can use either the low-level
259/// or the high-level functions. The low-level functions
260/// process a raw sequence of bytes, and cannot ensure that
261/// one call to Send will exactly match one call to Receive
262/// at the other end of the socket.
263///
264/// The high-level interface uses packets (see sf::Packet),
265/// which are easier to use and provide more safety regarding
266/// the data that is exchanged. You can look at the sf::Packet
267/// class to get more details about how they work.
268///
269/// The socket is automatically disconnected when it is destroyed,
270/// but if you want to explicitly close the connection while
271/// the socket instance is still alive, you can call disconnect.
272///
273/// Usage example:
274/// \code
275/// // ----- The client -----
276///
277/// // Create a socket and connect it to 192.168.1.50 on port 55001
278/// sf::TcpSocket socket;
279/// socket.connect("192.168.1.50", 55001);
280///
281/// // Send a message to the connected host
282/// std::string message = "Hi, I am a client";
283/// socket.send(message.c_str(), message.size() + 1);
284///
285/// // Receive an answer from the server
286/// char buffer[1024];
287/// std::size_t received = 0;
288/// socket.receive(buffer, sizeof(buffer), received);
289/// std::cout << "The server said: " << buffer << std::endl;
290///
291/// // ----- The server -----
292///
293/// // Create a listener to wait for incoming connections on port 55001
294/// sf::TcpListener listener;
295/// listener.listen(55001);
296///
297/// // Wait for a connection
298/// sf::TcpSocket socket;
299/// listener.accept(socket);
300/// std::cout << "New client connected: " << socket.getRemoteAddress() << std::endl;
301///
302/// // Receive a message from the client
303/// char buffer[1024];
304/// std::size_t received = 0;
305/// socket.receive(buffer, sizeof(buffer), received);
306/// std::cout << "The client said: " << buffer << std::endl;
307///
308/// // Send an answer
309/// std::string message = "Welcome, client";
310/// socket.send(message.c_str(), message.size() + 1);
311/// \endcode
312///
313/// \see sf::Socket, sf::UdpSocket, sf::Packet
314///
