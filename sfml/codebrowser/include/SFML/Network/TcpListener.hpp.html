41/// \brief Socket that listens to new TCP connections
42///
43////////////////////////////////////////////////////////////
44class SFML_NETWORK_API TcpListener : public Socket
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Default constructor
50 ///
51 ////////////////////////////////////////////////////////////
52 TcpListener();
53
54 ////////////////////////////////////////////////////////////
55 /// \brief Get the port to which the socket is bound locally
56 ///
57 /// If the socket is not listening to a port, this function
58 /// returns 0.
59 ///
60 /// \return Port to which the socket is bound
61 ///
62 /// \see listen
63 ///
64 ////////////////////////////////////////////////////////////
65 unsigned short getLocalPort() const;
66
67 ////////////////////////////////////////////////////////////
68 /// \brief Start listening for connections
69 ///
70 /// This functions makes the socket listen to the specified
71 /// port, waiting for new connections.
72 /// If the socket was previously listening to another port,
73 /// it will be stopped first and bound to the new port.
74 ///
75 /// \param port Port to listen for new connections
76 /// \param address Address of the interface to listen on
77 ///
78 /// \return Status code
79 ///
80 /// \see accept, close
81 ///
82 ////////////////////////////////////////////////////////////
83 Status listen(unsigned short port, const IpAddress& address = IpAddress::Any);
84
85 ////////////////////////////////////////////////////////////
86 /// \brief Stop listening and close the socket
87 ///
88 /// This function gracefully stops the listener. If the
89 /// socket is not listening, this function has no effect.
90 ///
91 /// \see listen
92 ///
93 ////////////////////////////////////////////////////////////
94 void close();
95
96 ////////////////////////////////////////////////////////////
97 /// \brief Accept a new connection
98 ///
99 /// If the socket is in blocking mode, this function will
100 /// not return until a connection is actually received.
101 ///
102 /// \param socket Socket that will hold the new connection
103 ///
104 /// \return Status code
105 ///
106 /// \see listen
107 ///
108 ////////////////////////////////////////////////////////////
109 Status accept(TcpSocket& socket);
110};

119////////////////////////////////////////////////////////////
120/// \class sf::TcpListener
121/// \ingroup network
122///
123/// A listener socket is a special type of socket that listens to
124/// a given port and waits for connections on that port.
125/// This is all it can do.
126///
127/// When a new connection is received, you must call accept and
128/// the listener returns a new instance of sf::TcpSocket that
129/// is properly initialized and can be used to communicate with
130/// the new client.
131///
132/// Listener sockets are specific to the TCP protocol,
133/// UDP sockets are connectionless and can therefore communicate
134/// directly. As a consequence, a listener socket will always
135/// return the new connections as sf::TcpSocket instances.
136///
137/// A listener is automatically closed on destruction, like all
138/// other types of socket. However if you want to stop listening
139/// before the socket is destroyed, you can call its close()
140/// function.
141///
142/// Usage example:
143/// \code
144/// // Create a listener socket and make it wait for new
145/// // connections on port 55001
146/// sf::TcpListener listener;
147/// listener.listen(55001);
148///
149/// // Endless loop that waits for new connections
150/// while (running)
151/// {
152/// sf::TcpSocket client;
153/// if (listener.accept(client) == sf::Socket::Done)
154/// {
155/// // A new client just connected!
156/// std::cout << "New connection received from " << client.getRemoteAddress() << std::endl;
157/// doSomethingWith(client);
158/// }
159/// }
160/// \endcode
161///
162/// \see sf::TcpSocket, sf::Socket
163///
