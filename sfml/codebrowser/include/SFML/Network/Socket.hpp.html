42/// \brief Base class for all the socket types
43///
44////////////////////////////////////////////////////////////
45class SFML_NETWORK_API Socket : NonCopyable
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 /// \brief Status codes that may be returned by socket functions
51 ///
52 ////////////////////////////////////////////////////////////
53 enum Status
54 {
55 Done, ///< The socket has sent / received the data
56 NotReady, ///< The socket is not ready to send / receive data yet
57 Partial, ///< The socket sent a part of the data
58 Disconnected, ///< The TCP socket has been disconnected
59 Error ///< An unexpected error happened
60 };
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Some special values used by sockets
64 ///
65 ////////////////////////////////////////////////////////////
66 enum
67 {
68 AnyPort = 0 ///< Special value that tells the system to pick any available port
69 };
70
71public:
72
73 ////////////////////////////////////////////////////////////
74 /// \brief Destructor
75 ///
76 ////////////////////////////////////////////////////////////
77 virtual ~Socket();
78
79 ////////////////////////////////////////////////////////////
80 /// \brief Set the blocking state of the socket
81 ///
82 /// In blocking mode, calls will not return until they have
83 /// completed their task. For example, a call to Receive in
84 /// blocking mode won't return until some data was actually
85 /// received.
86 /// In non-blocking mode, calls will always return immediately,
87 /// using the return code to signal whether there was data
88 /// available or not.
89 /// By default, all sockets are blocking.
90 ///
91 /// \param blocking True to set the socket as blocking, false for non-blocking
92 ///
93 /// \see isBlocking
94 ///
95 ////////////////////////////////////////////////////////////
96 void setBlocking(bool blocking);
97
98 ////////////////////////////////////////////////////////////
99 /// \brief Tell whether the socket is in blocking or non-blocking mode
100 ///
101 /// \return True if the socket is blocking, false otherwise
102 ///
103 /// \see setBlocking
104 ///
105 ////////////////////////////////////////////////////////////
106 bool isBlocking() const;
107
108protected:
109
110 ////////////////////////////////////////////////////////////
111 /// \brief Types of protocols that the socket can use
112 ///
113 ////////////////////////////////////////////////////////////
114 enum Type
115 {
116 Tcp, ///< TCP protocol
117 Udp ///< UDP protocol
118 };
119
120 ////////////////////////////////////////////////////////////
121 /// \brief Default constructor
122 ///
123 /// This constructor can only be accessed by derived classes.
124 ///
125 /// \param type Type of the socket (TCP or UDP)
126 ///
127 ////////////////////////////////////////////////////////////
128 Socket(Type type);
129
130 ////////////////////////////////////////////////////////////
131 /// \brief Return the internal handle of the socket
132 ///
133 /// The returned handle may be invalid if the socket
134 /// was not created yet (or already destroyed).
135 /// This function can only be accessed by derived classes.
136 ///
137 /// \return The internal (OS-specific) handle of the socket
138 ///
139 ////////////////////////////////////////////////////////////
140 SocketHandle getHandle() const;
141
142 ////////////////////////////////////////////////////////////
143 /// \brief Create the internal representation of the socket
144 ///
145 /// This function can only be accessed by derived classes.
146 ///
147 ////////////////////////////////////////////////////////////
148 void create();
149
150 ////////////////////////////////////////////////////////////
151 /// \brief Create the internal representation of the socket
152 /// from a socket handle
153 ///
154 /// This function can only be accessed by derived classes.
155 ///
156 /// \param handle OS-specific handle of the socket to wrap
157 ///
158 ////////////////////////////////////////////////////////////
159 void create(SocketHandle handle);
160
161 ////////////////////////////////////////////////////////////
162 /// \brief Close the socket gracefully
163 ///
164 /// This function can only be accessed by derived classes.
165 ///
166 ////////////////////////////////////////////////////////////
167 void close();
168
169private:
170
171 friend class SocketSelector;
172
173 ////////////////////////////////////////////////////////////
174 // Member data
175 ////////////////////////////////////////////////////////////
176 Type m_type; ///< Type of the socket (TCP or UDP)
177 SocketHandle m_socket; ///< Socket descriptor
178 bool m_isBlocking; ///< Current blocking mode of the socket
179};

187////////////////////////////////////////////////////////////
188/// \class sf::Socket
189/// \ingroup network
190///
191/// This class mainly defines internal stuff to be used by
192/// derived classes.
193///
194/// The only public features that it defines, and which
195/// is therefore common to all the socket classes, is the
196/// blocking state. All sockets can be set as blocking or
197/// non-blocking.
198///
199/// In blocking mode, socket functions will hang until
200/// the operation completes, which means that the entire
201/// program (well, in fact the current thread if you use
202/// multiple ones) will be stuck waiting for your socket
203/// operation to complete.
204///
205/// In non-blocking mode, all the socket functions will
206/// return immediately. If the socket is not ready to complete
207/// the requested operation, the function simply returns
208/// the proper status code (Socket::NotReady).
209///
210/// The default mode, which is blocking, is the one that is
211/// generally used, in combination with threads or selectors.
212/// The non-blocking mode is rather used in real-time
213/// applications that run an endless loop that can poll
214/// the socket often enough, and cannot afford blocking
215/// this loop.
216///
217/// \see sf::TcpListener, sf::TcpSocket, sf::UdpSocket
218///
