70 ////////////////////////////////////////////////////////////
71 /// \brief Status codes possibly returned by a FTP response
72 ///
73 ////////////////////////////////////////////////////////////
74 enum Status
75 {
76 // 1xx: the requested action is being initiated,
77 // expect another reply before proceeding with a new command
78 RestartMarkerReply = 110, ///< Restart marker reply
79 ServiceReadySoon = 120, ///< Service ready in N minutes
80 DataConnectionAlreadyOpened = 125, ///< Data connection already opened, transfer starting
81 OpeningDataConnection = 150, ///< File status ok, about to open data connection
82
83 // 2xx: the requested action has been successfully completed
84 Ok = 200, ///< Command ok
85 PointlessCommand = 202, ///< Command not implemented
86 SystemStatus = 211, ///< System status, or system help reply
87 DirectoryStatus = 212, ///< Directory status
88 FileStatus = 213, ///< File status
89 HelpMessage = 214, ///< Help message
90 SystemType = 215, ///< NAME system type, where NAME is an official system name from the list in the Assigned Numbers document
91 ServiceReady = 220, ///< Service ready for new user
92 ClosingConnection = 221, ///< Service closing control connection
93 DataConnectionOpened = 225, ///< Data connection open, no transfer in progress
94 ClosingDataConnection = 226, ///< Closing data connection, requested file action successful
95 EnteringPassiveMode = 227, ///< Entering passive mode
96 LoggedIn = 230, ///< User logged in, proceed. Logged out if appropriate
97 FileActionOk = 250, ///< Requested file action ok
98 DirectoryOk = 257, ///< PATHNAME created
99
100 // 3xx: the command has been accepted, but the requested action
101 // is dormant, pending receipt of further information
102 NeedPassword = 331, ///< User name ok, need password
103 NeedAccountToLogIn = 332, ///< Need account for login
104 NeedInformation = 350, ///< Requested file action pending further information
105
106 // 4xx: the command was not accepted and the requested action did not take place,
107 // but the error condition is temporary and the action may be requested again
108 ServiceUnavailable = 421, ///< Service not available, closing control connection
109 DataConnectionUnavailable = 425, ///< Can't open data connection
110 TransferAborted = 426, ///< Connection closed, transfer aborted
111 FileActionAborted = 450, ///< Requested file action not taken
112 LocalError = 451, ///< Requested action aborted, local error in processing
113 InsufficientStorageSpace = 452, ///< Requested action not taken; insufficient storage space in system, file unavailable
114
115 // 5xx: the command was not accepted and
116 // the requested action did not take place
117 CommandUnknown = 500, ///< Syntax error, command unrecognized
118 ParametersUnknown = 501, ///< Syntax error in parameters or arguments
119 CommandNotImplemented = 502, ///< Command not implemented
120 BadCommandSequence = 503, ///< Bad sequence of commands
121 ParameterNotImplemented = 504, ///< Command not implemented for that parameter
122 NotLoggedIn = 530, ///< Not logged in
123 NeedAccountToStore = 532, ///< Need account for storing files
124 FileUnavailable = 550, ///< Requested action not taken, file unavailable
125 PageTypeUnknown = 551, ///< Requested action aborted, page type unknown
126 NotEnoughMemory = 552, ///< Requested file action aborted, exceeded storage allocation
127 FilenameNotAllowed = 553, ///< Requested action not taken, file name not allowed
128
129 // 10xx: SFML custom codes
130 InvalidResponse = 1000, ///< Not part of the FTP standard, generated by SFML when a received response cannot be parsed
131 ConnectionFailed = 1001, ///< Not part of the FTP standard, generated by SFML when the low-level socket connection with the server fails
132 ConnectionClosed = 1002, ///< Not part of the FTP standard, generated by SFML when the low-level socket connection is unexpectedly closed
133 InvalidFile = 1003 ///< Not part of the FTP standard, generated by SFML when a local file cannot be read or written
134 };

261 /// \brief Connect to the specified FTP server
262 ///
263 /// The port has a default value of 21, which is the standard
264 /// port used by the FTP protocol. You shouldn't use a different
265 /// value, unless you really know what you do.
266 /// This function tries to connect to the server so it may take
267 /// a while to complete, especially if the server is not
268 /// reachable. To avoid blocking your application for too long,
269 /// you can use a timeout. The default value, Time::Zero, means that the
270 /// system timeout will be used (which is usually pretty long).
271 ///
272 /// \param server Name or address of the FTP server to connect to
273 /// \param port Port used for the connection
274 /// \param timeout Maximum time to wait
275 ///
276 /// \return Server response to the request
277 ///
278 /// \see disconnect
279 ///
280 ////////////////////////////////////////////////////////////
281 Response connect(const IpAddress& server, unsigned short port = 21, Time timeout = Time::Zero);

543/// \class sf::Ftp
544/// \ingroup network
545///
546/// sf::Ftp is a very simple FTP client that allows you
547/// to communicate with a FTP server. The FTP protocol allows
548/// you to manipulate a remote file system (list files,
549/// upload, download, create, remove, ...).
550///
551/// Using the FTP client consists of 4 parts:
552/// \li Connecting to the FTP server
553/// \li Logging in (either as a registered user or anonymously)
554/// \li Sending commands to the server
555/// \li Disconnecting (this part can be done implicitly by the destructor)
556///
557/// Every command returns a FTP response, which contains the
558/// status code as well as a message from the server. Some
559/// commands such as getWorkingDirectory() and getDirectoryListing()
560/// return additional data, and use a class derived from
561/// sf::Ftp::Response to provide this data. The most often used
562/// commands are directly provided as member functions, but it is
563/// also possible to use specific commands with the sendCommand() function.
564///
565/// Note that response statuses >= 1000 are not part of the FTP standard,
566/// they are generated by SFML when an internal error occurs.
567///
568/// All commands, especially upload and download, may take some
569/// time to complete. This is important to know if you don't want
570/// to block your application while the server is completing
571/// the task.
572///
573/// Usage example:
574/// \code
575/// // Create a new FTP client
576/// sf::Ftp ftp;
577///
578/// // Connect to the server
579/// sf::Ftp::Response response = ftp.connect("ftp://ftp.myserver.com");
580/// if (response.isOk())
581/// std::cout << "Connected" << std::endl;
582///
583/// // Log in
584/// response = ftp.login("laurent", "dF6Zm89D");
585/// if (response.isOk())
586/// std::cout << "Logged in" << std::endl;
587///
588/// // Print the working directory
589/// sf::Ftp::DirectoryResponse directory = ftp.getWorkingDirectory();
590/// if (directory.isOk())
591/// std::cout << "Working directory: " << directory.getDirectory() << std::endl;
592///
593/// // Create a new directory
594/// response = ftp.createDirectory("files");
595/// if (response.isOk())
596/// std::cout << "Created new directory" << std::endl;
597///
598/// // Upload a file to this new directory
599/// response = ftp.upload("local-path/file.txt", "files", sf::Ftp::Ascii);
600/// if (response.isOk())
601/// std::cout << "File uploaded" << std::endl;
602///
603/// // Send specific commands (here: FEAT to list supported FTP features)
604/// response = ftp.sendCommand("FEAT");
605/// if (response.isOk())
606/// std::cout << "Feature list:\n" << response.getMessage() << std::endl;
607///
608/// // Disconnect from the server (optional)
609/// ftp.disconnect();
610/// \endcode
611///
