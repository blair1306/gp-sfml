41/// \brief Encapsulate an IPv4 network address
42///
43////////////////////////////////////////////////////////////
44class SFML_NETWORK_API IpAddress
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Default constructor
50 ///
51 /// This constructor creates an empty (invalid) address
52 ///
53 ////////////////////////////////////////////////////////////
54 IpAddress();
55
56 ////////////////////////////////////////////////////////////
57 /// \brief Construct the address from a string
58 ///
59 /// Here \a address can be either a decimal address
60 /// (ex: "192.168.1.56") or a network name (ex: "localhost").
61 ///
62 /// \param address IP address or network name
63 ///
64 ////////////////////////////////////////////////////////////
65 IpAddress(const std::string& address);
66
67 ////////////////////////////////////////////////////////////
68 /// \brief Construct the address from a string
69 ///
70 /// Here \a address can be either a decimal address
71 /// (ex: "192.168.1.56") or a network name (ex: "localhost").
72 /// This is equivalent to the constructor taking a std::string
73 /// parameter, it is defined for convenience so that the
74 /// implicit conversions from literal strings to IpAddress work.
75 ///
76 /// \param address IP address or network name
77 ///
78 ////////////////////////////////////////////////////////////
79 IpAddress(const char* address);
80
81 ////////////////////////////////////////////////////////////
82 /// \brief Construct the address from 4 bytes
83 ///
84 /// Calling IpAddress(a, b, c, d) is equivalent to calling
85 /// IpAddress("a.b.c.d"), but safer as it doesn't have to
86 /// parse a string to get the address components.
87 ///
88 /// \param byte0 First byte of the address
89 /// \param byte1 Second byte of the address
90 /// \param byte2 Third byte of the address
91 /// \param byte3 Fourth byte of the address
92 ///
93 ////////////////////////////////////////////////////////////
94 IpAddress(Uint8 byte0, Uint8 byte1, Uint8 byte2, Uint8 byte3);
95
96 ////////////////////////////////////////////////////////////
97 /// \brief Construct the address from a 32-bits integer
98 ///
99 /// This constructor uses the internal representation of
100 /// the address directly. It should be used for optimization
101 /// purposes, and only if you got that representation from
102 /// IpAddress::toInteger().
103 ///
104 /// \param address 4 bytes of the address packed into a 32-bits integer
105 ///
106 /// \see toInteger
107 ///
108 ////////////////////////////////////////////////////////////
109 explicit IpAddress(Uint32 address);
110
111 ////////////////////////////////////////////////////////////
112 /// \brief Get a string representation of the address
113 ///
114 /// The returned string is the decimal representation of the
115 /// IP address (like "192.168.1.56"), even if it was constructed
116 /// from a host name.
117 ///
118 /// \return String representation of the address
119 ///
120 /// \see toInteger
121 ///
122 ////////////////////////////////////////////////////////////
123 std::string toString() const;
124
125 ////////////////////////////////////////////////////////////
126 /// \brief Get an integer representation of the address
127 ///
128 /// The returned number is the internal representation of the
129 /// address, and should be used for optimization purposes only
130 /// (like sending the address through a socket).
131 /// The integer produced by this function can then be converted
132 /// back to a sf::IpAddress with the proper constructor.
133 ///
134 /// \return 32-bits unsigned integer representation of the address
135 ///
136 /// \see toString
137 ///
138 ////////////////////////////////////////////////////////////
139 Uint32 toInteger() const;
140
141 ////////////////////////////////////////////////////////////
142 /// \brief Get the computer's local address
143 ///
144 /// The local address is the address of the computer from the
145 /// LAN point of view, i.e. something like 192.168.1.56. It is
146 /// meaningful only for communications over the local network.
147 /// Unlike getPublicAddress, this function is fast and may be
148 /// used safely anywhere.
149 ///
150 /// \return Local IP address of the computer
151 ///
152 /// \see getPublicAddress
153 ///
154 ////////////////////////////////////////////////////////////
155 static IpAddress getLocalAddress();
156
157 ////////////////////////////////////////////////////////////
158 /// \brief Get the computer's public address
159 ///
160 /// The public address is the address of the computer from the
161 /// internet point of view, i.e. something like 89.54.1.169.
162 /// It is necessary for communications over the world wide web.
163 /// The only way to get a public address is to ask it to a
164 /// distant website; as a consequence, this function depends on
165 /// both your network connection and the server, and may be
166 /// very slow. You should use it as few as possible. Because
167 /// this function depends on the network connection and on a distant
168 /// server, you may use a time limit if you don't want your program
169 /// to be possibly stuck waiting in case there is a problem; this
170 /// limit is deactivated by default.
171 ///
172 /// \param timeout Maximum time to wait
173 ///
174 /// \return Public IP address of the computer
175 ///
176 /// \see getLocalAddress
177 ///
178 ////////////////////////////////////////////////////////////
179 static IpAddress getPublicAddress(Time timeout = Time::Zero);
180
181 ////////////////////////////////////////////////////////////
182 // Static member data
183 ////////////////////////////////////////////////////////////
184 static const IpAddress None; ///< Value representing an empty/invalid address
185 static const IpAddress Any; ///< Value representing any address (0.0.0.0)
186 static const IpAddress LocalHost; ///< The "localhost" address (for connecting a computer to itself locally)
187 static const IpAddress Broadcast; ///< The "broadcast" address (for sending UDP messages to everyone on a local network)
188
189private:
190
191 friend SFML_NETWORK_API bool operator <(const IpAddress& left, const IpAddress& right);
192
193 ////////////////////////////////////////////////////////////
194 /// \brief Resolve the given address string
195 ///
196 /// \param address Address string
197 ///
198 ////////////////////////////////////////////////////////////
199 void resolve(const std::string& address);
200
201 ////////////////////////////////////////////////////////////
202 // Member data
203 ////////////////////////////////////////////////////////////
204 Uint32 m_address; ///< Address stored as an unsigned 32 bits integer
205 bool m_valid; ///< Is the address valid?
206};

302////////////////////////////////////////////////////////////
303/// \class sf::IpAddress
304/// \ingroup network
305///
306/// sf::IpAddress is a utility class for manipulating network
307/// addresses. It provides a set a implicit constructors and
308/// conversion functions to easily build or transform an IP
309/// address from/to various representations.
310///
311/// Usage example:
312/// \code
313/// sf::IpAddress a0; // an invalid address
314/// sf::IpAddress a1 = sf::IpAddress::None; // an invalid address (same as a0)
315/// sf::IpAddress a2("127.0.0.1"); // the local host address
316/// sf::IpAddress a3 = sf::IpAddress::Broadcast; // the broadcast address
317/// sf::IpAddress a4(192, 168, 1, 56); // a local address
318/// sf::IpAddress a5("my_computer"); // a local address created from a network name
319/// sf::IpAddress a6("89.54.1.169"); // a distant address
320/// sf::IpAddress a7("www.google.com"); // a distant address created from a network name
321/// sf::IpAddress a8 = sf::IpAddress::getLocalAddress(); // my address on the local network
322/// sf::IpAddress a9 = sf::IpAddress::getPublicAddress(); // my address on the internet
323/// \endcode
324///
325/// Note that sf::IpAddress currently doesn't support IPv6
326/// nor other types of network addresses.
327///
