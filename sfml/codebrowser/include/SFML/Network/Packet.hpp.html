42////////////////////////////////////////////////////////////
43/// \brief Utility class to build blocks of data to transfer
44/// over the network
45///
46////////////////////////////////////////////////////////////
47class SFML_NETWORK_API Packet
48{
49 // A bool-like type that cannot be converted to integer or pointer types
50 typedef bool (Packet::*BoolType)(std::size_t);
51
52public:
53
54 ////////////////////////////////////////////////////////////
55 /// \brief Default constructor
56 ///
57 /// Creates an empty packet.
58 ///
59 ////////////////////////////////////////////////////////////
60 Packet();
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Virtual destructor
64 ///
65 ////////////////////////////////////////////////////////////
66 virtual ~Packet();
67
68 ////////////////////////////////////////////////////////////
69 /// \brief Append data to the end of the packet
70 ///
71 /// \param data Pointer to the sequence of bytes to append
72 /// \param sizeInBytes Number of bytes to append
73 ///
74 /// \see clear
75 ///
76 ////////////////////////////////////////////////////////////
77 void append(const void* data, std::size_t sizeInBytes);
78
79 ////////////////////////////////////////////////////////////
80 /// \brief Clear the packet
81 ///
82 /// After calling Clear, the packet is empty.
83 ///
84 /// \see append
85 ///
86 ////////////////////////////////////////////////////////////
87 void clear();
88
89 ////////////////////////////////////////////////////////////
90 /// \brief Get a pointer to the data contained in the packet
91 ///
92 /// Warning: the returned pointer may become invalid after
93 /// you append data to the packet, therefore it should never
94 /// be stored.
95 /// The return pointer is NULL if the packet is empty.
96 ///
97 /// \return Pointer to the data
98 ///
99 /// \see getDataSize
100 ///
101 ////////////////////////////////////////////////////////////
102 const void* getData() const;
103
104 ////////////////////////////////////////////////////////////
105 /// \brief Get the size of the data contained in the packet
106 ///
107 /// This function returns the number of bytes pointed to by
108 /// what getData returns.
109 ///
110 /// \return Data size, in bytes
111 ///
112 /// \see getData
113 ///
114 ////////////////////////////////////////////////////////////
115 std::size_t getDataSize() const;
116
117 ////////////////////////////////////////////////////////////
118 /// \brief Tell if the reading position has reached the
119 /// end of the packet
120 ///
121 /// This function is useful to know if there is some data
122 /// left to be read, without actually reading it.
123 ///
124 /// \return True if all data was read, false otherwise
125 ///
126 /// \see operator bool
127 ///
128 ////////////////////////////////////////////////////////////
129 bool endOfPacket() const;
130
131public:
132
133 ////////////////////////////////////////////////////////////
134 /// \brief Test the validity of the packet, for reading
135 ///
136 /// This operator allows to test the packet as a boolean
137 /// variable, to check if a reading operation was successful.
138 ///
139 /// A packet will be in an invalid state if it has no more
140 /// data to read.
141 ///
142 /// This behavior is the same as standard C++ streams.
143 ///
144 /// Usage example:
145 /// \code
146 /// float x;
147 /// packet >> x;
148 /// if (packet)
149 /// {
150 /// // ok, x was extracted successfully
151 /// }
152 ///
153 /// // -- or --
154 ///
155 /// float x;
156 /// if (packet >> x)
157 /// {
158 /// // ok, x was extracted successfully
159 /// }
160 /// \endcode
161 ///
162 /// Don't focus on the return type, it's equivalent to bool but
163 /// it disallows unwanted implicit conversions to integer or
164 /// pointer types.
165 ///
166 /// \return True if last data extraction from packet was successful
167 ///
168 /// \see endOfPacket
169 ///
170 ////////////////////////////////////////////////////////////
171 operator BoolType() const;
172
173 ////////////////////////////////////////////////////////////
174 /// Overloads of operator >> to read data from the packet
175 ///
176 ////////////////////////////////////////////////////////////
177 Packet& operator >>(bool& data);
178 Packet& operator >>(Int8& data);
179 Packet& operator >>(Uint8& data);
180 Packet& operator >>(Int16& data);
181 Packet& operator >>(Uint16& data);
182 Packet& operator >>(Int32& data);
183 Packet& operator >>(Uint32& data);
184 Packet& operator >>(Int64& data);
185 Packet& operator >>(Uint64& data);
186 Packet& operator >>(float& data);
187 Packet& operator >>(double& data);
188 Packet& operator >>(char* data);
189 Packet& operator >>(std::string& data);
190 Packet& operator >>(wchar_t* data);
191 Packet& operator >>(std::wstring& data);
192 Packet& operator >>(String& data);
193
194 ////////////////////////////////////////////////////////////
195 /// Overloads of operator << to write data into the packet
196 ///
197 ////////////////////////////////////////////////////////////
198 Packet& operator <<(bool data);
199 Packet& operator <<(Int8 data);
200 Packet& operator <<(Uint8 data);
201 Packet& operator <<(Int16 data);
202 Packet& operator <<(Uint16 data);
203 Packet& operator <<(Int32 data);
204 Packet& operator <<(Uint32 data);
205 Packet& operator <<(Int64 data);
206 Packet& operator <<(Uint64 data);
207 Packet& operator <<(float data);
208 Packet& operator <<(double data);
209 Packet& operator <<(const char* data);
210 Packet& operator <<(const std::string& data);
211 Packet& operator <<(const wchar_t* data);
212 Packet& operator <<(const std::wstring& data);
213 Packet& operator <<(const String& data);
214
215protected:
216
217 friend class TcpSocket;
218 friend class UdpSocket;
219
220 ////////////////////////////////////////////////////////////
221 /// \brief Called before the packet is sent over the network
222 ///
223 /// This function can be defined by derived classes to
224 /// transform the data before it is sent; this can be
225 /// used for compression, encryption, etc.
226 /// The function must return a pointer to the modified data,
227 /// as well as the number of bytes pointed.
228 /// The default implementation provides the packet's data
229 /// without transforming it.
230 ///
231 /// \param size Variable to fill with the size of data to send
232 ///
233 /// \return Pointer to the array of bytes to send
234 ///
235 /// \see onReceive
236 ///
237 ////////////////////////////////////////////////////////////
238 virtual const void* onSend(std::size_t& size);
239
240 ////////////////////////////////////////////////////////////
241 /// \brief Called after the packet is received over the network
242 ///
243 /// This function can be defined by derived classes to
244 /// transform the data after it is received; this can be
245 /// used for decompression, decryption, etc.
246 /// The function receives a pointer to the received data,
247 /// and must fill the packet with the transformed bytes.
248 /// The default implementation fills the packet directly
249 /// without transforming the data.
250 ///
251 /// \param data Pointer to the received bytes
252 /// \param size Number of bytes
253 ///
254 /// \see onSend
255 ///
256 ////////////////////////////////////////////////////////////
257 virtual void onReceive(const void* data, std::size_t size);
258
259private:
260
261 ////////////////////////////////////////////////////////////
262 /// Disallow comparisons between packets
263 ///
264 ////////////////////////////////////////////////////////////
265 bool operator ==(const Packet& right) const;
266 bool operator !=(const Packet& right) const;
267
268 ////////////////////////////////////////////////////////////
269 /// \brief Check if the packet can extract a given number of bytes
270 ///
271 /// This function updates accordingly the state of the packet.
272 ///
273 /// \param size Size to check
274 ///
275 /// \return True if \a size bytes can be read from the packet
276 ///
277 ////////////////////////////////////////////////////////////
278 bool checkSize(std::size_t size);
279
280 ////////////////////////////////////////////////////////////
281 // Member data
282 ////////////////////////////////////////////////////////////
283 std::vector<char> m_data; ///< Data stored in the packet
284 std::size_t m_readPos; ///< Current reading position in the packet
285 std::size_t m_sendPos; ///< Current send position in the packet (for handling partial sends)
286 bool m_isValid; ///< Reading state of the packet
287};

296/// \class sf::Packet
297/// \ingroup network
298///
299/// Packets provide a safe and easy way to serialize data,
300/// in order to send it over the network using sockets
301/// (sf::TcpSocket, sf::UdpSocket).
302///
303/// Packets solve 2 fundamental problems that arise when
304/// transferring data over the network:
305/// \li data is interpreted correctly according to the endianness
306/// \li the bounds of the packet are preserved (one send == one receive)
307///
308/// The sf::Packet class provides both input and output modes.
309/// It is designed to follow the behavior of standard C++ streams,
310/// using operators >> and << to extract and insert data.
311///
312/// It is recommended to use only fixed-size types (like sf::Int32, etc.),
313/// to avoid possible differences between the sender and the receiver.
314/// Indeed, the native C++ types may have different sizes on two platforms
315/// and your data may be corrupted if that happens.
316///
317/// Usage example:
318/// \code
319/// sf::Uint32 x = 24;
320/// std::string s = "hello";
321/// double d = 5.89;
322///
323/// // Group the variables to send into a packet
324/// sf::Packet packet;
325/// packet << x << s << d;
326///
327/// // Send it over the network (socket is a valid sf::TcpSocket)
328/// socket.send(packet);
329///
330/// -----------------------------------------------------------------
331///
332/// // Receive the packet at the other end
333/// sf::Packet packet;
334/// socket.receive(packet);
335///
336/// // Extract the variables contained in the packet
337/// sf::Uint32 x;
338/// std::string s;
339/// double d;
340/// if (packet >> x >> s >> d)
341/// {
342/// // Data extracted successfully...
343/// }
344/// \endcode
345///
346/// Packets have built-in operator >> and << overloads for
347/// standard types:
348/// \li bool
349/// \li fixed-size integer types (sf::Int8/16/32, sf::Uint8/16/32)
350/// \li floating point numbers (float, double)
351/// \li string types (char*, wchar_t*, std::string, std::wstring, sf::String)
352///
353/// Like standard streams, it is also possible to define your own
354/// overloads of operators >> and << in order to handle your
355/// custom types.
356///
357/// \code
358/// struct MyStruct
359/// {
360/// float number;
361/// sf::Int8 integer;
362/// std::string str;
363/// };
364///
365/// sf::Packet& operator <<(sf::Packet& packet, const MyStruct& m)
366/// {
367/// return packet << m.number << m.integer << m.str;
368/// }
369///
370/// sf::Packet& operator >>(sf::Packet& packet, MyStruct& m)
371/// {
372/// return packet >> m.number >> m.integer >> m.str;
373/// }
374/// \endcode
375///
376/// Packets also provide an extra feature that allows to apply
377/// custom transformations to the data before it is sent,
378/// and after it is received. This is typically used to
379/// handle automatic compression or encryption of the data.
380/// This is achieved by inheriting from sf::Packet, and overriding
381/// the onSend and onReceive functions.
382///
383/// Here is an example:
384/// \code
385/// class ZipPacket : public sf::Packet
386/// {
387/// virtual const void* onSend(std::size_t& size)
388/// {
389/// const void* srcData = getData();
390/// std::size_t srcSize = getDataSize();
391///
392/// return MySuperZipFunction(srcData, srcSize, &size);
393/// }
394///
395/// virtual void onReceive(const void* data, std::size_t size)
396/// {
397/// std::size_t dstSize;
398/// const void* dstData = MySuperUnzipFunction(data, size, &dstSize);
399///
400/// append(dstData, dstSize);
401/// }
402/// };
403///
404/// // Use like regular packets:
405/// ZipPacket packet;
406/// packet << x << s << d;
407/// ...
408/// \endcode
409///
410/// \see sf::TcpSocket, sf::UdpSocket
411///
