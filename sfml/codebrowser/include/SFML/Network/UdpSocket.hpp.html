42/// \brief Specialized socket using the UDP protocol
43///
44////////////////////////////////////////////////////////////
45class SFML_NETWORK_API UdpSocket : public Socket
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 // Constants
51 ////////////////////////////////////////////////////////////
52 enum
53 {
54 MaxDatagramSize = 65507 ///< The maximum number of bytes that can be sent in a single UDP datagram
55 };
56
57 ////////////////////////////////////////////////////////////
58 /// \brief Default constructor
59 ///
60 ////////////////////////////////////////////////////////////
61 UdpSocket();
62
63 ////////////////////////////////////////////////////////////
64 /// \brief Get the port to which the socket is bound locally
65 ///
66 /// If the socket is not bound to a port, this function
67 /// returns 0.
68 ///
69 /// \return Port to which the socket is bound
70 ///
71 /// \see bind
72 ///
73 ////////////////////////////////////////////////////////////
74 unsigned short getLocalPort() const;
75
76 ////////////////////////////////////////////////////////////
77 /// \brief Bind the socket to a specific port
78 ///
79 /// Binding the socket to a port is necessary for being
80 /// able to receive data on that port.
81 /// You can use the special value Socket::AnyPort to tell the
82 /// system to automatically pick an available port, and then
83 /// call getLocalPort to retrieve the chosen port.
84 ///
85 /// \param port Port to bind the socket to
86 /// \param address Address of the interface to bind to
87 ///
88 /// \return Status code
89 ///
90 /// \see unbind, getLocalPort
91 ///
92 ////////////////////////////////////////////////////////////
93 Status bind(unsigned short port, const IpAddress& address = IpAddress::Any);
94
95 ////////////////////////////////////////////////////////////
96 /// \brief Unbind the socket from the local port to which it is bound
97 ///
98 /// The port that the socket was previously bound to is immediately
99 /// made available to the operating system after this function is called.
100 /// This means that a subsequent call to bind() will be able to re-bind
101 /// the port if no other process has done so in the mean time.
102 /// If the socket is not bound to a port, this function has no effect.
103 ///
104 /// \see bind
105 ///
106 ////////////////////////////////////////////////////////////
107 void unbind();
108
109 ////////////////////////////////////////////////////////////
110 /// \brief Send raw data to a remote peer
111 ///
112 /// Make sure that \a size is not greater than
113 /// UdpSocket::MaxDatagramSize, otherwise this function will
114 /// fail and no data will be sent.
115 ///
116 /// \param data Pointer to the sequence of bytes to send
117 /// \param size Number of bytes to send
118 /// \param remoteAddress Address of the receiver
119 /// \param remotePort Port of the receiver to send the data to
120 ///
121 /// \return Status code
122 ///
123 /// \see receive
124 ///
125 ////////////////////////////////////////////////////////////
126 Status send(const void* data, std::size_t size, const IpAddress& remoteAddress, unsigned short remotePort);
127
128 ////////////////////////////////////////////////////////////
129 /// \brief Receive raw data from a remote peer
130 ///
131 /// In blocking mode, this function will wait until some
132 /// bytes are actually received.
133 /// Be careful to use a buffer which is large enough for
134 /// the data that you intend to receive, if it is too small
135 /// then an error will be returned and *all* the data will
136 /// be lost.
137 ///
138 /// \param data Pointer to the array to fill with the received bytes
139 /// \param size Maximum number of bytes that can be received
140 /// \param received This variable is filled with the actual number of bytes received
141 /// \param remoteAddress Address of the peer that sent the data
142 /// \param remotePort Port of the peer that sent the data
143 ///
144 /// \return Status code
145 ///
146 /// \see send
147 ///
148 ////////////////////////////////////////////////////////////
149 Status receive(void* data, std::size_t size, std::size_t& received, IpAddress& remoteAddress, unsigned short& remotePort);
150
151 ////////////////////////////////////////////////////////////
152 /// \brief Send a formatted packet of data to a remote peer
153 ///
154 /// Make sure that the packet size is not greater than
155 /// UdpSocket::MaxDatagramSize, otherwise this function will
156 /// fail and no data will be sent.
157 ///
158 /// \param packet Packet to send
159 /// \param remoteAddress Address of the receiver
160 /// \param remotePort Port of the receiver to send the data to
161 ///
162 /// \return Status code
163 ///
164 /// \see receive
165 ///
166 ////////////////////////////////////////////////////////////
167 Status send(Packet& packet, const IpAddress& remoteAddress, unsigned short remotePort);
168
169 ////////////////////////////////////////////////////////////
170 /// \brief Receive a formatted packet of data from a remote peer
171 ///
172 /// In blocking mode, this function will wait until the whole packet
173 /// has been received.
174 ///
175 /// \param packet Packet to fill with the received data
176 /// \param remoteAddress Address of the peer that sent the data
177 /// \param remotePort Port of the peer that sent the data
178 ///
179 /// \return Status code
180 ///
181 /// \see send
182 ///
183 ////////////////////////////////////////////////////////////
184 Status receive(Packet& packet, IpAddress& remoteAddress, unsigned short& remotePort);
185
186private:
187
188 ////////////////////////////////////////////////////////////
189 // Member data
190 ////////////////////////////////////////////////////////////
191 std::vector<char> m_buffer; ///< Temporary buffer holding the received data in Receive(Packet)
192};
193

201/// \class sf::UdpSocket
202/// \ingroup network
203///
204/// A UDP socket is a connectionless socket. Instead of
205/// connecting once to a remote host, like TCP sockets,
206/// it can send to and receive from any host at any time.
207///
208/// It is a datagram protocol: bounded blocks of data (datagrams)
209/// are transfered over the network rather than a continuous
210/// stream of data (TCP). Therefore, one call to send will always
211/// match one call to receive (if the datagram is not lost),
212/// with the same data that was sent.
213///
214/// The UDP protocol is lightweight but unreliable. Unreliable
215/// means that datagrams may be duplicated, be lost or
216/// arrive reordered. However, if a datagram arrives, its
217/// data is guaranteed to be valid.
218///
219/// UDP is generally used for real-time communication
220/// (audio or video streaming, real-time games, etc.) where
221/// speed is crucial and lost data doesn't matter much.
222///
223/// Sending and receiving data can use either the low-level
224/// or the high-level functions. The low-level functions
225/// process a raw sequence of bytes, whereas the high-level
226/// interface uses packets (see sf::Packet), which are easier
227/// to use and provide more safety regarding the data that is
228/// exchanged. You can look at the sf::Packet class to get
229/// more details about how they work.
230///
231/// It is important to note that UdpSocket is unable to send
232/// datagrams bigger than MaxDatagramSize. In this case, it
233/// returns an error and doesn't send anything. This applies
234/// to both raw data and packets. Indeed, even packets are
235/// unable to split and recompose data, due to the unreliability
236/// of the protocol (dropped, mixed or duplicated datagrams may
237/// lead to a big mess when trying to recompose a packet).
238///
239/// If the socket is bound to a port, it is automatically
240/// unbound from it when the socket is destroyed. However,
241/// you can unbind the socket explicitly with the Unbind
242/// function if necessary, to stop receiving messages or
243/// make the port available for other sockets.
244///
245/// Usage example:
246/// \code
247/// // ----- The client -----
248///
249/// // Create a socket and bind it to the port 55001
250/// sf::UdpSocket socket;
251/// socket.bind(55001);
252///
253/// // Send a message to 192.168.1.50 on port 55002
254/// std::string message = "Hi, I am " + sf::IpAddress::getLocalAddress().toString();
255/// socket.send(message.c_str(), message.size() + 1, "192.168.1.50", 55002);
256///
257/// // Receive an answer (most likely from 192.168.1.50, but could be anyone else)
258/// char buffer[1024];
259/// std::size_t received = 0;
260/// sf::IpAddress sender;
261/// unsigned short port;
262/// socket.receive(buffer, sizeof(buffer), received, sender, port);
263/// std::cout << sender.ToString() << " said: " << buffer << std::endl;
264///
265/// // ----- The server -----
266///
267/// // Create a socket and bind it to the port 55002
268/// sf::UdpSocket socket;
269/// socket.bind(55002);
270///
271/// // Receive a message from anyone
272/// char buffer[1024];
273/// std::size_t received = 0;
274/// sf::IpAddress sender;
275/// unsigned short port;
276/// socket.receive(buffer, sizeof(buffer), received, sender, port);
277/// std::cout << sender.ToString() << " said: " << buffer << std::endl;
278///
279/// // Send an answer
280/// std::string message = "Welcome " + sender.toString();
281/// socket.send(message.c_str(), message.size() + 1, sender, port);
282/// \endcode
283///
284/// \see sf::Socket, sf::TcpSocket, sf::Packet
285///
