52/// \brief Implementation of input stream based on a file
53///
54////////////////////////////////////////////////////////////
55class SFML_SYSTEM_API FileInputStream : public InputStream, NonCopyable
56{
57public:
58 ////////////////////////////////////////////////////////////
59 /// \brief Default constructor
60 ///
61 ////////////////////////////////////////////////////////////
62 FileInputStream();
63
64 ////////////////////////////////////////////////////////////
65 /// \brief Default destructor
66 ///
67 ////////////////////////////////////////////////////////////
68 virtual ~FileInputStream();
69
70 ////////////////////////////////////////////////////////////
71 /// \brief Open the stream from a file path
72 ///
73 /// \param filename Name of the file to open
74 ///
75 /// \return True on success, false on error
76 ///
77 ////////////////////////////////////////////////////////////
78 bool open(const std::string& filename);
79
80 ////////////////////////////////////////////////////////////
81 /// \brief Read data from the stream
82 ///
83 /// After reading, the stream's reading position must be
84 /// advanced by the amount of bytes read.
85 ///
86 /// \param data Buffer where to copy the read data
87 /// \param size Desired number of bytes to read
88 ///
89 /// \return The number of bytes actually read, or -1 on error
90 ///
91 ////////////////////////////////////////////////////////////
92 virtual Int64 read(void* data, Int64 size);
93
94 ////////////////////////////////////////////////////////////
95 /// \brief Change the current reading position
96 ///
97 /// \param position The position to seek to, from the beginning
98 ///
99 /// \return The position actually sought to, or -1 on error
100 ///
101 ////////////////////////////////////////////////////////////
102 virtual Int64 seek(Int64 position);
103
104 ////////////////////////////////////////////////////////////
105 /// \brief Get the current reading position in the stream
106 ///
107 /// \return The current position, or -1 on error.
108 ///
109 ////////////////////////////////////////////////////////////
110 virtual Int64 tell();
111
112 ////////////////////////////////////////////////////////////
113 /// \brief Return the size of the stream
114 ///
115 /// \return The total number of bytes available in the stream, or -1 on error
116 ///
117 ////////////////////////////////////////////////////////////
118 virtual Int64 getSize();
119
120private:
121
122 ////////////////////////////////////////////////////////////
123 // Member data
124 ////////////////////////////////////////////////////////////
125#ifdef ANDROID
126 priv::ResourceStream* m_file;
127#else
128 std::FILE* m_file; ///< stdio file stream
129#endif
130};

138////////////////////////////////////////////////////////////
139/// \class sf::FileInputStream
140/// \ingroup system
141///
142/// This class is a specialization of InputStream that
143/// reads from a file on disk.
144///
145/// It wraps a file in the common InputStream interface
146/// and therefore allows to use generic classes or functions
147/// that accept such a stream, with a file on disk as the data
148/// source.
149///
150/// In addition to the virtual functions inherited from
151/// InputStream, FileInputStream adds a function to
152/// specify the file to open.
153///
154/// SFML resource classes can usually be loaded directly from
155/// a filename, so this class shouldn't be useful to you unless
156/// you create your own algorithms that operate on an InputStream.
157///
158/// Usage example:
159/// \code
160/// void process(InputStream& stream);
161///
162/// FileInputStream stream;
163/// if (stream.open("some_file.dat"))
164/// process(stream);
165/// \endcode
166///
167/// InputStream, MemoryInputStream
168///
