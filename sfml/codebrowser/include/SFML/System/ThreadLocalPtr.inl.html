26namespace sf
27{
28////////////////////////////////////////////////////////////
29template <typename T>
30ThreadLocalPtr<T>::ThreadLocalPtr(T* value) :
31ThreadLocal(value)
32{
33}
34
35
36////////////////////////////////////////////////////////////
37template <typename T>
38T& ThreadLocalPtr<T>::operator *() const
39{
40 return *static_cast<T*>(getValue());
41}
42
43
44////////////////////////////////////////////////////////////
45template <typename T>
46T* ThreadLocalPtr<T>::operator ->() const
47{
48 return static_cast<T*>(getValue());
49}
50
51
52////////////////////////////////////////////////////////////
53template <typename T>
54ThreadLocalPtr<T>::operator T*() const
55{
56 return static_cast<T*>(getValue());
57}
58
59
60////////////////////////////////////////////////////////////
61template <typename T>
62ThreadLocalPtr<T>& ThreadLocalPtr<T>::operator =(T* value)
63{
64 setValue(value);
65 return *this;
66}
67
68
69////////////////////////////////////////////////////////////
70template <typename T>
71ThreadLocalPtr<T>& ThreadLocalPtr<T>::operator =(const ThreadLocalPtr<T>& right)
72{
73 setValue(right.getValue());
74 return *this;
75}

77} // namespace sf
