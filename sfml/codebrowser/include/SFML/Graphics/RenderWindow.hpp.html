40////////////////////////////////////////////////////////////
41/// \brief Window that can serve as a target for 2D drawing
42///
43////////////////////////////////////////////////////////////
44class SFML_GRAPHICS_API RenderWindow : public Window, public RenderTarget
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Default constructor
50 ///
51 /// This constructor doesn't actually create the window,
52 /// use the other constructors or call create() to do so.
53 ///
54 ////////////////////////////////////////////////////////////
55 RenderWindow();
56
57 ////////////////////////////////////////////////////////////
58 /// \brief Construct a new window
59 ///
60 /// This constructor creates the window with the size and pixel
61 /// depth defined in \a mode. An optional style can be passed to
62 /// customize the look and behavior of the window (borders,
63 /// title bar, resizable, closable, ...).
64 ///
65 /// The fourth parameter is an optional structure specifying
66 /// advanced OpenGL context settings such as antialiasing,
67 /// depth-buffer bits, etc. You shouldn't care about these
68 /// parameters for a regular usage of the graphics module.
69 ///
70 /// \param mode Video mode to use (defines the width, height and depth of the rendering area of the window)
71 /// \param title Title of the window
72 /// \param style %Window style, a bitwise OR combination of sf::Style enumerators
73 /// \param settings Additional settings for the underlying OpenGL context
74 ///
75 ////////////////////////////////////////////////////////////
76 RenderWindow(VideoMode mode, const String& title, Uint32 style = Style::Default, const ContextSettings& settings = ContextSettings());
77
78 ////////////////////////////////////////////////////////////
79 /// \brief Construct the window from an existing control
80 ///
81 /// Use this constructor if you want to create an SFML
82 /// rendering area into an already existing control.
83 ///
84 /// The second parameter is an optional structure specifying
85 /// advanced OpenGL context settings such as antialiasing,
86 /// depth-buffer bits, etc. You shouldn't care about these
87 /// parameters for a regular usage of the graphics module.
88 ///
89 /// \param handle Platform-specific handle of the control (\a HWND on
90 /// Windows, \a %Window on Linux/FreeBSD, \a NSWindow on OS X)
91 /// \param settings Additional settings for the underlying OpenGL context
92 ///
93 ////////////////////////////////////////////////////////////
94 explicit RenderWindow(WindowHandle handle, const ContextSettings& settings = ContextSettings());
95
96 ////////////////////////////////////////////////////////////
97 /// \brief Destructor
98 ///
99 /// Closes the window and frees all the resources attached to it.
100 ///
101 ////////////////////////////////////////////////////////////
102 virtual ~RenderWindow();
103
104 ////////////////////////////////////////////////////////////
105 /// \brief Get the size of the rendering region of the window
106 ///
107 /// The size doesn't include the titlebar and borders
108 /// of the window.
109 ///
110 /// \return Size in pixels
111 ///
112 ////////////////////////////////////////////////////////////
113 virtual Vector2u getSize() const;
114
115 ////////////////////////////////////////////////////////////
116 /// \brief Activate or deactivate the window as the current target
117 /// for OpenGL rendering
118 ///
119 /// A window is active only on the current thread, if you want to
120 /// make it active on another thread you have to deactivate it
121 /// on the previous thread first if it was active.
122 /// Only one window can be active on a thread at a time, thus
123 /// the window previously active (if any) automatically gets deactivated.
124 /// This is not to be confused with requestFocus().
125 ///
126 /// \param active True to activate, false to deactivate
127 ///
128 /// \return True if operation was successful, false otherwise
129 ///
130 ////////////////////////////////////////////////////////////
131 bool setActive(bool active = true);
132
133 ////////////////////////////////////////////////////////////
134 /// \brief Copy the current contents of the window to an image
135 ///
136 /// \deprecated
137 /// Use a sf::Texture and its sf::Texture::update(const Window&)
138 /// function and copy its contents into an sf::Image instead.
139 /// \code
140 /// sf::Vector2u windowSize = window.getSize();
141 /// sf::Texture texture;
142 /// texture.create(windowSize.x, windowSize.y);
143 /// texture.update(window);
144 /// sf::Image screenshot = texture.copyToImage();
145 /// \endcode
146 ///
147 /// This is a slow operation, whose main purpose is to make
148 /// screenshots of the application. If you want to update an
149 /// image with the contents of the window and then use it for
150 /// drawing, you should rather use a sf::Texture and its
151 /// update(Window&) function.
152 /// You can also draw things directly to a texture with the
153 /// sf::RenderTexture class.
154 ///
155 /// \return Image containing the captured contents
156 ///
157 ////////////////////////////////////////////////////////////
158 SFML_DEPRECATED Image capture() const;
159
160protected:
161
162 ////////////////////////////////////////////////////////////
163 /// \brief Function called after the window has been created
164 ///
165 /// This function is called so that derived classes can
166 /// perform their own specific initialization as soon as
167 /// the window is created.
168 ///
169 ////////////////////////////////////////////////////////////
170 virtual void onCreate();
171
172 ////////////////////////////////////////////////////////////
173 /// \brief Function called after the window has been resized
174 ///
175 /// This function is called so that derived classes can
176 /// perform custom actions when the size of the window changes.
177 ///
178 ////////////////////////////////////////////////////////////
179 virtual void onResize();
180};
181

189/// \class sf::RenderWindow
190/// \ingroup graphics
191///
192/// sf::RenderWindow is the main class of the Graphics module.
193/// It defines an OS window that can be painted using the other
194/// classes of the graphics module.
195///
196/// sf::RenderWindow is derived from sf::Window, thus it inherits
197/// all its features: events, window management, OpenGL rendering,
198/// etc. See the documentation of sf::Window for a more complete
199/// description of all these features, as well as code examples.
200///
201/// On top of that, sf::RenderWindow adds more features related to
202/// 2D drawing with the graphics module (see its base class
203/// sf::RenderTarget for more details).
204/// Here is a typical rendering and event loop with a sf::RenderWindow:
205///
206/// \code
207/// // Declare and create a new render-window
208/// sf::RenderWindow window(sf::VideoMode(800, 600), "SFML window");
209///
210/// // Limit the framerate to 60 frames per second (this step is optional)
211/// window.setFramerateLimit(60);
212///
213/// // The main loop - ends as soon as the window is closed
214/// while (window.isOpen())
215/// {
216/// // Event processing
217/// sf::Event event;
218/// while (window.pollEvent(event))
219/// {
220/// // Request for closing the window
221/// if (event.type == sf::Event::Closed)
222/// window.close();
223/// }
224///
225/// // Clear the whole window before rendering a new frame
226/// window.clear();
227///
228/// // Draw some graphical entities
229/// window.draw(sprite);
230/// window.draw(circle);
231/// window.draw(text);
232///
233/// // End the current frame and display its contents on screen
234/// window.display();
235/// }
236/// \endcode
237///
238/// Like sf::Window, sf::RenderWindow is still able to render direct
239/// OpenGL stuff. It is even possible to mix together OpenGL calls
240/// and regular SFML drawing commands.
241///
242/// \code
243/// // Create the render window
244/// sf::RenderWindow window(sf::VideoMode(800, 600), "SFML OpenGL");
245///
246/// // Create a sprite and a text to display
247/// sf::Sprite sprite;
248/// sf::Text text;
249/// ...
250///
251/// // Perform OpenGL initializations
252/// glMatrixMode(GL_PROJECTION);
253/// ...
254///
255/// // Start the rendering loop
256/// while (window.isOpen())
257/// {
258/// // Process events
259/// ...
260///
261/// // Draw a background sprite
262/// window.pushGLStates();
263/// window.draw(sprite);
264/// window.popGLStates();
265///
266/// // Draw a 3D object using OpenGL
267/// glBegin(GL_QUADS);
268/// glVertex3f(...);
269/// ...
270/// glEnd();
271///
272/// // Draw text on top of the 3D object
273/// window.pushGLStates();
274/// window.draw(text);
275/// window.popGLStates();
276///
277/// // Finally, display the rendered frame on screen
278/// window.display();
279/// }
280/// \endcode
281///
282/// \see sf::Window, sf::RenderTarget, sf::RenderTexture, sf::View
283///
