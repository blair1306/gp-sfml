35namespace sf

38/// \brief Utility class for manipulating 2D axis aligned rectangles
39///
40////////////////////////////////////////////////////////////
41template <typename T>
42class Rect
43{
44public:
45
46 ////////////////////////////////////////////////////////////
47 /// \brief Default constructor
48 ///
49 /// Creates an empty rectangle (it is equivalent to calling
50 /// Rect(0, 0, 0, 0)).
51 ///
52 ////////////////////////////////////////////////////////////
53 Rect();
54
55 ////////////////////////////////////////////////////////////
56 /// \brief Construct the rectangle from its coordinates
57 ///
58 /// Be careful, the last two parameters are the width
59 /// and height, not the right and bottom coordinates!
60 ///
61 /// \param rectLeft Left coordinate of the rectangle
62 /// \param rectTop Top coordinate of the rectangle
63 /// \param rectWidth Width of the rectangle
64 /// \param rectHeight Height of the rectangle
65 ///
66 ////////////////////////////////////////////////////////////
67 Rect(T rectLeft, T rectTop, T rectWidth, T rectHeight);
68
69 ////////////////////////////////////////////////////////////
70 /// \brief Construct the rectangle from position and size
71 ///
72 /// Be careful, the last parameter is the size,
73 /// not the bottom-right corner!
74 ///
75 /// \param position Position of the top-left corner of the rectangle
76 /// \param size Size of the rectangle
77 ///
78 ////////////////////////////////////////////////////////////
79 Rect(const Vector2<T>& position, const Vector2<T>& size);
80
81 ////////////////////////////////////////////////////////////
82 /// \brief Construct the rectangle from another type of rectangle
83 ///
84 /// This constructor doesn't replace the copy constructor,
85 /// it's called only when U != T.
86 /// A call to this constructor will fail to compile if U
87 /// is not convertible to T.
88 ///
89 /// \param rectangle Rectangle to convert
90 ///
91 ////////////////////////////////////////////////////////////
92 template <typename U>
93 explicit Rect(const Rect<U>& rectangle);
94
95 ////////////////////////////////////////////////////////////
96 /// \brief Check if a point is inside the rectangle's area
97 ///
98 /// This check is non-inclusive. If the point lies on the
99 /// edge of the rectangle, this function will return false.
100 ///
101 /// \param x X coordinate of the point to test
102 /// \param y Y coordinate of the point to test
103 ///
104 /// \return True if the point is inside, false otherwise
105 ///
106 /// \see intersects
107 ///
108 ////////////////////////////////////////////////////////////
109 bool contains(T x, T y) const;
110
111 ////////////////////////////////////////////////////////////
112 /// \brief Check if a point is inside the rectangle's area
113 ///
114 /// This check is non-inclusive. If the point lies on the
115 /// edge of the rectangle, this function will return false.
116 ///
117 /// \param point Point to test
118 ///
119 /// \return True if the point is inside, false otherwise
120 ///
121 /// \see intersects
122 ///
123 ////////////////////////////////////////////////////////////
124 bool contains(const Vector2<T>& point) const;
125
126 ////////////////////////////////////////////////////////////
127 /// \brief Check the intersection between two rectangles
128 ///
129 /// \param rectangle Rectangle to test
130 ///
131 /// \return True if rectangles overlap, false otherwise
132 ///
133 /// \see contains
134 ///
135 ////////////////////////////////////////////////////////////
136 bool intersects(const Rect<T>& rectangle) const;
137
138 ////////////////////////////////////////////////////////////
139 /// \brief Check the intersection between two rectangles
140 ///
141 /// This overload returns the overlapped rectangle in the
142 /// \a intersection parameter.
143 ///
144 /// \param rectangle Rectangle to test
145 /// \param intersection Rectangle to be filled with the intersection
146 ///
147 /// \return True if rectangles overlap, false otherwise
148 ///
149 /// \see contains
150 ///
151 ////////////////////////////////////////////////////////////
152 bool intersects(const Rect<T>& rectangle, Rect<T>& intersection) const;
153
154 ////////////////////////////////////////////////////////////
155 // Member data
156 ////////////////////////////////////////////////////////////
157 T left; ///< Left coordinate of the rectangle
158 T top; ///< Top coordinate of the rectangle
159 T width; ///< Width of the rectangle
160 T height; ///< Height of the rectangle
161};
162
163////////////////////////////////////////////////////////////
164/// \relates Rect
165/// \brief Overload of binary operator ==
166///
167/// This operator compares strict equality between two rectangles.
168///
169/// \param left Left operand (a rectangle)
170/// \param right Right operand (a rectangle)
171///
172/// \return True if \a left is equal to \a right
173///
174////////////////////////////////////////////////////////////
175template <typename T>
176bool operator ==(const Rect<T>& left, const Rect<T>& right);
177
178////////////////////////////////////////////////////////////
179/// \relates Rect
180/// \brief Overload of binary operator !=
181///
182/// This operator compares strict difference between two rectangles.
183///
184/// \param left Left operand (a rectangle)
185/// \param right Right operand (a rectangle)
186///
187/// \return True if \a left is not equal to \a right
188///
189////////////////////////////////////////////////////////////
190template <typename T>
191bool operator !=(const Rect<T>& left, const Rect<T>& right);

205////////////////////////////////////////////////////////////
206/// \class sf::Rect
207/// \ingroup graphics
208///
209/// A rectangle is defined by its top-left corner and its size.
210/// It is a very simple class defined for convenience, so
211/// its member variables (left, top, width and height) are public
212/// and can be accessed directly, just like the vector classes
213/// (Vector2 and Vector3).
214///
215/// To keep things simple, sf::Rect doesn't define
216/// functions to emulate the properties that are not directly
217/// members (such as right, bottom, center, etc.), it rather
218/// only provides intersection functions.
219///
220/// sf::Rect uses the usual rules for its boundaries:
221/// \li The left and top edges are included in the rectangle's area
222/// \li The right (left + width) and bottom (top + height) edges are excluded from the rectangle's area
223///
224/// This means that sf::IntRect(0, 0, 1, 1) and sf::IntRect(1, 1, 1, 1)
225/// don't intersect.
226///
227/// sf::Rect is a template and may be used with any numeric type, but
228/// for simplicity the instantiations used by SFML are typedef'd:
229/// \li sf::Rect<int> is sf::IntRect
230/// \li sf::Rect<float> is sf::FloatRect
231///
232/// So that you don't have to care about the template syntax.
233///
234/// Usage example:
235/// \code
236/// // Define a rectangle, located at (0, 0) with a size of 20x5
237/// sf::IntRect r1(0, 0, 20, 5);
238///
239/// // Define another rectangle, located at (4, 2) with a size of 18x10
240/// sf::Vector2i position(4, 2);
241/// sf::Vector2i size(18, 10);
242/// sf::IntRect r2(position, size);
243///
244/// // Test intersections with the point (3, 1)
245/// bool b1 = r1.contains(3, 1); // true
246/// bool b2 = r2.contains(3, 1); // false
247///
248/// // Test the intersection between r1 and r2
249/// sf::IntRect result;
250/// bool b3 = r1.intersects(r2, result); // true
251/// // result == (4, 2, 16, 3)
252/// \endcode
253///
