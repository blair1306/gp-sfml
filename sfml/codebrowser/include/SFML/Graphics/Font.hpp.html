89 ////////////////////////////////////////////////////////////
90 /// \brief Load the font from a file
91 ///
92 /// The supported font formats are: TrueType, Type 1, CFF,
93 /// OpenType, SFNT, X11 PCF, Windows FNT, BDF, PFR and Type 42.
94 /// Note that this function knows nothing about the standard
95 /// fonts installed on the user's system, thus you can't
96 /// load them directly.
97 ///
98 /// \warning SFML cannot preload all the font data in this
99 /// function, so the file has to remain accessible until
100 /// the sf::Font object loads a new font or is destroyed.
101 ///
102 /// \param filename Path of the font file to load
103 ///
104 /// \return True if loading succeeded, false if it failed
105 ///
106 /// \see loadFromMemory, loadFromStream
107 ///
108 ////////////////////////////////////////////////////////////
109 bool loadFromFile(const std::string& filename);
110
111 ////////////////////////////////////////////////////////////
112 /// \brief Load the font from a file in memory
113 ///
114 /// The supported font formats are: TrueType, Type 1, CFF,
115 /// OpenType, SFNT, X11 PCF, Windows FNT, BDF, PFR and Type 42.
116 ///
117 /// \warning SFML cannot preload all the font data in this
118 /// function, so the buffer pointed by \a data has to remain
119 /// valid until the sf::Font object loads a new font or
120 /// is destroyed.
121 ///
122 /// \param data Pointer to the file data in memory
123 /// \param sizeInBytes Size of the data to load, in bytes
124 ///
125 /// \return True if loading succeeded, false if it failed
126 ///
127 /// \see loadFromFile, loadFromStream
128 ///
129 ////////////////////////////////////////////////////////////
130 bool loadFromMemory(const void* data, std::size_t sizeInBytes);
131
132 ////////////////////////////////////////////////////////////
133 /// \brief Load the font from a custom stream
134 ///
135 /// The supported font formats are: TrueType, Type 1, CFF,
136 /// OpenType, SFNT, X11 PCF, Windows FNT, BDF, PFR and Type 42.
137 /// Warning: SFML cannot preload all the font data in this
138 /// function, so the contents of \a stream have to remain
139 /// valid as long as the font is used.
140 ///
141 /// \warning SFML cannot preload all the font data in this
142 /// function, so the stream has to remain accessible until
143 /// the sf::Font object loads a new font or is destroyed.
144 ///
145 /// \param stream Source stream to read from
146 ///
147 /// \return True if loading succeeded, false if it failed
148 ///
149 /// \see loadFromFile, loadFromMemory
150 ///
151 ////////////////////////////////////////////////////////////
152 bool loadFromStream(InputStream& stream);
153
154 ////////////////////////////////////////////////////////////
155 /// \brief Get the font information
156 ///
157 /// \return A structure that holds the font information
158 ///
159 ////////////////////////////////////////////////////////////
160 const Info& getInfo() const;
161
162 ////////////////////////////////////////////////////////////
163 /// \brief Retrieve a glyph of the font
164 ///
165 /// If the font is a bitmap font, not all character sizes
166 /// might be available. If the glyph is not available at the
167 /// requested size, an empty glyph is returned.
168 ///
169 /// Be aware that using a negative value for the outline
170 /// thickness will cause distorted rendering.
171 ///
172 /// \param codePoint Unicode code point of the character to get
173 /// \param characterSize Reference character size
174 /// \param bold Retrieve the bold version or the regular one?
175 /// \param outlineThickness Thickness of outline (when != 0 the glyph will not be filled)
176 ///
177 /// \return The glyph corresponding to \a codePoint and \a characterSize
178 ///
179 ////////////////////////////////////////////////////////////
180 const Glyph& getGlyph(Uint32 codePoint, unsigned int characterSize, bool bold, float outlineThickness = 0) const;
181
182 ////////////////////////////////////////////////////////////
183 /// \brief Get the kerning offset of two glyphs
184 ///
185 /// The kerning is an extra offset (negative) to apply between two
186 /// glyphs when rendering them, to make the pair look more "natural".
187 /// For example, the pair "AV" have a special kerning to make them
188 /// closer than other characters. Most of the glyphs pairs have a
189 /// kerning offset of zero, though.
190 ///
191 /// \param first Unicode code point of the first character
192 /// \param second Unicode code point of the second character
193 /// \param characterSize Reference character size
194 ///
195 /// \return Kerning value for \a first and \a second, in pixels
196 ///
197 ////////////////////////////////////////////////////////////
198 float getKerning(Uint32 first, Uint32 second, unsigned int characterSize) const;
199
200 ////////////////////////////////////////////////////////////
201 /// \brief Get the line spacing
202 ///
203 /// Line spacing is the vertical offset to apply between two
204 /// consecutive lines of text.
205 ///
206 /// \param characterSize Reference character size
207 ///
208 /// \return Line spacing, in pixels
209 ///
210 ////////////////////////////////////////////////////////////
211 float getLineSpacing(unsigned int characterSize) const;
212
213 ////////////////////////////////////////////////////////////
214 /// \brief Get the position of the underline
215 ///
216 /// Underline position is the vertical offset to apply between the
217 /// baseline and the underline.
218 ///
219 /// \param characterSize Reference character size
220 ///
221 /// \return Underline position, in pixels
222 ///
223 /// \see getUnderlineThickness
224 ///
225 ////////////////////////////////////////////////////////////
226 float getUnderlinePosition(unsigned int characterSize) const;
227
228 ////////////////////////////////////////////////////////////
229 /// \brief Get the thickness of the underline
230 ///
231 /// Underline thickness is the vertical size of the underline.
232 ///
233 /// \param characterSize Reference character size
234 ///
235 /// \return Underline thickness, in pixels
236 ///
237 /// \see getUnderlinePosition
238 ///
239 ////////////////////////////////////////////////////////////
240 float getUnderlineThickness(unsigned int characterSize) const;

242 ////////////////////////////////////////////////////////////
243 /// \brief Retrieve the texture containing the loaded glyphs of a certain size

369/// \class sf::Font
370/// \ingroup graphics
371///
372/// Fonts can be loaded from a file, from memory or from a custom
373/// stream, and supports the most common types of fonts. See
374/// the loadFromFile function for the complete list of supported formats.
375///
376/// Once it is loaded, a sf::Font instance provides three
377/// types of information about the font:
378/// \li Global metrics, such as the line spacing
379/// \li Per-glyph metrics, such as bounding box or kerning
380/// \li Pixel representation of glyphs
381///
382/// Fonts alone are not very useful: they hold the font data
383/// but cannot make anything useful of it. To do so you need to
384/// use the sf::Text class, which is able to properly output text
385/// with several options such as character size, style, color,
386/// position, rotation, etc.
387/// This separation allows more flexibility and better performances:
388/// indeed a sf::Font is a heavy resource, and any operation on it
389/// is slow (often too slow for real-time applications). On the other
390/// side, a sf::Text is a lightweight object which can combine the
391/// glyphs data and metrics of a sf::Font to display any text on a
392/// render target.
393/// Note that it is also possible to bind several sf::Text instances
394/// to the same sf::Font.
395///
396/// It is important to note that the sf::Text instance doesn't
397/// copy the font that it uses, it only keeps a reference to it.
398/// Thus, a sf::Font must not be destructed while it is
399/// used by a sf::Text (i.e. never write a function that
400/// uses a local sf::Font instance for creating a text).
401///
402/// Usage example:
403/// \code
404/// // Declare a new font
405/// sf::Font font;
406///
407/// // Load it from a file
408/// if (!font.loadFromFile("arial.ttf"))
409/// {
410/// // error...
411/// }
412///
413/// // Create a text which uses our font
414/// sf::Text text1;
415/// text1.setFont(font);
416/// text1.setCharacterSize(30);
417/// text1.setStyle(sf::Text::Regular);
418///
419/// // Create another text using the same font, but with different parameters
420/// sf::Text text2;
421/// text2.setFont(font);
422/// text2.setCharacterSize(50);
423/// text2.setStyle(sf::Text::Italic);
424/// \endcode
425///
426/// Apart from loading font files, and passing them to instances
427/// of sf::Text, you should normally not have to deal directly
428/// with this class. However, it may be useful to access the
429/// font metrics or rasterized glyphs for advanced usage.
430///
431/// Note that if the font is a bitmap font, it is not scalable,
432/// thus not all requested sizes will be available to use. This
433/// needs to be taken into consideration when using sf::Text.
434/// If you need to display text of a certain size, make sure the
435/// corresponding bitmap font that supports that size is used.
436///
437/// \see sf::Text
438///
