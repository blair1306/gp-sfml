37/// \brief Utility class for manipulating RGBA colors
38///
39////////////////////////////////////////////////////////////
40class SFML_GRAPHICS_API Color
41{
42public:
43
44 ////////////////////////////////////////////////////////////
45 /// \brief Default constructor
46 ///
47 /// Constructs an opaque black color. It is equivalent to
48 /// sf::Color(0, 0, 0, 255).
49 ///
50 ////////////////////////////////////////////////////////////
51 Color();
52
53 ////////////////////////////////////////////////////////////
54 /// \brief Construct the color from its 4 RGBA components
55 ///
56 /// \param red Red component (in the range [0, 255])
57 /// \param green Green component (in the range [0, 255])
58 /// \param blue Blue component (in the range [0, 255])
59 /// \param alpha Alpha (opacity) component (in the range [0, 255])
60 ///
61 ////////////////////////////////////////////////////////////
62 Color(Uint8 red, Uint8 green, Uint8 blue, Uint8 alpha = 255);
63
64 ////////////////////////////////////////////////////////////
65 /// \brief Construct the color from 32-bit unsigned integer
66 ///
67 /// \param color Number containing the RGBA components (in that order)
68 ///
69 ////////////////////////////////////////////////////////////
70 explicit Color(Uint32 color);
71
72 ////////////////////////////////////////////////////////////
73 /// \brief Retrieve the color as a 32-bit unsigned integer
74 ///
75 /// \return Color represented as a 32-bit unsigned integer
76 ///
77 ////////////////////////////////////////////////////////////
78 Uint32 toInteger() const;
79
80 ////////////////////////////////////////////////////////////
81 // Static member data
82 ////////////////////////////////////////////////////////////
83 static const Color Black; ///< Black predefined color
84 static const Color White; ///< White predefined color
85 static const Color Red; ///< Red predefined color
86 static const Color Green; ///< Green predefined color
87 static const Color Blue; ///< Blue predefined color
88 static const Color Yellow; ///< Yellow predefined color
89 static const Color Magenta; ///< Magenta predefined color
90 static const Color Cyan; ///< Cyan predefined color
91 static const Color Transparent; ///< Transparent (black) predefined color
92
93 ////////////////////////////////////////////////////////////
94 // Member data
95 ////////////////////////////////////////////////////////////
96 Uint8 r; ///< Red component
97 Uint8 g; ///< Green component
98 Uint8 b; ///< Blue component
99 Uint8 a; ///< Alpha (opacity) component
100};

234/// \class sf::Color
235/// \ingroup graphics
236///
237/// sf::Color is a simple color class composed of 4 components:
238/// \li Red
239/// \li Green
240/// \li Blue
241/// \li Alpha (opacity)
242///
243/// Each component is a public member, an unsigned integer in
244/// the range [0, 255]. Thus, colors can be constructed and
245/// manipulated very easily:
246///
247/// \code
248/// sf::Color color(255, 0, 0); // red
249/// color.r = 0; // make it black
250/// color.b = 128; // make it dark blue
251/// \endcode
252///
253/// The fourth component of colors, named "alpha", represents
254/// the opacity of the color. A color with an alpha value of
255/// 255 will be fully opaque, while an alpha value of 0 will
256/// make a color fully transparent, whatever the value of the
257/// other components is.
258///
259/// The most common colors are already defined as static variables:
260/// \code
261/// sf::Color black = sf::Color::Black;
262/// sf::Color white = sf::Color::White;
263/// sf::Color red = sf::Color::Red;
264/// sf::Color green = sf::Color::Green;
265/// sf::Color blue = sf::Color::Blue;
266/// sf::Color yellow = sf::Color::Yellow;
267/// sf::Color magenta = sf::Color::Magenta;
268/// sf::Color cyan = sf::Color::Cyan;
269/// sf::Color transparent = sf::Color::Transparent;
270/// \endcode
271///
272/// Colors can also be added and modulated (multiplied) using the
273/// overloaded operators + and *.
274///
