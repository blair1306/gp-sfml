37////////////////////////////////////////////////////////////
38/// \brief Blending modes for drawing
39///
40////////////////////////////////////////////////////////////
41struct SFML_GRAPHICS_API BlendMode
42{
43 ////////////////////////////////////////////////////////
44 /// \brief Enumeration of the blending factors
45 ///
46 /// The factors are mapped directly to their OpenGL equivalents,
47 /// specified by glBlendFunc() or glBlendFuncSeparate().
48 ////////////////////////////////////////////////////////
49 enum Factor
50 {
51 Zero, ///< (0, 0, 0, 0)
52 One, ///< (1, 1, 1, 1)
53 SrcColor, ///< (src.r, src.g, src.b, src.a)
54 OneMinusSrcColor, ///< (1, 1, 1, 1) - (src.r, src.g, src.b, src.a)
55 DstColor, ///< (dst.r, dst.g, dst.b, dst.a)
56 OneMinusDstColor, ///< (1, 1, 1, 1) - (dst.r, dst.g, dst.b, dst.a)
57 SrcAlpha, ///< (src.a, src.a, src.a, src.a)
58 OneMinusSrcAlpha, ///< (1, 1, 1, 1) - (src.a, src.a, src.a, src.a)
59 DstAlpha, ///< (dst.a, dst.a, dst.a, dst.a)
60 OneMinusDstAlpha ///< (1, 1, 1, 1) - (dst.a, dst.a, dst.a, dst.a)
61 };
62
63 ////////////////////////////////////////////////////////
64 /// \brief Enumeration of the blending equations
65 ///
66 /// The equations are mapped directly to their OpenGL equivalents,
67 /// specified by glBlendEquation() or glBlendEquationSeparate().
68 ////////////////////////////////////////////////////////
69 enum Equation
70 {
71 Add, ///< Pixel = Src * SrcFactor + Dst * DstFactor
72 Subtract, ///< Pixel = Src * SrcFactor - Dst * DstFactor
73 ReverseSubtract ///< Pixel = Dst * DstFactor - Src * SrcFactor
74 };
75
76 ////////////////////////////////////////////////////////////
77 /// \brief Default constructor
78 ///
79 /// Constructs a blending mode that does alpha blending.
80 ///
81 ////////////////////////////////////////////////////////////
82 BlendMode();
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Construct the blend mode given the factors and equation.
86 ///
87 /// This constructor uses the same factors and equation for both
88 /// color and alpha components. It also defaults to the Add equation.
89 ///
90 /// \param sourceFactor Specifies how to compute the source factor for the color and alpha channels.
91 /// \param destinationFactor Specifies how to compute the destination factor for the color and alpha channels.
92 /// \param blendEquation Specifies how to combine the source and destination colors and alpha.
93 ///
94 ////////////////////////////////////////////////////////////
95 BlendMode(Factor sourceFactor, Factor destinationFactor, Equation blendEquation = Add);
96
97 ////////////////////////////////////////////////////////////
98 /// \brief Construct the blend mode given the factors and equation.
99 ///
100 /// \param colorSourceFactor Specifies how to compute the source factor for the color channels.
101 /// \param colorDestinationFactor Specifies how to compute the destination factor for the color channels.
102 /// \param colorBlendEquation Specifies how to combine the source and destination colors.
103 /// \param alphaSourceFactor Specifies how to compute the source factor.
104 /// \param alphaDestinationFactor Specifies how to compute the destination factor.
105 /// \param alphaBlendEquation Specifies how to combine the source and destination alphas.
106 ///
107 ////////////////////////////////////////////////////////////
108 BlendMode(Factor colorSourceFactor, Factor colorDestinationFactor,
109 Equation colorBlendEquation, Factor alphaSourceFactor,
110 Factor alphaDestinationFactor, Equation alphaBlendEquation);
111
112 ////////////////////////////////////////////////////////////
113 // Member Data
114 ////////////////////////////////////////////////////////////
115 Factor colorSrcFactor; ///< Source blending factor for the color channels
116 Factor colorDstFactor; ///< Destination blending factor for the color channels
117 Equation colorEquation; ///< Blending equation for the color channels
118 Factor alphaSrcFactor; ///< Source blending factor for the alpha channel
119 Factor alphaDstFactor; ///< Destination blending factor for the alpha channel
120 Equation alphaEquation; ///< Blending equation for the alpha channel
121};
122

125/// \brief Overload of the == operator
126///
127/// \param left Left operand
128/// \param right Right operand

130/// \return True if blending modes are equal, false if they are different
131///
132////////////////////////////////////////////////////////////
133SFML_GRAPHICS_API bool operator ==(const BlendMode& left, const BlendMode& right);

135////////////////////////////////////////////////////////////
136/// \relates BlendMode
137/// \brief Overload of the != operator
138///
139/// \param left Left operand
140/// \param right Right operand
141///
142/// \return True if blending modes are different, false if they are equal
143///
144////////////////////////////////////////////////////////////
145SFML_GRAPHICS_API bool operator !=(const BlendMode& left, const BlendMode& right);

147////////////////////////////////////////////////////////////
148// Commonly used blending modes
149////////////////////////////////////////////////////////////
150SFML_GRAPHICS_API extern const BlendMode BlendAlpha; ///< Blend source and dest according to dest alpha
151SFML_GRAPHICS_API extern const BlendMode BlendAdd; ///< Add source to dest
152SFML_GRAPHICS_API extern const BlendMode BlendMultiply; ///< Multiply source and dest
153SFML_GRAPHICS_API extern const BlendMode BlendNone; ///< Overwrite dest with source
154

161////////////////////////////////////////////////////////////
162/// \class sf::BlendMode
163/// \ingroup graphics
164///
165/// sf::BlendMode is a class that represents a blend mode. A blend
166/// mode determines how the colors of an object you draw are
167/// mixed with the colors that are already in the buffer.
168///
169/// The class is composed of 6 components, each of which has its
170/// own public member variable:
171/// \li %Color Source Factor (@ref colorSrcFactor)
172/// \li %Color Destination Factor (@ref colorDstFactor)
173/// \li %Color Blend Equation (@ref colorEquation)
174/// \li Alpha Source Factor (@ref alphaSrcFactor)
175/// \li Alpha Destination Factor (@ref alphaDstFactor)
176/// \li Alpha Blend Equation (@ref alphaEquation)
177///
178/// The source factor specifies how the pixel you are drawing contributes
179/// to the final color. The destination factor specifies how the pixel
180/// already drawn in the buffer contributes to the final color.
181///
182/// The color channels RGB (red, green, blue; simply referred to as
183/// color) and A (alpha; the transparency) can be treated separately. This
184/// separation can be useful for specific blend modes, but most often you
185/// won't need it and will simply treat the color as a single unit.
186///
187/// The blend factors and equations correspond to their OpenGL equivalents.
188/// In general, the color of the resulting pixel is calculated according
189/// to the following formula (\a src is the color of the source pixel, \a dst
190/// the color of the destination pixel, the other variables correspond to the
191/// public members, with the equations being + or - operators):
192/// \code
193/// dst.rgb = colorSrcFactor * src.rgb (colorEquation) colorDstFactor * dst.rgb
194/// dst.a = alphaSrcFactor * src.a (alphaEquation) alphaDstFactor * dst.a
195/// \endcode
196/// All factors and colors are represented as floating point numbers between
197/// 0 and 1. Where necessary, the result is clamped to fit in that range.
198///
199/// The most common blending modes are defined as constants
200/// in the sf namespace:
201///
202/// \code
203/// sf::BlendMode alphaBlending = sf::BlendAlpha;
204/// sf::BlendMode additiveBlending = sf::BlendAdd;
205/// sf::BlendMode multiplicativeBlending = sf::BlendMultiply;
206/// sf::BlendMode noBlending = sf::BlendNone;
207/// \endcode
208///
209/// In SFML, a blend mode can be specified every time you draw a sf::Drawable
210/// object to a render target. It is part of the sf::RenderStates compound
211/// that is passed to the member function sf::RenderTarget::draw().
212///
213/// \see sf::RenderStates, sf::RenderTarget
214///
