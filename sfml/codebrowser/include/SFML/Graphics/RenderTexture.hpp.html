43////////////////////////////////////////////////////////////
44/// \brief Target for off-screen 2D rendering into a texture
45///
46////////////////////////////////////////////////////////////
47class SFML_GRAPHICS_API RenderTexture : public RenderTarget
48{
49public:
50
51 ////////////////////////////////////////////////////////////
52 /// \brief Default constructor
53 ///
54 /// Constructs an empty, invalid render-texture. You must
55 /// call create to have a valid render-texture.
56 ///
57 /// \see create
58 ///
59 ////////////////////////////////////////////////////////////
60 RenderTexture();
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Destructor
64 ///
65 ////////////////////////////////////////////////////////////
66 virtual ~RenderTexture();
67
68 ////////////////////////////////////////////////////////////
69 /// \brief Create the render-texture
70 ///
71 /// Before calling this function, the render-texture is in
72 /// an invalid state, thus it is mandatory to call it before
73 /// doing anything with the render-texture.
74 /// The last parameter, \a depthBuffer, is useful if you want
75 /// to use the render-texture for 3D OpenGL rendering that requires
76 /// a depth buffer. Otherwise it is unnecessary, and you should
77 /// leave this parameter to false (which is its default value).
78 ///
79 /// \param width Width of the render-texture
80 /// \param height Height of the render-texture
81 /// \param depthBuffer Do you want this render-texture to have a depth buffer?
82 ///
83 /// \return True if creation has been successful
84 ///
85 ////////////////////////////////////////////////////////////
86 bool create(unsigned int width, unsigned int height, bool depthBuffer = false);
87
88 ////////////////////////////////////////////////////////////
89 /// \brief Enable or disable texture smoothing
90 ///
91 /// This function is similar to Texture::setSmooth.
92 /// This parameter is disabled by default.
93 ///
94 /// \param smooth True to enable smoothing, false to disable it
95 ///
96 /// \see isSmooth
97 ///
98 ////////////////////////////////////////////////////////////
99 void setSmooth(bool smooth);
100
101 ////////////////////////////////////////////////////////////
102 /// \brief Tell whether the smooth filtering is enabled or not
103 ///
104 /// \return True if texture smoothing is enabled
105 ///
106 /// \see setSmooth
107 ///
108 ////////////////////////////////////////////////////////////
109 bool isSmooth() const;
110
111 ////////////////////////////////////////////////////////////
112 /// \brief Enable or disable texture repeating
113 ///
114 /// This function is similar to Texture::setRepeated.
115 /// This parameter is disabled by default.
116 ///
117 /// \param repeated True to enable repeating, false to disable it
118 ///
119 /// \see isRepeated
120 ///
121 ////////////////////////////////////////////////////////////
122 void setRepeated(bool repeated);
123
124 ////////////////////////////////////////////////////////////
125 /// \brief Tell whether the texture is repeated or not
126 ///
127 /// \return True if texture is repeated
128 ///
129 /// \see setRepeated
130 ///
131 ////////////////////////////////////////////////////////////
132 bool isRepeated() const;
133
134 ////////////////////////////////////////////////////////////
135 /// \brief Generate a mipmap using the current texture data
136 ///
137 /// This function is similar to Texture::generateMipmap and operates
138 /// on the texture used as the target for drawing.
139 /// Be aware that any draw operation may modify the base level image data.
140 /// For this reason, calling this function only makes sense after all
141 /// drawing is completed and display has been called. Not calling display
142 /// after subsequent drawing will lead to undefined behavior if a mipmap
143 /// had been previously generated.
144 ///
145 /// \return True if mipmap generation was successful, false if unsuccessful
146 ///
147 ////////////////////////////////////////////////////////////
148 bool generateMipmap();
149
150 ////////////////////////////////////////////////////////////
151 /// \brief Activate or deactivate the render-texture for rendering
152 ///
153 /// This function makes the render-texture's context current for
154 /// future OpenGL rendering operations (so you shouldn't care
155 /// about it if you're not doing direct OpenGL stuff).
156 /// Only one context can be current in a thread, so if you
157 /// want to draw OpenGL geometry to another render target
158 /// (like a RenderWindow) don't forget to activate it again.
159 ///
160 /// \param active True to activate, false to deactivate
161 ///
162 /// \return True if operation was successful, false otherwise
163 ///
164 ////////////////////////////////////////////////////////////
165 bool setActive(bool active = true);
166
167 ////////////////////////////////////////////////////////////
168 /// \brief Update the contents of the target texture
169 ///
170 /// This function updates the target texture with what
171 /// has been drawn so far. Like for windows, calling this
172 /// function is mandatory at the end of rendering. Not calling
173 /// it may leave the texture in an undefined state.
174 ///
175 ////////////////////////////////////////////////////////////
176 void display();
177
178 ////////////////////////////////////////////////////////////
179 /// \brief Return the size of the rendering region of the texture
180 ///
181 /// The returned value is the size that you passed to
182 /// the create function.
183 ///
184 /// \return Size in pixels
185 ///
186 ////////////////////////////////////////////////////////////
187 virtual Vector2u getSize() const;
188
189 ////////////////////////////////////////////////////////////
190 /// \brief Get a read-only reference to the target texture
191 ///
192 /// After drawing to the render-texture and calling Display,
193 /// you can retrieve the updated texture using this function,
194 /// and draw it using a sprite (for example).
195 /// The internal sf::Texture of a render-texture is always the
196 /// same instance, so that it is possible to call this function
197 /// once and keep a reference to the texture even after it is
198 /// modified.
199 ///
200 /// \return Const reference to the texture
201 ///
202 ////////////////////////////////////////////////////////////
203 const Texture& getTexture() const;
204
205private:
206
207 ////////////////////////////////////////////////////////////
208 // Member data
209 ////////////////////////////////////////////////////////////
210 priv::RenderTextureImpl* m_impl; ///< Platform/hardware specific implementation
211 Texture m_texture; ///< Target texture to draw on
212};

221/// \class sf::RenderTexture
222/// \ingroup graphics
223///
224/// sf::RenderTexture is the little brother of sf::RenderWindow.
225/// It implements the same 2D drawing and OpenGL-related functions
226/// (see their base class sf::RenderTarget for more details),
227/// the difference is that the result is stored in an off-screen
228/// texture rather than being show in a window.
229///
230/// Rendering to a texture can be useful in a variety of situations:
231/// \li precomputing a complex static texture (like a level's background from multiple tiles)
232/// \li applying post-effects to the whole scene with shaders
233/// \li creating a sprite from a 3D object rendered with OpenGL
234/// \li etc.
235///
236/// Usage example:
237///
238/// \code
239/// // Create a new render-window
240/// sf::RenderWindow window(sf::VideoMode(800, 600), "SFML window");
241///
242/// // Create a new render-texture
243/// sf::RenderTexture texture;
244/// if (!texture.create(500, 500))
245/// return -1;
246///
247/// // The main loop
248/// while (window.isOpen())
249/// {
250/// // Event processing
251/// // ...
252///
253/// // Clear the whole texture with red color
254/// texture.clear(sf::Color::Red);
255///
256/// // Draw stuff to the texture
257/// texture.draw(sprite); // sprite is a sf::Sprite
258/// texture.draw(shape); // shape is a sf::Shape
259/// texture.draw(text); // text is a sf::Text
260///
261/// // We're done drawing to the texture
262/// texture.display();
263///
264/// // Now we start rendering to the window, clear it first
265/// window.clear();
266///
267/// // Draw the texture
268/// sf::Sprite sprite(texture.getTexture());
269/// window.draw(sprite);
270///
271/// // End the current frame and display its contents on screen
272/// window.display();
273/// }
274/// \endcode
275///
276/// Like sf::RenderWindow, sf::RenderTexture is still able to render direct
277/// OpenGL stuff. It is even possible to mix together OpenGL calls
278/// and regular SFML drawing commands. If you need a depth buffer for
279/// 3D rendering, don't forget to request it when calling RenderTexture::create.
280///
281/// \see sf::RenderTarget, sf::RenderWindow, sf::View, sf::Texture
282///
