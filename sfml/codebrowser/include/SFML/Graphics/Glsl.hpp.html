99 /// \brief 4D float vector (\p vec4 in GLSL)
100 ///
101 /// 4D float vectors can be implicitly converted from sf::Color
102 /// instances. Each color channel is normalized from integers
103 /// in [0, 255] to floating point values in [0, 1].
104 /// \code
105 /// sf::Glsl::Vec4 zeroVector;
106 /// sf::Glsl::Vec4 vector(1.f, 2.f, 3.f, 4.f);
107 /// sf::Glsl::Vec4 color = sf::Color::Cyan;
108 /// \endcode
109 ////////////////////////////////////////////////////////////
110 typedef implementation-defined Vec4;
111
112 ////////////////////////////////////////////////////////////
113 /// \brief 4D int vector (\p ivec4 in GLSL)
114 ///
115 /// 4D int vectors can be implicitly converted from sf::Color
116 /// instances. Each color channel remains unchanged inside
117 /// the integer interval [0, 255].
118 /// \code
119 /// sf::Glsl::Ivec4 zeroVector;
120 /// sf::Glsl::Ivec4 vector(1, 2, 3, 4);
121 /// sf::Glsl::Ivec4 color = sf::Color::Cyan;
122 /// \endcode

132 ////////////////////////////////////////////////////////////
133 /// \brief 3x3 float matrix (\p mat3 in GLSL)
134 ///
135 /// The matrix can be constructed from an array with 3x3
136 /// elements, aligned in column-major order. For example,
137 /// a translation by (x, y) looks as follows:
138 /// \code
139 /// float array[9] =
140 /// {
141 /// 1, 0, 0,
142 /// 0, 1, 0,
143 /// x, y, 1
144 /// };
145 ///
146 /// sf::Glsl::Mat3 matrix(array);
147 /// \endcode
148 ///
149 /// Mat3 can also be implicitly converted from sf::Transform:
150 /// \code
151 /// sf::Transform transform;
152 /// sf::Glsl::Mat3 matrix = transform;
153 /// \endcode
154 ////////////////////////////////////////////////////////////
155 typedef implementation-defined Mat3;
156
157 ////////////////////////////////////////////////////////////
158 /// \brief 4x4 float matrix (\p mat4 in GLSL)
159 ///
160 /// The matrix can be constructed from an array with 4x4
161 /// elements, aligned in column-major order. For example,
162 /// a translation by (x, y, z) looks as follows:
163 /// \code
164 /// float array[16] =
165 /// {
166 /// 1, 0, 0, 0,
167 /// 0, 1, 0, 0,
168 /// 0, 0, 1, 0,
169 /// x, y, z, 1
170 /// };
171 ///
172 /// sf::Glsl::Mat4 matrix(array);
173 /// \endcode
174 ///
175 /// Mat4 can also be implicitly converted from sf::Transform:
176 /// \code
177 /// sf::Transform transform;
178 /// sf::Glsl::Mat4 matrix = transform;
179 /// \endcode
180 ////////////////////////////////////////////////////////////
181 typedef implementation-defined Mat4;

200/// \namespace sf::Glsl
201/// \ingroup graphics
202///
203/// \details The sf::Glsl namespace contains types that match
204/// their equivalents in GLSL, the OpenGL shading language.
205/// These types are exclusively used by the sf::Shader class.
206///
207/// Types that already exist in SFML, such as \ref sf::Vector2<T>
208/// and \ref sf::Vector3<T>, are reused as typedefs, so you can use
209/// the types in this namespace as well as the original ones.
210/// Others are newly defined, such as Glsl::Vec4 or Glsl::Mat3. Their
211/// actual type is an implementation detail and should not be used.
212///
213/// All vector types support a default constructor that
214/// initializes every component to zero, in addition to a
215/// constructor with one parameter for each component.
216/// The components are stored in member variables called
217/// x, y, z, and w.
218///
219/// All matrix types support a constructor with a float*
220/// parameter that points to a float array of the appropriate
221/// size (that is, 9 in a 3x3 matrix, 16 in a 4x4 matrix).
222/// Furthermore, they can be converted from sf::Transform
223/// objects.
224///
225/// \see sf::Shader
226///
