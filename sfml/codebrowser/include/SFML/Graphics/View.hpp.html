40/// \brief 2D camera that defines what region is shown on screen

43class SFML_GRAPHICS_API View
44{
45public:
46
47 ////////////////////////////////////////////////////////////
48 /// \brief Default constructor
49 ///
50 /// This constructor creates a default view of (0, 0, 1000, 1000)
51 ///
52 ////////////////////////////////////////////////////////////
53 View();
54
55 ////////////////////////////////////////////////////////////
56 /// \brief Construct the view from a rectangle
57 ///
58 /// \param rectangle Rectangle defining the zone to display
59 ///
60 ////////////////////////////////////////////////////////////
61 explicit View(const FloatRect& rectangle);
62
63 ////////////////////////////////////////////////////////////
64 /// \brief Construct the view from its center and size
65 ///
66 /// \param center Center of the zone to display
67 /// \param size Size of zone to display
68 ///
69 ////////////////////////////////////////////////////////////
70 View(const Vector2f& center, const Vector2f& size);
71
72 ////////////////////////////////////////////////////////////
73 /// \brief Set the center of the view
74 ///
75 /// \param x X coordinate of the new center
76 /// \param y Y coordinate of the new center
77 ///
78 /// \see setSize, getCenter
79 ///
80 ////////////////////////////////////////////////////////////
81 void setCenter(float x, float y);
82
83 ////////////////////////////////////////////////////////////
84 /// \brief Set the center of the view
85 ///
86 /// \param center New center
87 ///
88 /// \see setSize, getCenter
89 ///
90 ////////////////////////////////////////////////////////////
91 void setCenter(const Vector2f& center);
92
93 ////////////////////////////////////////////////////////////
94 /// \brief Set the size of the view
95 ///
96 /// \param width New width of the view
97 /// \param height New height of the view
98 ///
99 /// \see setCenter, getCenter
100 ///
101 ////////////////////////////////////////////////////////////
102 void setSize(float width, float height);
103
104 ////////////////////////////////////////////////////////////
105 /// \brief Set the size of the view
106 ///
107 /// \param size New size
108 ///
109 /// \see setCenter, getCenter
110 ///
111 ////////////////////////////////////////////////////////////
112 void setSize(const Vector2f& size);
113
114 ////////////////////////////////////////////////////////////
115 /// \brief Set the orientation of the view
116 ///
117 /// The default rotation of a view is 0 degree.
118 ///
119 /// \param angle New angle, in degrees
120 ///
121 /// \see getRotation
122 ///
123 ////////////////////////////////////////////////////////////
124 void setRotation(float angle);
125
126 ////////////////////////////////////////////////////////////
127 /// \brief Set the target viewport
128 ///
129 /// The viewport is the rectangle into which the contents of the
130 /// view are displayed, expressed as a factor (between 0 and 1)
131 /// of the size of the RenderTarget to which the view is applied.
132 /// For example, a view which takes the left side of the target would
133 /// be defined with View.setViewport(sf::FloatRect(0, 0, 0.5, 1)).
134 /// By default, a view has a viewport which covers the entire target.
135 ///
136 /// \param viewport New viewport rectangle
137 ///
138 /// \see getViewport
139 ///
140 ////////////////////////////////////////////////////////////
141 void setViewport(const FloatRect& viewport);
142
143 ////////////////////////////////////////////////////////////
144 /// \brief Reset the view to the given rectangle
145 ///
146 /// Note that this function resets the rotation angle to 0.
147 ///
148 /// \param rectangle Rectangle defining the zone to display
149 ///
150 /// \see setCenter, setSize, setRotation
151 ///
152 ////////////////////////////////////////////////////////////
153 void reset(const FloatRect& rectangle);
154
155 ////////////////////////////////////////////////////////////
156 /// \brief Get the center of the view
157 ///
158 /// \return Center of the view
159 ///
160 /// \see getSize, setCenter
161 ///
162 ////////////////////////////////////////////////////////////
163 const Vector2f& getCenter() const;
164
165 ////////////////////////////////////////////////////////////
166 /// \brief Get the size of the view
167 ///
168 /// \return Size of the view
169 ///
170 /// \see getCenter, setSize
171 ///
172 ////////////////////////////////////////////////////////////
173 const Vector2f& getSize() const;
174
175 ////////////////////////////////////////////////////////////
176 /// \brief Get the current orientation of the view
177 ///
178 /// \return Rotation angle of the view, in degrees
179 ///
180 /// \see setRotation
181 ///
182 ////////////////////////////////////////////////////////////
183 float getRotation() const;
184
185 ////////////////////////////////////////////////////////////
186 /// \brief Get the target viewport rectangle of the view
187 ///
188 /// \return Viewport rectangle, expressed as a factor of the target size
189 ///
190 /// \see setViewport
191 ///
192 ////////////////////////////////////////////////////////////
193 const FloatRect& getViewport() const;
194
195 ////////////////////////////////////////////////////////////
196 /// \brief Move the view relatively to its current position
197 ///
198 /// \param offsetX X coordinate of the move offset
199 /// \param offsetY Y coordinate of the move offset
200 ///
201 /// \see setCenter, rotate, zoom
202 ///
203 ////////////////////////////////////////////////////////////
204 void move(float offsetX, float offsetY);
205
206 ////////////////////////////////////////////////////////////
207 /// \brief Move the view relatively to its current position
208 ///
209 /// \param offset Move offset
210 ///
211 /// \see setCenter, rotate, zoom
212 ///
213 ////////////////////////////////////////////////////////////
214 void move(const Vector2f& offset);
215
216 ////////////////////////////////////////////////////////////
217 /// \brief Rotate the view relatively to its current orientation
218 ///
219 /// \param angle Angle to rotate, in degrees
220 ///
221 /// \see setRotation, move, zoom
222 ///
223 ////////////////////////////////////////////////////////////
224 void rotate(float angle);
225
226 ////////////////////////////////////////////////////////////
227 /// \brief Resize the view rectangle relatively to its current size
228 ///
229 /// Resizing the view simulates a zoom, as the zone displayed on
230 /// screen grows or shrinks.
231 /// \a factor is a multiplier:
232 /// \li 1 keeps the size unchanged
233 /// \li > 1 makes the view bigger (objects appear smaller)
234 /// \li < 1 makes the view smaller (objects appear bigger)
235 ///
236 /// \param factor Zoom factor to apply
237 ///
238 /// \see setSize, move, rotate
239 ///
240 ////////////////////////////////////////////////////////////
241 void zoom(float factor);
242
243 ////////////////////////////////////////////////////////////
244 /// \brief Get the projection transform of the view
245 ///
246 /// This function is meant for internal use only.
247 ///
248 /// \return Projection transform defining the view
249 ///
250 /// \see getInverseTransform
251 ///
252 ////////////////////////////////////////////////////////////
253 const Transform& getTransform() const;
254
255 ////////////////////////////////////////////////////////////
256 /// \brief Get the inverse projection transform of the view
257 ///
258 /// This function is meant for internal use only.
259 ///
260 /// \return Inverse of the projection transform defining the view
261 ///
262 /// \see getTransform
263 ///
264 ////////////////////////////////////////////////////////////
265 const Transform& getInverseTransform() const;
266
267private:
268
269 ////////////////////////////////////////////////////////////
270 // Member data
271 ////////////////////////////////////////////////////////////
272 Vector2f m_center; ///< Center of the view, in scene coordinates
273 Vector2f m_size; ///< Size of the view, in scene coordinates
274 float m_rotation; ///< Angle of rotation of the view rectangle, in degrees
275 FloatRect m_viewport; ///< Viewport rectangle, expressed as a factor of the render-target's size
276 mutable Transform m_transform; ///< Precomputed projection transform corresponding to the view
277 mutable Transform m_inverseTransform; ///< Precomputed inverse projection transform corresponding to the view
278 mutable bool m_transformUpdated; ///< Internal state telling if the transform needs to be updated
279 mutable bool m_invTransformUpdated; ///< Internal state telling if the inverse transform needs to be updated
280};

289/// \class sf::View
290/// \ingroup graphics
291///
292/// sf::View defines a camera in the 2D scene. This is a
293/// very powerful concept: you can scroll, rotate or zoom
294/// the entire scene without altering the way that your
295/// drawable objects are drawn.
296///
297/// A view is composed of a source rectangle, which defines
298/// what part of the 2D scene is shown, and a target viewport,
299/// which defines where the contents of the source rectangle
300/// will be displayed on the render target (window or texture).
301///
302/// The viewport allows to map the scene to a custom part
303/// of the render target, and can be used for split-screen
304/// or for displaying a minimap, for example. If the source
305/// rectangle doesn't have the same size as the viewport, its
306/// contents will be stretched to fit in.
307///
308/// To apply a view, you have to assign it to the render target.
309/// Then, objects drawn in this render target will be
310/// affected by the view until you use another view.
311///
312/// Usage example:
313/// \code
314/// sf::RenderWindow window;
315/// sf::View view;
316///
317/// // Initialize the view to a rectangle located at (100, 100) and with a size of 400x200
318/// view.reset(sf::FloatRect(100, 100, 400, 200));
319///
320/// // Rotate it by 45 degrees
321/// view.rotate(45);
322///
323/// // Set its target viewport to be half of the window
324/// view.setViewport(sf::FloatRect(0.f, 0.f, 0.5f, 1.f));
325///
326/// // Apply it
327/// window.setView(view);
328///
329/// // Render stuff
330/// window.draw(someSprite);
331///
332/// // Set the default view back
333/// window.setView(window.getDefaultView());
334///
335/// // Render stuff not affected by the view
336/// window.draw(someText);
337/// \endcode
338///
339/// See also the note on coordinates and undistorted rendering in sf::Transformable.
340///
341/// \see sf::RenderWindow, sf::RenderTexture
342///
