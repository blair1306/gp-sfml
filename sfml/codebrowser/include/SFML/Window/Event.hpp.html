40////////////////////////////////////////////////////////////
41/// \brief Defines a system event and its parameters
42///
43////////////////////////////////////////////////////////////
44class Event
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Size events parameters (Resized)
50 ///
51 ////////////////////////////////////////////////////////////
52 struct SizeEvent
53 {
54 unsigned int width; ///< New width, in pixels
55 unsigned int height; ///< New height, in pixels
56 };
57
58 ////////////////////////////////////////////////////////////
59 /// \brief Keyboard event parameters (KeyPressed, KeyReleased)
60 ///
61 ////////////////////////////////////////////////////////////
62 struct KeyEvent
63 {
64 Keyboard::Key code; ///< Code of the key that has been pressed
65 bool alt; ///< Is the Alt key pressed?
66 bool control; ///< Is the Control key pressed?
67 bool shift; ///< Is the Shift key pressed?
68 bool system; ///< Is the System key pressed?
69 };
70
71 ////////////////////////////////////////////////////////////
72 /// \brief Text event parameters (TextEntered)
73 ///
74 ////////////////////////////////////////////////////////////
75 struct TextEvent
76 {
77 Uint32 unicode; ///< UTF-32 Unicode value of the character
78 };
79
80 ////////////////////////////////////////////////////////////
81 /// \brief Mouse move event parameters (MouseMoved)
82 ///
83 ////////////////////////////////////////////////////////////
84 struct MouseMoveEvent
85 {
86 int x; ///< X position of the mouse pointer, relative to the left of the owner window
87 int y; ///< Y position of the mouse pointer, relative to the top of the owner window
88 };
89
90 ////////////////////////////////////////////////////////////
91 /// \brief Mouse buttons events parameters
92 /// (MouseButtonPressed, MouseButtonReleased)
93 ///
94 ////////////////////////////////////////////////////////////
95 struct MouseButtonEvent
96 {
97 Mouse::Button button; ///< Code of the button that has been pressed
98 int x; ///< X position of the mouse pointer, relative to the left of the owner window
99 int y; ///< Y position of the mouse pointer, relative to the top of the owner window
100 };
101
102 ////////////////////////////////////////////////////////////
103 /// \brief Mouse wheel events parameters (MouseWheelMoved)
104 ///
105 /// \deprecated This event is deprecated and potentially inaccurate.
106 /// Use MouseWheelScrollEvent instead.
107 ///
108 ////////////////////////////////////////////////////////////
109 struct MouseWheelEvent
110 {
111 int delta; ///< Number of ticks the wheel has moved (positive is up, negative is down)
112 int x; ///< X position of the mouse pointer, relative to the left of the owner window
113 int y; ///< Y position of the mouse pointer, relative to the top of the owner window
114 };
115
116 ////////////////////////////////////////////////////////////
117 /// \brief Mouse wheel events parameters (MouseWheelScrolled)
118 ///
119 ////////////////////////////////////////////////////////////
120 struct MouseWheelScrollEvent
121 {
122 Mouse::Wheel wheel; ///< Which wheel (for mice with multiple ones)
123 float delta; ///< Wheel offset (positive is up/left, negative is down/right). High-precision mice may use non-integral offsets.
124 int x; ///< X position of the mouse pointer, relative to the left of the owner window
125 int y; ///< Y position of the mouse pointer, relative to the top of the owner window
126 };
127
128 ////////////////////////////////////////////////////////////
129 /// \brief Joystick connection events parameters
130 /// (JoystickConnected, JoystickDisconnected)
131 ///
132 ////////////////////////////////////////////////////////////
133 struct JoystickConnectEvent
134 {
135 unsigned int joystickId; ///< Index of the joystick (in range [0 .. Joystick::Count - 1])
136 };
137
138 ////////////////////////////////////////////////////////////
139 /// \brief Joystick axis move event parameters (JoystickMoved)
140 ///
141 ////////////////////////////////////////////////////////////
142 struct JoystickMoveEvent
143 {
144 unsigned int joystickId; ///< Index of the joystick (in range [0 .. Joystick::Count - 1])
145 Joystick::Axis axis; ///< Axis on which the joystick moved
146 float position; ///< New position on the axis (in range [-100 .. 100])
147 };
148
149 ////////////////////////////////////////////////////////////
150 /// \brief Joystick buttons events parameters
151 /// (JoystickButtonPressed, JoystickButtonReleased)
152 ///
153 ////////////////////////////////////////////////////////////
154 struct JoystickButtonEvent
155 {
156 unsigned int joystickId; ///< Index of the joystick (in range [0 .. Joystick::Count - 1])
157 unsigned int button; ///< Index of the button that has been pressed (in range [0 .. Joystick::ButtonCount - 1])
158 };
159
160 ////////////////////////////////////////////////////////////
161 /// \brief Touch events parameters (TouchBegan, TouchMoved, TouchEnded)
162 ///
163 ////////////////////////////////////////////////////////////
164 struct TouchEvent
165 {
166 unsigned int finger; ///< Index of the finger in case of multi-touch events
167 int x; ///< X position of the touch, relative to the left of the owner window
168 int y; ///< Y position of the touch, relative to the top of the owner window
169 };
170
171 ////////////////////////////////////////////////////////////
172 /// \brief Sensor event parameters (SensorChanged)
173 ///
174 ////////////////////////////////////////////////////////////
175 struct SensorEvent
176 {
177 Sensor::Type type; ///< Type of the sensor
178 float x; ///< Current value of the sensor on X axis
179 float y; ///< Current value of the sensor on Y axis
180 float z; ///< Current value of the sensor on Z axis
181 };
182
183 ////////////////////////////////////////////////////////////
184 /// \brief Enumeration of the different types of events
185 ///
186 ////////////////////////////////////////////////////////////
187 enum EventType
188 {
189 Closed, ///< The window requested to be closed (no data)
190 Resized, ///< The window was resized (data in event.size)
191 LostFocus, ///< The window lost the focus (no data)
192 GainedFocus, ///< The window gained the focus (no data)
193 TextEntered, ///< A character was entered (data in event.text)
194 KeyPressed, ///< A key was pressed (data in event.key)
195 KeyReleased, ///< A key was released (data in event.key)
196 MouseWheelMoved, ///< The mouse wheel was scrolled (data in event.mouseWheel) (deprecated)
197 MouseWheelScrolled, ///< The mouse wheel was scrolled (data in event.mouseWheelScroll)
198 MouseButtonPressed, ///< A mouse button was pressed (data in event.mouseButton)
199 MouseButtonReleased, ///< A mouse button was released (data in event.mouseButton)
200 MouseMoved, ///< The mouse cursor moved (data in event.mouseMove)
201 MouseEntered, ///< The mouse cursor entered the area of the window (no data)
202 MouseLeft, ///< The mouse cursor left the area of the window (no data)
203 JoystickButtonPressed, ///< A joystick button was pressed (data in event.joystickButton)
204 JoystickButtonReleased, ///< A joystick button was released (data in event.joystickButton)
205 JoystickMoved, ///< The joystick moved along an axis (data in event.joystickMove)
206 JoystickConnected, ///< A joystick was connected (data in event.joystickConnect)
207 JoystickDisconnected, ///< A joystick was disconnected (data in event.joystickConnect)
208 TouchBegan, ///< A touch event began (data in event.touch)
209 TouchMoved, ///< A touch moved (data in event.touch)
210 TouchEnded, ///< A touch event ended (data in event.touch)
211 SensorChanged, ///< A sensor value changed (data in event.sensor)
212
213 Count ///< Keep last -- the total number of event types
214 };
215
216 ////////////////////////////////////////////////////////////
217 // Member data
218 ////////////////////////////////////////////////////////////
219 EventType type; ///< Type of the event
220
221 union
222 {
223 SizeEvent size; ///< Size event parameters (Event::Resized)
224 KeyEvent key; ///< Key event parameters (Event::KeyPressed, Event::KeyReleased)
225 TextEvent text; ///< Text event parameters (Event::TextEntered)
226 MouseMoveEvent mouseMove; ///< Mouse move event parameters (Event::MouseMoved)
227 MouseButtonEvent mouseButton; ///< Mouse button event parameters (Event::MouseButtonPressed, Event::MouseButtonReleased)
228 MouseWheelEvent mouseWheel; ///< Mouse wheel event parameters (Event::MouseWheelMoved) (deprecated)
229 MouseWheelScrollEvent mouseWheelScroll; ///< Mouse wheel event parameters (Event::MouseWheelScrolled)
230 JoystickMoveEvent joystickMove; ///< Joystick move event parameters (Event::JoystickMoved)
231 JoystickButtonEvent joystickButton; ///< Joystick button event parameters (Event::JoystickButtonPressed, Event::JoystickButtonReleased)
232 JoystickConnectEvent joystickConnect; ///< Joystick (dis)connect event parameters (Event::JoystickConnected, Event::JoystickDisconnected)
233 TouchEvent touch; ///< Touch events parameters (Event::TouchBegan, Event::TouchMoved, Event::TouchEnded)
234 SensorEvent sensor; ///< Sensor event parameters (Event::SensorChanged)
235 };
236};
237
238} // namespace sf

241#endif // SFML_EVENT_HPP

244////////////////////////////////////////////////////////////
245/// \class sf::Event
246/// \ingroup window
247///
248/// sf::Event holds all the informations about a system event
249/// that just happened. Events are retrieved using the
250/// sf::Window::pollEvent and sf::Window::waitEvent functions.
251///
252/// A sf::Event instance contains the type of the event
253/// (mouse moved, key pressed, window closed, ...) as well
254/// as the details about this particular event. Please note that
255/// the event parameters are defined in a union, which means that
256/// only the member matching the type of the event will be properly
257/// filled; all other members will have undefined values and must not
258/// be read if the type of the event doesn't match. For example,
259/// if you received a KeyPressed event, then you must read the
260/// event.key member, all other members such as event.mouseMove
261/// or event.text will have undefined values.
262///
263/// Usage example:
264/// \code
265/// sf::Event event;
266/// while (window.pollEvent(event))
267/// {
268/// // Request for closing the window
269/// if (event.type == sf::Event::Closed)
270/// window.close();
271///
272/// // The escape key was pressed
273/// if ((event.type == sf::Event::KeyPressed) && (event.key.code == sf::Keyboard::Escape))
274/// window.close();
275///
276/// // The window was resized
277/// if (event.type == sf::Event::Resized)
278/// doSomethingWithTheNewSize(event.size.width, event.size.height);
279///
280/// // etc ...
281/// }
282/// \endcode
283///
