25#ifndef SFML_WINDOWSTYLE_HPP
26#define SFML_WINDOWSTYLE_HPP
27
28
29namespace sf
30{
31namespace Style
32{
33 ////////////////////////////////////////////////////////////
34 /// \ingroup window
35 /// \brief Enumeration of the window styles
36 ///
37 ////////////////////////////////////////////////////////////
38 enum
39 {
40 None = 0, ///< No border / title bar (this flag and all others are mutually exclusive)
41 Titlebar = 1 << 0, ///< Title bar + fixed border
42 Resize = 1 << 1, ///< Title bar + resizable border + maximize button
43 Close = 1 << 2, ///< Title bar + close button
44 Fullscreen = 1 << 3, ///< Fullscreen mode (this flag and all others are mutually exclusive)
45
46 Default = Titlebar | Resize | Close ///< Default window style
47 };
48}
49
50} // namespace sf
51
52
53#endif // SFML_WINDOWSTYLE_HPP
