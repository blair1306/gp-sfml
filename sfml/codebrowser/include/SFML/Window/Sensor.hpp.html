39/// \brief Give access to the real-time state of the sensors
40///
41////////////////////////////////////////////////////////////
42class SFML_WINDOW_API Sensor
43{
44public:
45
46 ////////////////////////////////////////////////////////////
47 /// \brief Sensor type
48 ///
49 ////////////////////////////////////////////////////////////
50 enum Type
51 {
52 Accelerometer, ///< Measures the raw acceleration (m/s^2)
53 Gyroscope, ///< Measures the raw rotation rates (degrees/s)
54 Magnetometer, ///< Measures the ambient magnetic field (micro-teslas)
55 Gravity, ///< Measures the direction and intensity of gravity, independent of device acceleration (m/s^2)
56 UserAcceleration, ///< Measures the direction and intensity of device acceleration, independent of the gravity (m/s^2)
57 Orientation, ///< Measures the absolute 3D orientation (degrees)
58
59 Count ///< Keep last -- the total number of sensor types
60 };
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Check if a sensor is available on the underlying platform
64 ///
65 /// \param sensor Sensor to check
66 ///
67 /// \return True if the sensor is available, false otherwise
68 ///
69 ////////////////////////////////////////////////////////////
70 static bool isAvailable(Type sensor);
71
72 ////////////////////////////////////////////////////////////
73 /// \brief Enable or disable a sensor
74 ///
75 /// All sensors are disabled by default, to avoid consuming too
76 /// much battery power. Once a sensor is enabled, it starts
77 /// sending events of the corresponding type.
78 ///
79 /// This function does nothing if the sensor is unavailable.
80 ///
81 /// \param sensor Sensor to enable
82 /// \param enabled True to enable, false to disable
83 ///
84 ////////////////////////////////////////////////////////////
85 static void setEnabled(Type sensor, bool enabled);
86
87 ////////////////////////////////////////////////////////////
88 /// \brief Get the current sensor value
89 ///
90 /// \param sensor Sensor to read
91 ///
92 /// \return The current sensor value
93 ///
94 ////////////////////////////////////////////////////////////
95 static Vector3f getValue(Type sensor);
96};

104////////////////////////////////////////////////////////////
105/// \class sf::Sensor
106/// \ingroup window
107///
108/// sf::Sensor provides an interface to the state of the
109/// various sensors that a device provides. It only contains static
110/// functions, so it's not meant to be instantiated.
111///
112/// This class allows users to query the sensors values at any
113/// time and directly, without having to deal with a window and
114/// its events. Compared to the SensorChanged event, sf::Sensor
115/// can retrieve the state of a sensor at any time (you don't need to
116/// store and update its current value on your side).
117///
118/// Depending on the OS and hardware of the device (phone, tablet, ...),
119/// some sensor types may not be available. You should always check
120/// the availability of a sensor before trying to read it, with the
121/// sf::Sensor::isAvailable function.
122///
123/// You may wonder why some sensor types look so similar, for example
124/// Accelerometer and Gravity / UserAcceleration. The first one
125/// is the raw measurement of the acceleration, and takes into account
126/// both the earth gravity and the user movement. The others are
127/// more precise: they provide these components separately, which is
128/// usually more useful. In fact they are not direct sensors, they
129/// are computed internally based on the raw acceleration and other sensors.
130/// This is exactly the same for Gyroscope vs Orientation.
131///
132/// Because sensors consume a non-negligible amount of current, they are
133/// all disabled by default. You must call sf::Sensor::setEnabled for each
134/// sensor in which you are interested.
135///
136/// Usage example:
137/// \code
138/// if (sf::Sensor::isAvailable(sf::Sensor::Gravity))
139/// {
140/// // gravity sensor is available
141/// }
142///
143/// // enable the gravity sensor
144/// sf::Sensor::setEnabled(sf::Sensor::Gravity, true);
145///
146/// // get the current value of gravity
147/// sf::Vector3f gravity = sf::Sensor::getValue(sf::Sensor::Gravity);
148/// \endcode
149///
