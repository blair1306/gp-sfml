47/// \brief Class holding a valid drawing context
48///
49////////////////////////////////////////////////////////////
50class SFML_WINDOW_API Context : GlResource, NonCopyable
51{
52public:
53
54 ////////////////////////////////////////////////////////////
55 /// \brief Default constructor
56 ///
57 /// The constructor creates and activates the context
58 ///
59 ////////////////////////////////////////////////////////////
60 Context();
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Destructor
64 ///
65 /// The destructor deactivates and destroys the context
66 ///
67 ////////////////////////////////////////////////////////////
68 ~Context();
69
70 ////////////////////////////////////////////////////////////
71 /// \brief Activate or deactivate explicitly the context
72 ///
73 /// \param active True to activate, false to deactivate
74 ///
75 /// \return True on success, false on failure
76 ///
77 ////////////////////////////////////////////////////////////
78 bool setActive(bool active);
79
80 ////////////////////////////////////////////////////////////
81 /// \brief Get the settings of the context
82 ///
83 /// Note that these settings may be different than the ones
84 /// passed to the constructor; they are indeed adjusted if the
85 /// original settings are not directly supported by the system.
86 ///
87 /// \return Structure containing the settings
88 ///
89 ////////////////////////////////////////////////////////////
90 const ContextSettings& getSettings() const;
91
92 ////////////////////////////////////////////////////////////
93 /// \brief Check whether a given OpenGL extension is available
94 ///
95 /// \param name Name of the extension to check for
96 ///
97 /// \return True if available, false if unavailable

99 ////////////////////////////////////////////////////////////
100 static bool isExtensionAvailable(const char* name);
101
102 ////////////////////////////////////////////////////////////
103 /// \brief Get the address of an OpenGL function
104 ///
105 /// \param name Name of the function to get the address of
106 ///
107 /// \return Address of the OpenGL function, 0 on failure
108 ///
109 ////////////////////////////////////////////////////////////
110 static GlFunctionPointer getFunction(const char* name);
111
112 ////////////////////////////////////////////////////////////
113 /// \brief Get the currently active context
114 ///
115 /// \return The currently active context or NULL if none is active
116 ///
117 ////////////////////////////////////////////////////////////
118 static const Context* getActiveContext();
119
120 ////////////////////////////////////////////////////////////
121 /// \brief Construct a in-memory context
122 ///
123 /// This constructor is for internal use, you don't need

146////////////////////////////////////////////////////////////
147/// \class sf::Context
148/// \ingroup window
149///
150/// If you need to make OpenGL calls without having an
151/// active window (like in a thread), you can use an
152/// instance of this class to get a valid context.
153///
154/// Having a valid context is necessary for *every* OpenGL call.
155///
156/// Note that a context is only active in its current thread,
157/// if you create a new thread it will have no valid context
158/// by default.
159///
160/// To use a sf::Context instance, just construct it and let it
161/// live as long as you need a valid context. No explicit activation
162/// is needed, all it has to do is to exist. Its destructor
163/// will take care of deactivating and freeing all the attached
164/// resources.
165///
166/// Usage example:
167/// \code
168/// void threadFunction(void*)
169/// {
170/// sf::Context context;
171/// // from now on, you have a valid context
172///
173/// // you can make OpenGL calls
174/// glClear(GL_DEPTH_BUFFER_BIT);
175/// }
176/// // the context is automatically deactivated and destroyed
177/// // by the sf::Context destructor
178/// \endcode
179///
