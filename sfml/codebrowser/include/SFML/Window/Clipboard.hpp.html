38/// \brief Give access to the system clipboard

40////////////////////////////////////////////////////////////
41class SFML_WINDOW_API Clipboard
42{
43public:
44
45 ////////////////////////////////////////////////////////////
46 /// \brief Get the content of the clipboard as string data
47 ///
48 /// This function returns the content of the clipboard
49 /// as a string. If the clipboard does not contain string
50 /// it returns an empty sf::String object.
51 ///
52 /// \return Clipboard contents as sf::String object
53 ///
54 ////////////////////////////////////////////////////////////
55 static String getString();
56
57 ////////////////////////////////////////////////////////////
58 /// \brief Set the content of the clipboard as string data
59 ///
60 /// This function sets the content of the clipboard as a
61 /// string.
62 ///
63 /// \param text sf::String containing the data to be sent
64 /// to the clipboard
65 ///
66 ////////////////////////////////////////////////////////////
67 static void setString(const String& text);
68};

76////////////////////////////////////////////////////////////
77/// \class sf::Clipboard
78/// \ingroup window
79///
80/// sf::Clipboard provides an interface for getting and
81/// setting the contents of the system clipboard.
82///
83/// Usage example:
84/// \code
85/// // get the clipboard content as a string
86/// sf::String string = sf::Clipboard::getString();
87///
88/// // or use it in the event loop
89/// sf::Event event;
90/// while(window.pollEvent(event))
91/// {
92/// if(event.type == sf::Event::Closed)
93/// window.close();
94/// if(event.type == sf::Event::KeyPressed)
95/// {
96/// // Using Ctrl + V to paste a string into SFML
97/// if(event.key.control && event.key.code == sf::Keyboard::V)
98/// string = sf::Clipboard::getString();
99/// }
100/// }
101///
102/// // set the clipboard to a string
103/// sf::Clipboard::setString("Hello World!");
104/// \endcode
105///
106/// \see sf::String, sf::Event
107///
