42////////////////////////////////////////////////////////////
43/// \brief Cursor defines the appearance of a system cursor
44///
45////////////////////////////////////////////////////////////
46class SFML_WINDOW_API Cursor : NonCopyable
47{
48public:
49
50 ////////////////////////////////////////////////////////////
51 /// \brief Enumeration of the native system cursor types
52 ///
53 /// Refer to the following table to determine which cursor
54 /// is available on which platform.
55 ///
56 /// Type | Linux | Mac OS X | Windows
57 /// ------------------------------------|:-----:|:--------:|:--------:
58 /// sf::Cursor::Arrow | yes | yes | yes
59 /// sf::Cursor::ArrowWait | no | no | yes
60 /// sf::Cursor::Wait | yes | no | yes
61 /// sf::Cursor::Text | yes | yes | yes
62 /// sf::Cursor::Hand | yes | yes | yes
63 /// sf::Cursor::SizeHorizontal | yes | yes | yes
64 /// sf::Cursor::SizeVertical | yes | yes | yes
65 /// sf::Cursor::SizeTopLeftBottomRight | no | no | yes
66 /// sf::Cursor::SizeBottomLeftTopRight | no | no | yes
67 /// sf::Cursor::SizeAll | yes | no | yes
68 /// sf::Cursor::Cross | yes | yes | yes
69 /// sf::Cursor::Help | yes | no | yes
70 /// sf::Cursor::NotAllowed | yes | yes | yes
71 ///
72 ////////////////////////////////////////////////////////////
73 enum Type
74 {
75 Arrow, ///< Arrow cursor (default)
76 ArrowWait, ///< Busy arrow cursor
77 Wait, ///< Busy cursor
78 Text, ///< I-beam, cursor when hovering over a field allowing text entry
79 Hand, ///< Pointing hand cursor
80 SizeHorizontal, ///< Horizontal double arrow cursor
81 SizeVertical, ///< Vertical double arrow cursor
82 SizeTopLeftBottomRight, ///< Double arrow cursor going from top-left to bottom-right
83 SizeBottomLeftTopRight, ///< Double arrow cursor going from bottom-left to top-right
84 SizeAll, ///< Combination of SizeHorizontal and SizeVertical
85 Cross, ///< Crosshair cursor
86 Help, ///< Help cursor
87 NotAllowed ///< Action not allowed cursor
88 };
89
90public:
91
92 ////////////////////////////////////////////////////////////
93 /// \brief Default constructor
94 ///
95 /// This constructor doesn't actually create the cursor;
96 /// initially the new instance is invalid and must not be
97 /// used until either loadFromPixels() or loadFromSystem()
98 /// is called and successfully created a cursor.
99 ///
100 ////////////////////////////////////////////////////////////
101 Cursor();
102
103 ////////////////////////////////////////////////////////////
104 /// \brief Destructor
105 ///
106 /// This destructor releases the system resources
107 /// associated with this cursor, if any.
108 ///
109 ////////////////////////////////////////////////////////////
110 ~Cursor();
111
112 ////////////////////////////////////////////////////////////
113 /// \brief Create a cursor with the provided image
114 ///
115 /// \a pixels must be an array of \a width by \a height pixels
116 /// in 32-bit RGBA format. If not, this will cause undefined behavior.
117 ///
118 /// If \a pixels is null or either \a width or \a height are 0,
119 /// the current cursor is left unchanged and the function will
120 /// return false.
121 ///
122 /// In addition to specifying the pixel data, you can also
123 /// specify the location of the hotspot of the cursor. The
124 /// hotspot is the pixel coordinate within the cursor image
125 /// which will be located exactly where the mouse pointer
126 /// position is. Any mouse actions that are performed will
127 /// return the window/screen location of the hotspot.
128 ///
129 /// \warning On Unix, the pixels are mapped into a monochrome
130 /// bitmap: pixels with an alpha channel to 0 are
131 /// transparent, black if the RGB channel are close
132 /// to zero, and white otherwise.
133 ///
134 /// \param pixels Array of pixels of the image
135 /// \param size Width and height of the image
136 /// \param hotspot (x,y) location of the hotspot
137 /// \return true if the cursor was successfully loaded;
138 /// false otherwise
139 ///
140 ////////////////////////////////////////////////////////////
141 bool loadFromPixels(const Uint8* pixels, Vector2u size, Vector2u hotspot);
142
143 ////////////////////////////////////////////////////////////
144 /// \brief Create a native system cursor
145 ///
146 /// Refer to the list of cursor available on each system
147 /// (see sf::Cursor::Type) to know whether a given cursor is
148 /// expected to load successfully or is not supported by
149 /// the operating system.
150 ///
151 /// \param type Native system cursor type
152 /// \return true if and only if the corresponding cursor is
153 /// natively supported by the operating system;
154 /// false otherwise
155 ///
156 ////////////////////////////////////////////////////////////
157 bool loadFromSystem(Type type);
158
159private:
160
161 friend class Window;
162
163 ////////////////////////////////////////////////////////////
164 /// \brief Get access to the underlying implementation
165 ///
166 /// This is primarily designed for sf::Window::setMouseCursor,
167 /// hence the friendship.
168 ///
169 /// \return a reference to the OS-specific implementation
170 ///
171 ////////////////////////////////////////////////////////////
172 const priv::CursorImpl& getImpl() const;
173
174private:
175
176 ////////////////////////////////////////////////////////////
177 // Member data
178 ////////////////////////////////////////////////////////////
179 priv::CursorImpl* m_impl; ///< Platform-specific implementation of the cursor
180};
181
182} // namespace sf
183

185#endif // SFML_CURSOR_HPP

188////////////////////////////////////////////////////////////
189/// \class sf::Cursor
190/// \ingroup window
191///
192/// \warning Features related to Cursor are not supported on
193/// iOS and Android.
194///
195/// This class abstracts the operating system resources
196/// associated with either a native system cursor or a custom
197/// cursor.
198///
199/// After loading the cursor the graphical appearance
200/// with either loadFromPixels() or loadFromSystem(), the
201/// cursor can be changed with sf::Window::setMouseCursor().
202///
203/// The behaviour is undefined if the cursor is destroyed while
204/// in use by the window.
205///
206/// Usage example:
207/// \code
208/// sf::Window window;
209///
210/// // ... create window as usual ...
211///
212/// sf::Cursor cursor;
213/// if (cursor.loadFromSystem(sf::Cursor::Hand))
214/// window.setMouseCursor(cursor);
215/// \endcode
216///
217/// \see sf::Window::setMouseCursor
218///
