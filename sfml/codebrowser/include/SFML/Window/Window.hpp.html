54////////////////////////////////////////////////////////////
55/// \brief Window that serves as a target for OpenGL rendering
56///
57////////////////////////////////////////////////////////////
58class SFML_WINDOW_API Window : GlResource, NonCopyable
59{
60public:
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Default constructor
64 ///
65 /// This constructor doesn't actually create the window,
66 /// use the other constructors or call create() to do so.
67 ///
68 ////////////////////////////////////////////////////////////
69 Window();
70
71 ////////////////////////////////////////////////////////////
72 /// \brief Construct a new window
73 ///
74 /// This constructor creates the window with the size and pixel
75 /// depth defined in \a mode. An optional style can be passed to
76 /// customize the look and behavior of the window (borders,
77 /// title bar, resizable, closable, ...). If \a style contains
78 /// Style::Fullscreen, then \a mode must be a valid video mode.
79 ///
80 /// The fourth parameter is an optional structure specifying
81 /// advanced OpenGL context settings such as antialiasing,
82 /// depth-buffer bits, etc.
83 ///
84 /// \param mode Video mode to use (defines the width, height and depth of the rendering area of the window)
85 /// \param title Title of the window
86 /// \param style %Window style, a bitwise OR combination of sf::Style enumerators
87 /// \param settings Additional settings for the underlying OpenGL context
88 ///
89 ////////////////////////////////////////////////////////////
90 Window(VideoMode mode, const String& title, Uint32 style = Style::Default, const ContextSettings& settings = ContextSettings());
91
92 ////////////////////////////////////////////////////////////
93 /// \brief Construct the window from an existing control
94 ///
95 /// Use this constructor if you want to create an OpenGL
96 /// rendering area into an already existing control.
97 ///
98 /// The second parameter is an optional structure specifying
99 /// advanced OpenGL context settings such as antialiasing,
100 /// depth-buffer bits, etc.
101 ///
102 /// \param handle Platform-specific handle of the control
103 /// \param settings Additional settings for the underlying OpenGL context
104 ///
105 ////////////////////////////////////////////////////////////
106 explicit Window(WindowHandle handle, const ContextSettings& settings = ContextSettings());
107
108 ////////////////////////////////////////////////////////////
109 /// \brief Destructor
110 ///
111 /// Closes the window and frees all the resources attached to it.
112 ///
113 ////////////////////////////////////////////////////////////
114 virtual ~Window();
115
116 ////////////////////////////////////////////////////////////
117 /// \brief Create (or recreate) the window
118 ///
119 /// If the window was already created, it closes it first.
120 /// If \a style contains Style::Fullscreen, then \a mode
121 /// must be a valid video mode.
122 ///
123 /// The fourth parameter is an optional structure specifying
124 /// advanced OpenGL context settings such as antialiasing,
125 /// depth-buffer bits, etc.
126 ///
127 /// \param mode Video mode to use (defines the width, height and depth of the rendering area of the window)
128 /// \param title Title of the window
129 /// \param style %Window style, a bitwise OR combination of sf::Style enumerators
130 /// \param settings Additional settings for the underlying OpenGL context
131 ///
132 ////////////////////////////////////////////////////////////
133 void create(VideoMode mode, const String& title, Uint32 style = Style::Default, const ContextSettings& settings = ContextSettings());
134
135 ////////////////////////////////////////////////////////////
136 /// \brief Create (or recreate) the window from an existing control
137 ///
138 /// Use this function if you want to create an OpenGL
139 /// rendering area into an already existing control.
140 /// If the window was already created, it closes it first.
141 ///
142 /// The second parameter is an optional structure specifying
143 /// advanced OpenGL context settings such as antialiasing,
144 /// depth-buffer bits, etc.
145 ///
146 /// \param handle Platform-specific handle of the control
147 /// \param settings Additional settings for the underlying OpenGL context
148 ///
149 ////////////////////////////////////////////////////////////
150 void create(WindowHandle handle, const ContextSettings& settings = ContextSettings());
151
152 ////////////////////////////////////////////////////////////
153 /// \brief Close the window and destroy all the attached resources
154 ///
155 /// After calling this function, the sf::Window instance remains
156 /// valid and you can call create() to recreate the window.
157 /// All other functions such as pollEvent() or display() will
158 /// still work (i.e. you don't have to test isOpen() every time),
159 /// and will have no effect on closed windows.
160 ///
161 ////////////////////////////////////////////////////////////
162 void close();
163
164 ////////////////////////////////////////////////////////////
165 /// \brief Tell whether or not the window is open
166 ///
167 /// This function returns whether or not the window exists.
168 /// Note that a hidden window (setVisible(false)) is open
169 /// (therefore this function would return true).
170 ///
171 /// \return True if the window is open, false if it has been closed
172 ///
173 ////////////////////////////////////////////////////////////
174 bool isOpen() const;
175
176 ////////////////////////////////////////////////////////////
177 /// \brief Get the settings of the OpenGL context of the window
178 ///
179 /// Note that these settings may be different from what was
180 /// passed to the constructor or the create() function,
181 /// if one or more settings were not supported. In this case,
182 /// SFML chose the closest match.
183 ///
184 /// \return Structure containing the OpenGL context settings
185 ///
186 ////////////////////////////////////////////////////////////
187 const ContextSettings& getSettings() const;
188
189 ////////////////////////////////////////////////////////////
190 /// \brief Pop the event on top of the event queue, if any, and return it
191 ///
192 /// This function is not blocking: if there's no pending event then
193 /// it will return false and leave \a event unmodified.
194 /// Note that more than one event may be present in the event queue,
195 /// thus you should always call this function in a loop
196 /// to make sure that you process every pending event.
197 /// \code
198 /// sf::Event event;
199 /// while (window.pollEvent(event))
200 /// {
201 /// // process event...
202 /// }
203 /// \endcode
204 ///
205 /// \param event Event to be returned
206 ///
207 /// \return True if an event was returned, or false if the event queue was empty
208 ///
209 /// \see waitEvent
210 ///
211 ////////////////////////////////////////////////////////////
212 bool pollEvent(Event& event);
213
214 ////////////////////////////////////////////////////////////
215 /// \brief Wait for an event and return it
216 ///
217 /// This function is blocking: if there's no pending event then
218 /// it will wait until an event is received.
219 /// After this function returns (and no error occurred),
220 /// the \a event object is always valid and filled properly.
221 /// This function is typically used when you have a thread that
222 /// is dedicated to events handling: you want to make this thread
223 /// sleep as long as no new event is received.
224 /// \code
225 /// sf::Event event;
226 /// if (window.waitEvent(event))
227 /// {
228 /// // process event...
229 /// }
230 /// \endcode
231 ///
232 /// \param event Event to be returned
233 ///
234 /// \return False if any error occurred
235 ///
236 /// \see pollEvent
237 ///
238 ////////////////////////////////////////////////////////////
239 bool waitEvent(Event& event);
240
241 ////////////////////////////////////////////////////////////
242 /// \brief Get the position of the window
243 ///
244 /// \return Position of the window, in pixels
245 ///
246 /// \see setPosition
247 ///
248 ////////////////////////////////////////////////////////////
249 Vector2i getPosition() const;
250
251 ////////////////////////////////////////////////////////////
252 /// \brief Change the position of the window on screen
253 ///
254 /// This function only works for top-level windows
255 /// (i.e. it will be ignored for windows created from
256 /// the handle of a child window/control).
257 ///
258 /// \param position New position, in pixels
259 ///
260 /// \see getPosition
261 ///
262 ////////////////////////////////////////////////////////////
263 void setPosition(const Vector2i& position);
264
265 ////////////////////////////////////////////////////////////
266 /// \brief Get the size of the rendering region of the window
267 ///
268 /// The size doesn't include the titlebar and borders
269 /// of the window.
270 ///
271 /// \return Size in pixels
272 ///
273 /// \see setSize
274 ///
275 ////////////////////////////////////////////////////////////
276 Vector2u getSize() const;
277
278 ////////////////////////////////////////////////////////////
279 /// \brief Change the size of the rendering region of the window
280 ///
281 /// \param size New size, in pixels
282 ///
283 /// \see getSize
284 ///
285 ////////////////////////////////////////////////////////////
286 void setSize(const Vector2u& size);
287
288 ////////////////////////////////////////////////////////////
289 /// \brief Change the title of the window
290 ///
291 /// \param title New title
292 ///
293 /// \see setIcon
294 ///
295 ////////////////////////////////////////////////////////////
296 void setTitle(const String& title);
297
298 ////////////////////////////////////////////////////////////
299 /// \brief Change the window's icon
300 ///
301 /// \a pixels must be an array of \a width x \a height pixels
302 /// in 32-bits RGBA format.
303 ///
304 /// The OS default icon is used by default.
305 ///
306 /// \param width Icon's width, in pixels
307 /// \param height Icon's height, in pixels
308 /// \param pixels Pointer to the array of pixels in memory. The
309 /// pixels are copied, so you need not keep the
310 /// source alive after calling this function.
311 ///
312 /// \see setTitle
313 ///
314 ////////////////////////////////////////////////////////////
315 void setIcon(unsigned int width, unsigned int height, const Uint8* pixels);
316
317 ////////////////////////////////////////////////////////////
318 /// \brief Show or hide the window
319 ///
320 /// The window is shown by default.
321 ///
322 /// \param visible True to show the window, false to hide it
323 ///
324 ////////////////////////////////////////////////////////////
325 void setVisible(bool visible);
326
327 ////////////////////////////////////////////////////////////
328 /// \brief Enable or disable vertical synchronization
329 ///
330 /// Activating vertical synchronization will limit the number
331 /// of frames displayed to the refresh rate of the monitor.
332 /// This can avoid some visual artifacts, and limit the framerate
333 /// to a good value (but not constant across different computers).
334 ///
335 /// Vertical synchronization is disabled by default.
336 ///
337 /// \param enabled True to enable v-sync, false to deactivate it
338 ///
339 ////////////////////////////////////////////////////////////
340 void setVerticalSyncEnabled(bool enabled);
341
342 ////////////////////////////////////////////////////////////
343 /// \brief Show or hide the mouse cursor
344 ///
345 /// The mouse cursor is visible by default.
346 ///
347 /// \param visible True to show the mouse cursor, false to hide it
348 ///
349 ////////////////////////////////////////////////////////////
350 void setMouseCursorVisible(bool visible);
351
352 ////////////////////////////////////////////////////////////
353 /// \brief Grab or release the mouse cursor
354 ///
355 /// If set, grabs the mouse cursor inside this window's client
356 /// area so it may no longer be moved outside its bounds.
357 /// Note that grabbing is only active while the window has
358 /// focus.
359 ///
360 /// \param grabbed True to enable, false to disable
361 ///
362 ////////////////////////////////////////////////////////////
363 void setMouseCursorGrabbed(bool grabbed);
364
365 ////////////////////////////////////////////////////////////
366 /// \brief Set the displayed cursor to a native system cursor
367 ///
368 /// Upon window creation, the arrow cursor is used by default.
369 ///
370 /// \warning The cursor must not be destroyed while in use by
371 /// the window.
372 ///
373 /// \warning Features related to Cursor are not supported on
374 /// iOS and Android.
375 ///
376 /// \param cursor Native system cursor type to display
377 ///
378 /// \see sf::Cursor::loadFromSystem
379 /// \see sf::Cursor::loadFromPixels
380 ///
381 ////////////////////////////////////////////////////////////
382 void setMouseCursor(const Cursor& cursor);
383
384 ////////////////////////////////////////////////////////////
385 /// \brief Enable or disable automatic key-repeat
386 ///
387 /// If key repeat is enabled, you will receive repeated
388 /// KeyPressed events while keeping a key pressed. If it is disabled,
389 /// you will only get a single event when the key is pressed.
390 ///
391 /// Key repeat is enabled by default.
392 ///
393 /// \param enabled True to enable, false to disable
394 ///
395 ////////////////////////////////////////////////////////////
396 void setKeyRepeatEnabled(bool enabled);
397
398 ////////////////////////////////////////////////////////////
399 /// \brief Limit the framerate to a maximum fixed frequency
400 ///
401 /// If a limit is set, the window will use a small delay after
402 /// each call to display() to ensure that the current frame
403 /// lasted long enough to match the framerate limit.
404 /// SFML will try to match the given limit as much as it can,
405 /// but since it internally uses sf::sleep, whose precision
406 /// depends on the underlying OS, the results may be a little
407 /// unprecise as well (for example, you can get 65 FPS when
408 /// requesting 60).
409 ///
410 /// \param limit Framerate limit, in frames per seconds (use 0 to disable limit)
411 ///
412 ////////////////////////////////////////////////////////////
413 void setFramerateLimit(unsigned int limit);
414
415 ////////////////////////////////////////////////////////////
416 /// \brief Change the joystick threshold
417 ///
418 /// The joystick threshold is the value below which
419 /// no JoystickMoved event will be generated.
420 ///
421 /// The threshold value is 0.1 by default.
422 ///
423 /// \param threshold New threshold, in the range [0, 100]
424 ///
425 ////////////////////////////////////////////////////////////
426 void setJoystickThreshold(float threshold);
427
428 ////////////////////////////////////////////////////////////
429 /// \brief Activate or deactivate the window as the current target
430 /// for OpenGL rendering
431 ///
432 /// A window is active only on the current thread, if you want to
433 /// make it active on another thread you have to deactivate it
434 /// on the previous thread first if it was active.
435 /// Only one window can be active on a thread at a time, thus
436 /// the window previously active (if any) automatically gets deactivated.
437 /// This is not to be confused with requestFocus().
438 ///
439 /// \param active True to activate, false to deactivate
440 ///
441 /// \return True if operation was successful, false otherwise
442 ///
443 ////////////////////////////////////////////////////////////
444 bool setActive(bool active = true) const;
445
446 ////////////////////////////////////////////////////////////
447 /// \brief Request the current window to be made the active
448 /// foreground window
449 ///
450 /// At any given time, only one window may have the input focus
451 /// to receive input events such as keystrokes or mouse events.
452 /// If a window requests focus, it only hints to the operating
453 /// system, that it would like to be focused. The operating system
454 /// is free to deny the request.
455 /// This is not to be confused with setActive().
456 ///
457 /// \see hasFocus
458 ///
459 ////////////////////////////////////////////////////////////
460 void requestFocus();
461
462 ////////////////////////////////////////////////////////////
463 /// \brief Check whether the window has the input focus
464 ///
465 /// At any given time, only one window may have the input focus
466 /// to receive input events such as keystrokes or most mouse
467 /// events.
468 ///
469 /// \return True if window has focus, false otherwise
470 /// \see requestFocus
471 ///
472 ////////////////////////////////////////////////////////////
473 bool hasFocus() const;
474
475 ////////////////////////////////////////////////////////////
476 /// \brief Display on screen what has been rendered to the window so far
477 ///
478 /// This function is typically called after all OpenGL rendering
479 /// has been done for the current frame, in order to show
480 /// it on screen.
481 ///
482 ////////////////////////////////////////////////////////////
483 void display();
484
485 ////////////////////////////////////////////////////////////
486 /// \brief Get the OS-specific handle of the window
487 ///
488 /// The type of the returned handle is sf::WindowHandle,
489 /// which is a typedef to the handle type defined by the OS.
490 /// You shouldn't need to use this function, unless you have
491 /// very specific stuff to implement that SFML doesn't support,
492 /// or implement a temporary workaround until a bug is fixed.
493 ///
494 /// \return System handle of the window
495 ///
496 ////////////////////////////////////////////////////////////
497 WindowHandle getSystemHandle() const;
498
499protected:
500
501 ////////////////////////////////////////////////////////////
502 /// \brief Function called after the window has been created
503 ///
504 /// This function is called so that derived classes can
505 /// perform their own specific initialization as soon as
506 /// the window is created.
507 ///
508 ////////////////////////////////////////////////////////////
509 virtual void onCreate();
510
511 ////////////////////////////////////////////////////////////
512 /// \brief Function called after the window has been resized
513 ///
514 /// This function is called so that derived classes can
515 /// perform custom actions when the size of the window changes.
516 ///
517 ////////////////////////////////////////////////////////////
518 virtual void onResize();
519
520private:
521
522 ////////////////////////////////////////////////////////////
523 /// \brief Processes an event before it is sent to the user
524 ///
525 /// This function is called every time an event is received
526 /// from the internal window (through pollEvent or waitEvent).
527 /// It filters out unwanted events, and performs whatever internal
528 /// stuff the window needs before the event is returned to the
529 /// user.
530 ///
531 /// \param event Event to filter
532 ///
533 ////////////////////////////////////////////////////////////
534 bool filterEvent(const Event& event);
535
536 ////////////////////////////////////////////////////////////
537 /// \brief Perform some common internal initializations
538 ///
539 ////////////////////////////////////////////////////////////
540 void initialize();
541

544 ////////////////////////////////////////////////////////////

548 Time m_frameTimeLimit; ///< Current framerate limit
549 Vector2u m_size; ///< Current size of the window
550};

558////////////////////////////////////////////////////////////
559/// \class sf::Window
560/// \ingroup window
561///
562/// sf::Window is the main class of the Window module. It defines
563/// an OS window that is able to receive an OpenGL rendering.
564///
565/// A sf::Window can create its own new window, or be embedded into
566/// an already existing control using the create(handle) function.
567/// This can be useful for embedding an OpenGL rendering area into
568/// a view which is part of a bigger GUI with existing windows,
569/// controls, etc. It can also serve as embedding an OpenGL rendering
570/// area into a window created by another (probably richer) GUI library
571/// like Qt or wxWidgets.
572///
573/// The sf::Window class provides a simple interface for manipulating
574/// the window: move, resize, show/hide, control mouse cursor, etc.
575/// It also provides event handling through its pollEvent() and waitEvent()
576/// functions.
577///
578/// Note that OpenGL experts can pass their own parameters (antialiasing
579/// level, bits for the depth and stencil buffers, etc.) to the
580/// OpenGL context attached to the window, with the sf::ContextSettings
581/// structure which is passed as an optional argument when creating the
582/// window.
583///
584/// Usage example:
585/// \code
586/// // Declare and create a new window
587/// sf::Window window(sf::VideoMode(800, 600), "SFML window");
588///
589/// // Limit the framerate to 60 frames per second (this step is optional)
590/// window.setFramerateLimit(60);
591///
592/// // The main loop - ends as soon as the window is closed
593/// while (window.isOpen())
594/// {
595/// // Event processing
596/// sf::Event event;
597/// while (window.pollEvent(event))
598/// {
599/// // Request for closing the window
600/// if (event.type == sf::Event::Closed)
601/// window.close();
602/// }
603///
604/// // Activate the window for OpenGL rendering
605/// window.setActive();
606///
607/// // OpenGL drawing commands go here...
608///
609/// // End the current frame and display its contents on screen
610/// window.display();
611/// }
612/// \endcode
613///
