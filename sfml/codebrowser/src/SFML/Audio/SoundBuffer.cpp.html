50////////////////////////////////////////////////////////////
51SoundBuffer::SoundBuffer(const SoundBuffer& copy) :
52m_buffer (0),
53m_samples (copy.m_samples),
54m_duration(copy.m_duration),
55m_sounds () // don't copy the attached sounds
56{
57 // Create the buffer
58 alCheck(alGenBuffers(1, &m_buffer));
59
60 // Update the internal buffer with the new samples
61 update(copy.getChannelCount(), copy.getSampleRate());
62}

65////////////////////////////////////////////////////////////
66SoundBuffer::~SoundBuffer()
67{
68 // To prevent the iterator from becoming invalid, move the entire buffer to another
69 // container. Otherwise calling resetBuffer would result in detachSound being
70 // called which removes the sound from the internal list.
71 SoundList sounds;
72 sounds.swap(m_sounds);
73
74 // Detach the buffer from the sounds that use it (to avoid OpenAL errors)
75 for (SoundList::const_iterator it = sounds.begin(); it != sounds.end(); ++it)
76 (*it)->resetBuffer();
77
78 // Destroy the buffer
79 if (m_buffer)
80 alCheck(alDeleteBuffers(1, &m_buffer));
81}

117////////////////////////////////////////////////////////////
118bool SoundBuffer::loadFromSamples(const Int16* samples, Uint64 sampleCount, unsigned int channelCount, unsigned int sampleRate)
119{
120 if (samples && sampleCount && channelCount && sampleRate)
121 {
122 // Copy the new audio samples
123 m_samples.assign(samples, samples + sampleCount);
124
125 // Update the internal buffer with the new samples
126 return update(channelCount, sampleRate);
127 }
128 else
129 {
130 // Error...
131 err() << "Failed to load sound buffer from samples ("
132 << "array: " << samples << ", "
133 << "count: " << sampleCount << ", "
134 << "channels: " << channelCount << ", "
135 << "samplerate: " << sampleRate << ")"
136 << std::endl;
137
138 return false;
139 }
140}

143////////////////////////////////////////////////////////////
144bool SoundBuffer::saveToFile(const std::string& filename) const
145{
146 // Create the sound file in write mode
147 OutputSoundFile file;
148 if (file.openFromFile(filename, getSampleRate(), getChannelCount()))
149 {
150 // Write the samples to the opened file
151 file.write(&m_samples[0], m_samples.size());
152
153 return true;
154 }
155 else
156 {
157 return false;
158 }
159}

203////////////////////////////////////////////////////////////
204SoundBuffer& SoundBuffer::operator =(const SoundBuffer& right)
205{
206 SoundBuffer temp(right);
207
208 std::swap(m_samples, temp.m_samples);
209 std::swap(m_buffer, temp.m_buffer);
210 std::swap(m_duration, temp.m_duration);
211 std::swap(m_sounds, temp.m_sounds); // swap sounds too, so that they are detached when temp is destroyed
212
213 return *this;
214}

217////////////////////////////////////////////////////////////
218bool SoundBuffer::initialize(InputSoundFile& file)
219{
220 // Retrieve the sound parameters
221 Uint64 sampleCount = file.getSampleCount();
222 unsigned int channelCount = file.getChannelCount();
223 unsigned int sampleRate = file.getSampleRate();
224
225 // Read the samples from the provided file
226 m_samples.resize(static_cast<std::size_t>(sampleCount));
227 if (file.read(&m_samples[0], sampleCount) == sampleCount)
228 {
229 // Update the internal buffer with the new samples
230 return update(channelCount, sampleRate);
231 }
232 else
233 {
234 return false;
235 }
236}

239////////////////////////////////////////////////////////////
240bool SoundBuffer::update(unsigned int channelCount, unsigned int sampleRate)
241{
242 // Check parameters
243 if (!channelCount || !sampleRate || m_samples.empty())
244 return false;
245
246 // Find the good format according to the number of channels
247 ALenum format = priv::AudioDevice::getFormatFromChannelCount(channelCount);
248
249 // Check if the format is valid
250 if (format == 0)
251 {
252 err() << "Failed to load sound buffer (unsupported number of channels: " << channelCount << ")" << std::endl;
253 return false;
254 }
255
256 // First make a copy of the list of sounds so we can reattach later
257 SoundList sounds(m_sounds);
258
259 // Detach the buffer from the sounds that use it (to avoid OpenAL errors)
260 for (SoundList::const_iterator it = sounds.begin(); it != sounds.end(); ++it)
261 (*it)->resetBuffer();
262
263 // Fill the buffer
264 ALsizei size = static_cast<ALsizei>(m_samples.size()) * sizeof(Int16);
265 alCheck(alBufferData(m_buffer, format, &m_samples[0], size, sampleRate));
266
267 // Compute the duration
268 m_duration = seconds(static_cast<float>(m_samples.size()) / sampleRate / channelCount);
269
270 // Now reattach the buffer to the sounds that use it
271 for (SoundList::const_iterator it = sounds.begin(); it != sounds.end(); ++it)
272 (*it)->setBuffer(*this);
273
274 return true;
275}
