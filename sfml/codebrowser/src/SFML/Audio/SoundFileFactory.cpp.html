42 // Register all the built-in readers and writers if not already done
43 void ensureDefaultReadersWritersRegistered()
44 {
45 static bool registered = false;
46 if (!registered)
47 {
48 sf::SoundFileFactory::registerReader<sf::priv::SoundFileReaderFlac>();
49 sf::SoundFileFactory::registerWriter<sf::priv::SoundFileWriterFlac>();
50 sf::SoundFileFactory::registerReader<sf::priv::SoundFileReaderOgg>();
51 sf::SoundFileFactory::registerWriter<sf::priv::SoundFileWriterOgg>();
52 sf::SoundFileFactory::registerReader<sf::priv::SoundFileReaderWav>();
53 sf::SoundFileFactory::registerWriter<sf::priv::SoundFileWriterWav>();
54 registered = true;
55 }
56 }
57}

65////////////////////////////////////////////////////////////
66SoundFileReader* SoundFileFactory::createReaderFromFilename(const std::string& filename)
67{
68 // Register the built-in readers/writers on first call
69 ensureDefaultReadersWritersRegistered();
70
71 // Wrap the input file into a file stream
72 FileInputStream stream;
73 if (!stream.open(filename)) {
74 err() << "Failed to open sound file \"" << filename << "\" (couldn't open stream)" << std::endl;
75 return NULL;
76 }
77
78 // Test the filename in all the registered factories
79 for (ReaderFactoryArray::const_iterator it = s_readers.begin(); it != s_readers.end(); ++it)
80 {
81 stream.seek(0);
82 if (it->check(stream))
83 return it->create();
84 }
85
86 // No suitable reader found
87 err() << "Failed to open sound file \"" << filename << "\" (format not supported)" << std::endl;
88 return NULL;
89}
