119////////////////////////////////////////////////////////////
120void Music::setLoopPoints(TimeSpan timePoints)
121{
122 Span<Uint64> samplePoints(timeToSamples(timePoints.offset), timeToSamples(timePoints.length));
123
124 // Check our state. This averts a divide-by-zero. GetChannelCount() is cheap enough to use often
125 if (getChannelCount() == 0 || m_file.getSampleCount() == 0)
126 {
127 sf::err() << "Music is not in a valid state to assign Loop Points." << std::endl;
128 return;
129 }
130
131 // Round up to the next even sample if needed
132 samplePoints.offset += (getChannelCount() - 1);
133 samplePoints.offset -= (samplePoints.offset % getChannelCount());
134 samplePoints.length += (getChannelCount() - 1);
135 samplePoints.length -= (samplePoints.length % getChannelCount());
136
137 // Validate
138 if (samplePoints.offset >= m_file.getSampleCount())
139 {
140 sf::err() << "LoopPoints offset val must be in range [0, Duration)." << std::endl;
141 return;
142 }
143 if (samplePoints.length == 0)
144 {
145 sf::err() << "LoopPoints length val must be nonzero." << std::endl;
146 return;
147 }
148
149 // Clamp End Point
150 samplePoints.length = std::min(samplePoints.length, m_file.getSampleCount() - samplePoints.offset);
151
152 // If this change has no effect, we can return without touching anything
153 if (samplePoints.offset == m_loopSpan.offset && samplePoints.length == m_loopSpan.length)
154 return;
155
156 // When we apply this change, we need to "reset" this instance and its buffer
157
158 // Get old playing status and position
159 Status oldStatus = getStatus();
160 Time oldPos = getPlayingOffset();
161
162 // Unload
163 stop();
164
165 // Set
166 m_loopSpan = samplePoints;
167
168 // Restore
169 if (oldPos != Time::Zero)
170 setPlayingOffset(oldPos);
171
172 // Resume
173 if (oldStatus == Playing)
174 play();
175}

178////////////////////////////////////////////////////////////
179bool Music::onGetData(SoundStream::Chunk& data)
180{
181 Lock lock(m_mutex);
182
183 std::size_t toFill = m_samples.size();
184 Uint64 currentOffset = m_file.getSampleOffset();
185 Uint64 loopEnd = m_loopSpan.offset + m_loopSpan.length;
186
187 // If the loop end is enabled and imminent, request less data.
188 // This will trip an "onLoop()" call from the underlying SoundStream,
189 // and we can then take action.
190 if (getLoop() && (m_loopSpan.length != 0) && (currentOffset <= loopEnd) && (currentOffset + toFill > loopEnd))
191 toFill = loopEnd - currentOffset;
192
193 // Fill the chunk parameters
194 data.samples = &m_samples[0];
195 data.sampleCount = static_cast<std::size_t>(m_file.read(&m_samples[0], toFill));
196 currentOffset += data.sampleCount;
197
198 // Check if we have stopped obtaining samples or reached either the EOF or the loop end point
199 return (data.sampleCount != 0) && (currentOffset < m_file.getSampleCount()) && !(currentOffset == loopEnd && m_loopSpan.length != 0);
200}

211////////////////////////////////////////////////////////////
212Int64 Music::onLoop()
213{
214 // Called by underlying SoundStream so we can determine where to loop.
215 Lock lock(m_mutex);
216 Uint64 currentOffset = m_file.getSampleOffset();
217 if (getLoop() && (m_loopSpan.length != 0) && (currentOffset == m_loopSpan.offset + m_loopSpan.length))
218 {
219 // Looping is enabled, and either we're at the loop end, or we're at the EOF
220 // when it's equivalent to the loop end (loop end takes priority). Send us to loop begin
221 m_file.seek(m_loopSpan.offset);
222 return m_file.getSampleOffset();
223 }
224 else if (getLoop() && (currentOffset >= m_file.getSampleCount()))
225 {
226 // If we're at the EOF, reset to 0
227 m_file.seek(0);
228 return 0;
229 }
230 return NoLoop;
231}

234////////////////////////////////////////////////////////////
235void Music::initialize()
236{
237 // Compute the music positions
238 m_loopSpan.offset = 0;
239 m_loopSpan.length = m_file.getSampleCount();
240
241 // Resize the internal buffer so that it can contain 1 second of audio samples
242 m_samples.resize(m_file.getSampleRate() * m_file.getChannelCount());
243
244 // Initialize the stream
245 SoundStream::initialize(m_file.getChannelCount(), m_file.getSampleRate());
246}

248////////////////////////////////////////////////////////////
249Uint64 Music::timeToSamples(Time position) const
250{
251 // Always ROUND, no unchecked truncation, hence the addition in the numerator.
252 // This avoids most precision errors arising from "samples => Time => samples" conversions
253 // Original rounding calculation is ((Micros * Freq * Channels) / 1000000) + 0.5
254 // We refactor it to keep Int64 as the data type throughout the whole operation.
255 return ((position.asMicroseconds() * getSampleRate() * getChannelCount()) + 500000) / 1000000;
256}

259////////////////////////////////////////////////////////////
260Time Music::samplesToTime(Uint64 samples) const
261{
262 Time position = Time::Zero;
263
264 // Make sure we don't divide by 0
265 if (getSampleRate() != 0 && getChannelCount() != 0)
266 position = microseconds((samples * 1000000) / (getChannelCount() * getSampleRate()));
267
268 return position;
269}
