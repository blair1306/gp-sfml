25#ifndef SFML_AUDIODEVICE_HPP
26#define SFML_AUDIODEVICE_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/System/Vector3.hpp>
32#include <set>
33#include <string>
34
35
36namespace sf
37{
38namespace priv
39{
40////////////////////////////////////////////////////////////
41/// \brief High-level wrapper around the audio API, it manages
42/// the creation and destruction of the audio device and
43/// context and stores the device capabilities
44///
45////////////////////////////////////////////////////////////
46class AudioDevice
47{
48public:
49
50 ////////////////////////////////////////////////////////////
51 /// \brief Default constructor
52 ///
53 ////////////////////////////////////////////////////////////
54 AudioDevice();
55
56 ////////////////////////////////////////////////////////////
57 /// \brief Destructor
58 ///
59 ////////////////////////////////////////////////////////////
60 ~AudioDevice();
61
62 ////////////////////////////////////////////////////////////
63 /// \brief Check if an OpenAL extension is supported
64 ///
65 /// This functions automatically finds whether it
66 /// is an AL or ALC extension, and calls the corresponding
67 /// function.
68 ///
69 /// \param extension Name of the extension to test
70 ///
71 /// \return True if the extension is supported, false if not
72 ///
73 ////////////////////////////////////////////////////////////
74 static bool isExtensionSupported(const std::string& extension);
75
76 ////////////////////////////////////////////////////////////
77 /// \brief Get the OpenAL format that matches the given number of channels
78 ///
79 /// \param channelCount Number of channels
80 ///
81 /// \return Corresponding format
82 ///
83 ////////////////////////////////////////////////////////////
84 static int getFormatFromChannelCount(unsigned int channelCount);
85
86 ////////////////////////////////////////////////////////////
87 /// \brief Change the global volume of all the sounds and musics
88 ///
89 /// The volume is a number between 0 and 100; it is combined with
90 /// the individual volume of each sound / music.
91 /// The default value for the volume is 100 (maximum).
92 ///
93 /// \param volume New global volume, in the range [0, 100]
94 ///
95 /// \see getGlobalVolume
96 ///
97 ////////////////////////////////////////////////////////////
98 static void setGlobalVolume(float volume);
99
100 ////////////////////////////////////////////////////////////
101 /// \brief Get the current value of the global volume
102 ///
103 /// \return Current global volume, in the range [0, 100]
104 ///
105 /// \see setGlobalVolume
106 ///
107 ////////////////////////////////////////////////////////////
108 static float getGlobalVolume();
109
110 ////////////////////////////////////////////////////////////
111 /// \brief Set the position of the listener in the scene
112 ///
113 /// The default listener's position is (0, 0, 0).
114 ///
115 /// \param position New listener's position
116 ///
117 /// \see getPosition, setDirection
118 ///
119 ////////////////////////////////////////////////////////////
120 static void setPosition(const Vector3f& position);
121
122 ////////////////////////////////////////////////////////////
123 /// \brief Get the current position of the listener in the scene
124 ///
125 /// \return Listener's position
126 ///
127 /// \see setPosition
128 ///
129 ////////////////////////////////////////////////////////////
130 static Vector3f getPosition();
131
132 ////////////////////////////////////////////////////////////
133 /// \brief Set the forward vector of the listener in the scene
134 ///
135 /// The direction (also called "at vector") is the vector
136 /// pointing forward from the listener's perspective. Together
137 /// with the up vector, it defines the 3D orientation of the
138 /// listener in the scene. The direction vector doesn't
139 /// have to be normalized.
140 /// The default listener's direction is (0, 0, -1).
141 ///
142 /// \param direction New listener's direction
143 ///
144 /// \see getDirection, setUpVector, setPosition
145 ///
146 ////////////////////////////////////////////////////////////
147 static void setDirection(const Vector3f& direction);
148
149 ////////////////////////////////////////////////////////////
150 /// \brief Get the current forward vector of the listener in the scene
151 ///
152 /// \return Listener's forward vector (not normalized)
153 ///
154 /// \see setDirection
155 ///
156 ////////////////////////////////////////////////////////////
157 static Vector3f getDirection();
158
159 ////////////////////////////////////////////////////////////
160 /// \brief Set the upward vector of the listener in the scene
161 ///
162 /// The up vector is the vector that points upward from the
163 /// listener's perspective. Together with the direction, it
164 /// defines the 3D orientation of the listener in the scene.
165 /// The up vector doesn't have to be normalized.
166 /// The default listener's up vector is (0, 1, 0). It is usually
167 /// not necessary to change it, especially in 2D scenarios.
168 ///
169 /// \param upVector New listener's up vector
170 ///
171 /// \see getUpVector, setDirection, setPosition
172 ///
173 ////////////////////////////////////////////////////////////
174 static void setUpVector(const Vector3f& upVector);
175
176 ////////////////////////////////////////////////////////////
177 /// \brief Get the current upward vector of the listener in the scene
178 ///
179 /// \return Listener's upward vector (not normalized)
180 ///
181 /// \see setUpVector
182 ///
183 ////////////////////////////////////////////////////////////
184 static Vector3f getUpVector();
185};
186
187} // namespace priv
188
189} // namespace sf
190
191
192#endif // SFML_AUDIODEVICE_HPP
