38 // The following functions read integers as little endian and
39 // return them in the host byte order
40
41 bool decode(sf::InputStream& stream, sf::Uint8& value)
42 {
43 return stream.read(&value, sizeof(value)) == sizeof(value);
44 }
45
46 bool decode(sf::InputStream& stream, sf::Int16& value)
47 {
48 unsigned char bytes[sizeof(value)];
49 if (stream.read(bytes, sizeof(bytes)) != sizeof(bytes))
50 return false;
51
52 value = bytes[0] | (bytes[1] << 8);
53
54 return true;
55 }
56
57 bool decode(sf::InputStream& stream, sf::Uint16& value)
58 {
59 unsigned char bytes[sizeof(value)];
60 if (stream.read(bytes, sizeof(bytes)) != sizeof(bytes))
61 return false;
62
63 value = bytes[0] | (bytes[1] << 8);
64
65 return true;
66 }
67
68 bool decode24bit(sf::InputStream& stream, sf::Uint32& value)
69 {
70 unsigned char bytes[3];
71 if (stream.read(bytes, sizeof(bytes)) != sizeof(bytes))
72 return false;
73
74 value = bytes[0] | (bytes[1] << 8) | (bytes[2] << 16);
75
76 return true;
77 }
78
79 bool decode(sf::InputStream& stream, sf::Uint32& value)
80 {
81 unsigned char bytes[sizeof(value)];
82 if (stream.read(bytes, sizeof(bytes)) != sizeof(bytes))
83 return false;
84
85 value = bytes[0] | (bytes[1] << 8) | (bytes[2] << 16) | (bytes[3] << 24);
86
87 return true;
88 }

97////////////////////////////////////////////////////////////
98bool SoundFileReaderWav::check(InputStream& stream)
99{
100 char header[mainChunkSize];
101 if (stream.read(header, sizeof(header)) < static_cast<Int64>(sizeof(header)))
102 return false;
103
104 return (header[0] == 'R') && (header[1] == 'I') && (header[2] == 'F') && (header[3] == 'F')
105 && (header[8] == 'W') && (header[9] == 'A') && (header[10] == 'V') && (header[11] == 'E');
106}

119////////////////////////////////////////////////////////////
120bool SoundFileReaderWav::open(InputStream& stream, Info& info)
121{
122 m_stream = &stream;
123
124 if (!parseHeader(info))
125 {
126 err() << "Failed to open WAV sound file (invalid or unsupported file)" << std::endl;
127 return false;
128 }
129
130 return true;
131}

134////////////////////////////////////////////////////////////
135void SoundFileReaderWav::seek(Uint64 sampleOffset)
136{
137 assert(m_stream);
138
139 m_stream->seek(m_dataStart + sampleOffset * m_bytesPerSample);
140}

143////////////////////////////////////////////////////////////
144Uint64 SoundFileReaderWav::read(Int16* samples, Uint64 maxCount)
145{
146 assert(m_stream);
147
148 Uint64 count = 0;
149 while ((count < maxCount) && (m_stream->tell() < m_dataEnd))
150 {
151 switch (m_bytesPerSample)
152 {
153 case 1:
154 {
155 Uint8 sample = 0;
156 if (decode(*m_stream, sample))
157 *samples++ = (static_cast<Int16>(sample) - 128) << 8;
158 else
159 return count;
160 break;
161 }
162
163 case 2:
164 {
165 Int16 sample = 0;
166 if (decode(*m_stream, sample))
167 *samples++ = sample;
168 else
169 return count;
170 break;
171 }
172
173 case 3:
174 {
175 Uint32 sample = 0;
176 if (decode24bit(*m_stream, sample))
177 *samples++ = sample >> 8;
178 else
179 return count;
180 break;
181 }
182
183 case 4:
184 {
185 Uint32 sample = 0;
186 if (decode(*m_stream, sample))
187 *samples++ = sample >> 16;
188 else
189 return count;
190 break;
191 }
192
193 default:
194 {
195 assert(false);
196 return 0;
197 }
198 }
199
200 ++count;
201 }
202
203 return count;
204}

207////////////////////////////////////////////////////////////
208bool SoundFileReaderWav::parseHeader(Info& info)
209{
210 assert(m_stream);
211
212 // If we are here, it means that the first part of the header
213 // (the format) has already been checked
214 char mainChunk[mainChunkSize];
215 if (m_stream->read(mainChunk, sizeof(mainChunk)) != sizeof(mainChunk))
216 return false;
217
218 // Parse all the sub-chunks
219 bool dataChunkFound = false;
220 while (!dataChunkFound)
221 {
222 // Parse the sub-chunk id and size
223 char subChunkId[4];
224 if (m_stream->read(subChunkId, sizeof(subChunkId)) != sizeof(subChunkId))
225 return false;
226 Uint32 subChunkSize = 0;
227 if (!decode(*m_stream, subChunkSize))
228 return false;
229
230 // Check which chunk it is
231 if ((subChunkId[0] == 'f') && (subChunkId[1] == 'm') && (subChunkId[2] == 't') && (subChunkId[3] == ' '))
232 {
233 // "fmt" chunk
234
235 // Audio format
236 Uint16 format = 0;
237 if (!decode(*m_stream, format))
238 return false;
239 if (format != 1) // PCM
240 return false;
241
242 // Channel count
243 Uint16 channelCount = 0;
244 if (!decode(*m_stream, channelCount))
245 return false;
246 info.channelCount = channelCount;
247
248 // Sample rate
249 Uint32 sampleRate = 0;
250 if (!decode(*m_stream, sampleRate))
251 return false;
252 info.sampleRate = sampleRate;
253
254 // Byte rate
255 Uint32 byteRate = 0;
256 if (!decode(*m_stream, byteRate))
257 return false;
258
259 // Block align
260 Uint16 blockAlign = 0;
261 if (!decode(*m_stream, blockAlign))
262 return false;
263
264 // Bits per sample
265 Uint16 bitsPerSample = 0;
266 if (!decode(*m_stream, bitsPerSample))
267 return false;
268 if (bitsPerSample != 8 && bitsPerSample != 16 && bitsPerSample != 24 && bitsPerSample != 32)
269 {
270 err() << "Unsupported sample size: " << bitsPerSample << " bit (Supported sample sizes are 8/16/24/32 bit)" << std::endl;
271 return false;
272 }
273 m_bytesPerSample = bitsPerSample / 8;
274
275 // Skip potential extra information (should not exist for PCM)
276 if (subChunkSize > 16)
277 {
278 if (m_stream->seek(m_stream->tell() + subChunkSize - 16) == -1)
279 return false;
280 }
281 }
282 else if ((subChunkId[0] == 'd') && (subChunkId[1] == 'a') && (subChunkId[2] == 't') && (subChunkId[3] == 'a'))
283 {
284 // "data" chunk
285
286 // Compute the total number of samples
287 info.sampleCount = subChunkSize / m_bytesPerSample;
288
289 // Store the start and end position of samples in the file
290 m_dataStart = m_stream->tell();
291 m_dataEnd = m_dataStart + info.sampleCount * m_bytesPerSample;
292
293 dataChunkFound = true;
294 }
295 else
296 {
297 // unknown chunk, skip it
298 if (m_stream->seek(m_stream->tell() + subChunkSize) == -1)
299 return false;
300 }
301 }
302
303 return true;
304}
