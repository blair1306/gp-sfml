60////////////////////////////////////////////////////////////
61bool InputSoundFile::openFromFile(const std::string& filename)
62{
63 // If the file is already open, first close it
64 close();
65
66 // Find a suitable reader for the file type
67 m_reader = SoundFileFactory::createReaderFromFilename(filename);
68 if (!m_reader)
69 return false;
70
71 // Wrap the file into a stream
72 FileInputStream* file = new FileInputStream;
73 m_stream = file;
74 m_streamOwned = true;
75
76 // Open it
77 if (!file->open(filename))
78 {
79 close();
80 return false;
81 }
82
83 // Pass the stream to the reader
84 SoundFileReader::Info info;
85 if (!m_reader->open(*file, info))
86 {
87 close();
88 return false;
89 }
90
91 // Retrieve the attributes of the open sound file
92 m_sampleCount = info.sampleCount;
93 m_channelCount = info.channelCount;
94 m_sampleRate = info.sampleRate;
95
96 return true;
97}

136////////////////////////////////////////////////////////////
137bool InputSoundFile::openFromStream(InputStream& stream)
138{
139 // If the file is already open, first close it
140 close();
141
142 // Find a suitable reader for the file type
143 m_reader = SoundFileFactory::createReaderFromStream(stream);
144 if (!m_reader)
145 return false;
146
147 // store the stream
148 m_stream = &stream;
149 m_streamOwned = false;
150
151 // Don't forget to reset the stream to its beginning before re-opening it
152 if (stream.seek(0) != 0)
153 {
154 err() << "Failed to open sound file from stream (cannot restart stream)" << std::endl;
155 return false;
156 }
157
158 // Pass the stream to the reader
159 SoundFileReader::Info info;
160 if (!m_reader->open(stream, info))
161 {
162 close();
163 return false;
164 }
165
166 // Retrieve the attributes of the open sound file
167 m_sampleCount = info.sampleCount;
168 m_channelCount = info.channelCount;
169 m_sampleRate = info.sampleRate;
170
171 return true;
172}

196////////////////////////////////////////////////////////////
197Time InputSoundFile::getDuration() const
198{
199 // Make sure we don't divide by 0
200 if (m_channelCount == 0 || m_sampleRate == 0)
201 return Time::Zero;
202
203 return seconds(static_cast<float>(m_sampleCount) / m_channelCount / m_sampleRate);
204}

208Time InputSoundFile::getTimeOffset() const

210 // Make sure we don't divide by 0
211 if (m_channelCount == 0 || m_sampleRate == 0)
212 return Time::Zero;
213
214 return seconds(static_cast<float>(m_sampleOffset) / m_channelCount / m_sampleRate);

225////////////////////////////////////////////////////////////
226void InputSoundFile::seek(Uint64 sampleOffset)
227{
228 if (m_reader)
229 {
230 // The reader handles an overrun gracefully, but we
231 // pre-check to keep our known position consistent
232 m_sampleOffset = std::min(sampleOffset, m_sampleCount);
233 m_reader->seek(m_sampleOffset);
234 }
235}
236
237
238////////////////////////////////////////////////////////////
239void InputSoundFile::seek(Time timeOffset)
240{
241 seek(static_cast<Uint64>(timeOffset.asSeconds() * m_sampleRate * m_channelCount));
242}

245////////////////////////////////////////////////////////////
246Uint64 InputSoundFile::read(Int16* samples, Uint64 maxCount)
247{
248 Uint64 readSamples = 0;
249 if (m_reader && samples && maxCount)
250 readSamples = m_reader->read(samples, maxCount);
251 m_sampleOffset += readSamples;
252 return readSamples;
253}

256////////////////////////////////////////////////////////////
257void InputSoundFile::close()
258{
259 // Destroy the reader
260 delete m_reader;
261 m_reader = NULL;
262
263 // Destroy the stream if we own it
264 if (m_streamOwned)
265 {
266 delete m_stream;
267 m_streamOwned = false;
268 }
269 m_stream = NULL;
270 m_sampleOffset = 0;
271
272 // Reset the sound file attributes
273 m_sampleCount = 0;
274 m_channelCount = 0;
275 m_sampleRate = 0;
276}
