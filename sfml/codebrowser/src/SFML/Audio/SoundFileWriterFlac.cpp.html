39////////////////////////////////////////////////////////////
40bool SoundFileWriterFlac::check(const std::string& filename)
41{
42 std::string extension = filename.substr(filename.find_last_of(".") + 1);
43 std::transform(extension.begin(), extension.end(), extension.begin(), ::tolower);
44
45 return extension == "flac";
46}

65////////////////////////////////////////////////////////////
66bool SoundFileWriterFlac::open(const std::string& filename, unsigned int sampleRate, unsigned int channelCount)
67{
68 // Create the encoder
69 m_encoder = FLAC__stream_encoder_new();
70 if (!m_encoder)
71 {
72 err() << "Failed to write flac file \"" << filename << "\" (failed to allocate encoder)" << std::endl;
73 return false;
74 }
75
76 // Setup the encoder
77 FLAC__stream_encoder_set_channels(m_encoder, channelCount);
78 FLAC__stream_encoder_set_bits_per_sample(m_encoder, 16);
79 FLAC__stream_encoder_set_sample_rate(m_encoder, sampleRate);
80
81 // Initialize the output stream
82 if (FLAC__stream_encoder_init_file(m_encoder, filename.c_str(), NULL, NULL) != FLAC__STREAM_ENCODER_INIT_STATUS_OK)
83 {
84 err() << "Failed to write flac file \"" << filename << "\" (failed to open the file)" << std::endl;
85 close();
86 return false;
87 }
88
89 // Store the channel count
90 m_channelCount = channelCount;
91
92 return true;
93}

96////////////////////////////////////////////////////////////
97void SoundFileWriterFlac::write(const Int16* samples, Uint64 count)
98{
99 while (count > 0)
100 {
101 // Make sure that we don't process too many samples at once
102 unsigned int frames = std::min(static_cast<unsigned int>(count / m_channelCount), 10000u);
103
104 // Convert the samples to 32-bits
105 m_samples32.assign(samples, samples + frames * m_channelCount);
106
107 // Write them to the FLAC stream
108 FLAC__stream_encoder_process_interleaved(m_encoder, &m_samples32[0], frames);
109
110 // Next chunk
111 count -= m_samples32.size();
112 samples += m_samples32.size();
113 }
114}

117////////////////////////////////////////////////////////////
118void SoundFileWriterFlac::close()
119{
120 if (m_encoder)
121 {
122 // Close the output stream
123 FLAC__stream_encoder_finish(m_encoder);
124
125 // Destroy the encoder
126 FLAC__stream_encoder_delete(m_encoder);
127 m_encoder = NULL;
128 }
129}
