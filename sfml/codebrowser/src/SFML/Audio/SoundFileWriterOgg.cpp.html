41bool SoundFileWriterOgg::check(const std::string& filename)
42{
43 std::string extension = filename.substr(filename.find_last_of(".") + 1);
44 std::transform(extension.begin(), extension.end(), extension.begin(), ::tolower);
45
46 return extension == "ogg";
47}

68////////////////////////////////////////////////////////////
69bool SoundFileWriterOgg::open(const std::string& filename, unsigned int sampleRate, unsigned int channelCount)
70{
71 // Save the channel count
72 m_channelCount = channelCount;
73
74 // Initialize the ogg/vorbis stream
75 ogg_stream_init(&m_ogg, std::rand());
76 vorbis_info_init(&m_vorbis);
77
78 // Setup the encoder: VBR, automatic bitrate management
79 // Quality is in range [-1 .. 1], 0.4 gives ~128 kbps for a 44 KHz stereo sound
80 int status = vorbis_encode_init_vbr(&m_vorbis, channelCount, sampleRate, 0.4f);
81 if (status < 0)
82 {
83 err() << "Failed to write ogg/vorbis file \"" << filename << "\" (unsupported bitrate)" << std::endl;
84 close();
85 return false;
86 }
87 vorbis_analysis_init(&m_state, &m_vorbis);
88
89 // Open the file after the vorbis setup is ok
90 m_file.open(filename.c_str(), std::ios::binary);
91 if (!m_file)
92 {
93 err() << "Failed to write ogg/vorbis file \"" << filename << "\" (cannot open file)" << std::endl;
94 close();
95 return false;
96 }
97
98 // Generate header metadata (leave it empty)
99 vorbis_comment comment;
100 vorbis_comment_init(&comment);
101
102 // Generate the header packets
103 ogg_packet header, headerComm, headerCode;
104 status = vorbis_analysis_headerout(&m_state, &comment, &header, &headerComm, &headerCode);
105 vorbis_comment_clear(&comment);
106 if (status < 0)
107 {
108 err() << "Failed to write ogg/vorbis file \"" << filename << "\" (cannot generate the headers)" << std::endl;
109 close();
110 return false;
111 }
112
113 // Write the header packets to the ogg stream
114 ogg_stream_packetin(&m_ogg, &header);
115 ogg_stream_packetin(&m_ogg, &headerComm);
116 ogg_stream_packetin(&m_ogg, &headerCode);
117
118 // This ensures the actual audio data will start on a new page, as per spec
119 ogg_page page;
120 while (ogg_stream_flush(&m_ogg, &page) > 0)
121 {
122 m_file.write(reinterpret_cast<const char*>(page.header), page.header_len);
123 m_file.write(reinterpret_cast<const char*>(page.body), page.body_len);
124 }
125
126 return true;
127}

130////////////////////////////////////////////////////////////
131void SoundFileWriterOgg::write(const Int16* samples, Uint64 count)
132{
133 // Vorbis has issues with buffers that are too large, so we ask for 64K
134 static const int bufferSize = 65536;
135
136 // A frame contains a sample from each channel
137 int frameCount = static_cast<int>(count / m_channelCount);
138
139 while (frameCount > 0)
140 {
141 // Prepare a buffer to hold our samples
142 float** buffer = vorbis_analysis_buffer(&m_state, bufferSize);
143 assert(buffer);
144
145 // Write the samples to the buffer, converted to float
146 for (int i = 0; i < std::min(frameCount, bufferSize); ++i)
147 for (unsigned int j = 0; j < m_channelCount; ++j)
148 buffer[j][i] = *samples++ / 32767.0f;
149
150 // Tell the library how many samples we've written
151 vorbis_analysis_wrote(&m_state, std::min(frameCount, bufferSize));
152
153 frameCount -= bufferSize;
154
155 // Flush any produced block
156 flushBlocks();
157 }
158}

161////////////////////////////////////////////////////////////
162void SoundFileWriterOgg::flushBlocks()
163{
164 // Let the library divide uncompressed data into blocks, and process them
165 vorbis_block block;
166 vorbis_block_init(&m_state, &block);
167 while (vorbis_analysis_blockout(&m_state, &block) == 1)
168 {
169 // Let the automatic bitrate management do its job
170 vorbis_analysis(&block, NULL);
171 vorbis_bitrate_addblock(&block);
172
173 // Get new packets from the bitrate management engine
174 ogg_packet packet;
175 while (vorbis_bitrate_flushpacket(&m_state, &packet))
176 {
177 // Write the packet to the ogg stream
178 ogg_stream_packetin(&m_ogg, &packet);
179
180 // If the stream produced new pages, write them to the output file
181 ogg_page page;
182 while (ogg_stream_flush(&m_ogg, &page) > 0)
183 {
184 m_file.write(reinterpret_cast<const char*>(page.header), page.header_len);
185 m_file.write(reinterpret_cast<const char*>(page.body), page.body_len);
186 }
187 }
188 }
189
190 // Clear the allocated block
191 vorbis_block_clear(&block);
192}

195////////////////////////////////////////////////////////////
196void SoundFileWriterOgg::close()
197{
198 if (m_file.is_open())
199 {
200 // Submit an empty packet to mark the end of stream
201 vorbis_analysis_wrote(&m_state, 0);
202 flushBlocks();
203
204 // Close the file
205 m_file.close();
206 }
207
208 // Clear all the ogg/vorbis structures
209 ogg_stream_clear(&m_ogg);
210 vorbis_dsp_clear(&m_state);
211 vorbis_info_clear(&m_vorbis);
212}
