25#ifndef SFML_SOUNDFILEWRITERFLAC_HPP
26#define SFML_SOUNDFILEWRITERFLAC_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Audio/SoundFileWriter.hpp>
32#include <FLAC/stream_encoder.h>
33#include <vector>
34
35
36namespace sf
37{
38namespace priv
39{
40////////////////////////////////////////////////////////////
41/// \brief Implementation of sound file writer that handles FLAC files
42///
43////////////////////////////////////////////////////////////
44class SoundFileWriterFlac : public SoundFileWriter
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Check if this writer can handle a file on disk
50 ///
51 /// \param filename Path of the sound file to check
52 ///
53 /// \return True if the file can be written by this writer
54 ///
55 ////////////////////////////////////////////////////////////
56 static bool check(const std::string& filename);
57
58public:
59
60 ////////////////////////////////////////////////////////////
61 /// \brief Default constructor
62 ///
63 ////////////////////////////////////////////////////////////
64 SoundFileWriterFlac();
65
66 ////////////////////////////////////////////////////////////
67 /// \brief Destructor
68 ///
69 ////////////////////////////////////////////////////////////
70 ~SoundFileWriterFlac();
71
72 ////////////////////////////////////////////////////////////
73 /// \brief Open a sound file for writing
74 ///
75 /// \param filename Path of the file to open
76 /// \param sampleRate Sample rate of the sound
77 /// \param channelCount Number of channels of the sound
78 ///
79 /// \return True if the file was successfully opened
80 ///
81 ////////////////////////////////////////////////////////////
82 virtual bool open(const std::string& filename, unsigned int sampleRate, unsigned int channelCount);
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Write audio samples to the open file
86 ///
87 /// \param samples Pointer to the sample array to write
88 /// \param count Number of samples to write
89 ///
90 ////////////////////////////////////////////////////////////
91 virtual void write(const Int16* samples, Uint64 count);
92
93private:
94
95 ////////////////////////////////////////////////////////////
96 /// \brief Close the file
97 ///
98 ////////////////////////////////////////////////////////////
99 void close();
100
101 ////////////////////////////////////////////////////////////
102 // Member data
103 ////////////////////////////////////////////////////////////
104 FLAC__StreamEncoder* m_encoder; ///< FLAC stream encoder
105 unsigned int m_channelCount; ///< Number of channels
106 std::vector<Int32> m_samples32; ///< Conversion buffer
107};
108
109} // namespace priv
110
111} // namespace sf
112
113
114#endif // SFML_SOUNDFILEWRITERFLAC_HPP
