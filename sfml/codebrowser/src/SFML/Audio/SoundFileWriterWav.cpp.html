37 // The following functions takes integers in host byte order
38 // and writes them to a stream as little endian
39
40 void encode(std::ostream& stream, sf::Int16 value)
41 {
42 unsigned char bytes[] =
43 {
44 static_cast<unsigned char>(value & 0xFF),
45 static_cast<unsigned char>(value >> 8)
46 };
47 stream.write(reinterpret_cast<const char*>(bytes), sizeof(bytes));
48 }
49
50 void encode(std::ostream& stream, sf::Uint16 value)
51 {
52 unsigned char bytes[] =
53 {
54 static_cast<unsigned char>(value & 0xFF),
55 static_cast<unsigned char>(value >> 8)
56 };
57 stream.write(reinterpret_cast<const char*>(bytes), sizeof(bytes));
58 }
59
60 void encode(std::ostream& stream, sf::Uint32 value)
61 {
62 unsigned char bytes[] =
63 {
64 static_cast<unsigned char>(value & 0x000000FF),
65 static_cast<unsigned char>((value & 0x0000FF00) >> 8),
66 static_cast<unsigned char>((value & 0x00FF0000) >> 16),
67 static_cast<unsigned char>((value & 0xFF000000) >> 24),
68 };
69 stream.write(reinterpret_cast<const char*>(bytes), sizeof(bytes));
70 }
71}

77////////////////////////////////////////////////////////////
78bool SoundFileWriterWav::check(const std::string& filename)
79{
80 std::string extension = filename.substr(filename.find_last_of(".") + 1);
81 std::transform(extension.begin(), extension.end(), extension.begin(), ::tolower);
82
83 return extension == "wav";
84}

103////////////////////////////////////////////////////////////
104bool SoundFileWriterWav::open(const std::string& filename, unsigned int sampleRate, unsigned int channelCount)
105{
106 // Open the file
107 m_file.open(filename.c_str(), std::ios::binary);
108 if (!m_file)
109 {
110 err() << "Failed to open WAV sound file \"" << filename << "\" for writing" << std::endl;
111 return false;
112 }
113
114 // Write the header
115 if (!writeHeader(sampleRate, channelCount))
116 {
117 err() << "Failed to write header of WAV sound file \"" << filename << "\"" << std::endl;
118 return false;
119 }
120
121 // Save the channel count
122 m_channelCount = channelCount;
123
124 return true;
125}

128////////////////////////////////////////////////////////////
129void SoundFileWriterWav::write(const Int16* samples, Uint64 count)
130{
131 assert(m_file.good());
132
133 m_sampleCount += count;
134
135 while (count--)
136 encode(m_file, *samples++);
137}

140////////////////////////////////////////////////////////////
141bool SoundFileWriterWav::writeHeader(unsigned int sampleRate, unsigned int channelCount)
142{
143 assert(m_file.good());
144
145 // Write the main chunk ID
146 char mainChunkId[4] = {'R', 'I', 'F', 'F'};
147 m_file.write(mainChunkId, sizeof(mainChunkId));
148
149 // Write the main chunk header
150 Uint32 mainChunkSize = 0; // placeholder, will be written later
151 encode(m_file, mainChunkSize);
152 char mainChunkFormat[4] = {'W', 'A', 'V', 'E'};
153 m_file.write(mainChunkFormat, sizeof(mainChunkFormat));
154
155 // Write the sub-chunk 1 ("format") id and size
156 char fmtChunkId[4] = {'f', 'm', 't', ' '};
157 m_file.write(fmtChunkId, sizeof(fmtChunkId));
158 Uint32 fmtChunkSize = 16;
159 encode(m_file, fmtChunkSize);
160
161 // Write the format (PCM)
162 Uint16 format = 1;
163 encode(m_file, format);
164
165 // Write the sound attributes
166 encode(m_file, static_cast<Uint16>(channelCount));
167 encode(m_file, static_cast<Uint32>(sampleRate));
168 Uint32 byteRate = sampleRate * channelCount * 2;
169 encode(m_file, byteRate);
170 Uint16 blockAlign = channelCount * 2;
171 encode(m_file, blockAlign);
172 Uint16 bitsPerSample = 16;
173 encode(m_file, bitsPerSample);
174
175 // Write the sub-chunk 2 ("data") id and size
176 char dataChunkId[4] = {'d', 'a', 't', 'a'};
177 m_file.write(dataChunkId, sizeof(dataChunkId));
178 Uint32 dataChunkSize = 0; // placeholder, will be written later
179 encode(m_file, dataChunkSize);
180
181 return true;
182}

185////////////////////////////////////////////////////////////
186void SoundFileWriterWav::close()
187{
188 // If the file is open, finalize the header and close it
189 if (m_file.is_open())
190 {
191 m_file.flush();
192
193 // Update the main chunk size and data sub-chunk size
194 Uint32 dataChunkSize = static_cast<Uint32>(m_sampleCount * m_channelCount * 2);
195 Uint32 mainChunkSize = dataChunkSize + 36;
196 m_file.seekp(4);
197 encode(m_file, mainChunkSize);
198 m_file.seekp(40);
199 encode(m_file, dataChunkSize);
200
201 m_file.close();
202 }
