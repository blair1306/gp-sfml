36 FLAC__StreamDecoderReadStatus streamRead(const FLAC__StreamDecoder*, FLAC__byte buffer[], std::size_t* bytes, void* clientData)
37 {
38 sf::priv::SoundFileReaderFlac::ClientData* data = static_cast<sf::priv::SoundFileReaderFlac::ClientData*>(clientData);
39
40 sf::Int64 count = data->stream->read(buffer, *bytes);
41 if (count > 0)
42 {
43 *bytes = static_cast<std::size_t>(count);
44 return FLAC__STREAM_DECODER_READ_STATUS_CONTINUE;
45 }
46 else if (count == 0)
47 {
48 return FLAC__STREAM_DECODER_READ_STATUS_END_OF_STREAM;
49 }
50 else
51 {
52 return FLAC__STREAM_DECODER_READ_STATUS_ABORT;
53 }
54 }
55
56 FLAC__StreamDecoderSeekStatus streamSeek(const FLAC__StreamDecoder*, FLAC__uint64 absoluteByteOffset, void* clientData)
57 {
58 sf::priv::SoundFileReaderFlac::ClientData* data = static_cast<sf::priv::SoundFileReaderFlac::ClientData*>(clientData);
59
60 sf::Int64 position = data->stream->seek(absoluteByteOffset);
61 if (position >= 0)
62 return FLAC__STREAM_DECODER_SEEK_STATUS_OK;
63 else
64 return FLAC__STREAM_DECODER_SEEK_STATUS_ERROR;
65 }

99 FLAC__bool streamEof(const FLAC__StreamDecoder*, void* clientData)
100 {
101 sf::priv::SoundFileReaderFlac::ClientData* data = static_cast<sf::priv::SoundFileReaderFlac::ClientData*>(clientData);

106 FLAC__StreamDecoderWriteStatus streamWrite(const FLAC__StreamDecoder*, const FLAC__Frame* frame, const FLAC__int32* const buffer[], void* clientData)
107 {
108 sf::priv::SoundFileReaderFlac::ClientData* data = static_cast<sf::priv::SoundFileReaderFlac::ClientData*>(clientData);
109
110 // Reserve memory if we're going to use the leftovers buffer
111 unsigned int frameSamples = frame->header.blocksize * frame->header.channels;
112 if (data->remaining < frameSamples)
113 data->leftovers.reserve(static_cast<std::size_t>(frameSamples - data->remaining));
114
115 // Decode the samples
116 for (unsigned i = 0; i < frame->header.blocksize; ++i)
117 {
118 for (unsigned int j = 0; j < frame->header.channels; ++j)
119 {
120 // Decode the current sample
121 sf::Int16 sample = 0;
122 switch (frame->header.bits_per_sample)
123 {
124 case 8:
125 sample = buffer[j][i] << 8;
126 break;
127 case 16:
128 sample = buffer[j][i];
129 break;
130 case 24:
131 sample = buffer[j][i] >> 8;
132 break;
133 case 32:
134 sample = buffer[j][i] >> 16;
135 break;
136 default:
137 assert(false);
138 break;
139 }
140
141 if (data->buffer && data->remaining > 0)
142 {
143 // If there's room in the output buffer, copy the sample there
144 *data->buffer++ = sample;
145 data->remaining--;
146 }
147 else
148 {
149 // We are either seeking (null buffer) or have decoded all the requested samples during a
150 // normal read (0 remaining), so we put the sample in a temporary buffer until next call
151 data->leftovers.push_back(sample);
152 }
153 }
154 }
155
156 return FLAC__STREAM_DECODER_WRITE_STATUS_CONTINUE;
157 }

159 void streamMetadata(const FLAC__StreamDecoder*, const FLAC__StreamMetadata* meta, void* clientData)
160 {
161 sf::priv::SoundFileReaderFlac::ClientData* data = static_cast<sf::priv::SoundFileReaderFlac::ClientData*>(clientData);
162
163 if (meta->type == FLAC__METADATA_TYPE_STREAMINFO)
164 {
165 data->info.sampleCount = meta->data.stream_info.total_samples * meta->data.stream_info.channels;
166 data->info.sampleRate = meta->data.stream_info.sample_rate;
167 data->info.channelCount = meta->data.stream_info.channels;
168 }
169 }

171 void streamError(const FLAC__StreamDecoder*, FLAC__StreamDecoderErrorStatus, void* clientData)
172 {
173 sf::priv::SoundFileReaderFlac::ClientData* data = static_cast<sf::priv::SoundFileReaderFlac::ClientData*>(clientData);
174 data->error = true;

182////////////////////////////////////////////////////////////
183bool SoundFileReaderFlac::check(InputStream& stream)
184{
185 // Create a decoder
186 FLAC__StreamDecoder* decoder = FLAC__stream_decoder_new();
187 if (!decoder)
188 return false;
189
190 // Initialize the decoder with our callbacks
191 ClientData data;
192 data.stream = &stream;
193 data.error = false;
194 FLAC__stream_decoder_init_stream(decoder, &streamRead, &streamSeek, &streamTell, &streamLength, &streamEof, &streamWrite, NULL, &streamError, &data);
195
196 // Read the header
197 bool valid = FLAC__stream_decoder_process_until_end_of_metadata(decoder) != 0;
198
199 // Destroy the decoder
200 FLAC__stream_decoder_finish(decoder);
201 FLAC__stream_decoder_delete(decoder);
202
203 return valid && !data.error;
204}

222////////////////////////////////////////////////////////////
223bool SoundFileReaderFlac::open(InputStream& stream, Info& info)
224{
225 // Create the decoder
226 m_decoder = FLAC__stream_decoder_new();
227 if (!m_decoder)
228 {
229 err() << "Failed to open FLAC file (failed to allocate the decoder)" << std::endl;
230 return false;
231 }
232
233 // Initialize the decoder with our callbacks
234 m_clientData.stream = &stream;
235 FLAC__stream_decoder_init_stream(m_decoder, &streamRead, &streamSeek, &streamTell, &streamLength, &streamEof, &streamWrite, &streamMetadata, &streamError, &m_clientData);
236
237 // Read the header
238 if (!FLAC__stream_decoder_process_until_end_of_metadata(m_decoder))
239 {
240 close();
241 err() << "Failed to open FLAC file (failed to read metadata)" << std::endl;
242 return false;
243 }
244
245 // Retrieve the sound properties
246 info = m_clientData.info; // was filled in the "metadata" callback
247
248 return true;
249}

252////////////////////////////////////////////////////////////
253void SoundFileReaderFlac::seek(Uint64 sampleOffset)
254{
255 assert(m_decoder);
256
257 // Reset the callback data (the "write" callback will be called)
258 m_clientData.buffer = NULL;
259 m_clientData.remaining = 0;
260 m_clientData.leftovers.clear();
261
262 // FLAC decoder expects absolute sample offset, so we take the channel count out
263 if (sampleOffset < m_clientData.info.sampleCount)
264 {
265 // The "write" callback will populate the leftovers buffer with the first batch of samples from the
266 // seek destination, and since we want that data in this typical case, we don't re-clear it afterward
267 FLAC__stream_decoder_seek_absolute(m_decoder, sampleOffset / m_clientData.info.channelCount);
268 }
269 else
270 {
271 // FLAC decoder can't skip straight to EOF, so we short-seek by one sample and skip the rest
272 FLAC__stream_decoder_seek_absolute(m_decoder, (m_clientData.info.sampleCount / m_clientData.info.channelCount) - 1);
273 FLAC__stream_decoder_skip_single_frame(m_decoder);
274
275 // This was re-populated during the seek, but we're skipping everything in this, so we need it emptied
276 m_clientData.leftovers.clear();
277 }
278}

281////////////////////////////////////////////////////////////
282Uint64 SoundFileReaderFlac::read(Int16* samples, Uint64 maxCount)
283{
284 assert(m_decoder);
285
286 // If there are leftovers from previous call, use it first
287 Uint64 left = m_clientData.leftovers.size();
288 if (left > 0)
289 {
290 if (left > maxCount)
291 {
292 // There are more leftovers than needed
293 std::copy(m_clientData.leftovers.begin(), m_clientData.leftovers.begin() + maxCount, samples);
294 std::vector<Int16> leftovers(m_clientData.leftovers.begin() + maxCount, m_clientData.leftovers.end());
295 m_clientData.leftovers.swap(leftovers);
296 return maxCount;
297 }
298 else
299 {
300 // We can use all the leftovers and decode new frames
301 std::copy(m_clientData.leftovers.begin(), m_clientData.leftovers.end(), samples);
302 }
303 }
304
305 // Reset the data that will be used in the callback
306 m_clientData.buffer = samples + left;
307 m_clientData.remaining = maxCount - left;
308 m_clientData.leftovers.clear();
309
310 // Decode frames one by one until we reach the requested sample count, the end of file or an error
311 while (m_clientData.remaining > 0)
312 {
313 // Everything happens in the "write" callback
314 // This will break on any fatal error (does not include EOF)
315 if (!FLAC__stream_decoder_process_single(m_decoder))
316 break;
317
318 // Break on EOF
319 if (FLAC__stream_decoder_get_state(m_decoder) == FLAC__STREAM_DECODER_END_OF_STREAM)
320 break;
321 }
322
323 return maxCount - m_clientData.remaining;
324}
