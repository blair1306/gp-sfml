28#include <SFML/Audio/SoundBufferRecorder.hpp>
29#include <algorithm>
30#include <iterator>
31
32
33namespace sf
34{
35////////////////////////////////////////////////////////////
36SoundBufferRecorder::~SoundBufferRecorder()
37{
38 // Make sure to stop the recording thread
39 stop();
40}
41
42
43////////////////////////////////////////////////////////////
44bool SoundBufferRecorder::onStart()
45{
46 m_samples.clear();
47 m_buffer = SoundBuffer();
48
49 return true;
50}
51
52
53////////////////////////////////////////////////////////////
54bool SoundBufferRecorder::onProcessSamples(const Int16* samples, std::size_t sampleCount)
55{
56 std::copy(samples, samples + sampleCount, std::back_inserter(m_samples));
57
58 return true;
59}
60
61
62////////////////////////////////////////////////////////////
63void SoundBufferRecorder::onStop()
64{
65 if (!m_samples.empty())
66 m_buffer.loadFromSamples(&m_samples[0], m_samples.size(), getChannelCount(), getSampleRate());
67}
68
69
70////////////////////////////////////////////////////////////
71const SoundBuffer& SoundBufferRecorder::getBuffer() const
72{
73 return m_buffer;
74}
