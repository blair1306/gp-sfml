35////////////////////////////////////////////////////////////
36OutputSoundFile::OutputSoundFile() :
37m_writer(NULL)
38{
39}
40
41
42////////////////////////////////////////////////////////////
43OutputSoundFile::~OutputSoundFile()
44{
45 // Close the file in case it was open
46 close();
47}
48
49
50////////////////////////////////////////////////////////////
51bool OutputSoundFile::openFromFile(const std::string& filename, unsigned int sampleRate, unsigned int channelCount)
52{
53 // If the file is already open, first close it
54 close();
55
56 // Find a suitable writer for the file type
57 m_writer = SoundFileFactory::createWriterFromFilename(filename);
58 if (!m_writer)
59 return false;
60
61 // Pass the stream to the reader
62 if (!m_writer->open(filename, sampleRate, channelCount))
63 {
64 close();
65 return false;
66 }
67
68 return true;
69}
70
71
72////////////////////////////////////////////////////////////
73void OutputSoundFile::write(const Int16* samples, Uint64 count)
74{
75 if (m_writer && samples && count)
76 m_writer->write(samples, count);
77}
78
79
80////////////////////////////////////////////////////////////
81void OutputSoundFile::close()
82{
83 // Destroy the reader
84 delete m_writer;
85 m_writer = NULL;
86}
