50////////////////////////////////////////////////////////////
51AudioDevice::AudioDevice()
52{
53 // Create the device
54 audioDevice = alcOpenDevice(NULL);
55
56 if (audioDevice)
57 {
58 // Create the context
59 audioContext = alcCreateContext(audioDevice, NULL);
60
61 if (audioContext)
62 {
63 // Set the context as the current one (we'll only need one)
64 alcMakeContextCurrent(audioContext);
65
66 // Apply the listener properties the user might have set
67 float orientation[] = {listenerDirection.x,
68 listenerDirection.y,
69 listenerDirection.z,
70 listenerUpVector.x,
71 listenerUpVector.y,
72 listenerUpVector.z};
73 alCheck(alListenerf(AL_GAIN, listenerVolume * 0.01f));
74 alCheck(alListener3f(AL_POSITION, listenerPosition.x, listenerPosition.y, listenerPosition.z));
75 alCheck(alListenerfv(AL_ORIENTATION, orientation));
76 }
77 else
78 {
79 err() << "Failed to create the audio context" << std::endl;
80 }
81 }
82 else
83 {
84 err() << "Failed to open the audio device" << std::endl;
85 }
86}

89////////////////////////////////////////////////////////////
90AudioDevice::~AudioDevice()
91{
92 // Destroy the context
93 alcMakeContextCurrent(NULL);
94 if (audioContext)
95 alcDestroyContext(audioContext);
96
97 // Destroy the device
98 if (audioDevice)
99 alcCloseDevice(audioDevice);
100}

104bool AudioDevice::isExtensionSupported(const std::string& extension)
105{
106 // Create a temporary audio device in case none exists yet.
107 // This device will not be used in this function and merely
108 // makes sure there is a valid OpenAL device for extension
109 // queries if none has been created yet.
110 std::auto_ptr<AudioDevice> device;
111 if (!audioDevice)
112 device.reset(new AudioDevice);
113
114 if ((extension.length() > 2) && (extension.substr(0, 3) == "ALC"))
115 return alcIsExtensionPresent(audioDevice, extension.c_str()) != AL_FALSE;
116 else
117 return alIsExtensionPresent(extension.c_str()) != AL_FALSE;
118}

121////////////////////////////////////////////////////////////
122int AudioDevice::getFormatFromChannelCount(unsigned int channelCount)
123{
124 // Create a temporary audio device in case none exists yet.
125 // This device will not be used in this function and merely
126 // makes sure there is a valid OpenAL device for format
127 // queries if none has been created yet.
128 std::auto_ptr<AudioDevice> device;
129 if (!audioDevice)
130 device.reset(new AudioDevice);
131
132 // Find the good format according to the number of channels
133 int format = 0;
134 switch (channelCount)
135 {
136 case 1: format = AL_FORMAT_MONO16; break;
137 case 2: format = AL_FORMAT_STEREO16; break;
138 case 4: format = alGetEnumValue("AL_FORMAT_QUAD16"); break;
139 case 6: format = alGetEnumValue("AL_FORMAT_51CHN16"); break;
140 case 7: format = alGetEnumValue("AL_FORMAT_61CHN16"); break;
141 case 8: format = alGetEnumValue("AL_FORMAT_71CHN16"); break;
142 default: format = 0; break;
143 }
144
145 // Fixes a bug on OS X
146 if (format == -1)
147 format = 0;
148
149 return format;
150}

154void AudioDevice::setGlobalVolume(float volume)
155{
156 if (audioContext)
157 alCheck(alListenerf(AL_GAIN, volume * 0.01f));
158
159 listenerVolume = volume;
160}

188void AudioDevice::setDirection(const Vector3f& direction)
189{
190 if (audioContext)
191 {
192 float orientation[] = {direction.x, direction.y, direction.z, listenerUpVector.x, listenerUpVector.y, listenerUpVector.z};
193 alCheck(alListenerfv(AL_ORIENTATION, orientation));
194 }
195
196 listenerDirection = direction;
197}
