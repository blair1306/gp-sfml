25#ifndef SFML_SOUNDFILEREADERFLAC_HPP
26#define SFML_SOUNDFILEREADERFLAC_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Audio/SoundFileReader.hpp>
32#include <FLAC/stream_decoder.h>
33#include <string>
34#include <vector>
35
36
37namespace sf
38{
39namespace priv
40{
41////////////////////////////////////////////////////////////
42/// \brief Implementation of sound file reader that handles FLAC files
43///
44////////////////////////////////////////////////////////////
45class SoundFileReaderFlac : public SoundFileReader
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 /// \brief Check if this reader can handle a file given by an input stream
51 ///
52 /// \param stream Source stream to check
53 ///
54 /// \return True if the file is supported by this reader
55 ///
56 ////////////////////////////////////////////////////////////
57 static bool check(InputStream& stream);
58
59public:
60
61 ////////////////////////////////////////////////////////////
62 /// \brief Default constructor
63 ///
64 ////////////////////////////////////////////////////////////
65 SoundFileReaderFlac();
66
67 ////////////////////////////////////////////////////////////
68 /// \brief Default constructor
69 ///
70 ////////////////////////////////////////////////////////////
71 ~SoundFileReaderFlac();
72
73 ////////////////////////////////////////////////////////////
74 /// \brief Open a sound file for reading
75 ///
76 /// \param stream Stream to open
77 /// \param info Structure to fill with the attributes of the loaded sound
78 ///
79 ////////////////////////////////////////////////////////////
80 virtual bool open(sf::InputStream& stream, Info& info);
81
82 ////////////////////////////////////////////////////////////
83 /// \brief Change the current read position to the given sample offset
84 ///
85 /// The sample offset takes the channels into account.
86 /// Offsets can be calculated like this:
87 /// `sampleNumber * sampleRate * channelCount`
88 /// If the given offset exceeds to total number of samples,
89 /// this function must jump to the end of the file.
90 ///
91 /// \param sampleOffset Index of the sample to jump to, relative to the beginning
92 ///
93 ////////////////////////////////////////////////////////////
94 virtual void seek(Uint64 sampleOffset);
95
96 ////////////////////////////////////////////////////////////
97 /// \brief Read audio samples from the open file
98 ///
99 /// \param samples Pointer to the sample array to fill
100 /// \param maxCount Maximum number of samples to read
101 ///
102 /// \return Number of samples actually read (may be less than \a maxCount)
103 ///
104 ////////////////////////////////////////////////////////////
105 virtual Uint64 read(Int16* samples, Uint64 maxCount);
106
107public:
108
109 ////////////////////////////////////////////////////////////
110 /// \brief Hold the state that is passed to the decoder callbacks
111 ///
112 ////////////////////////////////////////////////////////////
113 struct ClientData
114 {
115 InputStream* stream;
116 SoundFileReader::Info info;
117 Int16* buffer;
118 Uint64 remaining;
119 std::vector<Int16> leftovers;
120 bool error;
121 };
122
123private:
124
125 ////////////////////////////////////////////////////////////
126 /// \brief Close the open FLAC file
127 ///
128 ////////////////////////////////////////////////////////////
129 void close();
130
131 ////////////////////////////////////////////////////////////
132 // Member data
133 ////////////////////////////////////////////////////////////
134 FLAC__StreamDecoder* m_decoder; ///< FLAC decoder
135 ClientData m_clientData; ///< Structure passed to the decoder callbacks
136};
137
138} // namespace priv
139
140} // namespace sf
141
142
143#endif // SFML_SOUNDFILEREADERFLAC_HPP
