38 size_t read(void* ptr, size_t size, size_t nmemb, void* data)
39 {
40 sf::InputStream* stream = static_cast<sf::InputStream*>(data);
41 return static_cast<std::size_t>(stream->read(ptr, size * nmemb));
42 }
43
44 int seek(void* data, ogg_int64_t offset, int whence)
45 {
46 sf::InputStream* stream = static_cast<sf::InputStream*>(data);
47 switch (whence)
48 {
49 case SEEK_SET:
50 break;
51 case SEEK_CUR:
52 offset += stream->tell();
53 break;
54 case SEEK_END:
55 offset = stream->getSize() - offset;
56 }
57 return static_cast<int>(stream->seek(offset));
58 }
59
60 long tell(void* data)
61 {
62 sf::InputStream* stream = static_cast<sf::InputStream*>(data);
63 return static_cast<long>(stream->tell());
64 }
65
66 static ov_callbacks callbacks = {&read, &seek, NULL, &tell};
67}

73////////////////////////////////////////////////////////////
74bool SoundFileReaderOgg::check(InputStream& stream)
75{
76 OggVorbis_File file;
77 if (ov_test_callbacks(&stream, &file, NULL, 0, callbacks) == 0)
78 {
79 ov_clear(&file);
80 return true;
81 }
82 else
83 {
84 return false;
85 }
86}

105////////////////////////////////////////////////////////////
106bool SoundFileReaderOgg::open(InputStream& stream, Info& info)
107{
108 // Open the Vorbis stream
109 int status = ov_open_callbacks(&stream, &m_vorbis, NULL, 0, callbacks);
110 if (status < 0)
111 {
112 err() << "Failed to open Vorbis file for reading" << std::endl;
113 return false;
114 }
115
116 // Retrieve the music attributes
117 vorbis_info* vorbisInfo = ov_info(&m_vorbis, -1);
118 info.channelCount = vorbisInfo->channels;
119 info.sampleRate = vorbisInfo->rate;
120 info.sampleCount = static_cast<std::size_t>(ov_pcm_total(&m_vorbis, -1) * vorbisInfo->channels);
121
122 // We must keep the channel count for the seek function
123 m_channelCount = info.channelCount;
124
125 return true;
126}

129////////////////////////////////////////////////////////////
130void SoundFileReaderOgg::seek(Uint64 sampleOffset)
131{
132 assert(m_vorbis.datasource);
133
134 ov_pcm_seek(&m_vorbis, sampleOffset / m_channelCount);
135}

138////////////////////////////////////////////////////////////
139Uint64 SoundFileReaderOgg::read(Int16* samples, Uint64 maxCount)
140{
141 assert(m_vorbis.datasource);
142
143 // Try to read the requested number of samples, stop only on error or end of file
144 Uint64 count = 0;
145 while (count < maxCount)
146 {
147 int bytesToRead = static_cast<int>(maxCount - count) * sizeof(Int16);
148 long bytesRead = ov_read(&m_vorbis, reinterpret_cast<char*>(samples), bytesToRead, 0, 2, 1, NULL);
149 if (bytesRead > 0)
150 {
151 long samplesRead = bytesRead / sizeof(Int16);
152 count += samplesRead;
153 samples += samplesRead;
154 }
155 else
156 {
157 // error or end of file
158 break;
159 }
160 }
161
162 return count;
163}

166////////////////////////////////////////////////////////////
167void SoundFileReaderOgg::close()
168{
169 if (m_vorbis.datasource)
170 {
171 ov_clear(&m_vorbis);
172 m_vorbis.datasource = NULL;
173 m_channelCount = 0;
174 }
175}
