60////////////////////////////////////////////////////////////
61SoundStream::~SoundStream()
62{
63 // Stop the sound if it was playing
64
65 // Request the thread to terminate
66 {
67 Lock lock(m_threadMutex);
68 m_isStreaming = false;
69 }
70
71 // Wait for the thread to terminate
72 m_thread.wait();
73}

76////////////////////////////////////////////////////////////
77void SoundStream::initialize(unsigned int channelCount, unsigned int sampleRate)
78{
79 m_channelCount = channelCount;
80 m_sampleRate = sampleRate;
81 m_samplesProcessed = 0;
82 m_isStreaming = false;
83
84 // Deduce the format from the number of channels
85 m_format = priv::AudioDevice::getFormatFromChannelCount(channelCount);
86
87 // Check if the format is valid
88 if (m_format == 0)
89 {
90 m_channelCount = 0;
91 m_sampleRate = 0;
92 err() << "Unsupported number of channels (" << m_channelCount << ")" << std::endl;
93 }
94}

97////////////////////////////////////////////////////////////
98void SoundStream::play()
99{
100 // Check if the sound parameters have been set
101 if (m_format == 0)
102 {
103 err() << "Failed to play audio stream: sound parameters have not been initialized (call initialize() first)" << std::endl;
104 return;
105 }
106
107 bool isStreaming = false;
108 Status threadStartState = Stopped;
109
110 {
111 Lock lock(m_threadMutex);
112
113 isStreaming = m_isStreaming;
114 threadStartState = m_threadStartState;
115 }
116
117
118 if (isStreaming && (threadStartState == Paused))
119 {
120 // If the sound is paused, resume it
121 Lock lock(m_threadMutex);
122 m_threadStartState = Playing;
123 alCheck(alSourcePlay(m_source));
124 return;
125 }
126 else if (isStreaming && (threadStartState == Playing))
127 {
128 // If the sound is playing, stop it and continue as if it was stopped
129 stop();
130 }
131
132 // Start updating the stream in a separate thread to avoid blocking the application
133 m_isStreaming = true;
134 m_threadStartState = Playing;
135 m_thread.launch();
136}

139////////////////////////////////////////////////////////////
140void SoundStream::pause()
141{
142 // Handle pause() being called before the thread has started
143 {
144 Lock lock(m_threadMutex);
145
146 if (!m_isStreaming)
147 return;
148
149 m_threadStartState = Paused;
150 }
151
152 alCheck(alSourcePause(m_source));
153}

156////////////////////////////////////////////////////////////
157void SoundStream::stop()
158{
159 // Request the thread to terminate
160 {
161 Lock lock(m_threadMutex);
162 m_isStreaming = false;
163 }
164
165 // Wait for the thread to terminate
166 m_thread.wait();
167
168 // Move to the beginning
169 onSeek(Time::Zero);
170}

187////////////////////////////////////////////////////////////
188SoundStream::Status SoundStream::getStatus() const
189{
190 Status status = SoundSource::getStatus();
191
192 // To compensate for the lag between play() and alSourceplay()
193 if (status == Stopped)
194 {
195 Lock lock(m_threadMutex);
196
197 if (m_isStreaming)
198 status = m_threadStartState;
199 }
200
201 return status;
202}

205////////////////////////////////////////////////////////////
206void SoundStream::setPlayingOffset(Time timeOffset)
207{
208 // Get old playing status
209 Status oldStatus = getStatus();
210
211 // Stop the stream
212 stop();
213
214 // Let the derived class update the current position
215 onSeek(timeOffset);
216
217 // Restart streaming
218 m_samplesProcessed = static_cast<Uint64>(timeOffset.asSeconds() * m_sampleRate * m_channelCount);
219
220 if (oldStatus == Stopped)
221 return;
222
223 m_isStreaming = true;
224 m_threadStartState = oldStatus;
225 m_thread.launch();
226}

229////////////////////////////////////////////////////////////
230Time SoundStream::getPlayingOffset() const
231{
232 if (m_sampleRate && m_channelCount)
233 {
234 ALfloat secs = 0.f;
235 alCheck(alGetSourcef(m_source, AL_SEC_OFFSET, &secs));
236
237 return seconds(secs + static_cast<float>(m_samplesProcessed) / m_sampleRate / m_channelCount);
238 }
239 else
240 {
241 return Time::Zero;
242 }
243}

268////////////////////////////////////////////////////////////
269void SoundStream::streamData()
270{
271 bool requestStop = false;
272
273 {
274 Lock lock(m_threadMutex);
275
276 // Check if the thread was launched Stopped
277 if (m_threadStartState == Stopped)
278 {
279 m_isStreaming = false;
280 return;
281 }
282 }
283
284 // Create the buffers
285 alCheck(alGenBuffers(BufferCount, m_buffers));
286 for (int i = 0; i < BufferCount; ++i)
287 m_bufferSeeks[i] = NoLoop;
288
289 // Fill the queue
290 requestStop = fillQueue();
291
292 // Play the sound
293 alCheck(alSourcePlay(m_source));
294
295 {
296 Lock lock(m_threadMutex);
297
298 // Check if the thread was launched Paused
299 if (m_threadStartState == Paused)
300 alCheck(alSourcePause(m_source));
301 }
302
303 for (;;)
304 {
305 {
306 Lock lock(m_threadMutex);
307 if (!m_isStreaming)
308 break;
309 }
310
311 // The stream has been interrupted!
312 if (SoundSource::getStatus() == Stopped)
313 {
314 if (!requestStop)
315 {
316 // Just continue
317 alCheck(alSourcePlay(m_source));
318 }
319 else
320 {
321 // End streaming
322 Lock lock(m_threadMutex);
323 m_isStreaming = false;
324 }
325 }
326
327 // Get the number of buffers that have been processed (i.e. ready for reuse)
328 ALint nbProcessed = 0;
329 alCheck(alGetSourcei(m_source, AL_BUFFERS_PROCESSED, &nbProcessed));
330
331 while (nbProcessed--)
332 {
333 // Pop the first unused buffer from the queue
334 ALuint buffer;
335 alCheck(alSourceUnqueueBuffers(m_source, 1, &buffer));
336
337 // Find its number
338 unsigned int bufferNum = 0;
339 for (int i = 0; i < BufferCount; ++i)
340 if (m_buffers[i] == buffer)
341 {
342 bufferNum = i;
343 break;
344 }
345
346 // Retrieve its size and add it to the samples count
347 if (m_bufferSeeks[bufferNum] != NoLoop)
348 {
349 // This was the last buffer before EOF or Loop End: reset the sample count
350 m_samplesProcessed = m_bufferSeeks[bufferNum];
351 m_bufferSeeks[bufferNum] = NoLoop;
352 }
353 else
354 {
355 ALint size, bits;
356 alCheck(alGetBufferi(buffer, AL_SIZE, &size));
357 alCheck(alGetBufferi(buffer, AL_BITS, &bits));
358
359 // Bits can be 0 if the format or parameters are corrupt, avoid division by zero
360 if (bits == 0)
361 {
362 err() << "Bits in sound stream are 0: make sure that the audio format is not corrupt "
363 << "and initialize() has been called correctly" << std::endl;
364
365 // Abort streaming (exit main loop)
366 Lock lock(m_threadMutex);
367 m_isStreaming = false;
368 requestStop = true;
369 break;
370 }
371 else
372 {
373 m_samplesProcessed += size / (bits / 8);
374 }
375 }
376
377 // Fill it and push it back into the playing queue
378 if (!requestStop)
379 {
380 if (fillAndPushBuffer(bufferNum))
381 requestStop = true;
382 }
383 }
384
385 // Leave some time for the other threads if the stream is still playing
386 if (SoundSource::getStatus() != Stopped)
387 sleep(milliseconds(10));
388 }
389
390 // Stop the playback
391 alCheck(alSourceStop(m_source));
392
393 // Dequeue any buffer left in the queue
394 clearQueue();
395
396 // Reset the playing position
397 m_samplesProcessed = 0;
398
399 // Delete the buffers
400 alCheck(alSourcei(m_source, AL_BUFFER, 0));
401 alCheck(alDeleteBuffers(BufferCount, m_buffers));
402}

405////////////////////////////////////////////////////////////
406bool SoundStream::fillAndPushBuffer(unsigned int bufferNum, bool immediateLoop)
407{
408 bool requestStop = false;
409
410 // Acquire audio data, also address EOF and error cases if they occur
411 Chunk data = {NULL, 0};
412 for (Uint32 retryCount = 0; !onGetData(data) && (retryCount < BufferRetries); ++retryCount)
413 {
414 // Check if the stream must loop or stop
415 if (!m_loop)
416 {
417 // Not looping: Mark this buffer as ending with 0 and request stop
418 if (data.samples != NULL && data.sampleCount != 0)
419 m_bufferSeeks[bufferNum] = 0;
420 requestStop = true;
421 break;
422 }
423
424 // Return to the beginning or loop-start of the stream source using onLoop(), and store the result in the buffer seek array
425 // This marks the buffer as the "last" one (so that we know where to reset the playing position)
426 m_bufferSeeks[bufferNum] = onLoop();
427
428 // If we got data, break and process it, else try to fill the buffer once again
429 if (data.samples != NULL && data.sampleCount != 0)
430 break;
431
432 // If immediateLoop is specified, we have to immediately adjust the sample count
433 if (immediateLoop && (m_bufferSeeks[bufferNum] != NoLoop))
434 {
435 // We just tried to begin preloading at EOF or Loop End: reset the sample count
436 m_samplesProcessed = m_bufferSeeks[bufferNum];
437 m_bufferSeeks[bufferNum] = NoLoop;
438 }
439
440 // We're a looping sound that got no data, so we retry onGetData()
441 }
442
443 // Fill the buffer if some data was returned
444 if (data.samples && data.sampleCount)
445 {
446 unsigned int buffer = m_buffers[bufferNum];
447
448 // Fill the buffer
449 ALsizei size = static_cast<ALsizei>(data.sampleCount) * sizeof(Int16);
450 alCheck(alBufferData(buffer, m_format, data.samples, size, m_sampleRate));
451
452 // Push it into the sound queue
453 alCheck(alSourceQueueBuffers(m_source, 1, &buffer));
454 }
455 else
456 {
457 // If we get here, we most likely ran out of retries
458 requestStop = true;
459 }
460
461 return requestStop;
462}

465////////////////////////////////////////////////////////////
466bool SoundStream::fillQueue()
467{
468 // Fill and enqueue all the available buffers
469 bool requestStop = false;
470 for (int i = 0; (i < BufferCount) && !requestStop; ++i)
471 {
472 // Since no sound has been loaded yet, we can't schedule loop seeks preemptively,
473 // So if we start on EOF or Loop End, we let fillAndPushBuffer() adjust the sample count
474 if (fillAndPushBuffer(i, (i == 0)))
475 requestStop = true;
476 }
477
478 return requestStop;
479}

482////////////////////////////////////////////////////////////
483void SoundStream::clearQueue()
484{
485 // Get the number of buffers still in the queue
486 ALint nbQueued;
487 alCheck(alGetSourcei(m_source, AL_BUFFERS_QUEUED, &nbQueued));
488
489 // Dequeue them all
490 ALuint buffer;
491 for (ALint i = 0; i < nbQueued; ++i)
492 alCheck(alSourceUnqueueBuffers(m_source, 1, &buffer));
493}
