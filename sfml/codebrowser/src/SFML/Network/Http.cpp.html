39 // Convert a string to lower case
40 std::string toLower(std::string str)
41 {
42 for (std::string::iterator i = str.begin(); i != str.end(); ++i)
43 *i = static_cast<char>(std::tolower(*i));
44 return str;
45 }

101////////////////////////////////////////////////////////////
102std::string Http::Request::prepare() const
103{
104 std::ostringstream out;
105
106 // Convert the method to its string representation
107 std::string method;
108 switch (m_method)
109 {
110 case Get: method = "GET"; break;
111 case Post: method = "POST"; break;
112 case Head: method = "HEAD"; break;
113 case Put: method = "PUT"; break;
114 case Delete: method = "DELETE"; break;
115 }
116
117 // Write the first line containing the request type
118 out << method << " " << m_uri << " ";
119 out << "HTTP/" << m_majorVersion << "." << m_minorVersion << "\r\n";
120
121 // Write fields
122 for (FieldTable::const_iterator i = m_fields.begin(); i != m_fields.end(); ++i)
123 {
124 out << i->first << ": " << i->second << "\r\n";
125 }
126
127 // Use an extra \r\n to separate the header from the body
128 out << "\r\n";
129
130 // Add the body
131 out << m_body;
132
133 return out.str();
134}

198////////////////////////////////////////////////////////////
199void Http::Response::parse(const std::string& data)
200{
201 std::istringstream in(data);
202
203 // Extract the HTTP version from the first line
204 std::string version;
205 if (in >> version)
206 {
207 if ((version.size() >= 8) && (version[6] == '.') &&
208 (toLower(version.substr(0, 5)) == "http/") &&
209 isdigit(version[5]) && isdigit(version[7]))
210 {
211 m_majorVersion = version[5] - '0';
212 m_minorVersion = version[7] - '0';
213 }
214 else
215 {
216 // Invalid HTTP version
217 m_status = InvalidResponse;
218 return;
219 }
220 }
221
222 // Extract the status code from the first line
223 int status;
224 if (in >> status)
225 {
226 m_status = static_cast<Status>(status);
227 }
228 else
229 {
230 // Invalid status code
231 m_status = InvalidResponse;
232 return;
233 }
234
235 // Ignore the end of the first line
236 in.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
237
238 // Parse the other lines, which contain fields, one by one
239 parseFields(in);
240
241 m_body.clear();
242
243 // Determine whether the transfer is chunked
244 if (toLower(getField("transfer-encoding")) != "chunked")
245 {
246 // Not chunked - just read everything at once
247 std::copy(std::istreambuf_iterator<char>(in), std::istreambuf_iterator<char>(), std::back_inserter(m_body));
248 }
249 else
250 {
251 // Chunked - have to read chunk by chunk
252 std::size_t length;
253
254 // Read all chunks, identified by a chunk-size not being 0
255 while (in >> std::hex >> length)
256 {
257 // Drop the rest of the line (chunk-extension)
258 in.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
259
260 // Copy the actual content data
261 std::istreambuf_iterator<char> it(in);
262 std::istreambuf_iterator<char> itEnd;
263 for (std::size_t i = 0; ((i < length) && (it != itEnd)); i++)
264 m_body.push_back(*it++);
265 }
266
267 // Drop the rest of the line (chunk-extension)
268 in.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
269
270 // Read all trailers (if present)
271 parseFields(in);
272 }
273}

276////////////////////////////////////////////////////////////
277void Http::Response::parseFields(std::istream &in)
278{
279 std::string line;
280 while (std::getline(in, line) && (line.size() > 2))
281 {
282 std::string::size_type pos = line.find(": ");
283 if (pos != std::string::npos)
284 {
285 // Extract the field name and its value
286 std::string field = line.substr(0, pos);
287 std::string value = line.substr(pos + 2);
288
289 // Remove any trailing \r
290 if (!value.empty() && (*value.rbegin() == '\r'))
291 value.erase(value.size() - 1);
292
293 // Add the field
294 m_fields[toLower(field)] = value;
295 }
296 }
297}

316////////////////////////////////////////////////////////////
317void Http::setHost(const std::string& host, unsigned short port)
318{
319 // Check the protocol
320 if (toLower(host.substr(0, 7)) == "http://")
321 {
322 // HTTP protocol
323 m_hostName = host.substr(7);
324 m_port = (port != 0 ? port : 80);
325 }
326 else if (toLower(host.substr(0, 8)) == "https://")
327 {
328 // HTTPS protocol -- unsupported (requires encryption and certificates and stuff...)
329 err() << "HTTPS protocol is not supported by sf::Http" << std::endl;
330 m_hostName = "";
331 m_port = 0;
332 }
333 else
334 {
335 // Undefined protocol - use HTTP
336 m_hostName = host;
337 m_port = (port != 0 ? port : 80);
338 }
339
340 // Remove any trailing '/' from the host name
341 if (!m_hostName.empty() && (*m_hostName.rbegin() == '/'))
342 m_hostName.erase(m_hostName.size() - 1);
343
344 m_host = IpAddress(m_hostName);
345}

348////////////////////////////////////////////////////////////
349Http::Response Http::sendRequest(const Http::Request& request, Time timeout)
350{
351 // First make sure that the request is valid -- add missing mandatory fields
352 Request toSend(request);
353 if (!toSend.hasField("From"))
354 {
355 toSend.setField("From", "user@sfml-dev.org");
356 }
357 if (!toSend.hasField("User-Agent"))
358 {
359 toSend.setField("User-Agent", "libsfml-network/2.x");
360 }
361 if (!toSend.hasField("Host"))
362 {
363 toSend.setField("Host", m_hostName);
364 }
365 if (!toSend.hasField("Content-Length"))
366 {
367 std::ostringstream out;
368 out << toSend.m_body.size();
369 toSend.setField("Content-Length", out.str());
370 }
371 if ((toSend.m_method == Request::Post) && !toSend.hasField("Content-Type"))
372 {
373 toSend.setField("Content-Type", "application/x-www-form-urlencoded");
374 }
375 if ((toSend.m_majorVersion * 10 + toSend.m_minorVersion >= 11) && !toSend.hasField("Connection"))
376 {
377 toSend.setField("Connection", "close");
378 }
379
380 // Prepare the response
381 Response received;
382
383 // Connect the socket to the host
384 if (m_connection.connect(m_host, m_port, timeout) == Socket::Done)
385 {
386 // Convert the request to string and send it through the connected socket
387 std::string requestStr = toSend.prepare();
388
389 if (!requestStr.empty())
390 {
391 // Send it through the socket
392 if (m_connection.send(requestStr.c_str(), requestStr.size()) == Socket::Done)
393 {
394 // Wait for the server's response
395 std::string receivedStr;
396 std::size_t size = 0;
397 char buffer[1024];
398 while (m_connection.receive(buffer, sizeof(buffer), size) == Socket::Done)
399 {
400 receivedStr.append(buffer, buffer + size);
401 }
402
403 // Build the Response object from the received data
404 received.parse(receivedStr);
405 }
406 }
407
408 // Close the connection
409 m_connection.disconnect();
410 }
411
412 return received;
413}
