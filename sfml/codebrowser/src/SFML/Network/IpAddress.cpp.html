104IpAddress IpAddress::getLocalAddress()
105{
106 // The method here is to connect a UDP socket to anyone (here to localhost),
107 // and get the local socket address with the getsockname function.
108 // UDP connection will not send anything to the network, so this function won't cause any overhead.
109
110 IpAddress localAddress;
111
112 // Create the socket
113 SocketHandle sock = socket(PF_INET, SOCK_DGRAM, 0);
114 if (sock == priv::SocketImpl::invalidSocket())
115 return localAddress;
116
117 // Connect the socket to localhost on any port
118 sockaddr_in address = priv::SocketImpl::createAddress(ntohl(INADDR_LOOPBACK), 9);
119 if (connect(sock, reinterpret_cast<sockaddr*>(&address), sizeof(address)) == -1)
120 {
121 priv::SocketImpl::close(sock);
122 return localAddress;
123 }
124
125 // Get the local address of the socket connection
126 priv::SocketImpl::AddrLength size = sizeof(address);
127 if (getsockname(sock, reinterpret_cast<sockaddr*>(&address), &size) == -1)
128 {
129 priv::SocketImpl::close(sock);
130 return localAddress;
131 }
132
133 // Close the socket
134 priv::SocketImpl::close(sock);
135
136 // Finally build the IP address
137 localAddress = IpAddress(ntohl(address.sin_addr.s_addr));
138
139 return localAddress;
140}

144IpAddress IpAddress::getPublicAddress(Time timeout)
145{
146 // The trick here is more complicated, because the only way
147 // to get our public IP address is to get it from a distant computer.
148 // Here we get the web page from http://www.sfml-dev.org/ip-provider.php
149 // and parse the result to extract our IP address
150 // (not very hard: the web page contains only our IP address).
151
152 Http server("www.sfml-dev.org");
153 Http::Request request("/ip-provider.php", Http::Request::Get);
154 Http::Response page = server.sendRequest(request, timeout);
155 if (page.getStatus() == Http::Response::Ok)
156 return IpAddress(page.getBody());
157
158 // Something failed: return an invalid address
159 return IpAddress();
160}

163////////////////////////////////////////////////////////////
164void IpAddress::resolve(const std::string& address)
165{
166 m_address = 0;
167 m_valid = false;
168
169 if (address == "255.255.255.255")
170 {
171 // The broadcast address needs to be handled explicitly,
172 // because it is also the value returned by inet_addr on error
173 m_address = INADDR_BROADCAST;
174 m_valid = true;
175 }
176 else if (address == "0.0.0.0")
177 {
178 m_address = INADDR_ANY;
179 m_valid = true;
180 }
181 else
182 {
183 // Try to convert the address as a byte representation ("xxx.xxx.xxx.xxx")
184 Uint32 ip = inet_addr(address.c_str());
185 if (ip != INADDR_NONE)
186 {
187 m_address = ip;
188 m_valid = true;
189 }
190 else
191 {
192 // Not a valid address, try to convert it as a host name
193 addrinfo hints;
194 std::memset(&hints, 0, sizeof(hints));
195 hints.ai_family = AF_INET;
196 addrinfo* result = NULL;
197 if (getaddrinfo(address.c_str(), NULL, &hints, &result) == 0)
198 {
199 if (result)
200 {
201 ip = reinterpret_cast<sockaddr_in*>(result->ai_addr)->sin_addr.s_addr;
202 freeaddrinfo(result);
203 m_address = ip;
204 m_valid = true;
205 }
206 }
207 }
208 }
209}

212////////////////////////////////////////////////////////////
213bool operator ==(const IpAddress& left, const IpAddress& right)
214{
215 return !(left < right) && !(right < left);
216}
217
218
219////////////////////////////////////////////////////////////
220bool operator !=(const IpAddress& left, const IpAddress& right)
221{
222 return !(left == right);
223}
224
225
226////////////////////////////////////////////////////////////
227bool operator <(const IpAddress& left, const IpAddress& right)
228{
229 return std::make_pair(left.m_valid, left.m_address) < std::make_pair(right.m_valid, right.m_address);
230}
