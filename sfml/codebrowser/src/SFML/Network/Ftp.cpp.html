99Ftp::DirectoryResponse::DirectoryResponse(const Ftp::Response& response) :
100Ftp::Response(response)
101{
102 if (isOk())
103 {
104 // Extract the directory from the server response
105 std::string::size_type begin = getMessage().find('"', 0);
106 std::string::size_type end = getMessage().find('"', begin + 1);
107 m_directory = getMessage().substr(begin + 1, end - begin - 1);
108 }
109}

120Ftp::ListingResponse::ListingResponse(const Ftp::Response& response, const std::string& data) :
121Ftp::Response(response)
122{
123 if (isOk())
124 {
125 // Fill the array of strings
126 std::string::size_type lastPos = 0;
127 for (std::string::size_type pos = data.find("\r\n"); pos != std::string::npos; pos = data.find("\r\n", lastPos))
128 {
129 m_listing.push_back(data.substr(lastPos, pos - lastPos));
130 lastPos = pos + 2;
131 }
132 }
133}

206////////////////////////////////////////////////////////////
207Ftp::ListingResponse Ftp::getDirectoryListing(const std::string& directory)
208{
209 // Open a data channel on default port (20) using ASCII transfer mode
210 std::ostringstream directoryData;
211 DataChannel data(*this);
212 Response response = data.open(Ascii);
213 if (response.isOk())
214 {
215 // Tell the server to send us the listing
216 response = sendCommand("NLST", directory);
217 if (response.isOk())
218 {
219 // Receive the listing
220 data.receive(directoryData);
221
222 // Get the response from the server
223 response = getResponse();
224 }
225 }
226
227 return ListingResponse(response, directoryData.str());
228}

277////////////////////////////////////////////////////////////
278Ftp::Response Ftp::download(const std::string& remoteFile, const std::string& localPath, TransferMode mode)
279{
280 // Open a data channel using the given transfer mode
281 DataChannel data(*this);
282 Response response = data.open(mode);
283 if (response.isOk())
284 {
285 // Tell the server to start the transfer
286 response = sendCommand("RETR", remoteFile);
287 if (response.isOk())
288 {
289 // Extract the filename from the file path
290 std::string filename = remoteFile;
291 std::string::size_type pos = filename.find_last_of("/\\");
292 if (pos != std::string::npos)
293 filename = filename.substr(pos + 1);
294
295 // Make sure the destination path ends with a slash
296 std::string path = localPath;
297 if (!path.empty() && (path[path.size() - 1] != '\\') && (path[path.size() - 1] != '/'))
298 path += "/";
299
300 // Create the file and truncate it if necessary
301 std::ofstream file((path + filename).c_str(), std::ios_base::binary | std::ios_base::trunc);
302 if (!file)
303 return Response(Response::InvalidFile);
304
305 // Receive the file data
306 data.receive(file);
307
308 // Close the file
309 file.close();
310
311 // Get the response from the server
312 response = getResponse();
313
314 // If the download was unsuccessful, delete the partial file
315 if (!response.isOk())
316 std::remove((path + filename).c_str());
317 }
318 }
319
320 return response;
321}

324////////////////////////////////////////////////////////////
325Ftp::Response Ftp::upload(const std::string& localFile, const std::string& remotePath, TransferMode mode)
326{
327 // Get the contents of the file to send
328 std::ifstream file(localFile.c_str(), std::ios_base::binary);
329 if (!file)
330 return Response(Response::InvalidFile);
331
332 // Extract the filename from the file path
333 std::string filename = localFile;
334 std::string::size_type pos = filename.find_last_of("/\\");
335 if (pos != std::string::npos)
336 filename = filename.substr(pos + 1);
337
338 // Make sure the destination path ends with a slash
339 std::string path = remotePath;
340 if (!path.empty() && (path[path.size() - 1] != '\\') && (path[path.size() - 1] != '/'))
341 path += "/";
342
343 // Open a data channel using the given transfer mode
344 DataChannel data(*this);
345 Response response = data.open(mode);
346 if (response.isOk())
347 {
348 // Tell the server to start the transfer
349 response = sendCommand("STOR", path + filename);
350 if (response.isOk())
351 {
352 // Send the file data
353 data.send(file);
354
355 // Get the response from the server
356 response = getResponse();
357 }
358 }
359
360 return response;
361}

364////////////////////////////////////////////////////////////
365Ftp::Response Ftp::sendCommand(const std::string& command, const std::string& parameter)
366{
367 // Build the command string
368 std::string commandStr;
369 if (parameter != "")
370 commandStr = command + " " + parameter + "\r\n";
371 else
372 commandStr = command + "\r\n";
373
374 // Send it to the server
375 if (m_commandSocket.send(commandStr.c_str(), commandStr.length()) != Socket::Done)
376 return Response(Response::ConnectionClosed);
377
378 // Get the response
379 return getResponse();
380}

383////////////////////////////////////////////////////////////
384Ftp::Response Ftp::getResponse()
385{
386 // We'll use a variable to keep track of the last valid code.
387 // It is useful in case of multi-lines responses, because the end of such a response
388 // will start by the same code
389 unsigned int lastCode = 0;
390 bool isInsideMultiline = false;
391 std::string message;
392
393 for (;;)
394 {
395 // Receive the response from the server
396 char buffer[1024];
397 std::size_t length;
398
399 if (m_receiveBuffer.empty())
400 {
401 if (m_commandSocket.receive(buffer, sizeof(buffer), length) != Socket::Done)
402 return Response(Response::ConnectionClosed);
403 }
404 else
405 {
406 std::copy(m_receiveBuffer.begin(), m_receiveBuffer.end(), buffer);
407 length = m_receiveBuffer.size();
408 m_receiveBuffer.clear();
409 }
410
411 // There can be several lines inside the received buffer, extract them all
412 std::istringstream in(std::string(buffer, length), std::ios_base::binary);
413 while (in)
414 {
415 // Try to extract the code
416 unsigned int code;
417 if (in >> code)
418 {
419 // Extract the separator
420 char separator;
421 in.get(separator);
422
423 // The '-' character means a multiline response
424 if ((separator == '-') && !isInsideMultiline)
425 {
426 // Set the multiline flag
427 isInsideMultiline = true;
428
429 // Keep track of the code
430 if (lastCode == 0)
431 lastCode = code;
432
433 // Extract the line
434 std::getline(in, message);
435
436 // Remove the ending '\r' (all lines are terminated by "\r\n")
437 message.erase(message.length() - 1);
438 message = separator + message + "\n";
439 }
440 else
441 {
442 // We must make sure that the code is the same, otherwise it means
443 // we haven't reached the end of the multiline response
444 if ((separator != '-') && ((code == lastCode) || (lastCode == 0)))
445 {
446 // Extract the line
447 std::string line;
448 std::getline(in, line);
449
450 // Remove the ending '\r' (all lines are terminated by "\r\n")
451 line.erase(line.length() - 1);
452
453 // Append it to the message
454 if (code == lastCode)
455 {
456 std::ostringstream out;
457 out << code << separator << line;
458 message += out.str();
459 }
460 else
461 {
462 message = separator + line;
463 }
464
465 // Save the remaining data for the next time getResponse() is called
466 m_receiveBuffer.assign(buffer + in.tellg(), length - in.tellg());
467
468 // Return the response code and message
469 return Response(static_cast<Response::Status>(code), message);
470 }
471 else
472 {
473 // The line we just read was actually not a response,
474 // only a new part of the current multiline response
475
476 // Extract the line
477 std::string line;
478 std::getline(in, line);
479
480 if (!line.empty())
481 {
482 // Remove the ending '\r' (all lines are terminated by "\r\n")
483 line.erase(line.length() - 1);
484
485 // Append it to the current message
486 std::ostringstream out;
487 out << code << separator << line << "\n";
488 message += out.str();
489 }
490 }
491 }
492 }
493 else if (lastCode != 0)
494 {
495 // It seems we are in the middle of a multiline response
496
497 // Clear the error bits of the stream
498 in.clear();
499
500 // Extract the line
501 std::string line;
502 std::getline(in, line);
503
504 if (!line.empty())
505 {
506 // Remove the ending '\r' (all lines are terminated by "\r\n")
507 line.erase(line.length() - 1);
508
509 // Append it to the current message
510 message += line + "\n";
511 }
512 }
513 else
514 {
515 // Error: cannot extract the code, and we are not in a multiline response
516 return Response(Response::InvalidResponse);
517 }
518 }
519 }
520
521 // We never reach there
522}

533////////////////////////////////////////////////////////////
534Ftp::Response Ftp::DataChannel::open(Ftp::TransferMode mode)
535{
536 // Open a data connection in active mode (we connect to the server)
537 Ftp::Response response = m_ftp.sendCommand("PASV");
538 if (response.isOk())
539 {
540 // Extract the connection address and port from the response
541 std::string::size_type begin = response.getMessage().find_first_of("0123456789");
542 if (begin != std::string::npos)
543 {
544 Uint8 data[6] = {0, 0, 0, 0, 0, 0};
545 std::string str = response.getMessage().substr(begin);
546 std::size_t index = 0;
547 for (int i = 0; i < 6; ++i)
548 {
549 // Extract the current number
550 while (isdigit(str[index]))
551 {
552 data[i] = data[i] * 10 + (str[index] - '0');
553 index++;
554 }
555
556 // Skip separator
557 index++;
558 }
559
560 // Reconstruct connection port and address
561 unsigned short port = data[4] * 256 + data[5];
562 IpAddress address(static_cast<Uint8>(data[0]),
563 static_cast<Uint8>(data[1]),
564 static_cast<Uint8>(data[2]),
565 static_cast<Uint8>(data[3]));
566
567 // Connect the data channel to the server
568 if (m_dataSocket.connect(address, port) == Socket::Done)
569 {
570 // Translate the transfer mode to the corresponding FTP parameter
571 std::string modeStr;
572 switch (mode)
573 {
574 case Ftp::Binary: modeStr = "I"; break;
575 case Ftp::Ascii: modeStr = "A"; break;
576 case Ftp::Ebcdic: modeStr = "E"; break;
577 }
578
579 // Set the transfer mode
580 response = m_ftp.sendCommand("TYPE", modeStr);
581 }
582 else
583 {
584 // Failed to connect to the server
585 response = Ftp::Response(Ftp::Response::ConnectionFailed);
586 }
587 }
588 }
589
590 return response;
591}

594////////////////////////////////////////////////////////////
595void Ftp::DataChannel::receive(std::ostream& stream)
596{
597 // Receive data
598 char buffer[1024];
599 std::size_t received;
600 while (m_dataSocket.receive(buffer, sizeof(buffer), received) == Socket::Done)
601 {
602 stream.write(buffer, static_cast<std::streamsize>(received));
603
604 if (!stream.good())
605 {
606 err() << "FTP Error: Writing to the file has failed" << std::endl;
607 break;
608 }
609 }
610
611 // Close the data socket
612 m_dataSocket.disconnect();
613}

616////////////////////////////////////////////////////////////
617void Ftp::DataChannel::send(std::istream& stream)
618{
619 // Send data
620 char buffer[1024];
621 std::size_t count;
622
623 for (;;)
624 {
625 // read some data from the stream
626 stream.read(buffer, sizeof(buffer));
627
628 if (!stream.good() && !stream.eof())
629 {
630 err() << "FTP Error: Reading from the file has failed" << std::endl;
631 break;
632 }
633
634 count = stream.gcount();
635
636 if (count > 0)
637 {
638 // we could read more data from the stream: send them
639 if (m_dataSocket.send(buffer, count) != Socket::Done)
640 break;
641 }
642 else
643 {
644 // no more data: exit the loop
645 break;
646 }
647 }
648
649 // Close the data socket
650 m_dataSocket.disconnect();
651}
