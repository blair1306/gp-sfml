62unsigned short TcpSocket::getLocalPort() const
63{
64 if (getHandle() != priv::SocketImpl::invalidSocket())
65 {
66 // Retrieve informations about the local end of the socket
67 sockaddr_in address;
68 priv::SocketImpl::AddrLength size = sizeof(address);
69 if (getsockname(getHandle(), reinterpret_cast<sockaddr*>(&address), &size) != -1)
70 {
71 return ntohs(address.sin_port);
72 }
73 }
74
75 // We failed to retrieve the port
76 return 0;
77}

81IpAddress TcpSocket::getRemoteAddress() const
82{
83 if (getHandle() != priv::SocketImpl::invalidSocket())
84 {
85 // Retrieve informations about the remote end of the socket
86 sockaddr_in address;
87 priv::SocketImpl::AddrLength size = sizeof(address);
88 if (getpeername(getHandle(), reinterpret_cast<sockaddr*>(&address), &size) != -1)
89 {
90 return IpAddress(ntohl(address.sin_addr.s_addr));
91 }
92 }
93
94 // We failed to retrieve the address
95 return IpAddress::None;

119Socket::Status TcpSocket::connect(const IpAddress& remoteAddress, unsigned short remotePort, Time timeout)
120{
121 // Create the internal socket if it doesn't exist
122 create();
123
124 // Create the remote address
125 sockaddr_in address = priv::SocketImpl::createAddress(remoteAddress.toInteger(), remotePort);
126
127 if (timeout <= Time::Zero)
128 {
129 // ----- We're not using a timeout: just try to connect -----
130
131 // Connect the socket
132 if (::connect(getHandle(), reinterpret_cast<sockaddr*>(&address), sizeof(address)) == -1)
133 return priv::SocketImpl::getErrorStatus();
134
135 // Connection succeeded
136 return Done;
137 }
138 else
139 {
140 // ----- We're using a timeout: we'll need a few tricks to make it work -----
141
142 // Save the previous blocking state
143 bool blocking = isBlocking();
144
145 // Switch to non-blocking to enable our connection timeout
146 if (blocking)
147 setBlocking(false);
148
149 // Try to connect to the remote address
150 if (::connect(getHandle(), reinterpret_cast<sockaddr*>(&address), sizeof(address)) >= 0)
151 {
152 // We got instantly connected! (it may no happen a lot...)
153 setBlocking(blocking);
154 return Done;
155 }
156
157 // Get the error status
158 Status status = priv::SocketImpl::getErrorStatus();
159
160 // If we were in non-blocking mode, return immediately
161 if (!blocking)
162 return status;
163
164 // Otherwise, wait until something happens to our socket (success, timeout or error)
165 if (status == Socket::NotReady)
166 {
167 // Setup the selector
168 fd_set selector;
169 FD_ZERO(&selector);
170 FD_SET(getHandle(), &selector);
171
172 // Setup the timeout
173 timeval time;
174 time.tv_sec = static_cast<long>(timeout.asMicroseconds() / 1000000);
175 time.tv_usec = static_cast<long>(timeout.asMicroseconds() % 1000000);
176
177 // Wait for something to write on our socket (which means that the connection request has returned)
178 if (select(static_cast<int>(getHandle() + 1), NULL, &selector, NULL, &time) > 0)
179 {
180 // At this point the connection may have been either accepted or refused.
181 // To know whether it's a success or a failure, we must check the address of the connected peer
182 if (getRemoteAddress() != IpAddress::None)
183 {
184 // Connection accepted
185 status = Done;
186 }
187 else
188 {
189 // Connection refused
190 status = priv::SocketImpl::getErrorStatus();
191 }
192 }
193 else
194 {
195 // Failed to connect before timeout is over
196 status = priv::SocketImpl::getErrorStatus();
197 }
198 }
199
200 // Switch back to blocking mode
201 setBlocking(true);
202
203 return status;
204 }
205}

231////////////////////////////////////////////////////////////
232Socket::Status TcpSocket::send(const void* data, std::size_t size, std::size_t& sent)
233{
234 // Check the parameters
235 if (!data || (size == 0))
236 {
237 err() << "Cannot send data over the network (no data to send)" << std::endl;
238 return Error;
239 }
240
241 // Loop until every byte has been sent
242 int result = 0;
243 for (sent = 0; sent < size; sent += result)
244 {
245 // Send a chunk of data
246 result = ::send(getHandle(), static_cast<const char*>(data) + sent, size - sent, flags);
247
248 // Check for errors
249 if (result < 0)
250 {
251 Status status = priv::SocketImpl::getErrorStatus();
252
253 if ((status == NotReady) && sent)
254 return Partial;
255
256 return status;
257 }
258 }
259
260 return Done;
261}

264////////////////////////////////////////////////////////////
265Socket::Status TcpSocket::receive(void* data, std::size_t size, std::size_t& received)
266{
267 // First clear the variables to fill
268 received = 0;
269
270 // Check the destination buffer
271 if (!data)
272 {
273 err() << "Cannot receive data from the network (the destination buffer is invalid)" << std::endl;
274 return Error;
275 }
276
277 // Receive a chunk of bytes
278 int sizeReceived = recv(getHandle(), static_cast<char*>(data), static_cast<int>(size), flags);
279
280 // Check the number of bytes received
281 if (sizeReceived > 0)
282 {
283 received = static_cast<std::size_t>(sizeReceived);
284 return Done;
285 }
286 else if (sizeReceived == 0)
287 {
288 return Socket::Disconnected;
289 }
290 else
291 {
292 return priv::SocketImpl::getErrorStatus();
293 }
294}

297////////////////////////////////////////////////////////////
298Socket::Status TcpSocket::send(Packet& packet)
299{
300 // TCP is a stream protocol, it doesn't preserve messages boundaries.
301 // This means that we have to send the packet size first, so that the
302 // receiver knows the actual end of the packet in the data stream.
303
304 // We allocate an extra memory block so that the size can be sent
305 // together with the data in a single call. This may seem inefficient,
306 // but it is actually required to avoid partial send, which could cause
307 // data corruption on the receiving end.
308
309 // Get the data to send from the packet
310 std::size_t size = 0;
311 const void* data = packet.onSend(size);
312
313 // First convert the packet size to network byte order
314 Uint32 packetSize = htonl(static_cast<Uint32>(size));
315
316 // Allocate memory for the data block to send
317 std::vector<char> blockToSend(sizeof(packetSize) + size);
318
319 // Copy the packet size and data into the block to send
320 std::memcpy(&blockToSend[0], &packetSize, sizeof(packetSize));
321 if (size > 0)
322 std::memcpy(&blockToSend[0] + sizeof(packetSize), data, size);
323
324 // Send the data block
325 std::size_t sent;
326 Status status = send(&blockToSend[0] + packet.m_sendPos, blockToSend.size() - packet.m_sendPos, sent);
327
328 // In the case of a partial send, record the location to resume from
329 if (status == Partial)
330 {
331 packet.m_sendPos += sent;
332 }
333 else if (status == Done)
334 {
335 packet.m_sendPos = 0;
336 }
337
338 return status;
339}

342////////////////////////////////////////////////////////////
343Socket::Status TcpSocket::receive(Packet& packet)
344{
345 // First clear the variables to fill
346 packet.clear();
347
348 // We start by getting the size of the incoming packet
349 Uint32 packetSize = 0;
350 std::size_t received = 0;
351 if (m_pendingPacket.SizeReceived < sizeof(m_pendingPacket.Size))
352 {
353 // Loop until we've received the entire size of the packet
354 // (even a 4 byte variable may be received in more than one call)
355 while (m_pendingPacket.SizeReceived < sizeof(m_pendingPacket.Size))
356 {
357 char* data = reinterpret_cast<char*>(&m_pendingPacket.Size) + m_pendingPacket.SizeReceived;
358 Status status = receive(data, sizeof(m_pendingPacket.Size) - m_pendingPacket.SizeReceived, received);
359 m_pendingPacket.SizeReceived += received;
360
361 if (status != Done)
362 return status;
363 }
364
365 // The packet size has been fully received
366 packetSize = ntohl(m_pendingPacket.Size);
367 }
368 else
369 {
370 // The packet size has already been received in a previous call
371 packetSize = ntohl(m_pendingPacket.Size);
372 }
373
374 // Loop until we receive all the packet data
375 char buffer[1024];
376 while (m_pendingPacket.Data.size() < packetSize)
377 {
378 // Receive a chunk of data
379 std::size_t sizeToGet = std::min(static_cast<std::size_t>(packetSize - m_pendingPacket.Data.size()), sizeof(buffer));
380 Status status = receive(buffer, sizeToGet, received);
381 if (status != Done)
382 return status;
383
384 // Append it into the packet
385 if (received > 0)
386 {
387 m_pendingPacket.Data.resize(m_pendingPacket.Data.size() + received);
388 char* begin = &m_pendingPacket.Data[0] + m_pendingPacket.Data.size() - received;
389 std::memcpy(begin, buffer, received);
390 }
391 }
392
393 // We have received all the packet data: we can copy it to the user packet
394 if (!m_pendingPacket.Data.empty())
395 packet.onReceive(&m_pendingPacket.Data[0], m_pendingPacket.Data.size());
396
397 // Clear the pending packet data
398 m_pendingPacket = PendingPacket();
399
400 return Done;
401}
