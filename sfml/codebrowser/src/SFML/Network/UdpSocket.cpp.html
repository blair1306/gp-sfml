48unsigned short UdpSocket::getLocalPort() const
49{
50 if (getHandle() != priv::SocketImpl::invalidSocket())
51 {
52 // Retrieve informations about the local end of the socket
53 sockaddr_in address;
54 priv::SocketImpl::AddrLength size = sizeof(address);
55 if (getsockname(getHandle(), reinterpret_cast<sockaddr*>(&address), &size) != -1)
56 {
57 return ntohs(address.sin_port);
58 }
59 }
60
61 // We failed to retrieve the port
62 return 0;
63}

66////////////////////////////////////////////////////////////
67Socket::Status UdpSocket::bind(unsigned short port, const IpAddress& address)
68{
69 // Create the internal socket if it doesn't exist
70 create();
71
72 // Check if the address is valid
73 if ((address == IpAddress::None) || (address == IpAddress::Broadcast))
74 return Error;
75
76 // Bind the socket
77 sockaddr_in addr = priv::SocketImpl::createAddress(address.toInteger(), port);
78 if (::bind(getHandle(), reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) == -1)
79 {
80 err() << "Failed to bind socket to port " << port << std::endl;
81 return Error;
82 }
83
84 return Done;
85}

97Socket::Status UdpSocket::send(const void* data, std::size_t size, const IpAddress& remoteAddress, unsigned short remotePort)
98{
99 // Create the internal socket if it doesn't exist
100 create();
101
102 // Make sure that all the data will fit in one datagram
103 if (size > MaxDatagramSize)
104 {
105 err() << "Cannot send data over the network "
106 << "(the number of bytes to send is greater than sf::UdpSocket::MaxDatagramSize)" << std::endl;
107 return Error;
108 }
109
110 // Build the target address
111 sockaddr_in address = priv::SocketImpl::createAddress(remoteAddress.toInteger(), remotePort);
112
113 // Send the data (unlike TCP, all the data is always sent in one call)
114 int sent = sendto(getHandle(), static_cast<const char*>(data), static_cast<int>(size), 0, reinterpret_cast<sockaddr*>(&address), sizeof(address));
115
116 // Check for errors
117 if (sent < 0)
118 return priv::SocketImpl::getErrorStatus();
119
120 return Done;
121}

124////////////////////////////////////////////////////////////
125Socket::Status UdpSocket::receive(void* data, std::size_t size, std::size_t& received, IpAddress& remoteAddress, unsigned short& remotePort)
126{
127 // First clear the variables to fill
128 received = 0;
129 remoteAddress = IpAddress();
130 remotePort = 0;
131
132 // Check the destination buffer
133 if (!data)
134 {
135 err() << "Cannot receive data from the network (the destination buffer is invalid)" << std::endl;
136 return Error;
137 }
138
139 // Data that will be filled with the other computer's address
140 sockaddr_in address = priv::SocketImpl::createAddress(INADDR_ANY, 0);
141
142 // Receive a chunk of bytes
143 priv::SocketImpl::AddrLength addressSize = sizeof(address);
144 int sizeReceived = recvfrom(getHandle(), static_cast<char*>(data), static_cast<int>(size), 0, reinterpret_cast<sockaddr*>(&address), &addressSize);
145
146 // Check for errors
147 if (sizeReceived < 0)
148 return priv::SocketImpl::getErrorStatus();
149
150 // Fill the sender informations
151 received = static_cast<std::size_t>(sizeReceived);
152 remoteAddress = IpAddress(ntohl(address.sin_addr.s_addr));
153 remotePort = ntohs(address.sin_port);
154
155 return Done;
156}

160Socket::Status UdpSocket::send(Packet& packet, const IpAddress& remoteAddress, unsigned short remotePort)
161{
162 // UDP is a datagram-oriented protocol (as opposed to TCP which is a stream protocol).
163 // Sending one datagram is almost safe: it may be lost but if it's received, then its data
164 // is guaranteed to be ok. However, splitting a packet into multiple datagrams would be highly
165 // unreliable, since datagrams may be reordered, dropped or mixed between different sources.
166 // That's why SFML imposes a limit on packet size so that they can be sent in a single datagram.
167 // This also removes the overhead associated to packets -- there's no size to send in addition
168 // to the packet's data.
169
170 // Get the data to send from the packet
171 std::size_t size = 0;
172 const void* data = packet.onSend(size);
173
174 // Send it
175 return send(data, size, remoteAddress, remotePort);
176}

179////////////////////////////////////////////////////////////
180Socket::Status UdpSocket::receive(Packet& packet, IpAddress& remoteAddress, unsigned short& remotePort)
181{
182 // See the detailed comment in send(Packet) above.
183
184 // Receive the datagram
185 std::size_t received = 0;
186 Status status = receive(&m_buffer[0], m_buffer.size(), received, remoteAddress, remotePort);
187
188 // If we received valid data, we can copy it to the user packet
189 packet.clear();
190 if ((status == Done) && (received > 0))
191 packet.onReceive(&m_buffer[0], received);
192
193 return status;
194}
