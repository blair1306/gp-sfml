45unsigned short TcpListener::getLocalPort() const
46{
47 if (getHandle() != priv::SocketImpl::invalidSocket())
48 {
49 // Retrieve informations about the local end of the socket
50 sockaddr_in address;
51 priv::SocketImpl::AddrLength size = sizeof(address);
52 if (getsockname(getHandle(), reinterpret_cast<sockaddr*>(&address), &size) != -1)
53 {
54 return ntohs(address.sin_port);
55 }
56 }
57
58 // We failed to retrieve the port
59 return 0;
60}

63////////////////////////////////////////////////////////////
64Socket::Status TcpListener::listen(unsigned short port, const IpAddress& address)
65{
66 // Create the internal socket if it doesn't exist
67 create();
68
69 // Check if the address is valid
70 if ((address == IpAddress::None) || (address == IpAddress::Broadcast))
71 return Error;
72
73 // Bind the socket to the specified port
74 sockaddr_in addr = priv::SocketImpl::createAddress(address.toInteger(), port);
75 if (bind(getHandle(), reinterpret_cast<sockaddr*>(&addr), sizeof(addr)) == -1)
76 {
77 // Not likely to happen, but...
78 err() << "Failed to bind listener socket to port " << port << std::endl;
79 return Error;
80 }
81
82 // Listen to the bound port
83 if (::listen(getHandle(), 0) == -1)
84 {
85 // Oops, socket is deaf
86 err() << "Failed to listen to port " << port << std::endl;
87 return Error;
88 }
89
90 return Done;
91}

103Socket::Status TcpListener::accept(TcpSocket& socket)
104{
105 // Make sure that we're listening
106 if (getHandle() == priv::SocketImpl::invalidSocket())
107 {
108 err() << "Failed to accept a new connection, the socket is not listening" << std::endl;
109 return Error;
110 }
111
112 // Accept a new connection
113 sockaddr_in address;
114 priv::SocketImpl::AddrLength length = sizeof(address);
115 SocketHandle remote = ::accept(getHandle(), reinterpret_cast<sockaddr*>(&address), &length);
116
117 // Check for errors
118 if (remote == priv::SocketImpl::invalidSocket())
119 return priv::SocketImpl::getErrorStatus();
120
121 // Initialize the new connected socket
122 socket.close();
123 socket.create(remote);
124
125 return Done;
126}
