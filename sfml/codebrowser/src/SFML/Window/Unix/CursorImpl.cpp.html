60bool CursorImpl::loadFromPixels(const Uint8* pixels, Vector2u size, Vector2u hotspot)
61{
62 release();
63
64 // Convert the image into a bitmap (monochrome!).
65 std::size_t bytes = (size.x + 7) / 8 * size.y;
66 std::vector<Uint8> mask(bytes, 0); // Defines which pixel is transparent.
67 std::vector<Uint8> data(bytes, 1); // Defines which pixel is white/black.
68
69 for (std::size_t j = 0; j < size.y; ++j)
70 {
71 for (std::size_t i = 0; i < size.x; ++i)
72 {
73 std::size_t pixelIndex = i + j * size.x;
74 std::size_t byteIndex = pixelIndex / 8;
75 std::size_t bitIndex = i % 8;
76
77 // Turn on pixel that are not transparent
78 Uint8 opacity = pixels[pixelIndex * 4 + 3] > 0 ? 1 : 0;
79 mask[byteIndex] |= opacity << bitIndex;
80
81 // Choose between black/background & white/foreground color for each pixel,
82 // based on the pixel color intensity: on average, if a channel is "active"
83 // at 25%, the bit is white.
84 int intensity = pixels[pixelIndex * 4 + 0] + pixels[pixelIndex * 4 + 1] + pixels[pixelIndex * 4 + 2];
85 Uint8 bit = intensity > 64 ? 1 : 0;
86 data[byteIndex] |= bit << bitIndex;
87 }
88 }
89
90 Pixmap maskPixmap = XCreateBitmapFromData(m_display, XDefaultRootWindow(m_display),
91 (char*)&mask[0], size.x, size.y);
92 Pixmap dataPixmap = XCreateBitmapFromData(m_display, XDefaultRootWindow(m_display),
93 (char*)&data[0], size.x, size.y);
94
95 // Define the foreground color as white and the background as black.
96 XColor fg, bg;
97 fg.red = fg.blue = fg.green = -1;
98 bg.red = bg.blue = bg.green = 0;
99
100 // Create the monochrome cursor.
101 m_cursor = XCreatePixmapCursor(m_display,
102 dataPixmap, maskPixmap,
103 &fg, &bg,
104 hotspot.x, hotspot.y);
105
106 // Free the resources
107 XFreePixmap(m_display, dataPixmap);
108 XFreePixmap(m_display, maskPixmap);
109
110 // We assume everything went fine...
111 return true;
112}

115////////////////////////////////////////////////////////////
116bool CursorImpl::loadFromSystem(Cursor::Type type)
117{
118 release();
119
120 unsigned int shape;
121 switch (type)
122 {
123 default: return false;
124
125 case Cursor::Arrow: shape = XC_arrow; break;
126 case Cursor::Wait: shape = XC_watch; break;
127 case Cursor::Text: shape = XC_xterm; break;
128 case Cursor::Hand: shape = XC_hand1; break;
129 case Cursor::SizeHorizontal: shape = XC_sb_h_double_arrow; break;
130 case Cursor::SizeVertical: shape = XC_sb_v_double_arrow; break;
131 case Cursor::SizeAll: shape = XC_fleur; break;
132 case Cursor::Cross: shape = XC_crosshair; break;
133 case Cursor::Help: shape = XC_question_arrow; break;
134 case Cursor::NotAllowed: shape = XC_X_cursor; break;
135 }
136
137 m_cursor = XCreateFontCursor(m_display, shape);
138 return true;
139}

142////////////////////////////////////////////////////////////
143void CursorImpl::release()
144{
145 if (m_cursor != None)
146 {
147 XFreeCursor(m_display, m_cursor);
148 m_cursor = None;
149 }
150}
