25#ifndef SFML_JOYSTICKIMPLLINUX_HPP
26#define SFML_JOYSTICKIMPLLINUX_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Window/JoystickImpl.hpp>
32#include <linux/input.h>
33
34
35namespace sf
36{
37namespace priv
38{
39////////////////////////////////////////////////////////////
40/// \brief Linux implementation of joysticks
41///
42////////////////////////////////////////////////////////////
43class JoystickImpl
44{
45public:
46
47 ////////////////////////////////////////////////////////////
48 /// \brief Constructor
49 ///
50 ////////////////////////////////////////////////////////////
51 JoystickImpl();
52
53 ////////////////////////////////////////////////////////////
54 /// \brief Perform the global initialization of the joystick module
55 ///
56 ////////////////////////////////////////////////////////////
57 static void initialize();
58
59 ////////////////////////////////////////////////////////////
60 /// \brief Perform the global cleanup of the joystick module
61 ///
62 ////////////////////////////////////////////////////////////
63 static void cleanup();
64
65 ////////////////////////////////////////////////////////////
66 /// \brief Check if a joystick is currently connected
67 ///
68 /// \param index Index of the joystick to check
69 ///
70 /// \return True if the joystick is connected, false otherwise
71 ///
72 ////////////////////////////////////////////////////////////
73 static bool isConnected(unsigned int index);
74
75 ////////////////////////////////////////////////////////////
76 /// \brief Open the joystick
77 ///
78 /// \param index Index assigned to the joystick
79 ///
80 /// \return True on success, false on failure
81 ///
82 ////////////////////////////////////////////////////////////
83 bool open(unsigned int index);
84
85 ////////////////////////////////////////////////////////////
86 /// \brief Close the joystick
87 ///
88 ////////////////////////////////////////////////////////////
89 void close();
90
91 ////////////////////////////////////////////////////////////
92 /// \brief Get the joystick capabilities
93 ///
94 /// \return Joystick capabilities
95 ///
96 ////////////////////////////////////////////////////////////
97 JoystickCaps getCapabilities() const;
98
99 ////////////////////////////////////////////////////////////
100 /// \brief Get the joystick identification
101 ///
102 /// \return Joystick identification
103 ///
104 ////////////////////////////////////////////////////////////
105 Joystick::Identification getIdentification() const;
106
107 ////////////////////////////////////////////////////////////
108 /// \brief Update the joystick and get its new state
109 ///
110 /// \return Joystick state
111 ///
112 ////////////////////////////////////////////////////////////
113 JoystickState update();
114
115private:
116
117 ////////////////////////////////////////////////////////////
118 // Member data
119 ////////////////////////////////////////////////////////////
120 int m_file; ///< File descriptor of the joystick
121 char m_mapping[ABS_MAX + 1]; ///< Axes mapping (index to axis id)
122 JoystickState m_state; ///< Current state of the joystick
123 sf::Joystick::Identification m_identification; ///< Identification of the joystick
124};
125
126} // namespace priv
127
128} // namespace sf
129
130
131#endif // SFML_JOYSTICKIMPLLINUX_HPP
