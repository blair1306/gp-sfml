25#ifndef SFML_SENSORIMPLUNIX_HPP
26#define SFML_SENSORIMPLUNIX_HPP
27
28
29namespace sf
30{
31namespace priv
32{
33////////////////////////////////////////////////////////////
34/// \brief Unix implementation of sensors
35///
36////////////////////////////////////////////////////////////
37class SensorImpl
38{
39public:
40
41 ////////////////////////////////////////////////////////////
42 /// \brief Perform the global initialization of the sensor module
43 ///
44 ////////////////////////////////////////////////////////////
45 static void initialize();
46
47 ////////////////////////////////////////////////////////////
48 /// \brief Perform the global cleanup of the sensor module
49 ///
50 ////////////////////////////////////////////////////////////
51 static void cleanup();
52
53 ////////////////////////////////////////////////////////////
54 /// \brief Check if a sensor is available
55 ///
56 /// \param sensor Sensor to check
57 ///
58 /// \return True if the sensor is available, false otherwise
59 ///
60 ////////////////////////////////////////////////////////////
61 static bool isAvailable(Sensor::Type sensor);
62
63 ////////////////////////////////////////////////////////////
64 /// \brief Open the sensor
65 ///
66 /// \param sensor Type of the sensor
67 ///
68 /// \return True on success, false on failure
69 ///
70 ////////////////////////////////////////////////////////////
71 bool open(Sensor::Type sensor);
72
73 ////////////////////////////////////////////////////////////
74 /// \brief Close the sensor
75 ///
76 ////////////////////////////////////////////////////////////
77 void close();
78
79 ////////////////////////////////////////////////////////////
80 /// \brief Update the sensor and get its new value
81 ///
82 /// \return Sensor value
83 ///
84 ////////////////////////////////////////////////////////////
85 Vector3f update();
86
87 ////////////////////////////////////////////////////////////
88 /// \brief Enable or disable the sensor
89 ///
90 /// \param enabled True to enable, false to disable
91 ///
92 ////////////////////////////////////////////////////////////
93 void setEnabled(bool enabled);
94};
95
96} // namespace priv
97
98} // namespace sf
99
100
101#endif // SFML_SENSORIMPLUNIX_HPP
