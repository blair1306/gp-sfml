80 // Predicate we use to find key repeat events in processEvent
81 struct KeyRepeatFinder
82 {
83 KeyRepeatFinder(unsigned int keycode, Time time) : keycode(keycode), time(time) {}
84
85 // Predicate operator that checks event type, keycode and timestamp
86 bool operator()(const XEvent& event)
87 {
88 return ((event.type == KeyPress) && (event.xkey.keycode == keycode) && (event.xkey.time - time < 2));
89 }
90
91 unsigned int keycode;
92 Time time;
93 };

102 // Find the name of the current executable
103 std::string findExecutableName()
104 {
105 // We use /proc/self/cmdline to get the command line
106 // the user used to invoke this instance of the application
107 int file = ::open("/proc/self/cmdline", O_RDONLY | O_NONBLOCK);
108
109 if (file < 0)
110 return "sfml";
111
112 std::vector<char> buffer(256, 0);
113 std::size_t offset = 0;
114 ssize_t result = 0;
115
116 while ((result = read(file, &buffer[offset], 256)) > 0)
117 {
118 buffer.resize(buffer.size() + result, 0);
119 offset += result;
120 }
121
122 ::close(file);
123
124 if (offset)
125 {
126 buffer[offset] = 0;
127
128 // Remove the path to keep the executable name only
129 return basename(&buffer[0]);
130 }
131
132 // Default fallback name
133 return "sfml";
134 }

136 // Check if Extended Window Manager Hints are supported
137 bool ewmhSupported()
138 {
139 static bool checked = false;
140 static bool ewmhSupported = false;
141
142 if (checked)
143 return ewmhSupported;
144
145 checked = true;
146
147 Atom netSupportingWmCheck = sf::priv::getAtom("_NET_SUPPORTING_WM_CHECK", true);
148 Atom netSupported = sf::priv::getAtom("_NET_SUPPORTED", true);
149
150 if (!netSupportingWmCheck || !netSupported)
151 return false;
152
153 ::Display* display = sf::priv::OpenDisplay();
154
155 Atom actualType;
156 int actualFormat;
157 unsigned long numItems;
158 unsigned long numBytes;
159 unsigned char* data;
160
161 int result = XGetWindowProperty(display,
162 DefaultRootWindow(display),
163 netSupportingWmCheck,
164 0,
165 1,
166 False,
167 XA_WINDOW,
168 &actualType,
169 &actualFormat,
170 &numItems,
171 &numBytes,
172 &data);
173
174 if (result != Success || actualType != XA_WINDOW || numItems != 1)
175 {
176 if(result == Success)
177 XFree(data);
178
179 sf::priv::CloseDisplay(display);
180 return false;
181 }
182
183 ::Window rootWindow = *reinterpret_cast< ::Window* >(data);
184
185 XFree(data);
186
187 if (!rootWindow)
188 {
189 sf::priv::CloseDisplay(display);
190 return false;
191 }
192
193 result = XGetWindowProperty(display,
194 rootWindow,
195 netSupportingWmCheck,
196 0,
197 1,
198 False,
199 XA_WINDOW,
200 &actualType,
201 &actualFormat,
202 &numItems,
203 &numBytes,
204 &data);
205
206 if (result != Success || actualType != XA_WINDOW || numItems != 1)
207 {
208 if(result == Success)
209 XFree(data);
210
211 sf::priv::CloseDisplay(display);
212 return false;
213 }
214
215 ::Window childWindow = *reinterpret_cast< ::Window* >(data);
216
217 XFree(data);
218
219 if (!childWindow)
220 {
221 sf::priv::CloseDisplay(display);
222 return false;
223 }
224
225 // Conforming window managers should return the same window for both queries
226 if (rootWindow != childWindow)
227 {
228 sf::priv::CloseDisplay(display);
229 return false;
230 }
231
232 ewmhSupported = true;
233
234 // We try to get the name of the window manager
235 // for window manager specific workarounds
236 Atom netWmName = sf::priv::getAtom("_NET_WM_NAME", true);
237
238 if (!netWmName)
239 {
240 sf::priv::CloseDisplay(display);
241 return true;
242 }
243
244 Atom utf8StringType = sf::priv::getAtom("UTF8_STRING");
245
246 if (!utf8StringType)
247 utf8StringType = XA_STRING;
248
249 result = XGetWindowProperty(display,
250 rootWindow,
251 netWmName,
252 0,
253 0x7fffffff,
254 False,
255 utf8StringType,
256 &actualType,
257 &actualFormat,
258 &numItems,
259 &numBytes,
260 &data);
261
262 if (actualType && numItems)
263 {
264 // It seems the wm name string reply is not necessarily
265 // null-terminated. The work around is to get its actual
266 // length to build a proper string
267 const char* begin = reinterpret_cast<const char*>(data);
268 const char* end = begin + numItems;
269 windowManagerName = sf::String::fromUtf8(begin, end);
270 }
271
272 if(result == Success)
273 XFree(data);
274
275 sf::priv::CloseDisplay(display);
276
277 return true;
278 }

296 // Get the Frame Extents from EWMH WMs that support it.
297 bool getEWMHFrameExtents(::Display* disp, ::Window win,
298 long& xFrameExtent, long& yFrameExtent)
299 {
300 if (!ewmhSupported())
301 return false;
302
303 Atom frameExtents = sf::priv::getAtom("_NET_FRAME_EXTENTS", true);
304
305 if (frameExtents == None)
306 return false;
307
308 bool gotFrameExtents = false;
309 Atom actualType;
310 int actualFormat;
311 unsigned long numItems;
312 unsigned long numBytesLeft;
313 unsigned char* data = NULL;
314
315 int result = XGetWindowProperty(disp,
316 win,
317 frameExtents,
318 0,
319 4,
320 False,
321 XA_CARDINAL,
322 &actualType,
323 &actualFormat,
324 &numItems,
325 &numBytesLeft,
326 &data);
327
328 if ((result == Success) && (actualType == XA_CARDINAL) &&
329 (actualFormat == 32) && (numItems == 4) && (numBytesLeft == 0) &&
330 (data != NULL))
331 {
332 gotFrameExtents = true;
333
334 long* extents = (long*) data;
335
336 xFrameExtent = extents[0]; // Left.
337 yFrameExtent = extents[2]; // Top.
338 }
339
340 // Always free data.
341 if (data != NULL)
342 XFree(data);
343
344 return gotFrameExtents;
345 }

347 // Check if the current WM is in the list of good WMs that provide
348 // a correct absolute position for the window when queried.
349 bool isWMAbsolutePositionGood()
350 {
351 // This can only work with EWMH, to get the name.
352 if (!ewmhSupported())
353 return false;

355 for (size_t i = 0; i < (sizeof(wmAbsPosGood) / sizeof(wmAbsPosGood[0])); i++)

357 if (wmAbsPosGood[i] == windowManagerName)
358 return true;
359 }
360
361 return false;

530WindowImplX11::WindowImplX11(VideoMode mode, const String& title, unsigned long style, const ContextSettings& settings) :
531m_window (0),
532m_screen (0),
533m_inputMethod (NULL),
534m_inputContext (NULL),
535m_isExternal (false),
536m_oldVideoMode (0),
537m_hiddenCursor (0),
538m_lastCursor (None),
539m_keyRepeat (true),
540m_previousSize (-1, -1),
541m_useSizeHints (false),
542m_fullscreen ((style & Style::Fullscreen) != 0),
543m_cursorGrabbed (m_fullscreen),
544m_windowMapped (false),
545m_iconPixmap (0),
546m_iconMaskPixmap (0),
547m_lastInputTime (0)
548{
549 // Open a connection with the X server
550 m_display = OpenDisplay();
551
552 // Make sure to check for EWMH support before we do anything
553 ewmhSupported();
554
555 m_screen = DefaultScreen(m_display);
556
557 // Compute position and size
558 int left = m_fullscreen ? 0 : (DisplayWidth(m_display, m_screen) - mode.width) / 2;
559 int top = m_fullscreen ? 0 : (DisplayHeight(m_display, m_screen) - mode.height) / 2;
560 int width = mode.width;
561 int height = mode.height;
562
563 // Choose the visual according to the context settings
564 XVisualInfo visualInfo = ContextType::selectBestVisual(m_display, mode.bitsPerPixel, settings);
565
566 // Define the window attributes
567 XSetWindowAttributes attributes;
568 attributes.colormap = XCreateColormap(m_display, DefaultRootWindow(m_display), visualInfo.visual, AllocNone);
569 attributes.event_mask = eventMask;
570 attributes.override_redirect = (m_fullscreen && !ewmhSupported()) ? True : False;
571
572 m_window = XCreateWindow(m_display,
573 DefaultRootWindow(m_display),
574 left, top,
575 width, height,
576 0,
577 visualInfo.depth,
578 InputOutput,
579 visualInfo.visual,
580 CWEventMask | CWOverrideRedirect | CWColormap,
581 &attributes);
582
583 if (!m_window)
584 {
585 err() << "Failed to create window" << std::endl;
586 return;
587 }
588
589 // Set the WM protocols
590 setProtocols();
591
592 // Set the WM initial state to the normal state
593 XWMHints* hints = XAllocWMHints();
594 hints->flags = StateHint;
595 hints->initial_state = NormalState;
596 XSetWMHints(m_display, m_window, hints);
597 XFree(hints);
598
599 // If not in fullscreen, set the window's style (tell the window manager to
600 // change our window's decorations and functions according to the requested style)
601 if (!m_fullscreen)
602 {
603 Atom WMHintsAtom = getAtom("_MOTIF_WM_HINTS", false);
604 if (WMHintsAtom)
605 {
606 static const unsigned long MWM_HINTS_FUNCTIONS = 1 << 0;
607 static const unsigned long MWM_HINTS_DECORATIONS = 1 << 1;
608
609 //static const unsigned long MWM_DECOR_ALL = 1 << 0;
610 static const unsigned long MWM_DECOR_BORDER = 1 << 1;
611 static const unsigned long MWM_DECOR_RESIZEH = 1 << 2;
612 static const unsigned long MWM_DECOR_TITLE = 1 << 3;
613 static const unsigned long MWM_DECOR_MENU = 1 << 4;
614 static const unsigned long MWM_DECOR_MINIMIZE = 1 << 5;
615 static const unsigned long MWM_DECOR_MAXIMIZE = 1 << 6;
616
617 //static const unsigned long MWM_FUNC_ALL = 1 << 0;
618 static const unsigned long MWM_FUNC_RESIZE = 1 << 1;
619 static const unsigned long MWM_FUNC_MOVE = 1 << 2;
620 static const unsigned long MWM_FUNC_MINIMIZE = 1 << 3;
621 static const unsigned long MWM_FUNC_MAXIMIZE = 1 << 4;
622 static const unsigned long MWM_FUNC_CLOSE = 1 << 5;
623
624 struct WMHints
625 {
626 unsigned long flags;
627 unsigned long functions;
628 unsigned long decorations;
629 long inputMode;
630 unsigned long state;
631 };
632
633 WMHints hints;
634 std::memset(&hints, 0, sizeof(hints));
635 hints.flags = MWM_HINTS_FUNCTIONS | MWM_HINTS_DECORATIONS;
636 hints.decorations = 0;
637 hints.functions = 0;
638
639 if (style & Style::Titlebar)
640 {
641 hints.decorations |= MWM_DECOR_BORDER | MWM_DECOR_TITLE | MWM_DECOR_MINIMIZE | MWM_DECOR_MENU;
642 hints.functions |= MWM_FUNC_MOVE | MWM_FUNC_MINIMIZE;
643 }
644 if (style & Style::Resize)
645 {
646 hints.decorations |= MWM_DECOR_MAXIMIZE | MWM_DECOR_RESIZEH;
647 hints.functions |= MWM_FUNC_MAXIMIZE | MWM_FUNC_RESIZE;
648 }
649 if (style & Style::Close)
650 {
651 hints.decorations |= 0;
652 hints.functions |= MWM_FUNC_CLOSE;
653 }
654
655 XChangeProperty(m_display,
656 m_window,
657 WMHintsAtom,
658 WMHintsAtom,
659 32,
660 PropModeReplace,
661 reinterpret_cast<const unsigned char*>(&hints),
662 5);
663 }
664 }
665
666 // This is a hack to force some windows managers to disable resizing
667 if (!(style & Style::Resize))
668 {
669 m_useSizeHints = true;
670 XSizeHints* sizeHints = XAllocSizeHints();
671 sizeHints->flags = PMinSize | PMaxSize;
672 sizeHints->min_width = sizeHints->max_width = width;
673 sizeHints->min_height = sizeHints->max_height = height;
674 XSetWMNormalHints(m_display, m_window, sizeHints);
675 XFree(sizeHints);
676 }
677
678 // Set the window's WM class (this can be used by window managers)
679 XClassHint* hint = XAllocClassHint();
680
681 // The instance name should be something unique to this invocation
682 // of the application but is rarely if ever used these days.
683 // For simplicity, we retrieve it via the base executable name.
684 std::string executableName = findExecutableName();
685 std::vector<char> windowInstance(executableName.size() + 1, 0);
686 std::copy(executableName.begin(), executableName.end(), windowInstance.begin());
687 hint->res_name = &windowInstance[0];
688
689 // The class name identifies a class of windows that
690 // "are of the same type". We simply use the initial window name as
691 // the class name.
692 std::string ansiTitle = title.toAnsiString();
693 std::vector<char> windowClass(ansiTitle.size() + 1, 0);
694 std::copy(ansiTitle.begin(), ansiTitle.end(), windowClass.begin());
695 hint->res_class = &windowClass[0];
696
697 XSetClassHint(m_display, m_window, hint);
698
699 XFree(hint);
700
701 // Set the window's name
702 setTitle(title);
703
704 // Do some common initializations
705 initialize();
706
707 // Set fullscreen video mode and switch to fullscreen if necessary
708 if (m_fullscreen)
709 {
710 setPosition(Vector2i(0, 0));
711 setVideoMode(mode);
712 switchToFullscreen();
713 }
714}

718WindowImplX11::~WindowImplX11()
719{
720 // Cleanup graphical resources
721 cleanup();
722
723 // Destroy icon pixmap
724 if(m_iconPixmap)
725 XFreePixmap(m_display, m_iconPixmap);
726
727 // Destroy icon mask pixmap
728 if(m_iconMaskPixmap)
729 XFreePixmap(m_display, m_iconMaskPixmap);
730
731 // Destroy the cursor
732 if (m_hiddenCursor)
733 XFreeCursor(m_display, m_hiddenCursor);
734
735 // Destroy the input context
736 if (m_inputContext)
737 XDestroyIC(m_inputContext);
738
739 // Destroy the window
740 if (m_window && !m_isExternal)
741 {
742 XDestroyWindow(m_display, m_window);
743 XFlush(m_display);
744 }
745
746 // Close the input method
747 if (m_inputMethod)
748 XCloseIM(m_inputMethod);
749
750 // Close the connection with the X server
751 CloseDisplay(m_display);
752
753 // Remove this window from the global list of windows (required for focus request)
754 Lock lock(allWindowsMutex);
755 allWindows.erase(std::find(allWindows.begin(), allWindows.end(), this));
756}

785////////////////////////////////////////////////////////////
786Vector2i WindowImplX11::getPosition() const
787{
788 // Get absolute position of our window relative to root window. This
789 // takes into account all information that X11 has, including X11
790 // border widths and any decorations. It corresponds to where the
791 // window actually is, but not necessarily to where we told it to
792 // go using setPosition() and XMoveWindow(). To have the two match
793 // as expected, we may have to subtract decorations and borders.
794 ::Window child;
795 int xAbsRelToRoot, yAbsRelToRoot;
796
797 XTranslateCoordinates(m_display, m_window, DefaultRootWindow(m_display),
798 0, 0, &xAbsRelToRoot, &yAbsRelToRoot, &child);
799
800 // CASE 1: some rare WMs actually put the window exactly where we tell
801 // it to, even with decorations and such, which get shifted back.
802 // In these rare cases, we can use the absolute value directly.
803 if (isWMAbsolutePositionGood())
804 return Vector2i(xAbsRelToRoot, yAbsRelToRoot);
805
806 // CASE 2: most modern WMs support EWMH and can define _NET_FRAME_EXTENTS
807 // with the exact frame size to subtract, so if present, we prefer it and
808 // query it first. According to spec, this already includes any borders.
809 long xFrameExtent, yFrameExtent;
810
811 if (getEWMHFrameExtents(m_display, m_window, xFrameExtent, yFrameExtent))
812 {
813 // Get final X/Y coordinates: subtract EWMH frame extents from
814 // absolute window position.
815 return Vector2i((xAbsRelToRoot - xFrameExtent), (yAbsRelToRoot - yFrameExtent));
816 }
817
818 // CASE 3: EWMH frame extents were not available, use geometry.
819 // We climb back up to the window before the root and use its
820 // geometry information to extract X/Y position. This because
821 // re-parenting WMs may re-parent the window multiple times, so
822 // we'd have to climb up to the furthest ancestor and sum the
823 // relative differences and borders anyway; and doing that to
824 // subtract those values from the absolute coordinates of the
825 // window is equivalent to going up the tree and asking the
826 // furthest ancestor what it's relative distance to the root is.
827 // So we use that approach because it's simpler.
828 // This approach assumes that any window between the root and
829 // our window is part of decorations/borders in some way. This
830 // seems to hold true for most reasonable WM implementations.
831 ::Window ancestor = m_window;
832 ::Window root = DefaultRootWindow(m_display);
833
834 while (getParentWindow(m_display, ancestor) != root)
835 {
836 // Next window up (parent window).
837 ancestor = getParentWindow(m_display, ancestor);
838 }
839
840 // Get final X/Y coordinates: take the relative position to
841 // the root of the furthest ancestor window.
842 int xRelToRoot, yRelToRoot;
843 unsigned int width, height, borderWidth, depth;
844
845 XGetGeometry(m_display, ancestor, &root, &xRelToRoot, &yRelToRoot,
846 &width, &height, &borderWidth, &depth);
847
848 return Vector2i(xRelToRoot, yRelToRoot);
849}

935////////////////////////////////////////////////////////////
936void WindowImplX11::setIcon(unsigned int width, unsigned int height, const Uint8* pixels)
937{
938 // X11 wants BGRA pixels: swap red and blue channels
939 // Note: this memory will be freed by XDestroyImage
940 Uint8* iconPixels = static_cast<Uint8*>(std::malloc(width * height * 4));
941 for (std::size_t i = 0; i < width * height; ++i)
942 {
943 iconPixels[i * 4 + 0] = pixels[i * 4 + 2];
944 iconPixels[i * 4 + 1] = pixels[i * 4 + 1];
945 iconPixels[i * 4 + 2] = pixels[i * 4 + 0];
946 iconPixels[i * 4 + 3] = pixels[i * 4 + 3];
947 }
948
949 // Create the icon pixmap
950 Visual* defVisual = DefaultVisual(m_display, m_screen);
951 unsigned int defDepth = DefaultDepth(m_display, m_screen);
952 XImage* iconImage = XCreateImage(m_display, defVisual, defDepth, ZPixmap, 0, (char*)iconPixels, width, height, 32, 0);
953 if (!iconImage)
954 {
955 err() << "Failed to set the window's icon" << std::endl;
956 return;
957 }
958
959 if(m_iconPixmap)
960 XFreePixmap(m_display, m_iconPixmap);
961
962 if(m_iconMaskPixmap)
963 XFreePixmap(m_display, m_iconMaskPixmap);
964
965 m_iconPixmap = XCreatePixmap(m_display, RootWindow(m_display, m_screen), width, height, defDepth);
966 XGCValues values;
967 GC iconGC = XCreateGC(m_display, m_iconPixmap, 0, &values);
968 XPutImage(m_display, m_iconPixmap, iconGC, iconImage, 0, 0, 0, 0, width, height);
969 XFreeGC(m_display, iconGC);
970 XDestroyImage(iconImage);
971
972 // Create the mask pixmap (must have 1 bit depth)
973 std::size_t pitch = (width + 7) / 8;
974 std::vector<Uint8> maskPixels(pitch * height, 0);
975 for (std::size_t j = 0; j < height; ++j)
976 {
977 for (std::size_t i = 0; i < pitch; ++i)
978 {
979 for (std::size_t k = 0; k < 8; ++k)
980 {
981 if (i * 8 + k < width)
982 {
983 Uint8 opacity = (pixels[(i * 8 + k + j * width) * 4 + 3] > 0) ? 1 : 0;
984 maskPixels[i + j * pitch] |= (opacity << k);
985 }
986 }
987 }
988 }
989 m_iconMaskPixmap = XCreatePixmapFromBitmapData(m_display, m_window, (char*)&maskPixels[0], width, height, 1, 0, 1);
990
991 // Send our new icon to the window through the WMHints
992 XWMHints* hints = XAllocWMHints();
993 hints->flags = IconPixmapHint | IconMaskHint;
994 hints->icon_pixmap = m_iconPixmap;
995 hints->icon_mask = m_iconMaskPixmap;
996 XSetWMHints(m_display, m_window, hints);
997 XFree(hints);
998
999 // ICCCM wants BGRA pixels: swap red and blue channels
1000 // ICCCM also wants the first 2 unsigned 32-bit values to be width and height
1001 std::vector<unsigned long> icccmIconPixels(2 + width * height, 0);
1002 unsigned long* ptr = &icccmIconPixels[0];
1003
1004 *ptr++ = width;
1005 *ptr++ = height;
1006
1007 for (std::size_t i = 0; i < width * height; ++i)
1008 {
1009 *ptr++ = (pixels[i * 4 + 2] << 0 ) |
1010 (pixels[i * 4 + 1] << 8 ) |
1011 (pixels[i * 4 + 0] << 16) |
1012 (pixels[i * 4 + 3] << 24);
1013 }
1014
1015 Atom netWmIcon = getAtom("_NET_WM_ICON");
1016
1017 XChangeProperty(m_display,
1018 m_window,
1019 netWmIcon,
1020 XA_CARDINAL,
1021 32,
1022 PropModeReplace,
1023 reinterpret_cast<const unsigned char*>(&icccmIconPixels[0]),
1024 2 + width * height);
1025
1026 XFlush(m_display);
1027}

1115////////////////////////////////////////////////////////////
1116void WindowImplX11::requestFocus()
1117{
1118 // Focus is only stolen among SFML windows, not between applications
1119 // Check the global list of windows to find out whether an SFML window has the focus
1120 // Note: can't handle console and other non-SFML windows belonging to the application.
1121 bool sfmlWindowFocused = false;
1122
1123 {
1124 Lock lock(allWindowsMutex);
1125 for (std::vector<WindowImplX11*>::iterator itr = allWindows.begin(); itr != allWindows.end(); ++itr)
1126 {
1127 if ((*itr)->hasFocus())
1128 {
1129 sfmlWindowFocused = true;
1130 break;
1131 }
1132 }
1133 }
1134
1135 // Check if window is viewable (not on other desktop, ...)
1136 // TODO: Check also if minimized
1137 XWindowAttributes attributes;
1138 if (XGetWindowAttributes(m_display, m_window, &attributes) == 0)
1139 {
1140 sf::err() << "Failed to check if window is viewable while requesting focus" << std::endl;
1141 return; // error getting attribute
1142 }
1143
1144 bool windowViewable = (attributes.map_state == IsViewable);
1145
1146 if (sfmlWindowFocused && windowViewable)
1147 {
1148 // Another SFML window of this application has the focus and the current window is viewable:
1149 // steal focus (i.e. bring window to the front and give it input focus)
1150 grabFocus();
1151 }
1152 else
1153 {
1154 // Otherwise: display urgency hint (flashing application logo)
1155 // Ensure WM hints exist, allocate if necessary
1156 XWMHints* hints = XGetWMHints(m_display, m_window);
1157 if (hints == NULL)
1158 hints = XAllocWMHints();
1159
1160 // Add urgency (notification) flag to hints
1161 hints->flags |= XUrgencyHint;
1162 XSetWMHints(m_display, m_window, hints);
1163 XFree(hints);
1164 }
1165}

1181{
1182 Atom netActiveWindow = None;
1183
1184 if (ewmhSupported())
1185 netActiveWindow = getAtom("_NET_ACTIVE_WINDOW");
1186

1189
1190 XGetWindowAttributes(m_display, m_window, &attr);
1191
1192 if (attr.map_state == IsUnmapped)
1193 return;
1194
1195 if (netActiveWindow)

1199

1204 event.xclient.data.l[0] = 1; // Normal application

1211 SubstructureNotifyMask | SubstructureRedirectMask,
1212 &event);
1213
1214 XFlush(m_display);
1215
1216 if (!result)
1217 err() << "Setting fullscreen failed, could not send \"_NET_ACTIVE_WINDOW\" event" << std::endl;
1218 }
1219 else
1220 {
1221 XRaiseWindow(m_display, m_window);
1222 XSetInputFocus(m_display, m_window, RevertToPointerRoot, CurrentTime);

1228////////////////////////////////////////////////////////////
1229void WindowImplX11::setVideoMode(const VideoMode& mode)
1230{
1231 // Skip mode switching if the new mode is equal to the desktop mode
1232 if (mode == VideoMode::getDesktopMode())
1233 return;
1234
1235 // Check if the XRandR extension is present
1236 int version;
1237 if (!XQueryExtension(m_display, "RANDR", &version, &version, &version))
1238 {
1239 // XRandR extension is not supported: we cannot use fullscreen mode
1240 err() << "Fullscreen is not supported, switching to window mode" << std::endl;
1241 return;
1242 }
1243
1244 // Get the current configuration
1245 XRRScreenConfiguration* config = XRRGetScreenInfo(m_display, RootWindow(m_display, m_screen));
1246
1247 if (!config)
1248 {
1249 // Failed to get the screen configuration
1250 err() << "Failed to get the current screen configuration for fullscreen mode, switching to window mode" << std::endl;
1251 return;
1252 }
1253
1254 // Save the current video mode before we switch to fullscreen
1255 Rotation currentRotation;
1256 m_oldVideoMode = XRRConfigCurrentConfiguration(config, &currentRotation);
1257
1258 // Get the available screen sizes
1259 int nbSizes;
1260 XRRScreenSize* sizes = XRRConfigSizes(config, &nbSizes);
1261
1262 // Search for a matching size
1263 for (int i = 0; (sizes && i < nbSizes); ++i)
1264 {
1265 XRRConfigRotations(config, &currentRotation);
1266
1267 if (currentRotation == RR_Rotate_90 || currentRotation == RR_Rotate_270)
1268 std::swap(sizes[i].height, sizes[i].width);
1269
1270 if ((sizes[i].width == static_cast<int>(mode.width)) && (sizes[i].height == static_cast<int>(mode.height)))
1271 {
1272 // Switch to fullscreen mode
1273 XRRSetScreenConfig(m_display, config, RootWindow(m_display, m_screen), i, currentRotation, CurrentTime);
1274
1275 // Set "this" as the current fullscreen window
1276 fullscreenWindow = this;
1277 break;
1278 }
1279 }
1280
1281 // Free the configuration instance
1282 XRRFreeScreenConfigInfo(config);
1283}

1457 err() << "Failed to create input context for window -- TextEntered event won't be able to return unicode" << std::endl;
1458
1459 Atom wmWindowType = getAtom("_NET_WM_WINDOW_TYPE", false);
1460 Atom wmWindowTypeNormal = getAtom("_NET_WM_WINDOW_TYPE_NORMAL", false);
1461
1462 if (wmWindowType && wmWindowTypeNormal)
1463 {
1464 XChangeProperty(m_display,
1465 m_window,
1466 wmWindowType,
1467 XA_ATOM,
1468 32,
1469 PropModeReplace,
1470 reinterpret_cast<const unsigned char*>(&wmWindowTypeNormal),
1471 1);
1472 }
1473
1474 // Show the window
1475 setVisible(true);
1476
1477 // Raise the window and grab input focus
1478 grabFocus();
1479
1480 // Create the hidden cursor
1481 createHiddenCursor();
1482
1483 // Flush the commands queue
1484 XFlush(m_display);
1485
1486 // Add this window to the global list of windows (required for focus request)
1487 Lock lock(allWindowsMutex);
1488 allWindows.push_back(this);
1489}

1547////////////////////////////////////////////////////////////
1548bool WindowImplX11::processEvent(XEvent& windowEvent)
1549{
1550 // This function implements a workaround to properly discard
1551 // repeated key events when necessary. The problem is that the
1552 // system's key events policy doesn't match SFML's one: X server will generate
1553 // both repeated KeyPress and KeyRelease events when maintaining a key down, while
1554 // SFML only wants repeated KeyPress events. Thus, we have to:
1555 // - Discard duplicated KeyRelease events when KeyRepeatEnabled is true
1556 // - Discard both duplicated KeyPress and KeyRelease events when KeyRepeatEnabled is false
1557
1558 // Detect repeated key events
1559 if (windowEvent.type == KeyRelease)
1560 {
1561 // Find the next KeyPress event with matching keycode and time
1562 std::deque<XEvent>::iterator iter = std::find_if(
1563 m_events.begin(),
1564 m_events.end(),
1565 KeyRepeatFinder(windowEvent.xkey.keycode, windowEvent.xkey.time)
1566 );
1567
1568 if (iter != m_events.end())
1569 {
1570 // If we don't want repeated events, remove the next KeyPress from the queue
1571 if (!m_keyRepeat)
1572 m_events.erase(iter);
1573
1574 // This KeyRelease is a repeated event and we don't want it
1575 return false;
1576 }
1577 }
1578
1579 // Convert the X11 event to a sf::Event
1580 switch (windowEvent.type)
1581 {
1582 // Destroy event
1583 case DestroyNotify:
1584 {
1585 // The window is about to be destroyed: we must cleanup resources
1586 cleanup();
1587 break;
1588 }
1589
1590 // Gain focus event
1591 case FocusIn:
1592 {
1593 // Update the input context
1594 if (m_inputContext)
1595 XSetICFocus(m_inputContext);
1596
1597 // Grab cursor
1598 if (m_cursorGrabbed)
1599 {
1600 // Try multiple times to grab the cursor
1601 for (unsigned int trial = 0; trial < maxTrialsCount; ++trial)
1602 {
1603 int result = XGrabPointer(m_display, m_window, True, None, GrabModeAsync, GrabModeAsync, m_window, None, CurrentTime);
1604
1605 if (result == GrabSuccess)
1606 {
1607 m_cursorGrabbed = true;
1608 break;
1609 }
1610
1611 // The cursor grab failed, trying again after a small sleep
1612 sf::sleep(sf::milliseconds(50));
1613 }
1614
1615 if (!m_cursorGrabbed)
1616 err() << "Failed to grab mouse cursor" << std::endl;
1617 }
1618
1619 Event event;
1620 event.type = Event::GainedFocus;
1621 pushEvent(event);
1622
1623 // If the window has been previously marked urgent (notification) as a result of a focus request, undo that
1624 XWMHints* hints = XGetWMHints(m_display, m_window);
1625 if (hints != NULL)
1626 {
1627 // Remove urgency (notification) flag from hints
1628 hints->flags &= ~XUrgencyHint;
1629 XSetWMHints(m_display, m_window, hints);
1630 XFree(hints);
1631 }
1632
1633 break;
1634 }
1635
1636 // Lost focus event
1637 case FocusOut:
1638 {
1639 // Update the input context
1640 if (m_inputContext)
1641 XUnsetICFocus(m_inputContext);
1642
1643 // Release cursor
1644 if (m_cursorGrabbed)
1645 XUngrabPointer(m_display, CurrentTime);
1646
1647 Event event;
1648 event.type = Event::LostFocus;
1649 pushEvent(event);
1650 break;
1651 }
1652
1653 // Resize event
1654 case ConfigureNotify:
1655 {
1656 // ConfigureNotify can be triggered for other reasons, check if the size has actually changed
1657 if ((windowEvent.xconfigure.width != m_previousSize.x) || (windowEvent.xconfigure.height != m_previousSize.y))
1658 {
1659 Event event;
1660 event.type = Event::Resized;
1661 event.size.width = windowEvent.xconfigure.width;
1662 event.size.height = windowEvent.xconfigure.height;
1663 pushEvent(event);
1664
1665 m_previousSize.x = windowEvent.xconfigure.width;
1666 m_previousSize.y = windowEvent.xconfigure.height;
1667 }
1668 break;
1669 }
1670
1671 // Close event
1672 case ClientMessage:
1673 {
1674 static Atom wmProtocols = getAtom("WM_PROTOCOLS");
1675
1676 // Handle window manager protocol messages we support
1677 if (windowEvent.xclient.message_type == wmProtocols)
1678 {
1679 static Atom wmDeleteWindow = getAtom("WM_DELETE_WINDOW");
1680 static Atom netWmPing = ewmhSupported() ? getAtom("_NET_WM_PING", true) : None;
1681
1682 if ((windowEvent.xclient.format == 32) && (windowEvent.xclient.data.l[0]) == static_cast<long>(wmDeleteWindow))
1683 {
1684 // Handle the WM_DELETE_WINDOW message
1685 Event event;
1686 event.type = Event::Closed;
1687 pushEvent(event);
1688 }
1689 else if (netWmPing && (windowEvent.xclient.format == 32) && (windowEvent.xclient.data.l[0]) == static_cast<long>(netWmPing))
1690 {
1691 // Handle the _NET_WM_PING message, send pong back to WM to show that we are responsive
1692 windowEvent.xclient.window = DefaultRootWindow(m_display);
1693
1694 XSendEvent(m_display, DefaultRootWindow(m_display), False, SubstructureNotifyMask | SubstructureRedirectMask, &windowEvent);
1695 }
1696 }
1697 break;
1698 }
1699
1700 // Key down event
1701 case KeyPress:
1702 {
1703 Keyboard::Key key = Keyboard::Unknown;
1704
1705 // Try each KeySym index (modifier group) until we get a match
1706 for (int i = 0; i < 4; ++i)
1707 {
1708 // Get the SFML keyboard code from the keysym of the key that has been pressed
1709 key = keysymToSF(XLookupKeysym(&windowEvent.xkey, i));
1710
1711 if (key != Keyboard::Unknown)
1712 break;
1713 }
1714
1715 // Fill the event parameters
1716 // TODO: if modifiers are wrong, use XGetModifierMapping to retrieve the actual modifiers mapping
1717 Event event;
1718 event.type = Event::KeyPressed;
1719 event.key.code = key;
1720 event.key.alt = windowEvent.xkey.state & Mod1Mask;
1721 event.key.control = windowEvent.xkey.state & ControlMask;
1722 event.key.shift = windowEvent.xkey.state & ShiftMask;
1723 event.key.system = windowEvent.xkey.state & Mod4Mask;
1724 pushEvent(event);
1725
1726 // Generate a TextEntered event
1727 if (!XFilterEvent(&windowEvent, None))
1728 {
1729 #ifdef X_HAVE_UTF8_STRING
1730 if (m_inputContext)
1731 {
1732 Status status;
1733 Uint8 keyBuffer[16];
1734
1735 int length = Xutf8LookupString(
1736 m_inputContext,
1737 &windowEvent.xkey,
1738 reinterpret_cast<char*>(keyBuffer),
1739 sizeof(keyBuffer),
1740 NULL,
1741 &status
1742 );
1743
1744 if (length > 0)
1745 {
1746 Uint32 unicode = 0;
1747 Utf8::decode(keyBuffer, keyBuffer + length, unicode, 0);
1748 if (unicode != 0)
1749 {
1750 Event textEvent;
1751 textEvent.type = Event::TextEntered;
1752 textEvent.text.unicode = unicode;
1753 pushEvent(textEvent);
1754 }
1755 }
1756 }
1757 else
1758 #endif
1759 {
1760 static XComposeStatus status;
1761 char keyBuffer[16];
1762 if (XLookupString(&windowEvent.xkey, keyBuffer, sizeof(keyBuffer), NULL, &status))
1763 {
1764 Event textEvent;
1765 textEvent.type = Event::TextEntered;
1766 textEvent.text.unicode = static_cast<Uint32>(keyBuffer[0]);
1767 pushEvent(textEvent);
1768 }
1769 }
1770 }
1771
1772 updateLastInputTime(windowEvent.xkey.time);
1773
1774 break;
1775 }
1776
1777 // Key up event
1778 case KeyRelease:
1779 {
1780 Keyboard::Key key = Keyboard::Unknown;
1781
1782 // Try each KeySym index (modifier group) until we get a match
1783 for (int i = 0; i < 4; ++i)
1784 {
1785 // Get the SFML keyboard code from the keysym of the key that has been released
1786 key = keysymToSF(XLookupKeysym(&windowEvent.xkey, i));
1787
1788 if (key != Keyboard::Unknown)
1789 break;
1790 }
1791
1792 // Fill the event parameters
1793 Event event;
1794 event.type = Event::KeyReleased;
1795 event.key.code = key;
1796 event.key.alt = windowEvent.xkey.state & Mod1Mask;
1797 event.key.control = windowEvent.xkey.state & ControlMask;
1798 event.key.shift = windowEvent.xkey.state & ShiftMask;
1799 event.key.system = windowEvent.xkey.state & Mod4Mask;
1800 pushEvent(event);
1801
1802 break;
1803 }
1804
1805 // Mouse button pressed
1806 case ButtonPress:
1807 {
1808 // XXX: Why button 8 and 9?
1809 // Because 4 and 5 are the vertical wheel and 6 and 7 are horizontal wheel ;)
1810 unsigned int button = windowEvent.xbutton.button;
1811 if ((button == Button1) ||
1812 (button == Button2) ||
1813 (button == Button3) ||
1814 (button == 8) ||
1815 (button == 9))
1816 {
1817 Event event;
1818 event.type = Event::MouseButtonPressed;
1819 event.mouseButton.x = windowEvent.xbutton.x;
1820 event.mouseButton.y = windowEvent.xbutton.y;
1821 switch(button)
1822 {
1823 case Button1: event.mouseButton.button = Mouse::Left; break;
1824 case Button2: event.mouseButton.button = Mouse::Middle; break;
1825 case Button3: event.mouseButton.button = Mouse::Right; break;
1826 case 8: event.mouseButton.button = Mouse::XButton1; break;
1827 case 9: event.mouseButton.button = Mouse::XButton2; break;
1828 }
1829 pushEvent(event);
1830 }
1831
1832 updateLastInputTime(windowEvent.xbutton.time);
1833
1834 break;
1835 }
1836
1837 // Mouse button released
1838 case ButtonRelease:
1839 {
1840 unsigned int button = windowEvent.xbutton.button;
1841 if ((button == Button1) ||
1842 (button == Button2) ||
1843 (button == Button3) ||
1844 (button == 8) ||
1845 (button == 9))
1846 {
1847 Event event;
1848 event.type = Event::MouseButtonReleased;
1849 event.mouseButton.x = windowEvent.xbutton.x;
1850 event.mouseButton.y = windowEvent.xbutton.y;
1851 switch(button)
1852 {
1853 case Button1: event.mouseButton.button = Mouse::Left; break;
1854 case Button2: event.mouseButton.button = Mouse::Middle; break;
1855 case Button3: event.mouseButton.button = Mouse::Right; break;
1856 case 8: event.mouseButton.button = Mouse::XButton1; break;
1857 case 9: event.mouseButton.button = Mouse::XButton2; break;
1858 }
1859 pushEvent(event);
1860 }
1861 else if ((button == Button4) || (button == Button5))
1862 {
1863 Event event;
1864
1865 event.type = Event::MouseWheelMoved;
1866 event.mouseWheel.delta = (button == Button4) ? 1 : -1;
1867 event.mouseWheel.x = windowEvent.xbutton.x;
1868 event.mouseWheel.y = windowEvent.xbutton.y;
1869 pushEvent(event);
1870
1871 event.type = Event::MouseWheelScrolled;
1872 event.mouseWheelScroll.wheel = Mouse::VerticalWheel;
1873 event.mouseWheelScroll.delta = (button == Button4) ? 1 : -1;
1874 event.mouseWheelScroll.x = windowEvent.xbutton.x;
1875 event.mouseWheelScroll.y = windowEvent.xbutton.y;
1876 pushEvent(event);
1877 }
1878 else if ((button == 6) || (button == 7))
1879 {
1880 Event event;
1881 event.type = Event::MouseWheelScrolled;
1882 event.mouseWheelScroll.wheel = Mouse::HorizontalWheel;
1883 event.mouseWheelScroll.delta = (button == 6) ? 1 : -1;
1884 event.mouseWheelScroll.x = windowEvent.xbutton.x;
1885 event.mouseWheelScroll.y = windowEvent.xbutton.y;
1886 pushEvent(event);
1887 }
1888 break;
1889 }
1890
1891 // Mouse moved
1892 case MotionNotify:
1893 {
1894 Event event;
1895 event.type = Event::MouseMoved;
1896 event.mouseMove.x = windowEvent.xmotion.x;
1897 event.mouseMove.y = windowEvent.xmotion.y;
1898 pushEvent(event);
1899 break;
1900 }
1901
1902 // Mouse entered
1903 case EnterNotify:
1904 {
1905 if (windowEvent.xcrossing.mode == NotifyNormal)
1906 {
1907 Event event;
1908 event.type = Event::MouseEntered;
1909 pushEvent(event);
1910 }
1911 break;
1912 }
1913
1914 // Mouse left
1915 case LeaveNotify:
1916 {
1917 if (windowEvent.xcrossing.mode == NotifyNormal)
1918 {
1919 Event event;
1920 event.type = Event::MouseLeft;
1921 pushEvent(event);
1922 }
1923 break;
1924 }
1925
1926 // Window unmapped
1927 case UnmapNotify:
1928 {
1929 if (windowEvent.xunmap.window == m_window)
1930 m_windowMapped = false;
1931
1932 break;
1933 }
1934
1935 // Window visibility change
1936 case VisibilityNotify:
1937 {
1938 // We prefer using VisibilityNotify over MapNotify because
1939 // some window managers like awesome don't internally flag a
1940 // window as viewable even after it is mapped but before it
1941 // is visible leading to certain function calls failing with
1942 // an unviewable error if called before VisibilityNotify arrives
1943
1944 // Empirical testing on most widely used window managers shows
1945 // that mapping a window will always lead to a VisibilityNotify
1946 // event that is not VisibilityFullyObscured
1947 if (windowEvent.xvisibility.window == m_window)
1948 {
1949 if (windowEvent.xvisibility.state != VisibilityFullyObscured)
1950 m_windowMapped = true;
1951 }
1952
1953 break;
1954 }
1955
1956 // Window property change
1957 case PropertyNotify:
1958 {
1959 if (!m_lastInputTime)
1960 m_lastInputTime = windowEvent.xproperty.time;
1961
1962 break;
1963 }
1964 }
1965
1966 return true;
1967}
