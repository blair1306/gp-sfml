43 sf::Mutex glxErrorMutex;
44 bool glxErrorOccurred = false;
45
46 int HandleXError(::Display*, XErrorEvent*)
47 {
48 glxErrorOccurred = true;
49 return 0;
50 }
51
52 class GlxErrorHandler
53 {
54 public:
55
56 GlxErrorHandler(::Display* display) :
57 m_lock (glxErrorMutex),
58 m_display(display)
59 {
60 glxErrorOccurred = false;
61 m_previousHandler = XSetErrorHandler(HandleXError);
62 }
63
64 ~GlxErrorHandler()
65 {
66 XSync(m_display, False);
67 XSetErrorHandler(m_previousHandler);
68 }
69
70 private:
71 sf::Lock m_lock;
72 ::Display* m_display;
73 int (*m_previousHandler)(::Display*, XErrorEvent*);
74 };
75}

82////////////////////////////////////////////////////////////
83void ensureExtensionsInit(::Display* display, int screen)
84{
85 static bool initialized = false;
86 if (!initialized)
87 {
88 initialized = true;
89
90 // We don't check the return value since the extension
91 // flags are cleared even if loading fails
92 sfglx_LoadFunctions(display, screen);
93 }
94}

97////////////////////////////////////////////////////////////
98GlxContext::GlxContext(GlxContext* shared) :
99m_display (NULL),
100m_window (0),
101m_context (NULL),
102m_pbuffer (0),
103m_ownsWindow(false)
104{
105 // Save the creation settings
106 m_settings = ContextSettings();
107
108 // Open the connection with the X server
109 m_display = OpenDisplay();
110
111 // Make sure that extensions are initialized
112 ensureExtensionsInit(m_display, DefaultScreen(m_display));
113
114 // Create the rendering surface (window or pbuffer if supported)
115 createSurface(shared, 1, 1, VideoMode::getDesktopMode().bitsPerPixel);
116
117 // Create the context
118 createContext(shared);
119}

172////////////////////////////////////////////////////////////
173GlxContext::~GlxContext()
174{
175 // Destroy the context
176 if (m_context)
177 {
178#if defined(GLX_DEBUGGING)
179 GlxErrorHandler handler(m_display);
180#endif
181
182 if (glXGetCurrentContext() == m_context)
183 glXMakeCurrent(m_display, None, NULL);
184 glXDestroyContext(m_display, m_context);
185
186#if defined(GLX_DEBUGGING)
187 if (glxErrorOccurred)
188 err() << "GLX error in GlxContext::~GlxContext()" << std::endl;
189#endif
190 }
191
192 if (m_pbuffer)
193 {
194 glXDestroyPbuffer(m_display, m_pbuffer);
195 }
196
197 // Destroy the window if we own it
198 if (m_window && m_ownsWindow)
199 {
200 XDestroyWindow(m_display, m_window);
201 XFlush(m_display);
202 }
203
204 // Close the connection with the X server
205 CloseDisplay(m_display);
206}

216////////////////////////////////////////////////////////////
217bool GlxContext::makeCurrent(bool current)
218{
219 if (!m_context)
220 return false;
221
222#if defined(GLX_DEBUGGING)
223 GlxErrorHandler handler(m_display);
224#endif
225
226 bool result = false;
227
228 if (current)
229 {
230 if (m_pbuffer)
231 {
232 result = glXMakeContextCurrent(m_display, m_pbuffer, m_pbuffer, m_context);
233 }
234 else if (m_window)
235 {
236 result = glXMakeCurrent(m_display, m_window, m_context);
237 }
238 }
239 else
240 {
241 result = glXMakeCurrent(m_display, None, NULL);
242 }
243
244#if defined(GLX_DEBUGGING)
245 if (glxErrorOccurred)
246 err() << "GLX error in GlxContext::makeCurrent()" << std::endl;
247#endif
248
249 return result;
250}

272////////////////////////////////////////////////////////////
273void GlxContext::setVerticalSyncEnabled(bool enabled)
274{
275 int result = 0;
276
277 // Prioritize the EXT variant and fall back to MESA or SGI if needed
278 // We use the direct pointer to the MESA entry point instead of the alias
279 // because glx.h declares the entry point as an external function
280 // which would require us to link in an additional library
281 if (sfglx_ext_EXT_swap_control == sfglx_LOAD_SUCCEEDED)
282 {
283 glXSwapIntervalEXT(m_display, m_pbuffer ? m_pbuffer : m_window, enabled ? 1 : 0);
284 }
285 else if (sfglx_ext_MESA_swap_control == sfglx_LOAD_SUCCEEDED)
286 {
287 result = sf_ptrc_glXSwapIntervalMESA(enabled ? 1 : 0);
288 }
289 else if (sfglx_ext_SGI_swap_control == sfglx_LOAD_SUCCEEDED)
290 {
291 result = glXSwapIntervalSGI(enabled ? 1 : 0);
292 }
293 else
294 {
295 static bool warned = false;
296
297 if (!warned)
298 {
299 err() << "Setting vertical sync not supported" << std::endl;
300
301 warned = true;
302 }
303 }
304
305 if (result != 0)
306 err() << "Setting vertical sync failed" << std::endl;
307}

310////////////////////////////////////////////////////////////
311XVisualInfo GlxContext::selectBestVisual(::Display* display, unsigned int bitsPerPixel, const ContextSettings& settings)
312{
313 // Make sure that extensions are initialized
314 ensureExtensionsInit(display, DefaultScreen(display));
315
316 // Retrieve all the visuals
317 int count;
318 XVisualInfo* visuals = XGetVisualInfo(display, 0, NULL, &count);
319 if (visuals)
320 {
321 // Evaluate all the returned visuals, and pick the best one
322 int bestScore = 0x7FFFFFFF;
323 XVisualInfo bestVisual = XVisualInfo();
324 for (int i = 0; i < count; ++i)
325 {
326 // Check mandatory attributes
327 int doubleBuffer;
328 glXGetConfig(display, &visuals[i], GLX_DOUBLEBUFFER, &doubleBuffer);
329 if (!doubleBuffer)
330 continue;
331
332 // Extract the components of the current visual
333 int red, green, blue, alpha, depth, stencil, multiSampling, samples, sRgb;
334 glXGetConfig(display, &visuals[i], GLX_RED_SIZE, &red);
335 glXGetConfig(display, &visuals[i], GLX_GREEN_SIZE, &green);
336 glXGetConfig(display, &visuals[i], GLX_BLUE_SIZE, &blue);
337 glXGetConfig(display, &visuals[i], GLX_ALPHA_SIZE, &alpha);
338 glXGetConfig(display, &visuals[i], GLX_DEPTH_SIZE, &depth);
339 glXGetConfig(display, &visuals[i], GLX_STENCIL_SIZE, &stencil);
340
341 if (sfglx_ext_ARB_multisample == sfglx_LOAD_SUCCEEDED)
342 {
343 glXGetConfig(display, &visuals[i], GLX_SAMPLE_BUFFERS_ARB, &multiSampling);
344 glXGetConfig(display, &visuals[i], GLX_SAMPLES_ARB, &samples);
345 }
346 else
347 {
348 multiSampling = 0;
349 samples = 0;
350 }
351
352 if ((sfglx_ext_EXT_framebuffer_sRGB == sfglx_LOAD_SUCCEEDED) || (sfglx_ext_ARB_framebuffer_sRGB == sfglx_LOAD_SUCCEEDED))
353 {
354 glXGetConfig(display, &visuals[i], GLX_FRAMEBUFFER_SRGB_CAPABLE_ARB, &sRgb);
355 }
356 else
357 {
358 sRgb = 0;
359 }
360
361 // TODO: Replace this with proper acceleration detection
362 bool accelerated = true;
363
364 // Evaluate the visual
365 int color = red + green + blue + alpha;
366 int score = evaluateFormat(bitsPerPixel, settings, color, depth, stencil, multiSampling ? samples : 0, accelerated, sRgb == True);
367
368 // If it's better than the current best, make it the new best
369 if (score < bestScore)
370 {
371 bestScore = score;
372 bestVisual = visuals[i];
373 }
374 }
375
376 // Free the array of visuals
377 XFree(visuals);
378
379 return bestVisual;
380 }
381 else
382 {
383 // Should never happen...
384 err() << "No GLX visual found. You should check your graphics driver" << std::endl;
385
386 return XVisualInfo();
387 }
388}

391////////////////////////////////////////////////////////////
392void GlxContext::updateSettingsFromVisualInfo(XVisualInfo* visualInfo)
393{
394 // Update the creation settings from the chosen format
395 int depth, stencil, multiSampling, samples, sRgb;
396 glXGetConfig(m_display, visualInfo, GLX_DEPTH_SIZE, &depth);
397 glXGetConfig(m_display, visualInfo, GLX_STENCIL_SIZE, &stencil);
398
399 if (sfglx_ext_ARB_multisample == sfglx_LOAD_SUCCEEDED)
400 {
401 glXGetConfig(m_display, visualInfo, GLX_SAMPLE_BUFFERS_ARB, &multiSampling);
402 glXGetConfig(m_display, visualInfo, GLX_SAMPLES_ARB, &samples);
403 }
404 else
405 {
406 multiSampling = 0;
407 samples = 0;
408 }
409
410 if ((sfglx_ext_EXT_framebuffer_sRGB == sfglx_LOAD_SUCCEEDED) || (sfglx_ext_ARB_framebuffer_sRGB == sfglx_LOAD_SUCCEEDED))
411 {
412 glXGetConfig(m_display, visualInfo, GLX_FRAMEBUFFER_SRGB_CAPABLE_ARB, &sRgb);
413 }
414 else
415 {
416 sRgb = 0;
417 }
418
419 m_settings.depthBits = static_cast<unsigned int>(depth);
420 m_settings.stencilBits = static_cast<unsigned int>(stencil);
421 m_settings.antialiasingLevel = multiSampling ? samples : 0;
422 m_settings.sRgbCapable = (sRgb == True);
423}

426////////////////////////////////////////////////////////////
427void GlxContext::updateSettingsFromWindow()
428{
429 // Retrieve the attributes of the target window
430 XWindowAttributes windowAttributes;
431 if (XGetWindowAttributes(m_display, m_window, &windowAttributes) == 0)
432 {
433 err() << "Failed to get the window attributes" << std::endl;
434 return;
435 }
436
437 // Get its visuals
438 XVisualInfo tpl;
439 tpl.screen = DefaultScreen(m_display);
440 tpl.visualid = XVisualIDFromVisual(windowAttributes.visual);
441 int nbVisuals = 0;
442 XVisualInfo* visualInfo = XGetVisualInfo(m_display, VisualIDMask | VisualScreenMask, &tpl, &nbVisuals);
443
444 if (!visualInfo)
445 return;
446
447 updateSettingsFromVisualInfo(visualInfo);
448
449 XFree(visualInfo);
450}

453////////////////////////////////////////////////////////////
454void GlxContext::createSurface(GlxContext* shared, unsigned int width, unsigned int height, unsigned int bitsPerPixel)
455{
456 // Choose the visual according to the context settings
457 XVisualInfo visualInfo = selectBestVisual(m_display, bitsPerPixel, m_settings);
458
459 // Check if the shared context already exists and pbuffers are supported
460 if (shared && (sfglx_ext_SGIX_pbuffer == sfglx_LOAD_SUCCEEDED))
461 {
462 // There are no GLX versions prior to 1.0
463 int major = 0;
464 int minor = 0;
465
466 glXQueryVersion(m_display, &major, &minor);
467
468 // Check if glXCreatePbuffer is available (requires GLX 1.3 or greater)
469 bool hasCreatePbuffer = ((major > 1) || (minor >= 3));
470
471 if (hasCreatePbuffer)
472 {
473 // Get a GLXFBConfig that matches the visual
474 GLXFBConfig* config = NULL;
475
476 // We don't supply attributes to match against, since
477 // the visual we are matching against was already
478 // deemed suitable in selectBestVisual()
479 int nbConfigs = 0;
480 GLXFBConfig* configs = glXChooseFBConfig(m_display, DefaultScreen(m_display), NULL, &nbConfigs);
481
482 for (int i = 0; configs && (i < nbConfigs); ++i)
483 {
484 XVisualInfo* visual = glXGetVisualFromFBConfig(m_display, configs[i]);
485
486 if (!visual)
487 continue;
488
489 if (visual->visualid == visualInfo.visualid)
490 {
491 config = &configs[i];
492 XFree(visual);
493 break;
494 }
495
496 XFree(visual);
497 }
498
499 if (config)
500 {
501 int attributes[] =
502 {
503 GLX_PBUFFER_WIDTH, static_cast<int>(width),
504 GLX_PBUFFER_HEIGHT, static_cast<int>(height),
505 0, 0
506 };
507
508 m_pbuffer = glXCreatePbuffer(m_display, *config, attributes);
509
510 updateSettingsFromVisualInfo(&visualInfo);
511
512 XFree(configs);
513
514 return;
515 }
516
517 if (configs)
518 XFree(configs);
519 }
520 }
521
522 // If pbuffers are not available we use a hidden window as the off-screen surface to draw to
523 int screen = DefaultScreen(m_display);
524
525 // Define the window attributes
526 XSetWindowAttributes attributes;
527 attributes.colormap = XCreateColormap(m_display, RootWindow(m_display, screen), visualInfo.visual, AllocNone);
528
529 m_window = XCreateWindow(m_display,
530 RootWindow(m_display, screen),
531 0, 0,
532 width, height,
533 0,
534 DefaultDepth(m_display, screen),
535 InputOutput,
536 visualInfo.visual,
537 CWColormap,
538 &attributes);
539
540 m_ownsWindow = true;
541
542 updateSettingsFromWindow();
543}

556////////////////////////////////////////////////////////////
557void GlxContext::createContext(GlxContext* shared)
558{
559 // Get a working copy of the context settings
560 ContextSettings settings = m_settings;
561
562 XVisualInfo* visualInfo = NULL;
563
564 if (m_pbuffer)
565 {
566 unsigned int fbConfigId = 0;
567
568 glXQueryDrawable(m_display, m_pbuffer, GLX_FBCONFIG_ID, &fbConfigId);
569
570 int attributes[] =
571 {
572 GLX_FBCONFIG_ID, static_cast<int>(fbConfigId),
573 0, 0
574 };
575
576 int count = 0;
577 GLXFBConfig* fbconfig = glXChooseFBConfig(m_display, DefaultScreen(m_display), attributes, &count);
578
579 if (count == 1)
580 visualInfo = glXGetVisualFromFBConfig(m_display, *fbconfig);
581
582 if (fbconfig)
583 XFree(fbconfig);
584 }
585 else
586 {
587 // Retrieve the attributes of the target window
588 XWindowAttributes windowAttributes;
589 if (XGetWindowAttributes(m_display, m_window, &windowAttributes) == 0)
590 {
591 err() << "Failed to get the window attributes" << std::endl;
592 return;
593 }
594
595 // Get its visuals
596 XVisualInfo tpl;
597 tpl.screen = DefaultScreen(m_display);
598 tpl.visualid = XVisualIDFromVisual(windowAttributes.visual);
599 int nbVisuals = 0;
600 visualInfo = XGetVisualInfo(m_display, VisualIDMask | VisualScreenMask, &tpl, &nbVisuals);
601 }
602
603 if (!visualInfo)
604 {
605 err() << "Failed to get visual info" << std::endl;
606 return;
607 }
608
609 // Get the context to share display lists with
610 GLXContext toShare = shared ? shared->m_context : NULL;
611
612 // There are no GLX versions prior to 1.0
613 int major = 0;
614 int minor = 0;
615
616 if (!glXQueryVersion(m_display, &major, &minor))
617 err() << "Failed to query GLX version, limited to legacy context creation" << std::endl;
618
619 // Check if glXCreateContextAttribsARB is available (requires GLX 1.3 or greater)
620 bool hasCreateContextArb = (sfglx_ext_ARB_create_context == sfglx_LOAD_SUCCEEDED) && ((major > 1) || (minor >= 3));
621
622 // Create the OpenGL context -- first try using glXCreateContextAttribsARB
623 if (hasCreateContextArb)
624 {
625 // Get a GLXFBConfig that matches the window's visual, for glXCreateContextAttribsARB
626 GLXFBConfig* config = NULL;
627
628 // We don't supply attributes to match against, since
629 // the visual we are matching against was already
630 // deemed suitable in selectBestVisual()
631 int nbConfigs = 0;
632 GLXFBConfig* configs = glXChooseFBConfig(m_display, DefaultScreen(m_display), NULL, &nbConfigs);
633
634 for (int i = 0; configs && (i < nbConfigs); ++i)
635 {
636 XVisualInfo* visual = glXGetVisualFromFBConfig(m_display, configs[i]);
637
638 if (!visual)
639 continue;
640
641 if (visual->visualid == visualInfo->visualid)
642 {
643 config = &configs[i];
644 XFree(visual);
645 break;
646 }
647
648 XFree(visual);
649 }
650
651 if (!config)
652 err() << "Failed to get GLXFBConfig which corresponds to the window's visual" << std::endl;
653
654 while (config && !m_context && m_settings.majorVersion)
655 {
656 std::vector<int> attributes;
657
658 // Check if the user requested a specific context version (anything > 1.1)
659 if ((m_settings.majorVersion > 1) || ((m_settings.majorVersion == 1) && (m_settings.minorVersion > 1)))
660 {
661 attributes.push_back(GLX_CONTEXT_MAJOR_VERSION_ARB);
662 attributes.push_back(m_settings.majorVersion);
663 attributes.push_back(GLX_CONTEXT_MINOR_VERSION_ARB);
664 attributes.push_back(m_settings.minorVersion);
665 }
666
667 // Check if setting the profile is supported
668 if (sfglx_ext_ARB_create_context_profile == sfglx_LOAD_SUCCEEDED)
669 {
670 int profile = (m_settings.attributeFlags & ContextSettings::Core) ? GLX_CONTEXT_CORE_PROFILE_BIT_ARB : GLX_CONTEXT_COMPATIBILITY_PROFILE_BIT_ARB;
671 int debug = (m_settings.attributeFlags & ContextSettings::Debug) ? GLX_CONTEXT_DEBUG_BIT_ARB : 0;
672
673 attributes.push_back(GLX_CONTEXT_PROFILE_MASK_ARB);
674 attributes.push_back(profile);
675 attributes.push_back(GLX_CONTEXT_FLAGS_ARB);
676 attributes.push_back(debug);
677 }
678 else
679 {
680 if ((m_settings.attributeFlags & ContextSettings::Core) || (m_settings.attributeFlags & ContextSettings::Debug))
681 err() << "Selecting a profile during context creation is not supported,"
682 << "disabling comptibility and debug" << std::endl;
683
684 m_settings.attributeFlags = ContextSettings::Default;
685 }
686
687 // Append the terminating 0
688 attributes.push_back(0);
689 attributes.push_back(0);
690
691 // RAII GLX error handler (we simply ignore errors here)
692 // On an error, glXCreateContextAttribsARB will return 0 anyway
693 GlxErrorHandler handler(m_display);
694
695 if (toShare)
696 {
697 if (!glXMakeCurrent(m_display, None, NULL))
698 {
699 err() << "Failed to deactivate shared context before sharing" << std::endl;
700 return;
701 }
702 }
703
704 // Create the context
705 m_context = glXCreateContextAttribsARB(m_display, *config, toShare, true, &attributes[0]);
706
707 if (!m_context)
708 {
709 // If we couldn't create the context, first try disabling flags,
710 // then lower the version number and try again -- stop at 0.0
711 // Invalid version numbers will be generated by this algorithm (like 3.9), but we really don't care
712 if (m_settings.attributeFlags != ContextSettings::Default)
713 {
714 m_settings.attributeFlags = ContextSettings::Default;
715 }
716 else if (m_settings.minorVersion > 0)
717 {
718 // If the minor version is not 0, we decrease it and try again
719 m_settings.minorVersion--;
720
721 m_settings.attributeFlags = settings.attributeFlags;
722 }
723 else
724 {
725 // If the minor version is 0, we decrease the major version
726 m_settings.majorVersion--;
727 m_settings.minorVersion = 9;
728
729 m_settings.attributeFlags = settings.attributeFlags;
730 }
731 }
732 }
733
734 if (configs)
735 XFree(configs);
736 }
737
738 // If glXCreateContextAttribsARB failed, use glXCreateContext
739 if (!m_context)
740 {
741 // set the context version to 2.1 (arbitrary) and disable flags
742 m_settings.majorVersion = 2;
743 m_settings.minorVersion = 1;
744 m_settings.attributeFlags = ContextSettings::Default;
745
746#if defined(GLX_DEBUGGING)
747 GlxErrorHandler handler(m_display);
748#endif
749
750 if (toShare)
751 {
752 if (!glXMakeCurrent(m_display, None, NULL))
753 {
754 err() << "Failed to deactivate shared context before sharing" << std::endl;
755 return;
756 }
757 }
758
759 // Create the context, using the target window's visual
760 m_context = glXCreateContext(m_display, visualInfo, toShare, true);
761
762#if defined(GLX_DEBUGGING)
763 if (glxErrorOccurred)
764 err() << "GLX error in GlxContext::createContext()" << std::endl;
765#endif
766 }
767
768 if (!m_context)
769 err() << "Failed to create an OpenGL context for this window" << std::endl;
770
771 // Free the visual info
772 XFree(visualInfo);
773}
