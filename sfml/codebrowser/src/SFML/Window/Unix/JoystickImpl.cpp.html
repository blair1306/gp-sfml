44 struct JoystickRecord
45 {
46 std::string deviceNode;
47 std::string systemPath;
48 bool plugged;
49 };

51 typedef std::vector<JoystickRecord> JoystickList;
52 JoystickList joystickList;

54 bool isJoystick(udev_device* udevDevice)
55 {
56 // If anything goes wrong, we go safe and return true
57
58 // No device to check, assume not a joystick
59 if (!udevDevice)
60 return false;
61
62 const char* devnode = udev_device_get_devnode(udevDevice);
63
64 // We only consider devices with a device node
65 if (!devnode)
66 return false;
67
68 // SFML doesn't support evdev yet, so make sure we only handle /js nodes
69 if (!std::strstr(devnode, "/js"))
70 return false;
71
72 // Check if this device is a joystick
73 if (udev_device_get_property_value(udevDevice, "ID_INPUT_JOYSTICK"))
74 return true;
75
76 // Check if this device is something that isn't a joystick
77 // We do this because the absence of any ID_INPUT_ property doesn't
78 // necessarily mean that the device isn't a joystick, whereas the
79 // presence of any ID_INPUT_ property that isn't ID_INPUT_JOYSTICK does
80 if (udev_device_get_property_value(udevDevice, "ID_INPUT_ACCELEROMETER") ||
81 udev_device_get_property_value(udevDevice, "ID_INPUT_KEY") ||
82 udev_device_get_property_value(udevDevice, "ID_INPUT_KEYBOARD") ||
83 udev_device_get_property_value(udevDevice, "ID_INPUT_MOUSE") ||
84 udev_device_get_property_value(udevDevice, "ID_INPUT_TABLET") ||
85 udev_device_get_property_value(udevDevice, "ID_INPUT_TOUCHPAD") ||
86 udev_device_get_property_value(udevDevice, "ID_INPUT_TOUCHSCREEN"))
87 return false;
88
89 // On some platforms (older udev), ID_INPUT_ properties are not present, instead
90 // the system makes use of the ID_CLASS property to identify the device class
91 const char* idClass = udev_device_get_property_value(udevDevice, "ID_CLASS");
92
93 if (idClass)
94 {
95 // Check if the device class matches joystick
96 if (std::strstr(idClass, "joystick"))
97 return true;
98
99 // Check if the device class matches something that isn't a joystick
100 // Rationale same as above
101 if (std::strstr(idClass, "accelerometer") ||
102 std::strstr(idClass, "key") ||
103 std::strstr(idClass, "keyboard") ||
104 std::strstr(idClass, "mouse") ||
105 std::strstr(idClass, "tablet") ||
106 std::strstr(idClass, "touchpad") ||
107 std::strstr(idClass, "touchscreen"))
108 return false;
109 }
110
111 // At this point, assume it is a joystick
112 return true;
113 }

115 void updatePluggedList(udev_device* udevDevice = NULL)
116 {
117 if (udevDevice)
118 {
119 const char* action = udev_device_get_action(udevDevice);
120
121 if (action)
122 {
123 if (isJoystick(udevDevice))
124 {
125 // Since isJoystick returned true, this has to succeed
126 const char* devnode = udev_device_get_devnode(udevDevice);
127
128 JoystickList::iterator record;
129
130 for (record = joystickList.begin(); record != joystickList.end(); ++record)
131 {
132 if (record->deviceNode == devnode)
133 {
134 if (std::strstr(action, "add"))
135 {
136 // The system path might have changed so update it
137 const char* syspath = udev_device_get_syspath(udevDevice);
138
139 record->plugged = true;
140 record->systemPath = syspath ? syspath : "";
141 break;
142 }
143 else if (std::strstr(action, "remove"))
144 {
145 record->plugged = false;
146 break;
147 }
148 }
149 }
150
151 if (record == joystickList.end())
152 {
153 if (std::strstr(action, "add"))
154 {
155 // If not mapped before and it got added, map it now
156 const char* syspath = udev_device_get_syspath(udevDevice);
157
158 JoystickRecord record;
159 record.deviceNode = devnode;
160 record.systemPath = syspath ? syspath : "";
161 record.plugged = true;
162
163 joystickList.push_back(record);
164 }
165 else if (std::strstr(action, "remove"))
166 {
167 // Not mapped during the initial scan, and removed (shouldn't happen)
168 sf::err() << "Trying to disconnect joystick that wasn't connected" << std::endl;
169 }
170 }
171 }
172
173 return;
174 }
175
176 // Do a full rescan if there was no action just to be sure
177 }
178
179 // Reset the plugged status of each mapping since we are doing a full rescan
180 for (JoystickList::iterator record = joystickList.begin(); record != joystickList.end(); ++record)
181 record->plugged = false;
182
183 udev_enumerate* udevEnumerator = udev_enumerate_new(udevContext);
184
185 if (!udevEnumerator)
186 {
187 sf::err() << "Error while creating udev enumerator" << std::endl;
188 return;
189 }
190
191 int result = 0;
192
193 result = udev_enumerate_add_match_subsystem(udevEnumerator, "input");
194
195 if (result < 0)
196 {
197 sf::err() << "Error while adding udev enumerator match" << std::endl;
198 return;
199 }
200
201 result = udev_enumerate_scan_devices(udevEnumerator);
202
203 if (result < 0)
204 {
205 sf::err() << "Error while enumerating udev devices" << std::endl;
206 return;
207 }
208
209 udev_list_entry* devices = udev_enumerate_get_list_entry(udevEnumerator);
210 udev_list_entry* device;
211
212 udev_list_entry_foreach(device, devices) {
213 const char* syspath = udev_list_entry_get_name(device);
214 udev_device* udevDevice = udev_device_new_from_syspath(udevContext, syspath);
215
216 if (udevDevice && isJoystick(udevDevice))
217 {
218 // Since isJoystick returned true, this has to succeed
219 const char* devnode = udev_device_get_devnode(udevDevice);
220
221 JoystickList::iterator record;
222
223 // Check if the device node has been mapped before
224 for (record = joystickList.begin(); record != joystickList.end(); ++record)
225 {
226 if (record->deviceNode == devnode)
227 {
228 record->plugged = true;
229 break;
230 }
231 }
232
233 // If not mapped before, map it now
234 if (record == joystickList.end())
235 {
236 JoystickRecord record;
237 record.deviceNode = devnode;
238 record.systemPath = syspath;
239 record.plugged = true;
240
241 joystickList.push_back(record);
242 }
243 }
244
245 udev_device_unref(udevDevice);
246 }
247
248 udev_enumerate_unref(udevEnumerator);
249 }

251 bool hasMonitorEvent()
252 {
253 // This will not fail since we make sure udevMonitor is valid
254 int monitorFd = udev_monitor_get_fd(udevMonitor);
255
256 fd_set descriptorSet;
257 FD_ZERO(&descriptorSet);
258 FD_SET(monitorFd, &descriptorSet);
259 timeval timeout = {0, 0};
260
261 return (select(monitorFd + 1, &descriptorSet, NULL, NULL, &timeout) > 0) &&
262 FD_ISSET(monitorFd, &descriptorSet);
263 }

265 // Get a property value from a udev device
266 const char* getUdevAttribute(udev_device* udevDevice, const std::string& attributeName)
267 {
268 return udev_device_get_property_value(udevDevice, attributeName.c_str());

271 // Get a system attribute from a USB device
272 const char* getUsbAttribute(udev_device* udevDevice, const std::string& attributeName)
273 {
274 udev_device* udevDeviceParent = udev_device_get_parent_with_subsystem_devtype(udevDevice, "usb", "usb_device");
275
276 if (!udevDeviceParent)
277 return NULL;
278
279 return udev_device_get_sysattr_value(udevDeviceParent, attributeName.c_str());
280 }

282 // Get a USB attribute for a joystick as an unsigned int
283 unsigned int getUsbAttributeUint(udev_device* udevDevice, const std::string& attributeName)
284 {
285 if (!udevDevice)
286 return 0;
287
288 const char* attribute = getUsbAttribute(udevDevice, attributeName);
289 unsigned int value = 0;
290
291 if (attribute)
292 value = static_cast<unsigned int>(std::strtoul(attribute, NULL, 16));
293
294 return value;
295 }

312 // Get the joystick vendor id
313 unsigned int getJoystickVendorId(unsigned int index)
314 {
315 if (!udevContext)
316 {
317 sf::err() << "Failed to get vendor ID of joystick " << joystickList[index].deviceNode << std::endl;
318 return 0;
319 }
320
321 udev_device* udevDevice = udev_device_new_from_syspath(udevContext, joystickList[index].systemPath.c_str());
322
323 if (!udevDevice)
324 {
325 sf::err() << "Failed to get vendor ID of joystick " << joystickList[index].deviceNode << std::endl;
326 return 0;
327 }
328
329 unsigned int id = 0;
330
331 // First try using udev
332 id = getUdevAttributeUint(udevDevice, "ID_VENDOR_ID");
333
334 if (id)
335 {
336 udev_device_unref(udevDevice);
337 return id;
338 }
339
340 // Fall back to using USB attribute
341 id = getUsbAttributeUint(udevDevice, "idVendor");
342
343 udev_device_unref(udevDevice);
344
345 if (id)
346 return id;
347
348 sf::err() << "Failed to get vendor ID of joystick " << joystickList[index].deviceNode << std::endl;
349
350 return 0;
351 }
352
353 // Get the joystick product id
354 unsigned int getJoystickProductId(unsigned int index)
355 {
356 if (!udevContext)
357 {
358 sf::err() << "Failed to get product ID of joystick " << joystickList[index].deviceNode << std::endl;
359 return 0;
360 }
361
362 udev_device* udevDevice = udev_device_new_from_syspath(udevContext, joystickList[index].systemPath.c_str());
363
364 if (!udevDevice)
365 {
366 sf::err() << "Failed to get product ID of joystick " << joystickList[index].deviceNode << std::endl;
367 return 0;
368 }
369
370 unsigned int id = 0;
371
372 // First try using udev
373 id = getUdevAttributeUint(udevDevice, "ID_MODEL_ID");
374
375 if (id)
376 {
377 udev_device_unref(udevDevice);
378 return id;
379 }
380
381 // Fall back to using USB attribute
382 id = getUsbAttributeUint(udevDevice, "idProduct");
383
384 udev_device_unref(udevDevice);
385
386 if (id)
387 return id;
388
389 sf::err() << "Failed to get product ID of joystick " << joystickList[index].deviceNode << std::endl;
390
391 return 0;
392 }

394 // Get the joystick name
395 std::string getJoystickName(unsigned int index)
396 {
397 std::string devnode = joystickList[index].deviceNode;
398
399 // First try using ioctl with JSIOCGNAME
400 int fd = ::open(devnode.c_str(), O_RDONLY | O_NONBLOCK);
401
402 if (fd >= 0)
403 {
404 // Get the name
405 char name[128];
406 std::memset(name, 0, sizeof(name));
407
408 int result = ioctl(fd, JSIOCGNAME(sizeof(name)), name);
409
410 ::close(fd);
411
412 if (result >= 0)
413 return std::string(name);
414 }
415
416 // Fall back to manual USB chain walk via udev
417 if (udevContext)
418 {
419 udev_device* udevDevice = udev_device_new_from_syspath(udevContext, joystickList[index].systemPath.c_str());
420
421 if (udevDevice)
422 {
423 const char* product = getUsbAttribute(udevDevice, "product");
424 udev_device_unref(udevDevice);
425
426 if (product)
427 return std::string(product);
428 }
429 }
430
431 sf::err() << "Unable to get name for joystick " << devnode << std::endl;
432
433 return std::string("Unknown Joystick");
434 }

450////////////////////////////////////////////////////////////
451void JoystickImpl::initialize()
452{
453 udevContext = udev_new();
454
455 if (!udevContext)
456 {
457 sf::err() << "Failed to create udev context, joystick support not available" << std::endl;
458 return;
459 }
460
461 udevMonitor = udev_monitor_new_from_netlink(udevContext, "udev");
462
463 if (!udevMonitor)
464 {
465 err() << "Failed to create udev monitor, joystick connections and disconnections won't be notified" << std::endl;
466 }
467 else
468 {
469 int error = udev_monitor_filter_add_match_subsystem_devtype(udevMonitor, "input", NULL);
470
471 if (error < 0)
472 {
473 err() << "Failed to add udev monitor filter, joystick connections and disconnections won't be notified: " << error << std::endl;
474
475 udev_monitor_unref(udevMonitor);
476 udevMonitor = 0;
477 }
478 else
479 {
480 error = udev_monitor_enable_receiving(udevMonitor);
481
482 if (error < 0)
483 {
484 err() << "Failed to enable udev monitor, joystick connections and disconnections won't be notified: " << error << std::endl;
485
486 udev_monitor_unref(udevMonitor);
487 udevMonitor = 0;
488 }
489 }
490 }
491
492 // Do an initial scan
493 updatePluggedList();
494}

516////////////////////////////////////////////////////////////
517bool JoystickImpl::isConnected(unsigned int index)
518{
519 // See if we can skip scanning if udev monitor is available
520 if (!udevMonitor)
521 {
522 // udev monitor is not available, perform a scan every query
523 updatePluggedList();
524 }
525 else if (hasMonitorEvent())
526 {
527 // Check if new joysticks were added/removed since last update
528 udev_device* udevDevice = udev_monitor_receive_device(udevMonitor);
529
530 // If we can get the specific device, we check that,
531 // otherwise just do a full scan if udevDevice == NULL
532 updatePluggedList(udevDevice);
533
534 if (udevDevice)
535 udev_device_unref(udevDevice);
536 }
537
538 if (index >= joystickList.size())
539 return false;
540
541 // Then check if the joystick is connected
542 return joystickList[index].plugged;
543}

545////////////////////////////////////////////////////////////
546bool JoystickImpl::open(unsigned int index)
547{
548 if (index >= joystickList.size())
549 return false;
550
551 if (joystickList[index].plugged)
552 {
553 std::string devnode = joystickList[index].deviceNode;
554
555 // Open the joystick's file descriptor (read-only and non-blocking)
556 m_file = ::open(devnode.c_str(), O_RDONLY | O_NONBLOCK);
557 if (m_file >= 0)
558 {
559 // Retrieve the axes mapping
560 ioctl(m_file, JSIOCGAXMAP, m_mapping);
561
562 // Get info
563 m_identification.name = getJoystickName(index);
564
565 if (udevContext)
566 {
567 m_identification.vendorId = getJoystickVendorId(index);
568 m_identification.productId = getJoystickProductId(index);
569 }
570
571 // Reset the joystick state
572 m_state = JoystickState();
573
574 return true;
575 }
576 else
577 {
578 err() << "Failed to open joystick " << devnode << ": " << errno << std::endl;
579 }
580 }
581
582 return false;
583}

594////////////////////////////////////////////////////////////
595JoystickCaps JoystickImpl::getCapabilities() const
596{
597 JoystickCaps caps;
598
599 if (m_file < 0)
600 return caps;
601
602 // Get the number of buttons
603 char buttonCount;
604 ioctl(m_file, JSIOCGBUTTONS, &buttonCount);
605 caps.buttonCount = buttonCount;
606 if (caps.buttonCount > Joystick::ButtonCount)
607 caps.buttonCount = Joystick::ButtonCount;
608
609 // Get the supported axes
610 char axesCount;
611 ioctl(m_file, JSIOCGAXES, &axesCount);
612 for (int i = 0; i < axesCount; ++i)
613 {
614 switch (m_mapping[i])
615 {
616 case ABS_X: caps.axes[Joystick::X] = true; break;
617 case ABS_Y: caps.axes[Joystick::Y] = true; break;
618 case ABS_Z:
619 case ABS_THROTTLE: caps.axes[Joystick::Z] = true; break;
620 case ABS_RZ:
621 case ABS_RUDDER: caps.axes[Joystick::R] = true; break;
622 case ABS_RX: caps.axes[Joystick::U] = true; break;
623 case ABS_RY: caps.axes[Joystick::V] = true; break;
624 case ABS_HAT0X: caps.axes[Joystick::PovX] = true; break;
625 case ABS_HAT0Y: caps.axes[Joystick::PovY] = true; break;
626 default: break;
627 }
628 }
629
630 return caps;
631}

641////////////////////////////////////////////////////////////
642JoystickState JoystickImpl::JoystickImpl::update()
643{
644 if (m_file < 0)
645 {
646 m_state = JoystickState();
647 return m_state;
648 }
649
650 // pop events from the joystick file
651 js_event joyState;
652 int result = read(m_file, &joyState, sizeof(joyState));
653 while (result > 0)
654 {
655 switch (joyState.type & ~JS_EVENT_INIT)
656 {
657 // An axis was moved
658 case JS_EVENT_AXIS:
659 {
660 float value = joyState.value * 100.f / 32767.f;
661
662 if (joyState.number < ABS_MAX + 1)
663 {
664 switch (m_mapping[joyState.number])
665 {
666 case ABS_X: m_state.axes[Joystick::X] = value; break;
667 case ABS_Y: m_state.axes[Joystick::Y] = value; break;
668 case ABS_Z:
669 case ABS_THROTTLE: m_state.axes[Joystick::Z] = value; break;
670 case ABS_RZ:
671 case ABS_RUDDER: m_state.axes[Joystick::R] = value; break;
672 case ABS_RX: m_state.axes[Joystick::U] = value; break;
673 case ABS_RY: m_state.axes[Joystick::V] = value; break;
674 case ABS_HAT0X: m_state.axes[Joystick::PovX] = value; break;
675 case ABS_HAT0Y: m_state.axes[Joystick::PovY] = value; break;
676 default: break;
677 }
678 }
679 break;
680 }
681
682 // A button was pressed
683 case JS_EVENT_BUTTON:
684 {
685 if (joyState.number < Joystick::ButtonCount)
686 m_state.buttons[joyState.number] = (joyState.value != 0);
687 break;
688 }
689 }
690
691 result = read(m_file, &joyState, sizeof(joyState));
692 }
693
694 // Check the connection state of the joystick
695 // read() returns -1 and errno != EGAIN if it's no longer connected
696 // We need to check the result of read() as well, since errno could
697 // have been previously set by some other function call that failed
698 // result can be either negative or 0 at this point
699 // If result is 0, assume the joystick is still connected
700 // If result is negative, check errno and disconnect if it is not EAGAIN
701 m_state.connected = (!result || (errno == EAGAIN));
702
703 return m_state;
704}
