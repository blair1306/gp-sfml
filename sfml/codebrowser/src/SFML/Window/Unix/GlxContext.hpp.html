25#ifndef SFML_GLXCONTEXT_HPP
26#define SFML_GLXCONTEXT_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Window/GlContext.hpp>
32#include <SFML/Window/Unix/GlxExtensions.hpp>
33#include <X11/Xlib.h>
34
35
36namespace sf
37{
38namespace priv
39{
40////////////////////////////////////////////////////////////
41/// \brief Linux (GLX) implementation of OpenGL contexts
42///
43////////////////////////////////////////////////////////////
44class GlxContext : public GlContext
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Create a new default context
50 ///
51 /// \param shared Context to share the new one with (can be NULL)
52 ///
53 ////////////////////////////////////////////////////////////
54 GlxContext(GlxContext* shared);
55
56 ////////////////////////////////////////////////////////////
57 /// \brief Create a new context attached to a window
58 ///
59 /// \param shared Context to share the new one with
60 /// \param settings Creation parameters
61 /// \param owner Pointer to the owner window
62 /// \param bitsPerPixel Pixel depth, in bits per pixel
63 ///
64 ////////////////////////////////////////////////////////////
65 GlxContext(GlxContext* shared, const ContextSettings& settings, const WindowImpl* owner, unsigned int bitsPerPixel);
66
67 ////////////////////////////////////////////////////////////
68 /// \brief Create a new context that embeds its own rendering target
69 ///
70 /// \param shared Context to share the new one with
71 /// \param settings Creation parameters
72 /// \param width Back buffer width, in pixels
73 /// \param height Back buffer height, in pixels
74 ///
75 ////////////////////////////////////////////////////////////
76 GlxContext(GlxContext* shared, const ContextSettings& settings, unsigned int width, unsigned int height);
77
78 ////////////////////////////////////////////////////////////
79 /// \brief Destructor
80 ///
81 ////////////////////////////////////////////////////////////
82 ~GlxContext();
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Get the address of an OpenGL function
86 ///
87 /// \param name Name of the function to get the address of
88 ///
89 /// \return Address of the OpenGL function, 0 on failure
90 ///
91 ////////////////////////////////////////////////////////////
92 static GlFunctionPointer getFunction(const char* name);
93
94 ////////////////////////////////////////////////////////////
95 /// \brief Activate the context as the current target for rendering
96 ///
97 /// \param current Whether to make the context current or no longer current
98 ///
99 /// \return True on success, false if any error happened
100 ///
101 ////////////////////////////////////////////////////////////
102 virtual bool makeCurrent(bool current);
103
104 ////////////////////////////////////////////////////////////
105 /// \brief Display what has been rendered to the context so far
106 ///
107 ////////////////////////////////////////////////////////////
108 virtual void display();
109
110 ////////////////////////////////////////////////////////////
111 /// \brief Enable or disable vertical synchronization
112 ///
113 /// Activating vertical synchronization will limit the number
114 /// of frames displayed to the refresh rate of the monitor.
115 /// This can avoid some visual artifacts, and limit the framerate
116 /// to a good value (but not constant across different computers).
117 ///
118 /// \param enabled True to enable v-sync, false to deactivate
119 ///
120 ////////////////////////////////////////////////////////////
121 virtual void setVerticalSyncEnabled(bool enabled);
122
123 ////////////////////////////////////////////////////////////
124 /// \brief Select the best GLX visual for a given set of settings
125 ///
126 /// \param display X display
127 /// \param bitsPerPixel Pixel depth, in bits per pixel
128 /// \param settings Requested context settings
129 ///
130 /// \return The best visual
131 ///
132 ////////////////////////////////////////////////////////////
133 static XVisualInfo selectBestVisual(::Display* display, unsigned int bitsPerPixel, const ContextSettings& settings);
134
135private:
136
137 ////////////////////////////////////////////////////////////
138 /// \brief Update the context visual settings from XVisualInfo
139 ///
140 /// \param visualInfo XVisualInfo to update settings from
141 ///
142 ////////////////////////////////////////////////////////////
143 void updateSettingsFromVisualInfo(XVisualInfo* visualInfo);
144
145 ////////////////////////////////////////////////////////////
146 /// \brief Update the context visual settings from the window
147 ///
148 ////////////////////////////////////////////////////////////
149 void updateSettingsFromWindow();
150
151 ////////////////////////////////////////////////////////////
152 /// \brief Create the context's drawing surface
153 ///
154 /// \param shared Context to share the new one with (can be NULL)
155 /// \param width Back buffer width, in pixels
156 /// \param height Back buffer height, in pixels
157 /// \param bitsPerPixel Pixel depth, in bits per pixel
158 ///
159 ////////////////////////////////////////////////////////////
160 void createSurface(GlxContext* shared, unsigned int width, unsigned int height, unsigned int bitsPerPixel);
161
162 ////////////////////////////////////////////////////////////
163 /// \brief Create the context's drawing surface from an existing window
164 ///
165 /// \param window Window ID of the owning window
166 ///
167 ////////////////////////////////////////////////////////////
168 void createSurface(::Window window);
169
170 ////////////////////////////////////////////////////////////
171 /// \brief Create the context
172 ///
173 /// \param shared Context to share the new one with (can be NULL)
174 ///
175 ////////////////////////////////////////////////////////////
176 void createContext(GlxContext* shared);
177
178 ////////////////////////////////////////////////////////////
179 // Member data
180 ////////////////////////////////////////////////////////////
181 ::Display* m_display; ///< Connection to the X server
182 ::Window m_window; ///< Window to which the context is attached
183 GLXContext m_context; ///< OpenGL context
184 GLXPbuffer m_pbuffer; ///< GLX pbuffer ID if one was created
185 bool m_ownsWindow; ///< Do we own the window associated to the context?
186};
187
188} // namespace priv
189
190} // namespace sf
191
192#endif // SFML_GLXCONTEXT_HPP
