121typedef int (*PFN_LOADFUNCPOINTERS)(void);
122typedef struct sfglx_StrToExtMap_s
123{
124 const char* extensionName;
125 int* extensionVariable;
126 PFN_LOADFUNCPOINTERS LoadExtension;
127} sfglx_StrToExtMap;
128
129static sfglx_StrToExtMap ExtensionMap[9] = {
130 {"GLX_EXT_swap_control", &sfglx_ext_EXT_swap_control, Load_EXT_swap_control},
131 {"GLX_MESA_swap_control", &sfglx_ext_MESA_swap_control, Load_MESA_swap_control},
132 {"GLX_SGI_swap_control", &sfglx_ext_SGI_swap_control, Load_SGI_swap_control},
133 {"GLX_EXT_framebuffer_sRGB", &sfglx_ext_EXT_framebuffer_sRGB, NULL},
134 {"GLX_ARB_framebuffer_sRGB", &sfglx_ext_ARB_framebuffer_sRGB, NULL},
135 {"GLX_ARB_multisample", &sfglx_ext_ARB_multisample, NULL},
136 {"GLX_SGIX_pbuffer", &sfglx_ext_SGIX_pbuffer, Load_SGIX_pbuffer},
137 {"GLX_ARB_create_context", &sfglx_ext_ARB_create_context, Load_ARB_create_context},
138 {"GLX_ARB_create_context_profile", &sfglx_ext_ARB_create_context_profile, NULL}
139};
140
141static int g_extensionMapSize = 9;
142
143
144static sfglx_StrToExtMap* FindExtEntry(const char* extensionName)
145{
146 sfglx_StrToExtMap* currLoc = ExtensionMap;
147 for (int loop = 0; loop < g_extensionMapSize; ++loop, ++currLoc)
148 {
149 if (std::strcmp(extensionName, currLoc->extensionName) == 0)
150 return currLoc;
151 }
152
153 return NULL;
154}

171static void LoadExtByName(const char* extensionName)
172{
173 sfglx_StrToExtMap* entry = NULL;
174 entry = FindExtEntry(extensionName);
175 if (entry)
176 {
177 if (entry->LoadExtension)
178 {
179 int numFailed = entry->LoadExtension();
180 if (numFailed == 0)
181 {
182 *(entry->extensionVariable) = sfglx_LOAD_SUCCEEDED;
183 }
184 else
185 {
186 *(entry->extensionVariable) = sfglx_LOAD_SUCCEEDED + numFailed;
187 }
188 }
189 else
190 {
191 *(entry->extensionVariable) = sfglx_LOAD_SUCCEEDED;
192 }
193 }
194}

197static void ProcExtsFromExtString(const char* strExtList)
198{
199 do
200 {
201 const char* begin = strExtList;
202
203 while ((*strExtList != ' ') && *strExtList)
204 strExtList++;
205
206 LoadExtByName(std::string(begin, strExtList).c_str());
207 } while (*strExtList++);
208}

210
211int sfglx_LoadFunctions(Display* display, int screen)
212{
213 ClearExtensionVars();
214
215
216 ProcExtsFromExtString(reinterpret_cast<const char*>(glXQueryExtensionsString(display, screen)));
217 return sfglx_LOAD_SUCCEEDED;
218}
