40/// \brief Get the shared Display
41///
42/// This function increments the reference count of the display,
43/// it must be matched with a call to CloseDisplay.
44///
45/// \return Pointer to the shared display
46///
47////////////////////////////////////////////////////////////

50////////////////////////////////////////////////////////////

52///
53/// \param display Display to release
54///
55////////////////////////////////////////////////////////////
56void CloseDisplay(Display* display);

58////////////////////////////////////////////////////////////

61/// \param name Name of the atom

63///
64/// \return Atom if it exists or None (0) if it doesn't
65///
66////////////////////////////////////////////////////////////
67Atom getAtom(const std::string& name, bool onlyIfExists = false);
