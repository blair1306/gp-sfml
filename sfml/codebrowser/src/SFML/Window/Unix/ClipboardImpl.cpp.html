40void initClipboard();
41void* hostSelection(void*);
42
43sf::String string;
44
45pthread_mutex_t mutex;
46pthread_t host_thread;
47
48bool is_fail = false;
49bool is_init = false;
50bool is_host = false;
51
52Display* display = NULL;
53Window window = 0;
54
55Atom selection = 0;
56Atom atom_targ = 0;
57Atom atom_text = 0;
58Atom utf8_text = 0;
59int xa_string = 31;
60int xa_atom = 4;
61
62////////////////////////////////////////////////////////////
63void initClipboard()
64{
65 is_init = true;
66
67 display = XOpenDisplay(NULL);
68 int screen = DefaultScreen(display);
69 window = XCreateSimpleWindow(display, RootWindow(display, screen),
70 0, 0, 1, 1, 0, BlackPixel(display, screen), WhitePixel(display, screen));
71
72 selection = XInternAtom(display, "CLIPBOARD", false);
73 atom_targ = XInternAtom(display, "TARGETS", false);
74 atom_text = XInternAtom(display, "TEXT", false);
75 utf8_text = XInternAtom(display, "UTF8_STRING", true);
76
77 if(utf8_text == None)
78 {
79 std::cerr << "UTF-8 format unavailable on clipboard." << std::endl;
80 utf8_text = xa_string;
81 }
82
83 if(pthread_mutex_init(&mutex, NULL))
84 {
85 is_fail = true;
86 std::cerr << "Unable to initialize mutex. Failed to initialize clipboard." << std::endl;
87 return;
88 }
89
90 if(pthread_create(&host_thread, NULL, hostSelection, NULL))
91 {
92 is_fail = true;
93 std::cerr << "Unable to create host thread. Failed to initialize clipboard." << std::endl;
94 return;
95 }
96}

98////////////////////////////////////////////////////////////
99void* hostSelection(void*)
100{
101 while(true)
102 {
103 if(XPending(display) && is_host)
104 {
105 XEvent event;
106
107 pthread_mutex_lock(&mutex);
108 XNextEvent(display, &event);
109 pthread_mutex_unlock(&mutex);
110
111 switch(event.type)
112 {
113 case SelectionClear:
114 {
115 pthread_mutex_lock(&mutex);
116 is_host = false;
117 pthread_mutex_unlock(&mutex);
118
119 break;
120 }
121 case SelectionRequest:
122 {
123 if(event.xselectionrequest.selection == selection)
124 {
125 XSelectionRequestEvent* sel_req_event = &event.xselectionrequest;
126 XSelectionEvent sel_event = {0};
127
128 int result = 0;
129 sel_event.type = SelectionNotify,
130 sel_event.display = sel_req_event->display,
131 sel_event.requestor = sel_req_event->requestor,
132 sel_event.selection = sel_req_event->selection,
133 sel_event.time = sel_req_event->time,
134 sel_event.target = sel_req_event->target,
135 sel_event.property = sel_req_event->property;
136
137 std::basic_string<unsigned char> str = string.toUtf8();
138
139 if(sel_event.target == atom_targ)
140 result = XChangeProperty(sel_event.display, sel_event.requestor,
141 sel_event.property, xa_atom, 32, PropModeReplace,
142 reinterpret_cast<unsigned char*>(&utf8_text), 1);
143 else if(sel_event.target == xa_string || sel_event.target == atom_text)
144 result = XChangeProperty(sel_event.display, sel_event.requestor,
145 sel_event.property, xa_string, 8, PropModeReplace,
146 reinterpret_cast<unsigned char*>(&str[0]), str.size());
147 else if(sel_event.target == utf8_text)
148 result = XChangeProperty(sel_event.display, sel_event.requestor,
149 sel_event.property, utf8_text, 8, PropModeReplace,
150 reinterpret_cast<unsigned char*>(&str[0]), str.size());
151 else
152 sel_event.property = None;
153
154 if((result & 2) == 0)
155 XSendEvent(display, sel_event.requestor, 0, 0,
156 reinterpret_cast<XEvent*>(&sel_event));
157 }
158 break;
159 }
160 default: break;
161 }
162 }
163 else
164 sf::sleep(sf::milliseconds(20));
165 }
166}

174////////////////////////////////////////////////////////////
175String ClipboardImpl::getString()
176{
177 if(!is_init)
178 initClipboard();
179
180 if(is_fail || is_host)
181 return string;
182
183 // Dangerous! Wipes all previous events!
184 XSync(display, true);
185 XConvertSelection(display, selection, utf8_text, atom_text, window, CurrentTime);
186
187 XEvent event;
188
189 pthread_mutex_lock(&mutex);
190 XNextEvent(display, &event);
191 pthread_mutex_unlock(&mutex);
192
193 if(event.type == SelectionNotify)
194 {
195 if(event.xselection.selection != selection || event.xselection.target != utf8_text)
196 {
197 std::cerr << "Failed to convert selection." << std::endl;
198 return string;
199 }
200
201 if(event.xselection.property)
202 {
203 Atom target;
204 int format;
205 unsigned long size;
206 unsigned long byte_left;
207 unsigned char* data;
208
209 XGetWindowProperty(event.xselection.display,
210 event.xselection.requestor, event.xselection.property,
211 0L, (~0L), false, AnyPropertyType,
212 &target, &format, &size, &byte_left, &data);
213
214 if(target == utf8_text)
215 {
216 std::basic_string<unsigned char> str(data, size);
217 string = sf::String::fromUtf8(str.begin(), str.end());
218
219 XFree(data);
220 }
221
222 XDeleteProperty(event.xselection.display, event.xselection.requestor, event.xselection.property);
223 }
224 }
225
226 return string;
227}

230////////////////////////////////////////////////////////////
231void ClipboardImpl::setString(const String& text)
232{
233 if(!is_init)
234 initClipboard();
235
236 if(is_fail)
237 return;
238
239 if(!is_host)
240 {
241 XSetSelectionOwner(display, selection, window, CurrentTime);
242
243 if(XGetSelectionOwner(display, selection) != window)
244 {
245 std::cerr << "Unable to get ownership of selection." << std::endl;
246 return;
247 }
248
249 pthread_mutex_lock(&mutex);
250 is_host = true;
251 pthread_mutex_unlock(&mutex);
252 }
253
254 pthread_mutex_lock(&mutex);
255 string = text;
256 pthread_mutex_unlock(&mutex);
257}
