25#ifndef SFML_WINDOWIMPLX11_HPP
26#define SFML_WINDOWIMPLX11_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Window/Event.hpp>
32#include <SFML/Window/WindowImpl.hpp>
33#include <SFML/System/String.hpp>
34#include <SFML/Window/WindowStyle.hpp> // Prevent conflict with macro None from Xlib
35#include <X11/Xlib.h>
36#include <deque>
37
38
39namespace sf
40{
41namespace priv
42{
43////////////////////////////////////////////////////////////
44/// \brief Linux (X11) implementation of WindowImpl
45///
46////////////////////////////////////////////////////////////
47class WindowImplX11 : public WindowImpl
48{
49public:
50
51 ////////////////////////////////////////////////////////////
52 /// \brief Construct the window implementation from an existing control
53 ///
54 /// \param handle Platform-specific handle of the control
55 ///
56 ////////////////////////////////////////////////////////////
57 WindowImplX11(WindowHandle handle);
58
59 ////////////////////////////////////////////////////////////
60 /// \brief Create the window implementation
61 ///
62 /// \param mode Video mode to use
63 /// \param title Title of the window
64 /// \param style Window style (resizable, fixed, or fullscren)
65 /// \param settings Additional settings for the underlying OpenGL context
66 ///
67 ////////////////////////////////////////////////////////////
68 WindowImplX11(VideoMode mode, const String& title, unsigned long style, const ContextSettings& settings);
69
70 ////////////////////////////////////////////////////////////
71 /// \brief Destructor
72 ///
73 ////////////////////////////////////////////////////////////
74 ~WindowImplX11();
75
76 ////////////////////////////////////////////////////////////
77 /// \brief Get the OS-specific handle of the window
78 ///
79 /// \return Handle of the window
80 ///
81 ////////////////////////////////////////////////////////////
82 virtual WindowHandle getSystemHandle() const;
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Get the position of the window
86 ///
87 /// \return Position of the window, in pixels
88 ///
89 ////////////////////////////////////////////////////////////
90 virtual Vector2i getPosition() const;
91
92 ////////////////////////////////////////////////////////////
93 /// \brief Change the position of the window on screen
94 ///
95 /// \param position New position of the window, in pixels
96 ///
97 ////////////////////////////////////////////////////////////
98 virtual void setPosition(const Vector2i& position);
99
100 ////////////////////////////////////////////////////////////
101 /// \brief Get the client size of the window
102 ///
103 /// \return Size of the window, in pixels
104 ///
105 ////////////////////////////////////////////////////////////
106 virtual Vector2u getSize() const;
107
108 ////////////////////////////////////////////////////////////
109 /// \brief Change the size of the rendering region of the window
110 ///
111 /// \param size New size, in pixels
112 ///
113 ////////////////////////////////////////////////////////////
114 virtual void setSize(const Vector2u& size);
115
116 ////////////////////////////////////////////////////////////
117 /// \brief Change the title of the window
118 ///
119 /// \param title New title
120 ///
121 ////////////////////////////////////////////////////////////
122 virtual void setTitle(const String& title);
123
124 ////////////////////////////////////////////////////////////
125 /// \brief Change the window's icon
126 ///
127 /// \param width Icon's width, in pixels
128 /// \param height Icon's height, in pixels
129 /// \param pixels Pointer to the pixels in memory, format must be RGBA 32 bits
130 ///
131 ////////////////////////////////////////////////////////////
132 virtual void setIcon(unsigned int width, unsigned int height, const Uint8* pixels);
133
134 ////////////////////////////////////////////////////////////
135 /// \brief Show or hide the window
136 ///
137 /// \param visible True to show, false to hide
138 ///
139 ////////////////////////////////////////////////////////////
140 virtual void setVisible(bool visible);
141
142 ////////////////////////////////////////////////////////////
143 /// \brief Show or hide the mouse cursor
144 ///
145 /// \param visible True to show, false to hide
146 ///
147 ////////////////////////////////////////////////////////////
148 virtual void setMouseCursorVisible(bool visible);
149
150 ////////////////////////////////////////////////////////////
151 /// \brief Grab or release the mouse cursor
152 ///
153 /// \param grabbed True to enable, false to disable
154 ///
155 ////////////////////////////////////////////////////////////
156 virtual void setMouseCursorGrabbed(bool grabbed);
157
158 ////////////////////////////////////////////////////////////
159 /// \brief Set the displayed cursor to a native system cursor
160 ///
161 /// \param cursor Native system cursor type to display
162 ///
163 ////////////////////////////////////////////////////////////
164 virtual void setMouseCursor(const CursorImpl& cursor);
165
166 ////////////////////////////////////////////////////////////
167 /// \brief Enable or disable automatic key-repeat
168 ///
169 /// \param enabled True to enable, false to disable
170 ///
171 ////////////////////////////////////////////////////////////
172 virtual void setKeyRepeatEnabled(bool enabled);
173
174 ////////////////////////////////////////////////////////////
175 /// \brief Request the current window to be made the active
176 /// foreground window
177 ///
178 ////////////////////////////////////////////////////////////
179 virtual void requestFocus();
180
181 ////////////////////////////////////////////////////////////
182 /// \brief Check whether the window has the input focus
183 ///
184 /// \return True if window has focus, false otherwise
185 ///
186 ////////////////////////////////////////////////////////////
187 virtual bool hasFocus() const;
188
189protected:
190
191 ////////////////////////////////////////////////////////////
192 /// \brief Process incoming events from the operating system
193 ///
194 ////////////////////////////////////////////////////////////
195 virtual void processEvents();
196
197private:
198
199 ////////////////////////////////////////////////////////////
200 /// \brief Request the WM to make the current window active
201 ///
202 ////////////////////////////////////////////////////////////
203 void grabFocus();
204
205 ////////////////////////////////////////////////////////////
206 /// \brief Set fullscreen video mode
207 ///
208 /// \param Mode video mode to switch to
209 ///
210 ////////////////////////////////////////////////////////////
211 void setVideoMode(const VideoMode& mode);
212
213 ////////////////////////////////////////////////////////////
214 /// \brief Reset to desktop video mode
215 ///
216 ////////////////////////////////////////////////////////////
217 void resetVideoMode();
218
219 ////////////////////////////////////////////////////////////
220 /// \brief Switch to fullscreen mode
221 ///
222 ////////////////////////////////////////////////////////////
223 void switchToFullscreen();
224
225 ////////////////////////////////////////////////////////////
226 /// \brief Set the WM protocols we support
227 ///
228 ////////////////////////////////////////////////////////////
229 void setProtocols();
230
231 ////////////////////////////////////////////////////////////
232 /// \brief Update the last time we received user input
233 ///
234 /// \param time Last time we received user input
235 ///
236 ////////////////////////////////////////////////////////////
237 void updateLastInputTime(::Time time);
238
239 ////////////////////////////////////////////////////////////
240 /// \brief Do some common initializations after the window has been created
241 ///
242 ////////////////////////////////////////////////////////////
243 void initialize();
244
245 ////////////////////////////////////////////////////////////
246 /// \brief Create a transparent mouse cursor
247 ///
248 ////////////////////////////////////////////////////////////
249 void createHiddenCursor();
250
251 ////////////////////////////////////////////////////////////
252 /// \brief Cleanup graphical resources attached to the window
253 ///
254 ////////////////////////////////////////////////////////////
255 void cleanup();
256
257 ////////////////////////////////////////////////////////////
258 /// \brief Process an incoming event from the window
259 ///
260 /// \param windowEvent Event which has been received
261 ///
262 /// \return True if the event was processed, false if it was discarded
263 ///
264 ////////////////////////////////////////////////////////////
265 bool processEvent(XEvent& windowEvent);
266
267 ////////////////////////////////////////////////////////////
268 // Member data
269 ////////////////////////////////////////////////////////////
270 ::Window m_window; ///< X identifier defining our window
271 ::Display* m_display; ///< Pointer to the display
272 int m_screen; ///< Screen identifier
273 XIM m_inputMethod; ///< Input method linked to the X display
274 XIC m_inputContext; ///< Input context used to get unicode input in our window
275 std::deque<XEvent> m_events; ///< Queue we use to store pending events for this window
276 bool m_isExternal; ///< Tell whether the window has been created externally or by SFML
277 int m_oldVideoMode; ///< Video mode in use before we switch to fullscreen
278 ::Cursor m_hiddenCursor; ///< As X11 doesn't provide cursor hiding, we must create a transparent one
279 ::Cursor m_lastCursor; ///< Last cursor used -- this data is not owned by the window and is required to be always valid
280 bool m_keyRepeat; ///< Is the KeyRepeat feature enabled?
281 Vector2i m_previousSize; ///< Previous size of the window, to find if a ConfigureNotify event is a resize event (could be a move event only)
282 bool m_useSizeHints; ///< Is the size of the window fixed with size hints?
283 bool m_fullscreen; ///< Is the window in fullscreen?
284 bool m_cursorGrabbed; ///< Is the mouse cursor trapped?
285 bool m_windowMapped; ///< Has the window been mapped by the window manager?
286 Pixmap m_iconPixmap; ///< The current icon pixmap if in use
287 Pixmap m_iconMaskPixmap; ///< The current icon mask pixmap if in use
288 ::Time m_lastInputTime; ///< Last time we received user input
289};
290
291} // namespace priv
292
293} // namespace sf
294
295
296#endif // SFML_WINDOWIMPLX11_HPP
