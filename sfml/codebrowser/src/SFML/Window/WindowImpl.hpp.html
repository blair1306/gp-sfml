31#include <SFML/Config.hpp>
32#include <SFML/System/NonCopyable.hpp>
33#include <SFML/System/String.hpp>
34#include <SFML/Window/ContextSettings.hpp>
35#include <SFML/Window/CursorImpl.hpp>
36#include <SFML/Window/Event.hpp>
37#include <SFML/Window/Joystick.hpp>
38#include <SFML/Window/JoystickImpl.hpp>
39#include <SFML/Window/Sensor.hpp>
40#include <SFML/Window/SensorImpl.hpp>
41#include <SFML/Window/VideoMode.hpp>
42#include <SFML/Window/WindowHandle.hpp>
43#include <SFML/Window/Window.hpp>
44#include <queue>
45#include <set>
46
47namespace sf
48{
49class WindowListener;
50
51namespace priv
52{
53////////////////////////////////////////////////////////////
54/// \brief Abstract base class for OS-specific window implementation
55///
56////////////////////////////////////////////////////////////
57class WindowImpl : NonCopyable
58{
59public:
60
61 ////////////////////////////////////////////////////////////
62 /// \brief Create a new window depending on the current OS
63 ///
64 /// \param mode Video mode to use
65 /// \param title Title of the window
66 /// \param style Window style
67 /// \param settings Additional settings for the underlying OpenGL context
68 ///
69 /// \return Pointer to the created window (don't forget to delete it)
70 ///
71 ////////////////////////////////////////////////////////////
72 static WindowImpl* create(VideoMode mode, const String& title, Uint32 style, const ContextSettings& settings);
73
74 ////////////////////////////////////////////////////////////
75 /// \brief Create a new window depending on to the current OS
76 ///
77 /// \param handle Platform-specific handle of the control
78 ///
79 /// \return Pointer to the created window (don't forget to delete it)
80 ///
81 ////////////////////////////////////////////////////////////
82 static WindowImpl* create(WindowHandle handle);
83
84public:
85
86 ////////////////////////////////////////////////////////////
87 /// \brief Destructor
88 ///
89 ////////////////////////////////////////////////////////////
90 virtual ~WindowImpl();
91
92 ////////////////////////////////////////////////////////////
93 /// \brief Change the joystick threshold, i.e. the value below which
94 /// no move event will be generated
95 ///
96 /// \param threshold New threshold, in range [0, 100]
97 ///
98 ////////////////////////////////////////////////////////////
99 void setJoystickThreshold(float threshold);
100
101 ////////////////////////////////////////////////////////////
102 /// \brief Return the next window event available
103 ///
104 /// If there's no event available, this function calls the
105 /// window's internal event processing function.
106 /// The \a block parameter controls the behavior of the function
107 /// if no event is available: if it is true then the function
108 /// doesn't return until a new event is triggered; otherwise it
109 /// returns false to indicate that no event is available.
110 ///
111 /// \param event Event to be returned
112 /// \param block Use true to block the thread until an event arrives
113 ///
114 ////////////////////////////////////////////////////////////
115 bool popEvent(Event& event, bool block);
116
117 ////////////////////////////////////////////////////////////
118 /// \brief Get the OS-specific handle of the window
119 ///
120 /// \return Handle of the window
121 ///
122 ////////////////////////////////////////////////////////////
123 virtual WindowHandle getSystemHandle() const = 0;
124
125 ////////////////////////////////////////////////////////////
126 /// \brief Get the position of the window
127 ///
128 /// \return Position of the window, in pixels
129 ///
130 ////////////////////////////////////////////////////////////
131 virtual Vector2i getPosition() const = 0;
132
133 ////////////////////////////////////////////////////////////
134 /// \brief Change the position of the window on screen
135 ///
136 /// \param position New position of the window, in pixels
137 ///
138 ////////////////////////////////////////////////////////////
139 virtual void setPosition(const Vector2i& position) = 0;
140
141 ////////////////////////////////////////////////////////////
142 /// \brief Get the client size of the window
143 ///
144 /// \return Size of the window, in pixels
145 ///
146 ////////////////////////////////////////////////////////////
147 virtual Vector2u getSize() const = 0;
148
149 ////////////////////////////////////////////////////////////
150 /// \brief Change the size of the rendering region of the window
151 ///
152 /// \param size New size, in pixels
153 ///
154 ////////////////////////////////////////////////////////////
155 virtual void setSize(const Vector2u& size) = 0;
156
157 ////////////////////////////////////////////////////////////
158 /// \brief Change the title of the window
159 ///
160 /// \param title New title
161 ///
162 ////////////////////////////////////////////////////////////
163 virtual void setTitle(const String& title) = 0;
164
165 ////////////////////////////////////////////////////////////
166 /// \brief Change the window's icon
167 ///
168 /// \param width Icon's width, in pixels
169 /// \param height Icon's height, in pixels
170 /// \param pixels Pointer to the pixels in memory, format must be RGBA 32 bits
171 ///
172 ////////////////////////////////////////////////////////////
173 virtual void setIcon(unsigned int width, unsigned int height, const Uint8* pixels) = 0;
174
175 ////////////////////////////////////////////////////////////
176 /// \brief Show or hide the window
177 ///
178 /// \param visible True to show, false to hide
179 ///
180 ////////////////////////////////////////////////////////////
181 virtual void setVisible(bool visible) = 0;
182
183 ////////////////////////////////////////////////////////////
184 /// \brief Show or hide the mouse cursor
185 ///
186 /// \param visible True to show, false to hide
187 ///
188 ////////////////////////////////////////////////////////////
189 virtual void setMouseCursorVisible(bool visible) = 0;
190
191 ////////////////////////////////////////////////////////////
192 /// \brief Grab or release the mouse cursor and keeps it from leaving
193 ///
194 /// \param grabbed True to enable, false to disable
195 ///
196 ////////////////////////////////////////////////////////////
197 virtual void setMouseCursorGrabbed(bool grabbed) = 0;
198
199 ////////////////////////////////////////////////////////////
200 /// \brief Set the displayed cursor to a native system cursor
201 ///
202 /// \param cursor Native system cursor type to display
203 ///
204 ////////////////////////////////////////////////////////////
205 virtual void setMouseCursor(const CursorImpl& cursor) = 0;
206
207 ////////////////////////////////////////////////////////////
208 /// \brief Enable or disable automatic key-repeat
209 ///
210 /// \param enabled True to enable, false to disable
211 ///
212 ////////////////////////////////////////////////////////////
213 virtual void setKeyRepeatEnabled(bool enabled) = 0;
214
215 ////////////////////////////////////////////////////////////
216 /// \brief Request the current window to be made the active
217 /// foreground window
218 ///
219 ////////////////////////////////////////////////////////////
220 virtual void requestFocus() = 0;
221
222 ////////////////////////////////////////////////////////////
223 /// \brief Check whether the window has the input focus
224 ///
225 /// \return True if window has focus, false otherwise
226 ///
227 ////////////////////////////////////////////////////////////
228 virtual bool hasFocus() const = 0;
229
230protected:
231
232 ////////////////////////////////////////////////////////////
233 /// \brief Default constructor
234 ///
235 ////////////////////////////////////////////////////////////
236 WindowImpl();
237
238 ////////////////////////////////////////////////////////////
239 /// \brief Push a new event into the event queue
240 ///
241 /// This function is to be used by derived classes, to
242 /// notify the SFML window that a new event was triggered
243 /// by the system.
244 ///
245 /// \param event Event to push
246 ///
247 ////////////////////////////////////////////////////////////
248 void pushEvent(const Event& event);
249
250 ////////////////////////////////////////////////////////////
251 /// \brief Process incoming events from the operating system
252 ///
253 ////////////////////////////////////////////////////////////
254 virtual void processEvents() = 0;
255
256private:
257
258 ////////////////////////////////////////////////////////////
259 /// \brief Read the joysticks state and generate the appropriate events
260 ///
261 ////////////////////////////////////////////////////////////
262 void processJoystickEvents();
263
264 ////////////////////////////////////////////////////////////
265 /// \brief Read the sensors state and generate the appropriate events
266 ///
267 ////////////////////////////////////////////////////////////
268 void processSensorEvents();
269
270 ////////////////////////////////////////////////////////////
271 // Member data
272 ////////////////////////////////////////////////////////////
273 std::queue<Event> m_events; ///< Queue of available events
274 JoystickState m_joystickStates[Joystick::Count]; ///< Previous state of the joysticks
275 Vector3f m_sensorValue[Sensor::Count]; ///< Previous value of the sensors
276 float m_joystickThreshold; ///< Joystick threshold (minimum motion for "move" event to be generated)
277};
278
279} // namespace priv
280
281} // namespace sf
282
283
284#endif // SFML_WINDOWIMPL_HPP
