83WindowImpl::WindowImpl() :
84m_joystickThreshold(0.1f)
85{
86 // Get the initial joystick states
87 JoystickManager::getInstance().update();
88 for (unsigned int i = 0; i < Joystick::Count; ++i)
89 m_joystickStates[i] = JoystickManager::getInstance().getState(i);
90
91 // Get the initial sensor states
92 for (unsigned int i = 0; i < Sensor::Count; ++i)
93 m_sensorValue[i] = Vector3f(0, 0, 0);
94}

112bool WindowImpl::popEvent(Event& event, bool block)
113{
114 // If the event queue is empty, let's first check if new events are available from the OS
115 if (m_events.empty())
116 {
117 // Get events from the system
118 processJoystickEvents();
119 processSensorEvents();
120 processEvents();
121
122 // In blocking mode, we must process events until one is triggered
123 if (block)
124 {
125 // Here we use a manual wait loop instead of the optimized
126 // wait-event provided by the OS, so that we don't skip joystick
127 // events (which require polling)
128 while (m_events.empty())
129 {
130 sleep(milliseconds(10));
131 processJoystickEvents();
132 processSensorEvents();
133 processEvents();
134 }
135 }
136 }
137
138 // Pop the first event of the queue, if it is not empty
139 if (!m_events.empty())
140 {
141 event = m_events.front();
142 m_events.pop();
143
144 return true;
145 }
146
147 return false;
148}

158////////////////////////////////////////////////////////////
159void WindowImpl::processJoystickEvents()
160{
161 // First update the global joystick states
162 JoystickManager::getInstance().update();
163
164 for (unsigned int i = 0; i < Joystick::Count; ++i)
165 {
166 // Copy the previous state of the joystick and get the new one
167 JoystickState previousState = m_joystickStates[i];
168 m_joystickStates[i] = JoystickManager::getInstance().getState(i);
169 JoystickCaps caps = JoystickManager::getInstance().getCapabilities(i);
170
171 // Connection state
172 bool connected = m_joystickStates[i].connected;
173 if (previousState.connected ^ connected)
174 {
175 Event event;
176 event.type = connected ? Event::JoystickConnected : Event::JoystickDisconnected;
177 event.joystickButton.joystickId = i;
178 pushEvent(event);
179 }
180
181 if (connected)
182 {
183 // Axes
184 for (unsigned int j = 0; j < Joystick::AxisCount; ++j)
185 {
186 if (caps.axes[j])
187 {
188 Joystick::Axis axis = static_cast<Joystick::Axis>(j);
189 float prevPos = previousState.axes[axis];
190 float currPos = m_joystickStates[i].axes[axis];
191 if (fabs(currPos - prevPos) >= m_joystickThreshold)
192 {
193 Event event;
194 event.type = Event::JoystickMoved;
195 event.joystickMove.joystickId = i;
196 event.joystickMove.axis = axis;
197 event.joystickMove.position = currPos;
198 pushEvent(event);
199 }
200 }
201 }
202
203 // Buttons
204 for (unsigned int j = 0; j < caps.buttonCount; ++j)
205 {
206 bool prevPressed = previousState.buttons[j];
207 bool currPressed = m_joystickStates[i].buttons[j];
208
209 if (prevPressed ^ currPressed)
210 {
211 Event event;
212 event.type = currPressed ? Event::JoystickButtonPressed : Event::JoystickButtonReleased;
213 event.joystickButton.joystickId = i;
214 event.joystickButton.button = j;
215 pushEvent(event);
216 }
217 }
218 }
219 }
220}

223////////////////////////////////////////////////////////////
224void WindowImpl::processSensorEvents()
225{
226 // First update the sensor states
227 SensorManager::getInstance().update();
228
229 for (unsigned int i = 0; i < Sensor::Count; ++i)
230 {
231 Sensor::Type sensor = static_cast<Sensor::Type>(i);
232
233 // Only process enabled sensors
234 if (SensorManager::getInstance().isEnabled(sensor))
235 {
236 // Copy the previous value of the sensor and get the new one
237 Vector3f previousValue = m_sensorValue[i];
238 m_sensorValue[i] = SensorManager::getInstance().getValue(sensor);
239
240 // If the value has changed, trigger an event
241 if (m_sensorValue[i] != previousValue) // @todo use a threshold?
242 {
243 Event event;
244 event.type = Event::SensorChanged;
245 event.sensor.type = sensor;
246 event.sensor.x = m_sensorValue[i].x;
247 event.sensor.y = m_sensorValue[i].y;
248 event.sensor.z = m_sensorValue[i].z;
249 pushEvent(event);
250 }
251 }
252 }
