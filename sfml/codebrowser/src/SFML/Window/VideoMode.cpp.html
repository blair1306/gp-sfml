60 return priv::VideoModeImpl::getDesktopMode();
61}

64////////////////////////////////////////////////////////////
65const std::vector<VideoMode>& VideoMode::getFullscreenModes()
66{
67 static std::vector<VideoMode> modes;
68
69 // Populate the array on first call
70 if (modes.empty())
71 {
72 modes = priv::VideoModeImpl::getFullscreenModes();
73 std::sort(modes.begin(), modes.end(), std::greater<VideoMode>());
74 }
75
76 return modes;
77}
78
79
80////////////////////////////////////////////////////////////
81bool VideoMode::isValid() const
82{
83 const std::vector<VideoMode>& modes = getFullscreenModes();
84
85 return std::find(modes.begin(), modes.end(), *this) != modes.end();
86}

89////////////////////////////////////////////////////////////
90bool operator ==(const VideoMode& left, const VideoMode& right)
91{
92 return (left.width == right.width) &&
93 (left.height == right.height) &&
94 (left.bitsPerPixel == right.bitsPerPixel);
95}
96
97
98////////////////////////////////////////////////////////////
99bool operator !=(const VideoMode& left, const VideoMode& right)
100{
101 return !(left == right);
102}
103
104
105////////////////////////////////////////////////////////////
106bool operator <(const VideoMode& left, const VideoMode& right)
107{
108 if (left.bitsPerPixel == right.bitsPerPixel)
109 {
110 if (left.width == right.width)
111 {
112 return left.height < right.height;
113 }
114 else
115 {
116 return left.width < right.width;
117 }
118 }
119 else
120 {
121 return left.bitsPerPixel < right.bitsPerPixel;
122 }
123}
124
125
126////////////////////////////////////////////////////////////
127bool operator >(const VideoMode& left, const VideoMode& right)
128{
129 return right < left;
130}
131
132
133////////////////////////////////////////////////////////////
134bool operator <=(const VideoMode& left, const VideoMode& right)
135{
136 return !(right < left);
137}
138
139
140////////////////////////////////////////////////////////////
141bool operator >=(const VideoMode& left, const VideoMode& right)
142{
143 return !(left < right);
144}
