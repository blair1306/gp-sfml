25#ifndef SFML_GLCONTEXT_HPP
26#define SFML_GLCONTEXT_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Config.hpp>
32#include <SFML/Window/Context.hpp>
33#include <SFML/Window/ContextSettings.hpp>
34#include <SFML/System/NonCopyable.hpp>
35
36
37namespace sf
38{
39namespace priv
40{
41class WindowImpl;
42
43////////////////////////////////////////////////////////////
44/// \brief Abstract class representing an OpenGL context
45///
46////////////////////////////////////////////////////////////
47class GlContext : NonCopyable
48{
49public:
50
51 ////////////////////////////////////////////////////////////
52 /// \brief Perform resource initialization
53 ///
54 /// This function is called every time an OpenGL resource is
55 /// created. When the first resource is initialized, it makes
56 /// sure that everything is ready for contexts to work properly.
57 ///
58 ////////////////////////////////////////////////////////////
59 static void initResource();
60
61 ////////////////////////////////////////////////////////////
62 /// \brief Perform resource cleanup
63 ///
64 /// This function is called every time an OpenGL resource is
65 /// destroyed. When the last resource is destroyed, it makes
66 /// sure that everything that was created by initResource()
67 /// is properly released.
68 ///
69 ////////////////////////////////////////////////////////////
70 static void cleanupResource();
71
72 ////////////////////////////////////////////////////////////
73 /// \brief Acquires a context for short-term use on the current thread
74 ///
75 ////////////////////////////////////////////////////////////
76 static void acquireTransientContext();
77
78 ////////////////////////////////////////////////////////////
79 /// \brief Releases a context after short-term use on the current thread
80 ///
81 ////////////////////////////////////////////////////////////
82 static void releaseTransientContext();
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Create a new context, not associated to a window
86 ///
87 /// This function automatically chooses the specialized class
88 /// to use according to the OS.
89 ///
90 /// \return Pointer to the created context (don't forget to delete it)
91 ///
92 ////////////////////////////////////////////////////////////
93 static GlContext* create();
94
95 ////////////////////////////////////////////////////////////
96 /// \brief Create a new context attached to a window
97 ///
98 /// This function automatically chooses the specialized class
99 /// to use according to the OS.
100 ///
101 /// \param settings Creation parameters
102 /// \param owner Pointer to the owner window
103 /// \param bitsPerPixel Pixel depth (in bits per pixel)
104 ///
105 /// \return Pointer to the created context
106 ///
107 ////////////////////////////////////////////////////////////
108 static GlContext* create(const ContextSettings& settings, const WindowImpl* owner, unsigned int bitsPerPixel);
109
110 ////////////////////////////////////////////////////////////
111 /// \brief Create a new context that embeds its own rendering target
112 ///
113 /// This function automatically chooses the specialized class
114 /// to use according to the OS.
115 ///
116 /// \param settings Creation parameters
117 /// \param width Back buffer width
118 /// \param height Back buffer height
119 ///
120 /// \return Pointer to the created context
121 ///
122 ////////////////////////////////////////////////////////////
123 static GlContext* create(const ContextSettings& settings, unsigned int width, unsigned int height);
124
125public:
126 ////////////////////////////////////////////////////////////
127 /// \brief Check whether a given OpenGL extension is available
128 ///
129 /// \param name Name of the extension to check for
130 ///
131 /// \return True if available, false if unavailable
132 ///
133 ////////////////////////////////////////////////////////////
134 static bool isExtensionAvailable(const char* name);
135
136 ////////////////////////////////////////////////////////////
137 /// \brief Get the address of an OpenGL function
138 ///
139 /// \param name Name of the function to get the address of
140 ///
141 /// \return Address of the OpenGL function, 0 on failure
142 ///
143 ////////////////////////////////////////////////////////////
144 static GlFunctionPointer getFunction(const char* name);
145
146 ////////////////////////////////////////////////////////////
147 /// \brief Destructor
148 ///
149 ////////////////////////////////////////////////////////////
150 virtual ~GlContext();
151
152 ////////////////////////////////////////////////////////////
153 /// \brief Get the settings of the context
154 ///
155 /// Note that these settings may be different than the ones
156 /// passed to the constructor; they are indeed adjusted if the
157 /// original settings are not directly supported by the system.
158 ///
159 /// \return Structure containing the settings
160 ///
161 ////////////////////////////////////////////////////////////
162 const ContextSettings& getSettings() const;
163
164 ////////////////////////////////////////////////////////////
165 /// \brief Activate or deactivate the context as the current target for rendering
166 ///
167 /// A context is active only on the current thread, if you want to
168 /// make it active on another thread you have to deactivate it
169 /// on the previous thread first if it was active.
170 /// Only one context can be active on a thread at a time, thus
171 /// the context previously active (if any) automatically gets deactivated.
172 ///
173 /// \param active True to activate, false to deactivate
174 ///
175 /// \return True if operation was successful, false otherwise
176 ///
177 ////////////////////////////////////////////////////////////
178 bool setActive(bool active);
179
180 ////////////////////////////////////////////////////////////
181 /// \brief Display what has been rendered to the context so far
182 ///
183 ////////////////////////////////////////////////////////////
184 virtual void display() = 0;
185
186 ////////////////////////////////////////////////////////////
187 /// \brief Enable or disable vertical synchronization
188 ///
189 /// Activating vertical synchronization will limit the number
190 /// of frames displayed to the refresh rate of the monitor.
191 /// This can avoid some visual artifacts, and limit the framerate
192 /// to a good value (but not constant across different computers).
193 ///
194 /// \param enabled True to enable v-sync, false to deactivate
195 ///
196 ////////////////////////////////////////////////////////////
197 virtual void setVerticalSyncEnabled(bool enabled) = 0;
198
199protected:
200
201 ////////////////////////////////////////////////////////////
202 /// \brief Default constructor
203 ///
204 /// This constructor is meant for derived classes only.
205 ///
206 ////////////////////////////////////////////////////////////
207 GlContext();
208
209 ////////////////////////////////////////////////////////////
210 /// \brief Activate the context as the current target
211 /// for rendering
212 ///
213 /// \param current Whether to make the context current or no longer current
214 ///
215 /// \return True on success, false if any error happened
216 ///
217 ////////////////////////////////////////////////////////////
218 virtual bool makeCurrent(bool current) = 0;
219
220 ////////////////////////////////////////////////////////////
221 /// \brief Evaluate a pixel format configuration
222 ///
223 /// This functions can be used by implementations that have
224 /// several valid formats and want to get the best one.
225 /// A score is returned for the given configuration: the
226 /// lower the score is, the better the configuration is.
227 ///
228 /// \param bitsPerPixel Requested pixel depth (bits per pixel)
229 /// \param settings Requested additional settings
230 /// \param colorBits Color bits of the configuration to evaluate
231 /// \param depthBits Depth bits of the configuration to evaluate
232 /// \param stencilBits Stencil bits of the configuration to evaluate
233 /// \param antialiasing Antialiasing level of the configuration to evaluate
234 /// \param accelerated Whether the pixel format is hardware accelerated
235 /// \param sRgb Whether the pixel format is sRGB capable
236 ///
237 /// \return Score of the configuration
238 ///
239 ////////////////////////////////////////////////////////////
240 static int evaluateFormat(unsigned int bitsPerPixel, const ContextSettings& settings, int colorBits, int depthBits, int stencilBits, int antialiasing, bool accelerated, bool sRgb);
241
242 ////////////////////////////////////////////////////////////
243 // Member data
244 ////////////////////////////////////////////////////////////
245 ContextSettings m_settings; ///< Creation settings of the context
246
247private:
248
249 ////////////////////////////////////////////////////////////
250 /// \brief Perform various initializations after the context construction
251 /// \param requestedSettings Requested settings during context creation
252 ///
253 ////////////////////////////////////////////////////////////
254 void initialize(const ContextSettings& requestedSettings);
255
256 ////////////////////////////////////////////////////////////
257 /// \brief Check whether the context is compatible with the requested settings
258 /// \param requestedSettings Requested settings during context creation
259 ///
260 ////////////////////////////////////////////////////////////
261 void checkSettings(const ContextSettings& requestedSettings);
262};
263
264} // namespace priv
265
266} // namespace sf
267
268
269#endif // SFML_GLCONTEXT_HPP
