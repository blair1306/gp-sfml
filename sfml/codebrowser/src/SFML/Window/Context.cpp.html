28#include <SFML/Window/Context.hpp>
29#include <SFML/Window/GlContext.hpp>
30#include <SFML/System/ThreadLocalPtr.hpp>
31
32
33namespace
34{
35 // This per-thread variable holds the current context for each thread
36 sf::ThreadLocalPtr<sf::Context> currentContext(NULL);
37}
38
39namespace sf
40{
41////////////////////////////////////////////////////////////
42Context::Context()
43{
44 m_context = priv::GlContext::create();
45 setActive(true);
46}
47
48
49////////////////////////////////////////////////////////////
50Context::~Context()
51{
52 setActive(false);
53 delete m_context;
54}
55
56
57////////////////////////////////////////////////////////////
58bool Context::setActive(bool active)
59{
60 bool result = m_context->setActive(active);
61
62 if (result)
63 currentContext = (active ? this : NULL);
64
65 return result;
66}
67
68
69////////////////////////////////////////////////////////////
70const ContextSettings& Context::getSettings() const
71{
72 return m_context->getSettings();
73}
74
75
76////////////////////////////////////////////////////////////
77const Context* Context::getActiveContext()
78{
79 return currentContext;
80}
81
82
83////////////////////////////////////////////////////////////

86 return priv::GlContext::isExtensionAvailable(name);
87}
88
89
90////////////////////////////////////////////////////////////
91GlFunctionPointer Context::getFunction(const char* name)
92{
93 return priv::GlContext::getFunction(name);
94}

97////////////////////////////////////////////////////////////
98Context::Context(const ContextSettings& settings, unsigned int width, unsigned int height)
99{
100 m_context = priv::GlContext::create(settings, width, height);
101 setActive(true);
102}
