37namespace sf
38{
39namespace priv
40{
41////////////////////////////////////////////////////////////
42/// \brief Structure holding a joystick's information
43///
44////////////////////////////////////////////////////////////
45struct JoystickCaps
46{
47 JoystickCaps()
48 {
49 buttonCount = 0;
50 std::fill(axes, axes + Joystick::AxisCount, false);
51 }
52
53 unsigned int buttonCount; ///< Number of buttons supported by the joystick
54 bool axes[Joystick::AxisCount]; ///< Support for each axis
55};
56
57
58////////////////////////////////////////////////////////////
59/// \brief Structure holding a joystick's state
60///
61////////////////////////////////////////////////////////////
62struct JoystickState
63{
64 JoystickState()
65 {
66 connected = false;
67 std::fill(axes, axes + Joystick::AxisCount, 0.f);
68 std::fill(buttons, buttons + Joystick::ButtonCount, false);
69 }
70
71 bool connected; ///< Is the joystick currently connected?
72 float axes[Joystick::AxisCount]; ///< Position of each axis, in range [-100, 100]
73 bool buttons[Joystick::ButtonCount]; ///< Status of each button (true = pressed)
74};
75
