25#ifndef SFML_SENSORMANAGER_HPP
26#define SFML_SENSORMANAGER_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Window/Sensor.hpp>
32#include <SFML/Window/SensorImpl.hpp>
33#include <SFML/System/NonCopyable.hpp>
34
35
36namespace sf
37{
38namespace priv
39{
40////////////////////////////////////////////////////////////
41/// \brief Global sensor manager
42///
43////////////////////////////////////////////////////////////
44class SensorManager : NonCopyable
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Get the global unique instance of the manager
50 ///
51 /// \return Unique instance of the sensor manager
52 ///
53 ////////////////////////////////////////////////////////////
54 static SensorManager& getInstance();
55
56 ////////////////////////////////////////////////////////////
57 /// \brief Check if a sensor is available on the underlying platform
58 ///
59 /// \param sensor Sensor to check
60 ///
61 /// \return True if the sensor is available, false otherwise
62 ///
63 ////////////////////////////////////////////////////////////
64 bool isAvailable(Sensor::Type sensor);
65
66 ////////////////////////////////////////////////////////////
67 /// \brief Enable or disable a sensor
68 ///
69 /// \param sensor Sensor to modify
70 /// \param enabled Whether it should be enabled or not
71 ///
72 ////////////////////////////////////////////////////////////
73 void setEnabled(Sensor::Type sensor, bool enabled);
74
75 ////////////////////////////////////////////////////////////
76 /// \brief Check if a sensor is enabled
77 ///
78 /// \param sensor Sensor to check
79 ///
80 /// \return True if the sensor is enabled, false otherwise
81 ///
82 ////////////////////////////////////////////////////////////
83 bool isEnabled(Sensor::Type sensor) const;
84
85 ////////////////////////////////////////////////////////////
86 /// \brief Get the current value of a sensor
87 ///
88 /// \param sensor Sensor to read
89 ///
90 /// \return Current value of the sensor
91 ///
92 ////////////////////////////////////////////////////////////
93 Vector3f getValue(Sensor::Type sensor) const;
94
95 ////////////////////////////////////////////////////////////
96 /// \brief Update the state of all the sensors
97 ///
98 ////////////////////////////////////////////////////////////
99 void update();
100
101private:
102
103 ////////////////////////////////////////////////////////////
104 /// \brief Default constructor
105 ///
106 ////////////////////////////////////////////////////////////
107 SensorManager();
108
109 ////////////////////////////////////////////////////////////
110 /// \brief Destructor
111 ///
112 ////////////////////////////////////////////////////////////
113 ~SensorManager();
114
115 ////////////////////////////////////////////////////////////
116 /// \brief Sensor information and state
117 ///
118 ////////////////////////////////////////////////////////////
119 struct Item
120 {
121 bool available; ///< Is the sensor available on this device?
122 bool enabled; ///< Current enable state of the sensor
123 SensorImpl sensor; ///< Sensor implementation
124 Vector3f value; ///< The current sensor value
125 };
126
127 ////////////////////////////////////////////////////////////
128 // Member data
129 ////////////////////////////////////////////////////////////
130 Item m_sensors[Sensor::Count]; ///< Sensors information and state
131};
132
133} // namespace priv
134
135} // namespace sf
136
137
138#endif // SFML_SENSORMANAGER_HPP
