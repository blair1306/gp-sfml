25#ifndef SFML_JOYSTICKMANAGER_HPP
26#define SFML_JOYSTICKMANAGER_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Window/Joystick.hpp>
32#include <SFML/Window/JoystickImpl.hpp>
33#include <SFML/System/NonCopyable.hpp>
34
35
36namespace sf
37{
38namespace priv
39{
40////////////////////////////////////////////////////////////
41/// \brief Global joystick manager
42///
43////////////////////////////////////////////////////////////
44class JoystickManager : NonCopyable
45{
46public:
47
48 ////////////////////////////////////////////////////////////
49 /// \brief Get the global unique instance of the manager
50 ///
51 /// \return Unique instance of the joystick manager
52 ///
53 ////////////////////////////////////////////////////////////
54 static JoystickManager& getInstance();
55
56 ////////////////////////////////////////////////////////////
57 /// \brief Get the capabilities for an open joystick
58 ///
59 /// \param joystick Index of the joystick
60 ///
61 /// \return Capabilities of the joystick
62 ///
63 ////////////////////////////////////////////////////////////
64 const JoystickCaps& getCapabilities(unsigned int joystick) const;
65
66 ////////////////////////////////////////////////////////////
67 /// \brief Get the current state of an open joystick
68 ///
69 /// \param joystick Index of the joystick
70 ///
71 /// \return Current state of the joystick
72 ///
73 ////////////////////////////////////////////////////////////
74 const JoystickState& getState(unsigned int joystick) const;
75
76 ////////////////////////////////////////////////////////////
77 /// \brief Get the identification for an open joystick
78 ///
79 /// \param joystick Index of the joystick
80 ///
81 /// \return Identification for the joystick
82 ///
83 ////////////////////////////////////////////////////////////
84 const Joystick::Identification& getIdentification(unsigned int joystick) const;
85
86 ////////////////////////////////////////////////////////////
87 /// \brief Update the state of all the joysticks
88 ///
89 ////////////////////////////////////////////////////////////
90 void update();
91
92private:
93
94 ////////////////////////////////////////////////////////////
95 /// \brief Default constructor
96 ///
97 ////////////////////////////////////////////////////////////
98 JoystickManager();
99
100 ////////////////////////////////////////////////////////////
101 /// \brief Destructor
102 ///
103 ////////////////////////////////////////////////////////////
104 ~JoystickManager();
105
106 ////////////////////////////////////////////////////////////
107 /// \brief Joystick information and state
108 ///
109 ////////////////////////////////////////////////////////////
110 struct Item
111 {
112 JoystickImpl joystick; ///< Joystick implementation
113 JoystickState state; ///< The current joystick state
114 JoystickCaps capabilities; ///< The joystick capabilities
115 Joystick::Identification identification; ///< The joystick identification
116 };
117
118 ////////////////////////////////////////////////////////////
119 // Member data
120 ////////////////////////////////////////////////////////////
121 Item m_joysticks[Joystick::Count]; ///< Joysticks information and state
122};
123
124} // namespace priv
125
126} // namespace sf
127
128
129#endif // SFML_JOYSTICKMANAGER_HPP
