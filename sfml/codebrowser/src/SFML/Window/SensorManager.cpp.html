44////////////////////////////////////////////////////////////
45bool SensorManager::isAvailable(Sensor::Type sensor)
46{
47 return m_sensors[sensor].available;
48}
49
50
51////////////////////////////////////////////////////////////
52void SensorManager::setEnabled(Sensor::Type sensor, bool enabled)
53{
54 if (m_sensors[sensor].available)
55 {
56 m_sensors[sensor].enabled = enabled;
57 m_sensors[sensor].sensor.setEnabled(enabled);
58 }
59 else
60 {
61 err() << "Warning: trying to enable a sensor that is not available (call Sensor::isAvailable to check it)" << std::endl;
62 }
63}
64
65
66////////////////////////////////////////////////////////////
67bool SensorManager::isEnabled(Sensor::Type sensor) const
68{
69 return m_sensors[sensor].enabled;
70}

73////////////////////////////////////////////////////////////
74Vector3f SensorManager::getValue(Sensor::Type sensor) const
75{
76 return m_sensors[sensor].value;

80////////////////////////////////////////////////////////////
81void SensorManager::update()
82{
83 for (int i = 0; i < Sensor::Count; ++i)
84 {
85 // Only process available sensors
86 if (m_sensors[i].available)
87 m_sensors[i].value = m_sensors[i].sensor.update();
88 }
89}
90
91
92////////////////////////////////////////////////////////////
93SensorManager::SensorManager()
94{
95 // Global sensor initialization
96 SensorImpl::initialize();
97
98 // Per sensor initialization
99 for (int i = 0; i < Sensor::Count; ++i)
100 {
101 // Check which sensors are available
102 m_sensors[i].available = SensorImpl::isAvailable(static_cast<Sensor::Type>(i));
103
104 // Open the available sensors
105 if (m_sensors[i].available)
106 {
107 m_sensors[i].sensor.open(static_cast<Sensor::Type>(i));
108 m_sensors[i].sensor.setEnabled(false);
109 }
110 }
111}
112
113////////////////////////////////////////////////////////////
114SensorManager::~SensorManager()
115{
116 // Per sensor cleanup
117 for (int i = 0; i < Sensor::Count; ++i)
118 {
119 if (m_sensors[i].available)
120 m_sensors[i].sensor.close();
121 }
122
123 // Global sensor cleanup
124 SensorImpl::cleanup();
125}
