64////////////////////////////////////////////////////////////
65void JoystickManager::update()
66{
67 for (int i = 0; i < Joystick::Count; ++i)
68 {
69 Item& item = m_joysticks[i];
70
71 if (item.state.connected)
72 {
73 // Get the current state of the joystick
74 item.state = item.joystick.update();
75
76 // Check if it's still connected
77 if (!item.state.connected)
78 {
79 item.joystick.close();
80 item.capabilities = JoystickCaps();
81 item.state = JoystickState();
82 item.identification = Joystick::Identification();
83 }
84 }
85 else
86 {
87 // Check if the joystick was connected since last update
88 if (JoystickImpl::isConnected(i))
89 {
90 if (item.joystick.open(i))
91 {
92 item.capabilities = item.joystick.getCapabilities();
93 item.state = item.joystick.update();
94 item.identification = item.joystick.getIdentification();
95 }
96 }
97 }
98 }
99}

109////////////////////////////////////////////////////////////
110JoystickManager::~JoystickManager()
111{
112 for (int i = 0; i < Joystick::Count; ++i)
113 {
114 if (m_joysticks[i].state.connected)
115 m_joysticks[i].joystick.close();
116 }
117
118 JoystickImpl::cleanup();
119}
