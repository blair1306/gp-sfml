34////////////////////////////////////////////////////////////
35bool Joystick::isConnected(unsigned int joystick)
36{
37 return priv::JoystickManager::getInstance().getState(joystick).connected;
38}
39
40
41////////////////////////////////////////////////////////////
42unsigned int Joystick::getButtonCount(unsigned int joystick)
43{
44 return priv::JoystickManager::getInstance().getCapabilities(joystick).buttonCount;
45}
46
47
48////////////////////////////////////////////////////////////
49bool Joystick::hasAxis(unsigned int joystick, Axis axis)
50{
51 return priv::JoystickManager::getInstance().getCapabilities(joystick).axes[axis];
52}
53
54
55////////////////////////////////////////////////////////////
56bool Joystick::isButtonPressed(unsigned int joystick, unsigned int button)
57{
58 return priv::JoystickManager::getInstance().getState(joystick).buttons[button];
59}
60
61
62////////////////////////////////////////////////////////////
63float Joystick::getAxisPosition(unsigned int joystick, Axis axis)
64{
65 return priv::JoystickManager::getInstance().getState(joystick).axes[axis];
66}

68
69////////////////////////////////////////////////////////////
70Joystick::Identification Joystick::getIdentification(unsigned int joystick)
71{
72 return priv::JoystickManager::getInstance().getIdentification(joystick);
73}

77void Joystick::update()

79 return priv::JoystickManager::getInstance().update();
