83////////////////////////////////////////////////////////////
84void Window::create(VideoMode mode, const String& title, Uint32 style, const ContextSettings& settings)
85{
86 // Destroy the previous window implementation
87 close();
88
89 // Fullscreen style requires some tests
90 if (style & Style::Fullscreen)
91 {
92 // Make sure there's not already a fullscreen window (only one is allowed)
93 if (fullscreenWindow)
94 {
95 err() << "Creating two fullscreen windows is not allowed, switching to windowed mode" << std::endl;
96 style &= ~Style::Fullscreen;
97 }
98 else
99 {
100 // Make sure that the chosen video mode is compatible
101 if (!mode.isValid())
102 {
103 err() << "The requested video mode is not available, switching to a valid mode" << std::endl;
104 mode = VideoMode::getFullscreenModes()[0];
105 }
106
107 // Update the fullscreen window
108 fullscreenWindow = this;
109 }
110 }
111
112 // Check validity of style according to the underlying platform
113 #if defined(SFML_SYSTEM_IOS) || defined(SFML_SYSTEM_ANDROID)
114 if (style & Style::Fullscreen)
115 style &= ~Style::Titlebar;
116 else
117 style |= Style::Titlebar;
118 #else
119 if ((style & Style::Close) || (style & Style::Resize))
120 style |= Style::Titlebar;
121 #endif
122
123 // Recreate the window implementation
124 m_impl = priv::WindowImpl::create(mode, title, style, settings);
125
126 // Recreate the context
127 m_context = priv::GlContext::create(settings, m_impl, mode.bitsPerPixel);
128
129 // Perform common initializations
130 initialize();
131}

134////////////////////////////////////////////////////////////
135void Window::create(WindowHandle handle, const ContextSettings& settings)
136{
137 // Destroy the previous window implementation
138 close();
139
140 // Recreate the window implementation
141 m_impl = priv::WindowImpl::create(handle);
142
143 // Recreate the context
144 m_context = priv::GlContext::create(settings, m_impl, VideoMode::getDesktopMode().bitsPerPixel);
145
146 // Perform common initializations
147 initialize();
148}

151////////////////////////////////////////////////////////////
152void Window::close()
153{
154 // Delete the context
155 delete m_context;
156 m_context = NULL;
157
158 // Delete the window implementation
159 delete m_impl;
160 m_impl = NULL;
161
162 // Update the fullscreen window
163 if (this == fullscreenWindow)
164 fullscreenWindow = NULL;
165}

184////////////////////////////////////////////////////////////
185bool Window::pollEvent(Event& event)
186{
187 if (m_impl && m_impl->popEvent(event, false))
188 {
189 return filterEvent(event);
190 }
191 else
192 {
193 return false;
194 }
195}

198////////////////////////////////////////////////////////////
199bool Window::waitEvent(Event& event)
200{
201 if (m_impl && m_impl->popEvent(event, true))
202 {
203 return filterEvent(event);
204 }
205 else
206 {
207 return false;
208 }
209}

371
372void Window::display()
373{
374 // Display the backbuffer on screen
375 if (setActive())
376 m_context->display();
377
378 // Limit the framerate if needed
379 if (m_frameTimeLimit != Time::Zero)
380 {
381 sleep(m_frameTimeLimit - m_clock.getElapsedTime());
382 m_clock.restart();
383 }
384}

408////////////////////////////////////////////////////////////
409bool Window::filterEvent(const Event& event)
410{
411 // Notify resize events to the derived class
412 if (event.type == Event::Resized)
413 {
414 // Cache the new size
415 m_size.x = event.size.width;
416 m_size.y = event.size.height;
417
418 // Notify the derived class
419 onResize();
420 }
421
422 return true;
423}

426////////////////////////////////////////////////////////////
427void Window::initialize()
428{
429 // Setup default behaviors (to get a consistent behavior across different implementations)
430 setVisible(true);
431 setMouseCursorVisible(true);
432 setVerticalSyncEnabled(false);
433 setKeyRepeatEnabled(true);
434 setFramerateLimit(0);
435
436 // Get and cache the initial size of the window
437 m_size = m_impl->getSize();
438
439 // Reset frame time
440 m_clock.restart();
441
442 // Activate the window
443 setActive();
444
445 // Notify the derived class
446 onCreate();
447}
