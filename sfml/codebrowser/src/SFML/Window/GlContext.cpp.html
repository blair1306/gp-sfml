131 // AMD drivers have issues with internal synchronization
132 // We need to make sure that no operating system context
133 // or pixel format operations are performed simultaneously
134 // This mutex is also used to protect the shared context
135 // from being locked on multiple threads and for managing
136 // the resource count
137 sf::Mutex mutex;
138
139 // OpenGL resources counter
140 unsigned int resourceCount = 0;
141
142 // This per-thread variable holds the current context for each thread
143 sf::ThreadLocalPtr<sf::priv::GlContext> currentContext(NULL);
144
145 // The hidden, inactive context that will be shared with all other contexts
146 ContextType* sharedContext = NULL;
147
148 // This structure contains all the state necessary to
149 // track TransientContext usage
150 struct TransientContext : private sf::NonCopyable
151 {
152 ////////////////////////////////////////////////////////////
153 /// \brief Constructor
154 ///
155 ////////////////////////////////////////////////////////////
156 TransientContext() :
157 referenceCount (0),
158 context (0),
159 sharedContextLock(0),
160 useSharedContext (false)
161 {
162 if (resourceCount == 0)
163 {
164 context = new sf::Context;
165 }
166 else if (!currentContext)
167 {
168 sharedContextLock = new sf::Lock(mutex);
169 useSharedContext = true;
170 sharedContext->setActive(true);
171 }
172 }
173
174 ////////////////////////////////////////////////////////////
175 /// \brief Destructor
176 ///
177 ////////////////////////////////////////////////////////////
178 ~TransientContext()
179 {
180 if (useSharedContext)
181 sharedContext->setActive(false);
182
183 delete sharedContextLock;
184 delete context;
185 }
186
187 ///////////////////////////////////////////////////////////
188 // Member data
189 ////////////////////////////////////////////////////////////
190 unsigned int referenceCount;
191 sf::Context* context;
192 sf::Lock* sharedContextLock;
193 bool useSharedContext;
194 };
195
196 // This per-thread variable tracks if and how a transient
197 // context is currently being used on the current thread
198 sf::ThreadLocalPtr<TransientContext> transientContext(NULL);
199
200 // Supported OpenGL extensions
201 std::vector<std::string> extensions;
202}

209////////////////////////////////////////////////////////////
210void GlContext::initResource()
211{
212 // Protect from concurrent access
213 Lock lock(mutex);
214
215 // If this is the very first resource, trigger the global context initialization
216 if (resourceCount == 0)
217 {
218 if (sharedContext)
219 {
220 // Increment the resources counter
221 resourceCount++;
222
223 return;
224 }
225
226 // Create the shared context
227 sharedContext = new ContextType(NULL);
228 sharedContext->initialize(ContextSettings());
229
230 // Load our extensions vector
231 extensions.clear();
232
233 // Check whether a >= 3.0 context is available
234 int majorVersion = 0;
235 glGetIntegerv(GL_MAJOR_VERSION, &majorVersion);
236
237 if (glGetError() == GL_INVALID_ENUM)
238 {
239 // Try to load the < 3.0 way
240 const char* extensionString = reinterpret_cast<const char*>(glGetString(GL_EXTENSIONS));
241
242 do
243 {
244 const char* extension = extensionString;
245
246 while(*extensionString && (*extensionString != ' '))
247 extensionString++;
248
249 extensions.push_back(std::string(extension, extensionString));
250 }
251 while (*extensionString++);
252 }
253 else
254 {
255 // Try to load the >= 3.0 way
256 glGetStringiFuncType glGetStringiFunc = NULL;
257 glGetStringiFunc = reinterpret_cast<glGetStringiFuncType>(getFunction("glGetStringi"));
258
259 if (glGetStringiFunc)
260 {
261 int numExtensions = 0;
262 glGetIntegerv(GL_NUM_EXTENSIONS, &numExtensions);
263
264 if (numExtensions)
265 {
266 for (unsigned int i = 0; i < static_cast<unsigned int>(numExtensions); ++i)
267 {
268 const char* extensionString = reinterpret_cast<const char*>(glGetStringiFunc(GL_EXTENSIONS, i));
269
270 extensions.push_back(extensionString);
271 }
272 }
273 }
274 }
275
276 // Deactivate the shared context so that others can activate it when necessary
277 sharedContext->setActive(false);
278 }
279
280 // Increment the resources counter
281 resourceCount++;
282}

285////////////////////////////////////////////////////////////
286void GlContext::cleanupResource()
287{
288 // Protect from concurrent access
289 Lock lock(mutex);
290
291 // Decrement the resources counter
292 resourceCount--;
293
294 // If there's no more resource alive, we can trigger the global context cleanup
295 if (resourceCount == 0)
296 {
297 if (!sharedContext)
298 return;
299
300 // Destroy the shared context
301 delete sharedContext;
302 sharedContext = NULL;
303 }
304}

307////////////////////////////////////////////////////////////
308void GlContext::acquireTransientContext()
309{
310 // Protect from concurrent access
311 Lock lock(mutex);
312
313 // If this is the first TransientContextLock on this thread
314 // construct the state object
315 if (!transientContext)
316 transientContext = new TransientContext;
317
318 // Increase the reference count
319 transientContext->referenceCount++;
320}

323////////////////////////////////////////////////////////////
324void GlContext::releaseTransientContext()
325{
326 // Protect from concurrent access
327 Lock lock(mutex);
328
329 // Make sure a matching acquireTransientContext() was called
330 assert(transientContext);
331
332 // Decrease the reference count
333 transientContext->referenceCount--;
334
335 // If this is the last TransientContextLock that is released
336 // destroy the state object
337 if (transientContext->referenceCount == 0)
338 {
339 delete transientContext;
340 transientContext = NULL;
341 }
342}

345////////////////////////////////////////////////////////////
346GlContext* GlContext::create()
347{
348 // Make sure that there's an active context (context creation may need extensions, and thus a valid context)
349 assert(sharedContext != NULL);
350
351 Lock lock(mutex);
352
353 GlContext* context = NULL;
354
355 // We don't use acquireTransientContext here since we have
356 // to ensure we have exclusive access to the shared context
357 // in order to make sure it is not active during context creation
358 {
359 sharedContext->setActive(true);
360
361 // Create the context
362 context = new ContextType(sharedContext);
363
364 sharedContext->setActive(false);
365 }
366
367 context->initialize(ContextSettings());
368
369 return context;
370}

474////////////////////////////////////////////////////////////
475bool GlContext::setActive(bool active)
476{
477 if (active)
478 {
479 if (this != currentContext)
480 {
481 Lock lock(mutex);
482
483 // Activate the context
484 if (makeCurrent(true))
485 {
486 // Set it as the new current context for this thread
487 currentContext = this;
488 return true;
489 }
490 else
491 {
492 return false;
493 }
494 }
495 else
496 {
497 // This context is already the active one on this thread, don't do anything
498 return true;
499 }
500 }
501 else
502 {
503 if (this == currentContext)
504 {
505 Lock lock(mutex);
506
507 // Deactivate the context
508 if (makeCurrent(false))
509 {
510 currentContext = NULL;
511 return true;
512 }
513 else
514 {
515 return false;
516 }
517 }
518 else
519 {
520 // This context is not the active one on this thread, don't do anything
521 return true;
522 }
523 }
524}

534////////////////////////////////////////////////////////////
535int GlContext::evaluateFormat(unsigned int bitsPerPixel, const ContextSettings& settings, int colorBits, int depthBits, int stencilBits, int antialiasing, bool accelerated, bool sRgb)
536{
537 int colorDiff = static_cast<int>(bitsPerPixel) - colorBits;
538 int depthDiff = static_cast<int>(settings.depthBits) - depthBits;
539 int stencilDiff = static_cast<int>(settings.stencilBits) - stencilBits;
540 int antialiasingDiff = static_cast<int>(settings.antialiasingLevel) - antialiasing;
541
542 // Weight sub-scores so that better settings don't score equally as bad as worse settings
543 colorDiff *= ((colorDiff > 0) ? 100000 : 1);
544 depthDiff *= ((depthDiff > 0) ? 100000 : 1);
545 stencilDiff *= ((stencilDiff > 0) ? 100000 : 1);
546 antialiasingDiff *= ((antialiasingDiff > 0) ? 100000 : 1);
547
548 // Aggregate the scores
549 int score = std::abs(colorDiff) + std::abs(depthDiff) + std::abs(stencilDiff) + std::abs(antialiasingDiff);
550
551 // If the user wants an sRGB capable format, try really hard to get one
552 if (settings.sRgbCapable && !sRgb)
553 score += 10000000;
554
555 // Make sure we prefer hardware acceleration over features
556 if (!accelerated)
557 score += 100000000;
558
559 return score;
560}

563////////////////////////////////////////////////////////////
564void GlContext::initialize(const ContextSettings& requestedSettings)
565{
566 // Activate the context
567 setActive(true);
568
569 // Retrieve the context version number
570 int majorVersion = 0;
571 int minorVersion = 0;
572
573 // Try the new way first
574 glGetIntegerv(GL_MAJOR_VERSION, &majorVersion);
575 glGetIntegerv(GL_MINOR_VERSION, &minorVersion);
576
577 if (glGetError() != GL_INVALID_ENUM)
578 {
579 m_settings.majorVersion = static_cast<unsigned int>(majorVersion);
580 m_settings.minorVersion = static_cast<unsigned int>(minorVersion);
581 }
582 else
583 {
584 // Try the old way
585 const GLubyte* version = glGetString(GL_VERSION);
586 if (version)
587 {
588 // The beginning of the returned string is "major.minor" (this is standard)
589 m_settings.majorVersion = version[0] - '0';
590 m_settings.minorVersion = version[2] - '0';
591 }
592 else
593 {
594 // Can't get the version number, assume 1.1
595 m_settings.majorVersion = 1;
596 m_settings.minorVersion = 1;
597 }
598 }
599
600 // 3.0 contexts only deprecate features, but do not remove them yet
601 // 3.1 contexts remove features if ARB_compatibility is not present
602 // 3.2+ contexts remove features only if a core profile is requested
603
604 // If the context was created with wglCreateContext, it is guaranteed to be compatibility.
605 // If a 3.0 context was created with wglCreateContextAttribsARB, it is guaranteed to be compatibility.
606 // If a 3.1 context was created with wglCreateContextAttribsARB, the compatibility flag
607 // is set only if ARB_compatibility is present
608 // If a 3.2+ context was created with wglCreateContextAttribsARB, the compatibility flag
609 // would have been set correctly already depending on whether ARB_create_context_profile is supported.
610
611 // If the user requests a 3.0 context, it will be a compatibility context regardless of the requested profile.
612 // If the user requests a 3.1 context and its creation was successful, the specification
613 // states that it will not be a compatibility profile context regardless of the requested
614 // profile unless ARB_compatibility is present.
615
616 m_settings.attributeFlags = ContextSettings::Default;
617
618 if (m_settings.majorVersion >= 3)
619 {
620 // Retrieve the context flags
621 int flags = 0;
622 glGetIntegerv(GL_CONTEXT_FLAGS, &flags);
623
624 if (flags & GL_CONTEXT_FLAG_DEBUG_BIT)
625 m_settings.attributeFlags |= ContextSettings::Debug;
626
627 if ((m_settings.majorVersion == 3) && (m_settings.minorVersion == 1))
628 {
629 m_settings.attributeFlags |= ContextSettings::Core;
630
631 glGetStringiFuncType glGetStringiFunc = reinterpret_cast<glGetStringiFuncType>(getFunction("glGetStringi"));
632
633 if (glGetStringiFunc)
634 {
635 int numExtensions = 0;
636 glGetIntegerv(GL_NUM_EXTENSIONS, &numExtensions);
637
638 for (unsigned int i = 0; i < static_cast<unsigned int>(numExtensions); ++i)
639 {
640 const char* extensionString = reinterpret_cast<const char*>(glGetStringiFunc(GL_EXTENSIONS, i));
641
642 if (std::strstr(extensionString, "GL_ARB_compatibility"))
643 {
644 m_settings.attributeFlags &= ~static_cast<Uint32>(ContextSettings::Core);
645 break;
646 }
647 }
648 }
649 }
650 else if ((m_settings.majorVersion > 3) || (m_settings.minorVersion >= 2))
651 {
652 // Retrieve the context profile
653 int profile = 0;
654 glGetIntegerv(GL_CONTEXT_PROFILE_MASK, &profile);
655
656 if (profile & GL_CONTEXT_CORE_PROFILE_BIT)
657 m_settings.attributeFlags |= ContextSettings::Core;
658 }
659 }
660
661 // Enable anti-aliasing if requested by the user and supported
662 if ((requestedSettings.antialiasingLevel > 0) && (m_settings.antialiasingLevel > 0))
663 {
664 glEnable(GL_MULTISAMPLE);
665 }
666 else
667 {
668 m_settings.antialiasingLevel = 0;
669 }
670
671 // Enable sRGB if requested by the user and supported
672 if (requestedSettings.sRgbCapable && m_settings.sRgbCapable)
673 {
674 glEnable(GL_FRAMEBUFFER_SRGB);
675
676 // Check to see if the enable was successful
677 if (glIsEnabled(GL_FRAMEBUFFER_SRGB) == GL_FALSE)
678 {
679 err() << "Warning: Failed to enable GL_FRAMEBUFFER_SRGB" << std::endl;
680 m_settings.sRgbCapable = false;
681 }
682 }
683 else
684 {
685 m_settings.sRgbCapable = false;
686 }
687}

690////////////////////////////////////////////////////////////
691void GlContext::checkSettings(const ContextSettings& requestedSettings)
692{
693 // Perform checks to inform the user if they are getting a context they might not have expected
694
695 // Detect any known non-accelerated implementations and warn
696 const char* vendorName = reinterpret_cast<const char*>(glGetString(GL_VENDOR));
697 const char* rendererName = reinterpret_cast<const char*>(glGetString(GL_RENDERER));
698
699 if (vendorName && rendererName)
700 {
701 if ((std::strcmp(vendorName, "Microsoft Corporation") == 0) && (std::strcmp(rendererName, "GDI Generic") == 0))
702 {
703 err() << "Warning: Detected \"Microsoft Corporation GDI Generic\" OpenGL implementation" << std::endl
704 << "The current OpenGL implementation is not hardware-accelerated" << std::endl;
705 }
706 }
707
708 int version = m_settings.majorVersion * 10 + m_settings.minorVersion;
709 int requestedVersion = requestedSettings.majorVersion * 10 + requestedSettings.minorVersion;
710
711 if ((m_settings.attributeFlags != requestedSettings.attributeFlags) ||
712 (version < requestedVersion) ||
713 (m_settings.stencilBits < requestedSettings.stencilBits) ||
714 (m_settings.antialiasingLevel < requestedSettings.antialiasingLevel) ||
715 (m_settings.depthBits < requestedSettings.depthBits) ||
716 (!m_settings.sRgbCapable && requestedSettings.sRgbCapable))
717 {
718 err() << "Warning: The created OpenGL context does not fully meet the settings that were requested" << std::endl;
719 err() << "Requested: version = " << requestedSettings.majorVersion << "." << requestedSettings.minorVersion
720 << " ; depth bits = " << requestedSettings.depthBits
721 << " ; stencil bits = " << requestedSettings.stencilBits
722 << " ; AA level = " << requestedSettings.antialiasingLevel
723 << std::boolalpha
724 << " ; core = " << ((requestedSettings.attributeFlags & ContextSettings::Core) != 0)
725 << " ; debug = " << ((requestedSettings.attributeFlags & ContextSettings::Debug) != 0)
726 << " ; sRGB = " << requestedSettings.sRgbCapable
727 << std::noboolalpha << std::endl;
728 err() << "Created: version = " << m_settings.majorVersion << "." << m_settings.minorVersion
729 << " ; depth bits = " << m_settings.depthBits
730 << " ; stencil bits = " << m_settings.stencilBits
731 << " ; AA level = " << m_settings.antialiasingLevel
732 << std::boolalpha
733 << " ; core = " << ((m_settings.attributeFlags & ContextSettings::Core) != 0)
734 << " ; debug = " << ((m_settings.attributeFlags & ContextSettings::Debug) != 0)
735 << " ; sRGB = " << m_settings.sRgbCapable
736 << std::noboolalpha << std::endl;
737 }
738}
