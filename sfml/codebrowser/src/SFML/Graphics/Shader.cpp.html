81 // Read the contents of a file into an array of char
82 bool getFileContents(const std::string& filename, std::vector<char>& buffer)
83 {
84 std::ifstream file(filename.c_str(), std::ios_base::binary);
85 if (file)
86 {
87 file.seekg(0, std::ios_base::end);
88 std::streamsize size = file.tellg();
89 if (size > 0)
90 {
91 file.seekg(0, std::ios_base::beg);
92 buffer.resize(static_cast<std::size_t>(size));
93 file.read(&buffer[0], size);
94 }
95 buffer.push_back('\0');
96 return true;
97 }
98 else
99 {
100 return false;
101 }
102 }
103
104 // Read the contents of a stream into an array of char
105 bool getStreamContents(sf::InputStream& stream, std::vector<char>& buffer)
106 {
107 bool success = true;
108 sf::Int64 size = stream.getSize();
109 if (size > 0)
110 {
111 buffer.resize(static_cast<std::size_t>(size));
112 stream.seek(0);
113 sf::Int64 read = stream.read(&buffer[0], size);
114 success = (read == size);
115 }
116 buffer.push_back('\0');
117 return success;
118 }

120 // Transforms an array of 2D vectors into a contiguous array of scalars
121 template <typename T>
122 std::vector<T> flatten(const sf::Vector2<T>* vectorArray, std::size_t length)
123 {
124 const std::size_t vectorSize = 2;
125
126 std::vector<T> contiguous(vectorSize * length);
127 for (std::size_t i = 0; i < length; ++i)
128 {
129 contiguous[vectorSize * i] = vectorArray[i].x;
130 contiguous[vectorSize * i + 1] = vectorArray[i].y;
131 }
132
133 return contiguous;
134 }

136 // Transforms an array of 3D vectors into a contiguous array of scalars
137 template <typename T>
138 std::vector<T> flatten(const sf::Vector3<T>* vectorArray, std::size_t length)
139 {
140 const std::size_t vectorSize = 3;
141
142 std::vector<T> contiguous(vectorSize * length);
143 for (std::size_t i = 0; i < length; ++i)
144 {
145 contiguous[vectorSize * i] = vectorArray[i].x;
146 contiguous[vectorSize * i + 1] = vectorArray[i].y;
147 contiguous[vectorSize * i + 2] = vectorArray[i].z;
148 }
149
150 return contiguous;
151 }

153 // Transforms an array of 4D vectors into a contiguous array of scalars
154 template <typename T>
155 std::vector<T> flatten(const sf::priv::Vector4<T>* vectorArray, std::size_t length)
156 {
157 const std::size_t vectorSize = 4;
158
159 std::vector<T> contiguous(vectorSize * length);
160 for (std::size_t i = 0; i < length; ++i)
161 {

163 contiguous[vectorSize * i + 1] = vectorArray[i].y;
164 contiguous[vectorSize * i + 2] = vectorArray[i].z;
165 contiguous[vectorSize * i + 3] = vectorArray[i].w;
166 }
167
168 return contiguous;
169 }

179////////////////////////////////////////////////////////////
180struct Shader::UniformBinder : private NonCopyable
181{
182 ////////////////////////////////////////////////////////////
183 /// \brief Constructor: set up state before uniform is set
184 ///
185 ////////////////////////////////////////////////////////////
186 UniformBinder(Shader& shader, const std::string& name) :
187 savedProgram(0),
188 currentProgram(castToGlHandle(shader.m_shaderProgram)),
189 location(-1)
190 {
191 if (currentProgram)
192 {
193 // Enable program object
194 glCheck(savedProgram = GLEXT_glGetHandle(GLEXT_GL_PROGRAM_OBJECT));
195 if (currentProgram != savedProgram)
196 glCheck(GLEXT_glUseProgramObject(currentProgram));
197
198 // Store uniform location for further use outside constructor
199 location = shader.getUniformLocation(name);
200 }
201 }
202
203 ////////////////////////////////////////////////////////////
204 /// \brief Destructor: restore state after uniform is set
205 ///
206 ////////////////////////////////////////////////////////////
207 ~UniformBinder()
208 {
209 // Disable program object
210 if (currentProgram && (currentProgram != savedProgram))
211 glCheck(GLEXT_glUseProgramObject(savedProgram));
212 }
213
214 TransientContextLock lock; ///< Lock to keep context active while uniform is bound
215 GLEXT_GLhandle savedProgram; ///< Handle to the previously active program object
216 GLEXT_GLhandle currentProgram; ///< Handle to the program object of the modified sf::Shader instance
217 GLint location; ///< Uniform location, used by the surrounding sf::Shader code
218};

243bool Shader::loadFromFile(const std::string& filename, Type type)
244{
245 // Read the file
246 std::vector<char> shader;
247 if (!getFileContents(filename, shader))
248 {
249 err() << "Failed to open shader file \"" << filename << "\"" << std::endl;
250 return false;
251 }
252
253 // Compile the shader program
254 if (type == Vertex)
255 return compile(&shader[0], NULL, NULL);
256 else if (type == Geometry)
257 return compile(NULL, &shader[0], NULL);
258 else
259 return compile(NULL, NULL, &shader[0]);
260}

543////////////////////////////////////////////////////////////
544void Shader::setUniform(const std::string& name, const Texture& texture)
545{
546 if (m_shaderProgram)
547 {
548 TransientContextLock lock;
549
550 // Find the location of the variable in the shader
551 int location = getUniformLocation(name);
552 if (location != -1)
553 {
554 // Store the location -> texture mapping
555 TextureTable::iterator it = m_textures.find(location);
556 if (it == m_textures.end())
557 {
558 // New entry, make sure there are enough texture units
559 GLint maxUnits = getMaxTextureUnits();
560 if (m_textures.size() + 1 >= static_cast<std::size_t>(maxUnits))
561 {
562 err() << "Impossible to use texture \"" << name << "\" for shader: all available texture units are used" << std::endl;
563 return;
564 }
565
566 m_textures[location] = &texture;
567 }
568 else
569 {
570 // Location already used, just replace the texture
571 it->second = &texture;
572 }
573 }
574 }
575}

740////////////////////////////////////////////////////////////
741void Shader::bind(const Shader* shader)
742{
743 TransientContextLock lock;
744
745 // Make sure that we can use shaders
746 if (!isAvailable())
747 {
748 err() << "Failed to bind or unbind shader: your system doesn't support shaders "
749 << "(you should test Shader::isAvailable() before trying to use the Shader class)" << std::endl;
750 return;
751 }
752
753 if (shader && shader->m_shaderProgram)
754 {
755 // Enable the program
756 glCheck(GLEXT_glUseProgramObject(castToGlHandle(shader->m_shaderProgram)));
757
758 // Bind the textures
759 shader->bindTextures();
760
761 // Bind the current texture
762 if (shader->m_currentTexture != -1)
763 glCheck(GLEXT_glUniform1i(shader->m_currentTexture, 0));
764 }
765 else
766 {
767 // Bind no shader
768 glCheck(GLEXT_glUseProgramObject(0));
769 }
770}

773////////////////////////////////////////////////////////////
774bool Shader::isAvailable()
775{
776 Lock lock(isAvailableMutex);
777
778 static bool checked = false;
779 static bool available = false;
780
781 if (!checked)
782 {
783 checked = true;
784
785 TransientContextLock contextLock;
786
787 // Make sure that extensions are initialized
788 sf::priv::ensureExtensionsInit();
789
790 available = GLEXT_multitexture &&
791 GLEXT_shading_language_100 &&
792 GLEXT_shader_objects &&
793 GLEXT_vertex_shader &&
794 GLEXT_fragment_shader;
795 }
796
797 return available;
798}

825////////////////////////////////////////////////////////////
826bool Shader::compile(const char* vertexShaderCode, const char* geometryShaderCode, const char* fragmentShaderCode)
827{
828 TransientContextLock lock;
829
830 // First make sure that we can use shaders
831 if (!isAvailable())
832 {
833 err() << "Failed to create a shader: your system doesn't support shaders "
834 << "(you should test Shader::isAvailable() before trying to use the Shader class)" << std::endl;
835 return false;
836 }
837
838 // Make sure we can use geometry shaders
839 if (geometryShaderCode && !isGeometryAvailable())
840 {
841 err() << "Failed to create a shader: your system doesn't support geometry shaders "
842 << "(you should test Shader::isGeometryAvailable() before trying to use geometry shaders)" << std::endl;
843 return false;
844 }
845
846 // Destroy the shader if it was already created
847 if (m_shaderProgram)
848 {
849 glCheck(GLEXT_glDeleteObject(castToGlHandle(m_shaderProgram)));
850 m_shaderProgram = 0;
851 }
852
853 // Reset the internal state
854 m_currentTexture = -1;
855 m_textures.clear();
856 m_uniforms.clear();
857
858 // Create the program
859 GLEXT_GLhandle shaderProgram;
860 glCheck(shaderProgram = GLEXT_glCreateProgramObject());
861
862 // Create the vertex shader if needed
863 if (vertexShaderCode)
864 {
865 // Create and compile the shader
866 GLEXT_GLhandle vertexShader;
867 glCheck(vertexShader = GLEXT_glCreateShaderObject(GLEXT_GL_VERTEX_SHADER));
868 glCheck(GLEXT_glShaderSource(vertexShader, 1, &vertexShaderCode, NULL));
869 glCheck(GLEXT_glCompileShader(vertexShader));
870
871 // Check the compile log
872 GLint success;
873 glCheck(GLEXT_glGetObjectParameteriv(vertexShader, GLEXT_GL_OBJECT_COMPILE_STATUS, &success));
874 if (success == GL_FALSE)
875 {
876 char log[1024];
877 glCheck(GLEXT_glGetInfoLog(vertexShader, sizeof(log), 0, log));
878 err() << "Failed to compile vertex shader:" << std::endl
879 << log << std::endl;
880 glCheck(GLEXT_glDeleteObject(vertexShader));
881 glCheck(GLEXT_glDeleteObject(shaderProgram));
882 return false;
883 }
884
885 // Attach the shader to the program, and delete it (not needed anymore)
886 glCheck(GLEXT_glAttachObject(shaderProgram, vertexShader));
887 glCheck(GLEXT_glDeleteObject(vertexShader));
888 }
889
890 // Create the geometry shader if needed
891 if (geometryShaderCode)
892 {
893 // Create and compile the shader
894 GLEXT_GLhandle geometryShader = GLEXT_glCreateShaderObject(GLEXT_GL_GEOMETRY_SHADER);
895 glCheck(GLEXT_glShaderSource(geometryShader, 1, &geometryShaderCode, NULL));
896 glCheck(GLEXT_glCompileShader(geometryShader));
897
898 // Check the compile log
899 GLint success;
900 glCheck(GLEXT_glGetObjectParameteriv(geometryShader, GLEXT_GL_OBJECT_COMPILE_STATUS, &success));
901 if (success == GL_FALSE)
902 {
903 char log[1024];
904 glCheck(GLEXT_glGetInfoLog(geometryShader, sizeof(log), 0, log));
905 err() << "Failed to compile geometry shader:" << std::endl
906 << log << std::endl;
907 glCheck(GLEXT_glDeleteObject(geometryShader));
908 glCheck(GLEXT_glDeleteObject(shaderProgram));
909 return false;
910 }
911
912 // Attach the shader to the program, and delete it (not needed anymore)
913 glCheck(GLEXT_glAttachObject(shaderProgram, geometryShader));
914 glCheck(GLEXT_glDeleteObject(geometryShader));
915 }
916
917 // Create the fragment shader if needed
918 if (fragmentShaderCode)
919 {
920 // Create and compile the shader
921 GLEXT_GLhandle fragmentShader;
922 glCheck(fragmentShader = GLEXT_glCreateShaderObject(GLEXT_GL_FRAGMENT_SHADER));
923 glCheck(GLEXT_glShaderSource(fragmentShader, 1, &fragmentShaderCode, NULL));
924 glCheck(GLEXT_glCompileShader(fragmentShader));
925
926 // Check the compile log
927 GLint success;
928 glCheck(GLEXT_glGetObjectParameteriv(fragmentShader, GLEXT_GL_OBJECT_COMPILE_STATUS, &success));
929 if (success == GL_FALSE)
930 {
931 char log[1024];
932 glCheck(GLEXT_glGetInfoLog(fragmentShader, sizeof(log), 0, log));
933 err() << "Failed to compile fragment shader:" << std::endl
934 << log << std::endl;
935 glCheck(GLEXT_glDeleteObject(fragmentShader));
936 glCheck(GLEXT_glDeleteObject(shaderProgram));
937 return false;
938 }
939
940 // Attach the shader to the program, and delete it (not needed anymore)
941 glCheck(GLEXT_glAttachObject(shaderProgram, fragmentShader));
942 glCheck(GLEXT_glDeleteObject(fragmentShader));
943 }
944
945 // Link the program
946 glCheck(GLEXT_glLinkProgram(shaderProgram));
947
948 // Check the link log
949 GLint success;
950 glCheck(GLEXT_glGetObjectParameteriv(shaderProgram, GLEXT_GL_OBJECT_LINK_STATUS, &success));
951 if (success == GL_FALSE)
952 {
953 char log[1024];
954 glCheck(GLEXT_glGetInfoLog(shaderProgram, sizeof(log), 0, log));
955 err() << "Failed to link shader:" << std::endl
956 << log << std::endl;
957 glCheck(GLEXT_glDeleteObject(shaderProgram));
958 return false;
959 }
960
961 m_shaderProgram = castFromGlHandle(shaderProgram);
962
963 // Force an OpenGL flush, so that the shader will appear updated
964 // in all contexts immediately (solves problems in multi-threaded apps)
965 glCheck(glFlush());
966
967 return true;
968}

971////////////////////////////////////////////////////////////
972void Shader::bindTextures() const
973{
974 TextureTable::const_iterator it = m_textures.begin();
975 for (std::size_t i = 0; i < m_textures.size(); ++i)
976 {
977 GLint index = static_cast<GLsizei>(i + 1);
978 glCheck(GLEXT_glUniform1i(it->first, index));
979 glCheck(GLEXT_glActiveTexture(GLEXT_GL_TEXTURE0 + index));
980 Texture::bind(it->second);
981 ++it;
982 }
983
984 // Make sure that the texture unit which is left active is the number 0
985 glCheck(GLEXT_glActiveTexture(GLEXT_GL_TEXTURE0));
986}

989////////////////////////////////////////////////////////////
990int Shader::getUniformLocation(const std::string& name)
991{
992 // Check the cache
993 UniformTable::const_iterator it = m_uniforms.find(name);
994 if (it != m_uniforms.end())
995 {
996 // Already in cache, return it
997 return it->second;
998 }
999 else
1000 {
1001 // Not in cache, request the location from OpenGL
1002 int location = GLEXT_glGetUniformLocation(castToGlHandle(m_shaderProgram), name.c_str());
1003 m_uniforms.insert(std::make_pair(name, location));
1004
1005 if (location == -1)
1006 err() << "Uniform \"" << name << "\" not found in shader" << std::endl;
1007
1008 return location;
1009 }
1010}
