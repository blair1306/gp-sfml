35CircleShape::CircleShape(float radius, std::size_t pointCount) :
36m_radius (radius),
37m_pointCount(pointCount)
38{
39 update();
40}

42
43////////////////////////////////////////////////////////////
44void CircleShape::setRadius(float radius)
45{
46 m_radius = radius;
47 update();
48}
49
50
51////////////////////////////////////////////////////////////

55}
56
57
58////////////////////////////////////////////////////////////
59void CircleShape::setPointCount(std::size_t count)
60{
61 m_pointCount = count;
62 update();

64
65////////////////////////////////////////////////////////////
66std::size_t CircleShape::getPointCount() const

68 return m_pointCount;

71
72////////////////////////////////////////////////////////////
73Vector2f CircleShape::getPoint(std::size_t index) const
74{
75 static const float pi = 3.141592654f;
76
77 float angle = index * 2 * pi / m_pointCount - pi / 2;
78 float x = std::cos(angle) * m_radius;
79 float y = std::sin(angle) * m_radius;
80
81 return Vector2f(m_radius + x, m_radius + y);
82}
