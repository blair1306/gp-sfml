36TextureSaver::TextureSaver()
37{
38 glCheck(glGetIntegerv(GL_TEXTURE_BINDING_2D, &m_textureBinding));
39}
40
41
42////////////////////////////////////////////////////////////
43TextureSaver::~TextureSaver()
44{
45 glCheck(glBindTexture(GL_TEXTURE_2D, m_textureBinding));
46}
