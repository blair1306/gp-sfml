72void Transformable::setRotation(float angle)
73{
74 m_rotation = static_cast<float>(fmod(angle, 360));
75 if (m_rotation < 0)
76 m_rotation += 360.f;
77
78 m_transformNeedUpdate = true;
79 m_inverseTransformNeedUpdate = true;

180////////////////////////////////////////////////////////////
181const Transform& Transformable::getTransform() const
182{
183 // Recompute the combined transform if needed
184 if (m_transformNeedUpdate)
185 {
186 float angle = -m_rotation * 3.141592654f / 180.f;
187 float cosine = static_cast<float>(std::cos(angle));
188 float sine = static_cast<float>(std::sin(angle));
189 float sxc = m_scale.x * cosine;
190 float syc = m_scale.y * cosine;
191 float sxs = m_scale.x * sine;
192 float sys = m_scale.y * sine;
193 float tx = -m_origin.x * sxc - m_origin.y * sys + m_position.x;
194 float ty = m_origin.x * sxs - m_origin.y * syc + m_position.y;
195
196 m_transform = Transform( sxc, sys, tx,
197 -sxs, syc, ty,
198 0.f, 0.f, 1.f);
199 m_transformNeedUpdate = false;
200 }
201
202 return m_transform;
203}

206////////////////////////////////////////////////////////////
207const Transform& Transformable::getInverseTransform() const
208{
209 // Recompute the inverse transform if needed
210 if (m_inverseTransformNeedUpdate)
211 {
212 m_inverseTransform = getTransform().getInverse();
213 m_inverseTransformNeedUpdate = false;
214 }
215
216 return m_inverseTransform;
217}
