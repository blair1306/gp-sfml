39////////////////////////////////////////////////////////////
40RenderTextureImplDefault::RenderTextureImplDefault() :
41m_context(0),
42m_width (0),
43m_height (0)
44{
45
46}
47
48
49////////////////////////////////////////////////////////////
50RenderTextureImplDefault::~RenderTextureImplDefault()
51{
52 // Destroy the context
53 delete m_context;
54}
55
56
57////////////////////////////////////////////////////////////
58bool RenderTextureImplDefault::create(unsigned int width, unsigned int height, unsigned int, bool depthBuffer)
59{
60 // Store the dimensions
61 m_width = width;
62 m_height = height;
63
64 // Create the in-memory OpenGL context
65 m_context = new Context(ContextSettings(depthBuffer ? 32 : 0), width, height);
66
67 return true;
68}
69
70
71////////////////////////////////////////////////////////////
72bool RenderTextureImplDefault::activate(bool active)
73{
74 return m_context->setActive(active);
75}
76
77
78////////////////////////////////////////////////////////////
79void RenderTextureImplDefault::updateTexture(unsigned int textureId)
80{
81 // Make sure that the current texture binding will be preserved
82 priv::TextureSaver save;
83
84 // Copy the rendered pixels to the texture
85 glCheck(glBindTexture(GL_TEXTURE_2D, textureId));
86 glCheck(glCopyTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, 0, 0, m_width, m_height));
87}
