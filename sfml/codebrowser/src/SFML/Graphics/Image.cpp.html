64////////////////////////////////////////////////////////////
65void Image::create(unsigned int width, unsigned int height, const Color& color)
66{
67 if (width && height)
68 {
69 // Create a new pixel buffer first for exception safety's sake
70 std::vector<Uint8> newPixels(width * height * 4);
71
72 // Fill it with the specified color
73 Uint8* ptr = &newPixels[0];
74 Uint8* end = ptr + newPixels.size();
75 while (ptr < end)
76 {
77 *ptr++ = color.r;
78 *ptr++ = color.g;
79 *ptr++ = color.b;
80 *ptr++ = color.a;
81 }
82
83 // Commit the new pixel buffer
84 m_pixels.swap(newPixels);
85
86 // Assign the new size
87 m_size.x = width;
88 m_size.y = height;
89 }
90 else
91 {
92 // Dump the pixel buffer
93 std::vector<Uint8>().swap(m_pixels);
94
95 // Assign the new size
96 m_size.x = 0;
97 m_size.y = 0;
98 }
99}

102////////////////////////////////////////////////////////////
103void Image::create(unsigned int width, unsigned int height, const Uint8* pixels)
104{
105 if (pixels && width && height)
106 {
107 // Create a new pixel buffer first for exception safety's sake
108 std::vector<Uint8> newPixels(pixels, pixels + width * height * 4);
109
110 // Commit the new pixel buffer
111 m_pixels.swap(newPixels);
112
113 // Assign the new size
114 m_size.x = width;
115 m_size.y = height;
116 }
117 else
118 {
119 // Dump the pixel buffer
120 std::vector<Uint8>().swap(m_pixels);
121
122 // Assign the new size
123 m_size.x = 0;
124 m_size.y = 0;
125 }
126}

176////////////////////////////////////////////////////////////
177void Image::createMaskFromColor(const Color& color, Uint8 alpha)
178{
179 // Make sure that the image is not empty
180 if (!m_pixels.empty())
181 {
182 // Replace the alpha of the pixels that match the transparent color
183 Uint8* ptr = &m_pixels[0];
184 Uint8* end = ptr + m_pixels.size();
185 while (ptr < end)
186 {
187 if ((ptr[0] == color.r) && (ptr[1] == color.g) && (ptr[2] == color.b) && (ptr[3] == color.a))
188 ptr[3] = alpha;
189 ptr += 4;
190 }
191 }
192}

196void Image::copy(const Image& source, unsigned int destX, unsigned int destY, const IntRect& sourceRect, bool applyAlpha)
197{
198 // Make sure that both images are valid
199 if ((source.m_size.x == 0) || (source.m_size.y == 0) || (m_size.x == 0) || (m_size.y == 0))
200 return;
201
202 // Adjust the source rectangle
203 IntRect srcRect = sourceRect;
204 if (srcRect.width == 0 || (srcRect.height == 0))
205 {
206 srcRect.left = 0;
207 srcRect.top = 0;
208 srcRect.width = source.m_size.x;
209 srcRect.height = source.m_size.y;
210 }
211 else
212 {
213 if (srcRect.left < 0) srcRect.left = 0;
214 if (srcRect.top < 0) srcRect.top = 0;
215 if (srcRect.width > static_cast<int>(source.m_size.x)) srcRect.width = source.m_size.x;
216 if (srcRect.height > static_cast<int>(source.m_size.y)) srcRect.height = source.m_size.y;
217 }
218
219 // Then find the valid bounds of the destination rectangle
220 int width = srcRect.width;
221 int height = srcRect.height;
222 if (destX + width > m_size.x) width = m_size.x - destX;
223 if (destY + height > m_size.y) height = m_size.y - destY;
224
225 // Make sure the destination area is valid
226 if ((width <= 0) || (height <= 0))
227 return;
228
229 // Precompute as much as possible
230 int pitch = width * 4;
231 int rows = height;
232 int srcStride = source.m_size.x * 4;
233 int dstStride = m_size.x * 4;
234 const Uint8* srcPixels = &source.m_pixels[0] + (srcRect.left + srcRect.top * source.m_size.x) * 4;
235 Uint8* dstPixels = &m_pixels[0] + (destX + destY * m_size.x) * 4;
236
237 // Copy the pixels
238 if (applyAlpha)
239 {
240 // Interpolation using alpha values, pixel by pixel (slower)
241 for (int i = 0; i < rows; ++i)
242 {
243 for (int j = 0; j < width; ++j)
244 {
245 // Get a direct pointer to the components of the current pixel
246 const Uint8* src = srcPixels + j * 4;
247 Uint8* dst = dstPixels + j * 4;
248
249 // Interpolate RGBA components using the alpha value of the source pixel
250 Uint8 alpha = src[3];
251 dst[0] = (src[0] * alpha + dst[0] * (255 - alpha)) / 255;
252 dst[1] = (src[1] * alpha + dst[1] * (255 - alpha)) / 255;
253 dst[2] = (src[2] * alpha + dst[2] * (255 - alpha)) / 255;
254 dst[3] = alpha + dst[3] * (255 - alpha) / 255;
255 }
256
257 srcPixels += srcStride;
258 dstPixels += dstStride;
259 }
260 }
261 else
262 {
263 // Optimized copy ignoring alpha values, row by row (faster)
264 for (int i = 0; i < rows; ++i)
265 {
266 std::memcpy(dstPixels, srcPixels, pitch);
267 srcPixels += srcStride;
268 dstPixels += dstStride;
269 }
270 }
271}

274////////////////////////////////////////////////////////////
275void Image::setPixel(unsigned int x, unsigned int y, const Color& color)
276{
277 Uint8* pixel = &m_pixels[(x + y * m_size.x) * 4];
278 *pixel++ = color.r;
279 *pixel++ = color.g;
280 *pixel++ = color.b;
281 *pixel++ = color.a;
282}

308////////////////////////////////////////////////////////////
309void Image::flipHorizontally()
310{
311 if (!m_pixels.empty())
312 {
313 std::size_t rowSize = m_size.x * 4;
314
315 for (std::size_t y = 0; y < m_size.y; ++y)
316 {
317 std::vector<Uint8>::iterator left = m_pixels.begin() + y * rowSize;
318 std::vector<Uint8>::iterator right = m_pixels.begin() + (y + 1) * rowSize - 4;
319
320 for (std::size_t x = 0; x < m_size.x / 2; ++x)
321 {
322 std::swap_ranges(left, left + 4, right);
323
324 left += 4;
325 right -= 4;
326 }
327 }
328 }
329}

332////////////////////////////////////////////////////////////
333void Image::flipVertically()
334{
335 if (!m_pixels.empty())
336 {
337 std::size_t rowSize = m_size.x * 4;
338
339 std::vector<Uint8>::iterator top = m_pixels.begin();
340 std::vector<Uint8>::iterator bottom = m_pixels.end() - rowSize;
341
342 for (std::size_t y = 0; y < m_size.y / 2; ++y)
343 {
344 std::swap_ranges(top, top + rowSize, bottom);
345
346 top += rowSize;
347 bottom -= rowSize;
348 }
349 }
