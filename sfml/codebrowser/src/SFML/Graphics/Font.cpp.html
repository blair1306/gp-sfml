47 // FreeType callbacks that operate on a sf::InputStream
48 unsigned long read(FT_Stream rec, unsigned long offset, unsigned char* buffer, unsigned long count)
49 {
50 sf::InputStream* stream = static_cast<sf::InputStream*>(rec->descriptor.pointer);
51 if (static_cast<unsigned long>(stream->seek(offset)) == offset)
52 {
53 if (count > 0)
54 return static_cast<unsigned long>(stream->read(reinterpret_cast<char*>(buffer), count));
55 else
56 return 0;
57 }
58 else
59 return count > 0 ? 0 : 1; // error code is 0 if we're reading, or nonzero if we're seeking
60 }
61 void close(FT_Stream)
62 {
63 }
64}

121////////////////////////////////////////////////////////////
122bool Font::loadFromFile(const std::string& filename)
123{
124 #ifndef SFML_SYSTEM_ANDROID
125
126 // Cleanup the previous resources
127 cleanup();
128 m_refCount = new int(1);
129
130 // Initialize FreeType
131 // Note: we initialize FreeType for every font instance in order to avoid having a single
132 // global manager that would create a lot of issues regarding creation and destruction order.
133 FT_Library library;
134 if (FT_Init_FreeType(&library) != 0)
135 {
136 err() << "Failed to load font \"" << filename << "\" (failed to initialize FreeType)" << std::endl;
137 return false;
138 }
139 m_library = library;
140
141 // Load the new font face from the specified file
142 FT_Face face;
143 if (FT_New_Face(static_cast<FT_Library>(m_library), filename.c_str(), 0, &face) != 0)
144 {
145 err() << "Failed to load font \"" << filename << "\" (failed to create the font face)" << std::endl;
146 return false;
147 }
148
149 // Load the stroker that will be used to outline the font
150 FT_Stroker stroker;
151 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0)
152 {
153 err() << "Failed to load font \"" << filename << "\" (failed to create the stroker)" << std::endl;
154 FT_Done_Face(face);
155 return false;
156 }
157
158 // Select the unicode character map
159 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0)
160 {
161 err() << "Failed to load font \"" << filename << "\" (failed to set the Unicode character set)" << std::endl;
162 FT_Stroker_Done(stroker);
163 FT_Done_Face(face);
164 return false;
165 }
166
167 // Store the loaded font in our ugly void* :)
168 m_stroker = stroker;
169 m_face = face;
170
171 // Store the font information
172 m_info.family = face->family_name ? face->family_name : std::string();
173
174 return true;
175
176 #else
177
178 if (m_stream)
179 delete (priv::ResourceStream*)m_stream;
180
181 m_stream = new priv::ResourceStream(filename);
182 return loadFromStream(*(priv::ResourceStream*)m_stream);
183
184 #endif
185}

243////////////////////////////////////////////////////////////
244bool Font::loadFromStream(InputStream& stream)
245{
246 // Cleanup the previous resources
247 cleanup();
248 m_refCount = new int(1);
249
250 // Initialize FreeType
251 // Note: we initialize FreeType for every font instance in order to avoid having a single
252 // global manager that would create a lot of issues regarding creation and destruction order.
253 FT_Library library;
254 if (FT_Init_FreeType(&library) != 0)
255 {
256 err() << "Failed to load font from stream (failed to initialize FreeType)" << std::endl;
257 return false;
258 }
259 m_library = library;
260
261 // Make sure that the stream's reading position is at the beginning
262 stream.seek(0);
263
264 // Prepare a wrapper for our stream, that we'll pass to FreeType callbacks
265 FT_StreamRec* rec = new FT_StreamRec;
266 std::memset(rec, 0, sizeof(*rec));
267 rec->base = NULL;
268 rec->size = static_cast<unsigned long>(stream.getSize());
269 rec->pos = 0;
270 rec->descriptor.pointer = &stream;
271 rec->read = &read;
272 rec->close = &close;
273
274 // Setup the FreeType callbacks that will read our stream
275 FT_Open_Args args;
276 args.flags = FT_OPEN_STREAM;
277 args.stream = rec;
278 args.driver = 0;
279
280 // Load the new font face from the specified stream
281 FT_Face face;
282 if (FT_Open_Face(static_cast<FT_Library>(m_library), &args, 0, &face) != 0)
283 {
284 err() << "Failed to load font from stream (failed to create the font face)" << std::endl;
285 delete rec;
286 return false;
287 }
288
289 // Load the stroker that will be used to outline the font
290 FT_Stroker stroker;
291 if (FT_Stroker_New(static_cast<FT_Library>(m_library), &stroker) != 0)
292 {
293 err() << "Failed to load font from stream (failed to create the stroker)" << std::endl;
294 FT_Done_Face(face);
295 delete rec;
296 return false;
297 }
298
299 // Select the Unicode character map
300 if (FT_Select_Charmap(face, FT_ENCODING_UNICODE) != 0)
301 {
302 err() << "Failed to load font from stream (failed to set the Unicode character set)" << std::endl;
303 FT_Done_Face(face);
304 FT_Stroker_Done(stroker);
305 delete rec;
306 return false;
307 }
308
309 // Store the loaded font in our ugly void* :)
310 m_stroker = stroker;
311 m_face = face;
312 m_streamRec = rec;
313
314 // Store the font information
315 m_info.family = face->family_name ? face->family_name : std::string();
316
317 return true;
318}

329const Glyph& Font::getGlyph(Uint32 codePoint, unsigned int characterSize, bool bold, float outlineThickness) const
330{
331 // Get the page corresponding to the character size
332 GlyphTable& glyphs = m_pages[characterSize].glyphs;
333
334 // Build the key by combining the code point, bold flag, and outline thickness
335 Uint64 key = (static_cast<Uint64>(*reinterpret_cast<Uint32*>(&outlineThickness)) << 32)
336 | (static_cast<Uint64>(bold ? 1 : 0) << 31)
337 | static_cast<Uint64>(codePoint);
338
339 // Search the glyph into the cache
340 GlyphTable::const_iterator it = glyphs.find(key);
341 if (it != glyphs.end())
342 {
343 // Found: just return it
344 return it->second;
345 }
346 else
347 {
348 // Not found: we have to load it
349 Glyph glyph = loadGlyph(codePoint, characterSize, bold, outlineThickness);
350 return glyphs.insert(std::make_pair(key, glyph)).first->second;
351 }
352}

355////////////////////////////////////////////////////////////
356float Font::getKerning(Uint32 first, Uint32 second, unsigned int characterSize) const
357{
358 // Special case where first or second is 0 (null character)
359 if (first == 0 || second == 0)
360 return 0.f;
361
362 FT_Face face = static_cast<FT_Face>(m_face);
363
364 if (face && FT_HAS_KERNING(face) && setCurrentSize(characterSize))
365 {
366 // Convert the characters to indices
367 FT_UInt index1 = FT_Get_Char_Index(face, first);
368 FT_UInt index2 = FT_Get_Char_Index(face, second);
369
370 // Get the kerning vector
371 FT_Vector kerning;
372 FT_Get_Kerning(face, index1, index2, FT_KERNING_DEFAULT, &kerning);
373
374 // X advance is already in pixels for bitmap fonts
375 if (!FT_IS_SCALABLE(face))
376 return static_cast<float>(kerning.x);
377
378 // Return the X advance
379 return static_cast<float>(kerning.x) / static_cast<float>(1 << 6);
380 }
381 else
382 {
383 // Invalid font, or no kerning
384 return 0.f;
385 }
386}

405////////////////////////////////////////////////////////////
406float Font::getUnderlinePosition(unsigned int characterSize) const
407{
408 FT_Face face = static_cast<FT_Face>(m_face);
409
410 if (face && setCurrentSize(characterSize))
411 {
412 // Return a fixed position if font is a bitmap font
413 if (!FT_IS_SCALABLE(face))
414 return characterSize / 10.f;
415
416 return -static_cast<float>(FT_MulFix(face->underline_position, face->size->metrics.y_scale)) / static_cast<float>(1 << 6);
417 }
418 else
419 {
420 return 0.f;
421 }
422}

474////////////////////////////////////////////////////////////
475void Font::cleanup()
476{
477 // Check if we must destroy the FreeType pointers
478 if (m_refCount)
479 {
480 // Decrease the reference counter
481 (*m_refCount)--;
482
483 // Free the resources only if we are the last owner
484 if (*m_refCount == 0)
485 {
486 // Delete the reference counter
487 delete m_refCount;
488
489 // Destroy the stroker
490 if (m_stroker)
491 FT_Stroker_Done(static_cast<FT_Stroker>(m_stroker));
492
493 // Destroy the font face
494 if (m_face)
495 FT_Done_Face(static_cast<FT_Face>(m_face));
496
497 // Destroy the stream rec instance, if any (must be done after FT_Done_Face!)
498 if (m_streamRec)
499 delete static_cast<FT_StreamRec*>(m_streamRec);
500
501 // Close the library
502 if (m_library)
503 FT_Done_FreeType(static_cast<FT_Library>(m_library));
504 }
505 }
506
507 // Reset members
508 m_library = NULL;
509 m_face = NULL;
510 m_stroker = NULL;
511 m_streamRec = NULL;
512 m_refCount = NULL;
513 m_pages.clear();
514 std::vector<Uint8>().swap(m_pixelBuffer);
515}

518////////////////////////////////////////////////////////////
519Glyph Font::loadGlyph(Uint32 codePoint, unsigned int characterSize, bool bold, float outlineThickness) const
520{
521 // The glyph to return
522 Glyph glyph;
523
524 // First, transform our ugly void* to a FT_Face
525 FT_Face face = static_cast<FT_Face>(m_face);
526 if (!face)
527 return glyph;
528
529 // Set the character size
530 if (!setCurrentSize(characterSize))
531 return glyph;
532
533 // Load the glyph corresponding to the code point
534 FT_Int32 flags = FT_LOAD_TARGET_NORMAL | FT_LOAD_FORCE_AUTOHINT;
535 if (outlineThickness != 0)
536 flags |= FT_LOAD_NO_BITMAP;
537 if (FT_Load_Char(face, codePoint, flags) != 0)
538 return glyph;
539
540 // Retrieve the glyph
541 FT_Glyph glyphDesc;
542 if (FT_Get_Glyph(face->glyph, &glyphDesc) != 0)
543 return glyph;
544
545 // Apply bold and outline (there is no fallback for outline) if necessary -- first technique using outline (highest quality)
546 FT_Pos weight = 1 << 6;
547 bool outline = (glyphDesc->format == FT_GLYPH_FORMAT_OUTLINE);
548 if (outline)
549 {
550 if (bold)
551 {
552 FT_OutlineGlyph outlineGlyph = (FT_OutlineGlyph)glyphDesc;
553 FT_Outline_Embolden(&outlineGlyph->outline, weight);
554 }
555
556 if (outlineThickness != 0)
557 {
558 FT_Stroker stroker = static_cast<FT_Stroker>(m_stroker);
559
560 FT_Stroker_Set(stroker, static_cast<FT_Fixed>(outlineThickness * static_cast<float>(1 << 6)), FT_STROKER_LINECAP_ROUND, FT_STROKER_LINEJOIN_ROUND, 0);
561 FT_Glyph_Stroke(&glyphDesc, stroker, false);
562 }
563 }
564
565 // Convert the glyph to a bitmap (i.e. rasterize it)
566 FT_Glyph_To_Bitmap(&glyphDesc, FT_RENDER_MODE_NORMAL, 0, 1);
567 FT_Bitmap& bitmap = reinterpret_cast<FT_BitmapGlyph>(glyphDesc)->bitmap;
568
569 // Apply bold if necessary -- fallback technique using bitmap (lower quality)
570 if (!outline)
571 {
572 if (bold)
573 FT_Bitmap_Embolden(static_cast<FT_Library>(m_library), &bitmap, weight, weight);
574
575 if (outlineThickness != 0)
576 err() << "Failed to outline glyph (no fallback available)" << std::endl;
577 }
578
579 // Compute the glyph's advance offset
580 glyph.advance = static_cast<float>(face->glyph->metrics.horiAdvance) / static_cast<float>(1 << 6);
581 if (bold)
582 glyph.advance += static_cast<float>(weight) / static_cast<float>(1 << 6);
583
584 int width = bitmap.width;
585 int height = bitmap.rows;
586
587 if ((width > 0) && (height > 0))
588 {
589 // Leave a small padding around characters, so that filtering doesn't
590 // pollute them with pixels from neighbors
591 const unsigned int padding = 1;
592
593 width += 2 * padding;
594 height += 2 * padding;
595
596 // Get the glyphs page corresponding to the character size
597 Page& page = m_pages[characterSize];
598
599 // Find a good position for the new glyph into the texture
600 glyph.textureRect = findGlyphRect(page, width, height);
601
602 // Make sure the texture data is positioned in the center
603 // of the allocated texture rectangle
604 glyph.textureRect.left += padding;
605 glyph.textureRect.top += padding;
606 glyph.textureRect.width -= 2 * padding;
607 glyph.textureRect.height -= 2 * padding;
608
609 // Compute the glyph's bounding box
610 glyph.bounds.left = static_cast<float>(face->glyph->metrics.horiBearingX) / static_cast<float>(1 << 6);
611 glyph.bounds.top = -static_cast<float>(face->glyph->metrics.horiBearingY) / static_cast<float>(1 << 6);
612 glyph.bounds.width = static_cast<float>(face->glyph->metrics.width) / static_cast<float>(1 << 6) + outlineThickness * 2;
613 glyph.bounds.height = static_cast<float>(face->glyph->metrics.height) / static_cast<float>(1 << 6) + outlineThickness * 2;
614
615 // Resize the pixel buffer to the new size and fill it with transparent white pixels
616 m_pixelBuffer.resize(width * height * 4);
617
618 Uint8* current = &m_pixelBuffer[0];
619 Uint8* end = current + width * height * 4;
620
621 while (current != end)
622 {
623 (*current++) = 255;
624 (*current++) = 255;
625 (*current++) = 255;
626 (*current++) = 0;
627 }
628
629 // Extract the glyph's pixels from the bitmap
630 const Uint8* pixels = bitmap.buffer;
631 if (bitmap.pixel_mode == FT_PIXEL_MODE_MONO)
632 {
633 // Pixels are 1 bit monochrome values
634 for (int y = padding; y < height - padding; ++y)
635 {
636 for (int x = padding; x < width - padding; ++x)
637 {
638 // The color channels remain white, just fill the alpha channel
639 std::size_t index = x + y * width;
640 m_pixelBuffer[index * 4 + 3] = ((pixels[(x - padding) / 8]) & (1 << (7 - ((x - padding) % 8)))) ? 255 : 0;
641 }
642 pixels += bitmap.pitch;
643 }
644 }
645 else
646 {
647 // Pixels are 8 bits gray levels
648 for (int y = padding; y < height - padding; ++y)
649 {
650 for (int x = padding; x < width - padding; ++x)
651 {
652 // The color channels remain white, just fill the alpha channel
653 std::size_t index = x + y * width;
654 m_pixelBuffer[index * 4 + 3] = pixels[x - padding];
655 }
656 pixels += bitmap.pitch;
657 }
658 }
659
660 // Write the pixels to the texture
661 unsigned int x = glyph.textureRect.left - padding;
662 unsigned int y = glyph.textureRect.top - padding;
663 unsigned int w = glyph.textureRect.width + 2 * padding;
664 unsigned int h = glyph.textureRect.height + 2 * padding;
665 page.texture.update(&m_pixelBuffer[0], w, h, x, y);
666 }
667
668 // Delete the FT glyph
669 FT_Done_Glyph(glyphDesc);
670
671 // Done :)
672 return glyph;
673}

676////////////////////////////////////////////////////////////
677IntRect Font::findGlyphRect(Page& page, unsigned int width, unsigned int height) const
678{
679 // Find the line that fits well the glyph
680 Row* row = NULL;
681 float bestRatio = 0;
682 for (std::vector<Row>::iterator it = page.rows.begin(); it != page.rows.end() && !row; ++it)
683 {
684 float ratio = static_cast<float>(height) / it->height;
685
686 // Ignore rows that are either too small or too high
687 if ((ratio < 0.7f) || (ratio > 1.f))
688 continue;
689
690 // Check if there's enough horizontal space left in the row
691 if (width > page.texture.getSize().x - it->width)
692 continue;
693
694 // Make sure that this new row is the best found so far
695 if (ratio < bestRatio)
696 continue;
697
698 // The current row passed all the tests: we can select it
699 row = &*it;
700 bestRatio = ratio;
701 }
702
703 // If we didn't find a matching row, create a new one (10% taller than the glyph)
704 if (!row)
705 {
706 int rowHeight = height + height / 10;
707 while ((page.nextRow + rowHeight >= page.texture.getSize().y) || (width >= page.texture.getSize().x))
708 {
709 // Not enough space: resize the texture if possible
710 unsigned int textureWidth = page.texture.getSize().x;
711 unsigned int textureHeight = page.texture.getSize().y;
712 if ((textureWidth * 2 <= Texture::getMaximumSize()) && (textureHeight * 2 <= Texture::getMaximumSize()))
713 {
714 // Make the texture 2 times bigger
715 Texture newTexture;
716 newTexture.create(textureWidth * 2, textureHeight * 2);
717 newTexture.update(page.texture);
718 page.texture.swap(newTexture);
719 }
720 else
721 {
722 // Oops, we've reached the maximum texture size...
723 err() << "Failed to add a new character to the font: the maximum texture size has been reached" << std::endl;
724 return IntRect(0, 0, 2, 2);
725 }
726 }
727
728 // We can now create the new row
729 page.rows.push_back(Row(page.nextRow, rowHeight));
730 page.nextRow += rowHeight;
731 row = &page.rows.back();
732 }
733
734 // Find the glyph's rectangle on the selected row
735 IntRect rect(row->width, row->top, width, height);
736
737 // Update the row informations
738 row->width += width;
739
740 return rect;
741}

744////////////////////////////////////////////////////////////
745bool Font::setCurrentSize(unsigned int characterSize) const
746{
747 // FT_Set_Pixel_Sizes is an expensive function, so we must call it
748 // only when necessary to avoid killing performances
749
750 FT_Face face = static_cast<FT_Face>(m_face);
751 FT_UShort currentSize = face->size->metrics.x_ppem;
752
753 if (currentSize != characterSize)
754 {
755 FT_Error result = FT_Set_Pixel_Sizes(face, 0, characterSize);
756
757 if (result == FT_Err_Invalid_Pixel_Size)
758 {
759 // In the case of bitmap fonts, resizing can
760 // fail if the requested size is not available
761 if (!FT_IS_SCALABLE(face))
762 {
763 err() << "Failed to set bitmap font size to " << characterSize << std::endl;
764 err() << "Available sizes are: ";
765 for (int i = 0; i < face->num_fixed_sizes; ++i)
