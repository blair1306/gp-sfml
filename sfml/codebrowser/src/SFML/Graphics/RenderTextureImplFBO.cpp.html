38////////////////////////////////////////////////////////////
39RenderTextureImplFBO::RenderTextureImplFBO() :
40m_context (NULL),
41m_frameBuffer(0),
42m_depthBuffer(0)
43{
44
45}
46
47
48////////////////////////////////////////////////////////////
49RenderTextureImplFBO::~RenderTextureImplFBO()
50{
51 m_context->setActive(true);
52
53 // Destroy the depth buffer
54 if (m_depthBuffer)
55 {
56 GLuint depthBuffer = static_cast<GLuint>(m_depthBuffer);
57 glCheck(GLEXT_glDeleteRenderbuffers(1, &depthBuffer));
58 }
59
60 // Destroy the frame buffer
61 if (m_frameBuffer)
62 {
63 GLuint frameBuffer = static_cast<GLuint>(m_frameBuffer);
64 glCheck(GLEXT_glDeleteFramebuffers(1, &frameBuffer));
65 }
66
67 // Delete the context
68 delete m_context;
69}

84////////////////////////////////////////////////////////////
85bool RenderTextureImplFBO::create(unsigned int width, unsigned int height, unsigned int textureId, bool depthBuffer)
86{
87 // Create the context
88 m_context = new Context;
89
90 // Create the framebuffer object
91 GLuint frameBuffer = 0;
92 glCheck(GLEXT_glGenFramebuffers(1, &frameBuffer));
93 m_frameBuffer = static_cast<unsigned int>(frameBuffer);
94 if (!m_frameBuffer)
95 {
96 err() << "Impossible to create render texture (failed to create the frame buffer object)" << std::endl;
97 return false;
98 }
99 glCheck(GLEXT_glBindFramebuffer(GLEXT_GL_FRAMEBUFFER, m_frameBuffer));
100
101 // Create the depth buffer if requested
102 if (depthBuffer)
103 {
104 GLuint depth = 0;
105 glCheck(GLEXT_glGenRenderbuffers(1, &depth));
106 m_depthBuffer = static_cast<unsigned int>(depth);
107 if (!m_depthBuffer)
108 {
109 err() << "Impossible to create render texture (failed to create the attached depth buffer)" << std::endl;
110 return false;
111 }
112 glCheck(GLEXT_glBindRenderbuffer(GLEXT_GL_RENDERBUFFER, m_depthBuffer));
113 glCheck(GLEXT_glRenderbufferStorage(GLEXT_GL_RENDERBUFFER, GLEXT_GL_DEPTH_COMPONENT, width, height));
114 glCheck(GLEXT_glFramebufferRenderbuffer(GLEXT_GL_FRAMEBUFFER, GLEXT_GL_DEPTH_ATTACHMENT, GLEXT_GL_RENDERBUFFER, m_depthBuffer));
115 }
116
117 // Link the texture to the frame buffer
118 glCheck(GLEXT_glFramebufferTexture2D(GLEXT_GL_FRAMEBUFFER, GLEXT_GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, textureId, 0));
119
120 // A final check, just to be sure...
121 GLenum status;
122 glCheck(status = GLEXT_glCheckFramebufferStatus(GLEXT_GL_FRAMEBUFFER));
123 if (status != GLEXT_GL_FRAMEBUFFER_COMPLETE)
124 {
125 glCheck(GLEXT_glBindFramebuffer(GLEXT_GL_FRAMEBUFFER, 0));
126 err() << "Impossible to create render texture (failed to link the target texture to the frame buffer)" << std::endl;
127 return false;
128 }
129
130 return true;
131}
