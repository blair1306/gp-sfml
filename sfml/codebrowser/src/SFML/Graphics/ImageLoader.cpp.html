40 // Convert a string to lower case
41 std::string toLower(std::string str)
42 {
43 for (std::string::iterator i = str.begin(); i != str.end(); ++i)
44 *i = static_cast<char>(std::tolower(*i));
45 return str;
46 }
47
48 // stb_image callbacks that operate on a sf::InputStream
49 int read(void* user, char* data, int size)
50 {
51 sf::InputStream* stream = static_cast<sf::InputStream*>(user);
52 return static_cast<int>(stream->read(data, size));
53 }
54 void skip(void* user, int size)
55 {
56 sf::InputStream* stream = static_cast<sf::InputStream*>(user);
57 stream->seek(stream->tell() + size);
58 }
59 int eof(void* user)
60 {
61 sf::InputStream* stream = static_cast<sf::InputStream*>(user);
62 return stream->tell() >= stream->getSize();
63 }
64}

95bool ImageLoader::loadImageFromFile(const std::string& filename, std::vector<Uint8>& pixels, Vector2u& size)
96{
97 // Clear the array (just in case)
98 pixels.clear();
99
100 // Load the image and get a pointer to the pixels in memory
101 int width = 0;
102 int height = 0;
103 int channels = 0;
104 unsigned char* ptr = stbi_load(filename.c_str(), &width, &height, &channels, STBI_rgb_alpha);
105
106 if (ptr)
107 {
108 // Assign the image properties
109 size.x = width;
110 size.y = height;
111
112 if (width && height)
113 {
114 // Copy the loaded pixels to the pixel buffer
115 pixels.resize(width * height * 4);
116 memcpy(&pixels[0], ptr, pixels.size());
117 }
118
119 // Free the loaded pixels (they are now in our own pixel buffer)
120 stbi_image_free(ptr);
121
122 return true;
123 }
124 else
125 {
126 // Error, failed to load the image
127 err() << "Failed to load image \"" << filename << "\". Reason: " << stbi_failure_reason() << std::endl;
128
129 return false;
130 }
131}

135bool ImageLoader::loadImageFromMemory(const void* data, std::size_t dataSize, std::vector<Uint8>& pixels, Vector2u& size)
136{
137 // Check input parameters
138 if (data && dataSize)
139 {
140 // Clear the array (just in case)
141 pixels.clear();
142
143 // Load the image and get a pointer to the pixels in memory
144 int width = 0;
145 int height = 0;
146 int channels = 0;
147 const unsigned char* buffer = static_cast<const unsigned char*>(data);
148 unsigned char* ptr = stbi_load_from_memory(buffer, static_cast<int>(dataSize), &width, &height, &channels, STBI_rgb_alpha);
149
150 if (ptr)
151 {
152 // Assign the image properties
153 size.x = width;
154 size.y = height;
155
156 if (width && height)
157 {
158 // Copy the loaded pixels to the pixel buffer
159 pixels.resize(width * height * 4);
160 memcpy(&pixels[0], ptr, pixels.size());
161 }
162
163 // Free the loaded pixels (they are now in our own pixel buffer)
164 stbi_image_free(ptr);
165
166 return true;
167 }
168 else
169 {
170 // Error, failed to load the image
171 err() << "Failed to load image from memory. Reason: " << stbi_failure_reason() << std::endl;
172
173 return false;
174 }
175 }
176 else
177 {
178 err() << "Failed to load image from memory, no data provided" << std::endl;
179 return false;

185bool ImageLoader::loadImageFromStream(InputStream& stream, std::vector<Uint8>& pixels, Vector2u& size)
186{
187 // Clear the array (just in case)
188 pixels.clear();
189
190 // Make sure that the stream's reading position is at the beginning
191 stream.seek(0);
192
193 // Setup the stb_image callbacks
194 stbi_io_callbacks callbacks;
195 callbacks.read = &read;
196 callbacks.skip = &skip;
197 callbacks.eof = &eof;
198
199 // Load the image and get a pointer to the pixels in memory
200 int width = 0;
201 int height = 0;
202 int channels = 0;
203 unsigned char* ptr = stbi_load_from_callbacks(&callbacks, &stream, &width, &height, &channels, STBI_rgb_alpha);
204
205 if (ptr)
206 {
207 // Assign the image properties
208 size.x = width;
209 size.y = height;
210
211 if (width && height)
212 {
213 // Copy the loaded pixels to the pixel buffer
214 pixels.resize(width * height * 4);
215 memcpy(&pixels[0], ptr, pixels.size());
216 }
217
218 // Free the loaded pixels (they are now in our own pixel buffer)
219 stbi_image_free(ptr);
220
221 return true;
222 }
223 else
224 {
225 // Error, failed to load the image
226 err() << "Failed to load image from stream. Reason: " << stbi_failure_reason() << std::endl;
227
228 return false;
229 }
230}

234bool ImageLoader::saveImageToFile(const std::string& filename, const std::vector<Uint8>& pixels, const Vector2u& size)
235{
236 // Make sure the image is not empty
237 if (!pixels.empty() && (size.x > 0) && (size.y > 0))
238 {
239 // Deduce the image type from its extension
240
241 // Extract the extension
242 const std::size_t dot = filename.find_last_of('.');
243 const std::string extension = dot != std::string::npos ? toLower(filename.substr(dot + 1)) : "";
244
245 if (extension == "bmp")
246 {
247 // BMP format
248 if (stbi_write_bmp(filename.c_str(), size.x, size.y, 4, &pixels[0]))
249 return true;
250 }
251 else if (extension == "tga")
252 {
253 // TGA format
254 if (stbi_write_tga(filename.c_str(), size.x, size.y, 4, &pixels[0]))
255 return true;
256 }
257 else if (extension == "png")
258 {
259 // PNG format
260 if (stbi_write_png(filename.c_str(), size.x, size.y, 4, &pixels[0], 0))
261 return true;
262 }
263 else if (extension == "jpg" || extension == "jpeg")
264 {
265 // JPG format
266 if (stbi_write_jpg(filename.c_str(), size.x, size.y, 4, &pixels[0], 90))
267 return true;
268 }
269 }
270
271 err() << "Failed to save image \"" << filename << "\"" << std::endl;
272 return false;
273}
