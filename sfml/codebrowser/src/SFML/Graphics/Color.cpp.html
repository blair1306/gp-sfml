48////////////////////////////////////////////////////////////
49Color::Color() :
50r(0),
51g(0),
52b(0),
53a(255)
54{
55
56}
57
58
59////////////////////////////////////////////////////////////
60Color::Color(Uint8 red, Uint8 green, Uint8 blue, Uint8 alpha) :
61r(red),
62g(green),
63b(blue),
64a(alpha)
65{
66
67}
68
69
70////////////////////////////////////////////////////////////
71Color::Color(Uint32 color) :
72r((color & 0xff000000) >> 24),
73g((color & 0x00ff0000) >> 16),
74b((color & 0x0000ff00) >> 8 ),
75a((color & 0x000000ff) >> 0 )
76{
77
78}
79
80
81////////////////////////////////////////////////////////////
82Uint32 Color::toInteger() const
83{
84 return (r << 24) | (g << 16) | (b << 8) | a;
85}
86
87
88////////////////////////////////////////////////////////////
89bool operator ==(const Color& left, const Color& right)
90{
91 return (left.r == right.r) &&
92 (left.g == right.g) &&
93 (left.b == right.b) &&
94 (left.a == right.a);
95}
96
97
98////////////////////////////////////////////////////////////
99bool operator !=(const Color& left, const Color& right)
100{
101 return !(left == right);
102}
103
104
105////////////////////////////////////////////////////////////
106Color operator +(const Color& left, const Color& right)
107{
108 return Color(Uint8(std::min(int(left.r) + right.r, 255)),
109 Uint8(std::min(int(left.g) + right.g, 255)),
110 Uint8(std::min(int(left.b) + right.b, 255)),
111 Uint8(std::min(int(left.a) + right.a, 255)));
112}
113
114
115////////////////////////////////////////////////////////////
116Color operator -(const Color& left, const Color& right)
117{
118 return Color(Uint8(std::max(int(left.r) - right.r, 0)),
119 Uint8(std::max(int(left.g) - right.g, 0)),
120 Uint8(std::max(int(left.b) - right.b, 0)),
121 Uint8(std::max(int(left.a) - right.a, 0)));
122}
123
124
125////////////////////////////////////////////////////////////
126Color operator *(const Color& left, const Color& right)
127{
128 return Color(Uint8(int(left.r) * right.r / 255),
129 Uint8(int(left.g) * right.g / 255),
130 Uint8(int(left.b) * right.b / 255),
131 Uint8(int(left.a) * right.a / 255));
132}
133
134
135////////////////////////////////////////////////////////////
136Color& operator +=(Color& left, const Color& right)
137{
138 return left = left + right;
139}
140
141
142////////////////////////////////////////////////////////////
143Color& operator -=(Color& left, const Color& right)
144{
145 return left = left - right;
146}
147
148
149////////////////////////////////////////////////////////////
150Color& operator *=(Color& left, const Color& right)
151{
152 return left = left * right;
153}
