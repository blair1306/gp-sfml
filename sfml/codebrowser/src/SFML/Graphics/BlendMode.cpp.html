34// Commonly used blending modes
35////////////////////////////////////////////////////////////
36const BlendMode BlendAlpha(BlendMode::SrcAlpha, BlendMode::OneMinusSrcAlpha, BlendMode::Add,
37 BlendMode::One, BlendMode::OneMinusSrcAlpha, BlendMode::Add);
38const BlendMode BlendAdd(BlendMode::SrcAlpha, BlendMode::One, BlendMode::Add,
39 BlendMode::One, BlendMode::One, BlendMode::Add);
40const BlendMode BlendMultiply(BlendMode::DstColor, BlendMode::Zero);
41const BlendMode BlendNone(BlendMode::One, BlendMode::Zero);

44////////////////////////////////////////////////////////////
45BlendMode::BlendMode() :
46colorSrcFactor(BlendMode::SrcAlpha),
47colorDstFactor(BlendMode::OneMinusSrcAlpha),
48colorEquation (BlendMode::Add),
49alphaSrcFactor(BlendMode::One),
50alphaDstFactor(BlendMode::OneMinusSrcAlpha),
51alphaEquation (BlendMode::Add)
52{
53
54}
55
56
57////////////////////////////////////////////////////////////
58BlendMode::BlendMode(Factor sourceFactor, Factor destinationFactor, Equation blendEquation) :
59colorSrcFactor(sourceFactor),
60colorDstFactor(destinationFactor),
61colorEquation (blendEquation),
62alphaSrcFactor(sourceFactor),
63alphaDstFactor(destinationFactor),
64alphaEquation (blendEquation)
65{
66
67}
68
69
70////////////////////////////////////////////////////////////
71BlendMode::BlendMode(Factor colorSourceFactor, Factor colorDestinationFactor,
72 Equation colorBlendEquation, Factor alphaSourceFactor,
73 Factor alphaDestinationFactor, Equation alphaBlendEquation) :
74colorSrcFactor(colorSourceFactor),
75colorDstFactor(colorDestinationFactor),
76colorEquation (colorBlendEquation),
77alphaSrcFactor(alphaSourceFactor),
78alphaDstFactor(alphaDestinationFactor),
79alphaEquation (alphaBlendEquation)
80{
81
82}

85////////////////////////////////////////////////////////////
86bool operator ==(const BlendMode& left, const BlendMode& right)
87{
88 return (left.colorSrcFactor == right.colorSrcFactor) &&
89 (left.colorDstFactor == right.colorDstFactor) &&
90 (left.colorEquation == right.colorEquation) &&
91 (left.alphaSrcFactor == right.alphaSrcFactor) &&
92 (left.alphaDstFactor == right.alphaDstFactor) &&
93 (left.alphaEquation == right.alphaEquation);
94}
95
96
97////////////////////////////////////////////////////////////
98bool operator !=(const BlendMode& left, const BlendMode& right)
99{
100 return !(left == right);
101}
