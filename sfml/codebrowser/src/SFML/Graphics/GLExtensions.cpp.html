46void ensureExtensionsInit()
47{
48#if !defined(SFML_OPENGL_ES)
49 static bool initialized = false;
50 if (!initialized)
51 {
52 initialized = true;
53
54 sfogl_LoadFunctions();
55
56 // Retrieve the context version number
57 int majorVersion = 0;
58 int minorVersion = 0;
59
60 // Try the new way first
61 glGetIntegerv(GL_MAJOR_VERSION, &majorVersion);
62 glGetIntegerv(GL_MINOR_VERSION, &minorVersion);
63
64 if (glGetError() == GL_INVALID_ENUM)
65 {
66 // Try the old way
67 const GLubyte* version = glGetString(GL_VERSION);
68 if (version)
69 {
70 // The beginning of the returned string is "major.minor" (this is standard)
71 majorVersion = version[0] - '0';
72 minorVersion = version[2] - '0';
73 }
74 else
75 {
76 // Can't get the version number, assume 1.1
77 majorVersion = 1;
78 minorVersion = 1;
79 }
80 }
81
82 if ((majorVersion < 1) || ((majorVersion == 1) && (minorVersion < 1)))
83 {
84 err() << "sfml-graphics requires support for OpenGL 1.1 or greater" << std::endl;
85 err() << "Ensure that hardware acceleration is enabled if available" << std::endl;
86 }
87 }
88#endif
89}
