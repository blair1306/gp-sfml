36 // Add an underline or strikethrough line to the vertex array
37 void addLine(sf::VertexArray& vertices, float lineLength, float lineTop, const sf::Color& color, float offset, float thickness, float outlineThickness = 0)
38 {
39 float top = std::floor(lineTop + offset - (thickness / 2) + 0.5f);
40 float bottom = top + std::floor(thickness + 0.5f);
41
42 vertices.append(sf::Vertex(sf::Vector2f(-outlineThickness, top - outlineThickness), color, sf::Vector2f(1, 1)));
43 vertices.append(sf::Vertex(sf::Vector2f(lineLength + outlineThickness, top - outlineThickness), color, sf::Vector2f(1, 1)));
44 vertices.append(sf::Vertex(sf::Vector2f(-outlineThickness, bottom + outlineThickness), color, sf::Vector2f(1, 1)));
45 vertices.append(sf::Vertex(sf::Vector2f(-outlineThickness, bottom + outlineThickness), color, sf::Vector2f(1, 1)));
46 vertices.append(sf::Vertex(sf::Vector2f(lineLength + outlineThickness, top - outlineThickness), color, sf::Vector2f(1, 1)));
47 vertices.append(sf::Vertex(sf::Vector2f(lineLength + outlineThickness, bottom + outlineThickness), color, sf::Vector2f(1, 1)));
48 }
49
50 // Add a glyph quad to the vertex array
51 void addGlyphQuad(sf::VertexArray& vertices, sf::Vector2f position, const sf::Color& color, const sf::Glyph& glyph, float italic, float outlineThickness = 0)
52 {
53 float left = glyph.bounds.left;
54 float top = glyph.bounds.top;
55 float right = glyph.bounds.left + glyph.bounds.width;
56 float bottom = glyph.bounds.top + glyph.bounds.height;
57
58 float u1 = static_cast<float>(glyph.textureRect.left);
59 float v1 = static_cast<float>(glyph.textureRect.top);
60 float u2 = static_cast<float>(glyph.textureRect.left + glyph.textureRect.width);
61 float v2 = static_cast<float>(glyph.textureRect.top + glyph.textureRect.height);
62
63 vertices.append(sf::Vertex(sf::Vector2f(position.x + left - italic * top - outlineThickness, position.y + top - outlineThickness), color, sf::Vector2f(u1, v1)));
64 vertices.append(sf::Vertex(sf::Vector2f(position.x + right - italic * top - outlineThickness, position.y + top - outlineThickness), color, sf::Vector2f(u2, v1)));
65 vertices.append(sf::Vertex(sf::Vector2f(position.x + left - italic * bottom - outlineThickness, position.y + bottom - outlineThickness), color, sf::Vector2f(u1, v2)));
66 vertices.append(sf::Vertex(sf::Vector2f(position.x + left - italic * bottom - outlineThickness, position.y + bottom - outlineThickness), color, sf::Vector2f(u1, v2)));
67 vertices.append(sf::Vertex(sf::Vector2f(position.x + right - italic * top - outlineThickness, position.y + top - outlineThickness), color, sf::Vector2f(u2, v1)));
68 vertices.append(sf::Vertex(sf::Vector2f(position.x + right - italic * bottom - outlineThickness, position.y + bottom - outlineThickness), color, sf::Vector2f(u2, v2)));
69 }
70}

162////////////////////////////////////////////////////////////
163void Text::setFillColor(const Color& color)
164{
165 if (color != m_fillColor)
166 {
167 m_fillColor = color;
168
169 // Change vertex colors directly, no need to update whole geometry
170 // (if geometry is updated anyway, we can skip this step)
171 if (!m_geometryNeedUpdate)
172 {
173 for (std::size_t i = 0; i < m_vertices.getVertexCount(); ++i)
174 m_vertices[i].color = m_fillColor;
175 }
176 }
177}

266Vector2f Text::findCharacterPos(std::size_t index) const
267{
268 // Make sure that we have a valid font
269 if (!m_font)
270 return Vector2f();
271
272 // Adjust the index if it's out of range
273 if (index > m_string.getSize())
274 index = m_string.getSize();
275
276 // Precompute the variables needed by the algorithm
277 bool bold = (m_style & Bold) != 0;
278 float hspace = static_cast<float>(m_font->getGlyph(L' ', m_characterSize, bold).advance);
279 float vspace = static_cast<float>(m_font->getLineSpacing(m_characterSize));
280
281 // Compute the position
282 Vector2f position;
283 Uint32 prevChar = 0;
284 for (std::size_t i = 0; i < index; ++i)
285 {
286 Uint32 curChar = m_string[i];
287
288 // Apply the kerning offset
289 position.x += static_cast<float>(m_font->getKerning(prevChar, curChar, m_characterSize));
290 prevChar = curChar;
291
292 // Handle special characters
293 switch (curChar)
294 {
295 case ' ': position.x += hspace; continue;
296 case '\t': position.x += hspace * 4; continue;
297 case '\n': position.y += vspace; position.x = 0; continue;
298 }
299
300 // For regular characters, add the advance offset of the glyph
301 position.x += static_cast<float>(m_font->getGlyph(curChar, m_characterSize, bold).advance);
302 }
303
304 // Transform the position to global coordinates
305 position = getTransform().transformPoint(position);
306
307 return position;
308}

328void Text::draw(RenderTarget& target, RenderStates states) const
329{
330 if (m_font)
331 {
332 ensureGeometryUpdate();
333
334 states.transform *= getTransform();
335 states.texture = &m_font->getTexture(m_characterSize);
336
337 // Only draw the outline if there is something to draw
338 if (m_outlineThickness != 0)
339 target.draw(m_outlineVertices, states);
340
341 target.draw(m_vertices, states);
342 }
343}

346////////////////////////////////////////////////////////////
347void Text::ensureGeometryUpdate() const
348{
349 // Do nothing, if geometry has not changed
350 if (!m_geometryNeedUpdate)
351 return;
352
353 // Mark geometry as updated
354 m_geometryNeedUpdate = false;
355
356 // Clear the previous geometry
357 m_vertices.clear();
358 m_outlineVertices.clear();
359 m_bounds = FloatRect();
360
361 // No font or text: nothing to draw
362 if (!m_font || m_string.isEmpty())
363 return;
364
365 // Compute values related to the text style
366 bool bold = (m_style & Bold) != 0;
367 bool underlined = (m_style & Underlined) != 0;
368 bool strikeThrough = (m_style & StrikeThrough) != 0;
369 float italic = (m_style & Italic) ? 0.208f : 0.f; // 12 degrees
370 float underlineOffset = m_font->getUnderlinePosition(m_characterSize);
371 float underlineThickness = m_font->getUnderlineThickness(m_characterSize);
372
373 // Compute the location of the strike through dynamically
374 // We use the center point of the lowercase 'x' glyph as the reference
375 // We reuse the underline thickness as the thickness of the strike through as well
376 FloatRect xBounds = m_font->getGlyph(L'x', m_characterSize, bold).bounds;
377 float strikeThroughOffset = xBounds.top + xBounds.height / 2.f;
378
379 // Precompute the variables needed by the algorithm
380 float hspace = static_cast<float>(m_font->getGlyph(L' ', m_characterSize, bold).advance);
381 float vspace = static_cast<float>(m_font->getLineSpacing(m_characterSize));
382 float x = 0.f;
383 float y = static_cast<float>(m_characterSize);
384
385 // Create one quad for each character
386 float minX = static_cast<float>(m_characterSize);
387 float minY = static_cast<float>(m_characterSize);
388 float maxX = 0.f;
389 float maxY = 0.f;
390 Uint32 prevChar = 0;
391 for (std::size_t i = 0; i < m_string.getSize(); ++i)
392 {
393 Uint32 curChar = m_string[i];
394
395 // Apply the kerning offset
396 x += m_font->getKerning(prevChar, curChar, m_characterSize);
397 prevChar = curChar;
398
399 // If we're using the underlined style and there's a new line, draw a line
400 if (underlined && (curChar == L'\n'))
401 {
402 addLine(m_vertices, x, y, m_fillColor, underlineOffset, underlineThickness);
403
404 if (m_outlineThickness != 0)
405 addLine(m_outlineVertices, x, y, m_outlineColor, underlineOffset, underlineThickness, m_outlineThickness);
406 }
407
408 // If we're using the strike through style and there's a new line, draw a line across all characters
409 if (strikeThrough && (curChar == L'\n'))
410 {
411 addLine(m_vertices, x, y, m_fillColor, strikeThroughOffset, underlineThickness);
412
413 if (m_outlineThickness != 0)
414 addLine(m_outlineVertices, x, y, m_outlineColor, strikeThroughOffset, underlineThickness, m_outlineThickness);
415 }
416
417 // Handle special characters
418 if ((curChar == ' ') || (curChar == '\t') || (curChar == '\n'))
419 {
420 // Update the current bounds (min coordinates)
421 minX = std::min(minX, x);
422 minY = std::min(minY, y);
423
424 switch (curChar)
425 {
426 case ' ': x += hspace; break;
427 case '\t': x += hspace * 4; break;
428 case '\n': y += vspace; x = 0; break;
429 }
430
431 // Update the current bounds (max coordinates)
432 maxX = std::max(maxX, x);
433 maxY = std::max(maxY, y);
434
435 // Next glyph, no need to create a quad for whitespace
436 continue;
437 }
438
439
440 // Apply the outline
441 if (m_outlineThickness != 0)
442 {
443 const Glyph& glyph = m_font->getGlyph(curChar, m_characterSize, bold, m_outlineThickness);
444
445 float left = glyph.bounds.left;
446 float top = glyph.bounds.top;
447 float right = glyph.bounds.left + glyph.bounds.width;
448 float bottom = glyph.bounds.top + glyph.bounds.height;
449
450 // Add the outline glyph to the vertices
451 addGlyphQuad(m_outlineVertices, Vector2f(x, y), m_outlineColor, glyph, italic, m_outlineThickness);
452
453 // Update the current bounds with the outlined glyph bounds
454 minX = std::min(minX, x + left - italic * bottom - m_outlineThickness);
455 maxX = std::max(maxX, x + right - italic * top - m_outlineThickness);
456 minY = std::min(minY, y + top - m_outlineThickness);
457 maxY = std::max(maxY, y + bottom - m_outlineThickness);
458 }
459
460 // Extract the current glyph's description
461 const Glyph& glyph = m_font->getGlyph(curChar, m_characterSize, bold);
462
463 // Add the glyph to the vertices
464 addGlyphQuad(m_vertices, Vector2f(x, y), m_fillColor, glyph, italic);
465
466 // Update the current bounds with the non outlined glyph bounds
467 if (m_outlineThickness == 0)
468 {
469 float left = glyph.bounds.left;
470 float top = glyph.bounds.top;
471 float right = glyph.bounds.left + glyph.bounds.width;
472 float bottom = glyph.bounds.top + glyph.bounds.height;
473
474 minX = std::min(minX, x + left - italic * bottom);
475 maxX = std::max(maxX, x + right - italic * top);
476 minY = std::min(minY, y + top);
477 maxY = std::max(maxY, y + bottom);
478 }
479
480 // Advance to the next character
481 x += glyph.advance;
482 }
483
484 // If we're using the underlined style, add the last line
485 if (underlined && (x > 0))
486 {
487 addLine(m_vertices, x, y, m_fillColor, underlineOffset, underlineThickness);
488
489 if (m_outlineThickness != 0)
490 addLine(m_outlineVertices, x, y, m_outlineColor, underlineOffset, underlineThickness, m_outlineThickness);
491 }
492
493 // If we're using the strike through style, add the last line across all characters
494 if (strikeThrough && (x > 0))
495 {
496 addLine(m_vertices, x, y, m_fillColor, strikeThroughOffset, underlineThickness);
497
498 if (m_outlineThickness != 0)
499 addLine(m_outlineVertices, x, y, m_outlineColor, strikeThroughOffset, underlineThickness, m_outlineThickness);
500 }
501
502 // Update the bounding rectangle
503 m_bounds.left = minX;
504 m_bounds.top = minY;
505 m_bounds.width = maxX - minX;
506 m_bounds.height = maxY - minY;
507}
