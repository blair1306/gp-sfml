40 // Convert an sf::BlendMode::Factor constant to the corresponding OpenGL constant.
41 sf::Uint32 factorToGlConstant(sf::BlendMode::Factor blendFactor)
42 {
43 switch (blendFactor)
44 {
45 case sf::BlendMode::Zero: return GL_ZERO;
46 case sf::BlendMode::One: return GL_ONE;
47 case sf::BlendMode::SrcColor: return GL_SRC_COLOR;
48 case sf::BlendMode::OneMinusSrcColor: return GL_ONE_MINUS_SRC_COLOR;
49 case sf::BlendMode::DstColor: return GL_DST_COLOR;
50 case sf::BlendMode::OneMinusDstColor: return GL_ONE_MINUS_DST_COLOR;
51 case sf::BlendMode::SrcAlpha: return GL_SRC_ALPHA;
52 case sf::BlendMode::OneMinusSrcAlpha: return GL_ONE_MINUS_SRC_ALPHA;
53 case sf::BlendMode::DstAlpha: return GL_DST_ALPHA;
54 case sf::BlendMode::OneMinusDstAlpha: return GL_ONE_MINUS_DST_ALPHA;
55 }
56
57 sf::err() << "Invalid value for sf::BlendMode::Factor! Fallback to sf::BlendMode::Zero." << std::endl;
58 assert(false);
59 return GL_ZERO;
60 }
61
62
63 // Convert an sf::BlendMode::BlendEquation constant to the corresponding OpenGL constant.
64 sf::Uint32 equationToGlConstant(sf::BlendMode::Equation blendEquation)
65 {
66 switch (blendEquation)
67 {
68 case sf::BlendMode::Add: return GLEXT_GL_FUNC_ADD;
69 case sf::BlendMode::Subtract: return GLEXT_GL_FUNC_SUBTRACT;
70 case sf::BlendMode::ReverseSubtract: return GLEXT_GL_FUNC_REVERSE_SUBTRACT;
71 }
72
73 sf::err() << "Invalid value for sf::BlendMode::Equation! Fallback to sf::BlendMode::Add." << std::endl;
74 assert(false);
75 return GLEXT_GL_FUNC_ADD;
76 }
77}

99void RenderTarget::clear(const Color& color)
100{
101 if (setActive(true))
102 {
103 // Unbind texture to fix RenderTexture preventing clear
104 applyTexture(NULL);
105
106 glCheck(glClearColor(color.r / 255.f, color.g / 255.f, color.b / 255.f, color.a / 255.f));
107 glCheck(glClear(GL_COLOR_BUFFER_BIT));
108 }
109}

134////////////////////////////////////////////////////////////
135IntRect RenderTarget::getViewport(const View& view) const
136{
137 float width = static_cast<float>(getSize().x);
138 float height = static_cast<float>(getSize().y);
139 const FloatRect& viewport = view.getViewport();
140
141 return IntRect(static_cast<int>(0.5f + width * viewport.left),
142 static_cast<int>(0.5f + height * viewport.top),
143 static_cast<int>(0.5f + width * viewport.width),
144 static_cast<int>(0.5f + height * viewport.height));
145}

155////////////////////////////////////////////////////////////
156Vector2f RenderTarget::mapPixelToCoords(const Vector2i& point, const View& view) const
157{
158 // First, convert from viewport coordinates to homogeneous coordinates
159 Vector2f normalized;
160 IntRect viewport = getViewport(view);
161 normalized.x = -1.f + 2.f * (point.x - viewport.left) / viewport.width;
162 normalized.y = 1.f - 2.f * (point.y - viewport.top) / viewport.height;
163
164 // Then transform by the inverse of the view matrix
165 return view.getInverseTransform().transformPoint(normalized);
166}

176////////////////////////////////////////////////////////////
177Vector2i RenderTarget::mapCoordsToPixel(const Vector2f& point, const View& view) const
178{
179 // First, transform the point by the view matrix
180 Vector2f normalized = view.getTransform().transformPoint(point);
181
182 // Then convert to viewport coordinates
183 Vector2i pixel;
184 IntRect viewport = getViewport(view);
185 pixel.x = static_cast<int>(( normalized.x + 1.f) / 2.f * viewport.width + viewport.left);
186 pixel.y = static_cast<int>((-normalized.y + 1.f) / 2.f * viewport.height + viewport.top);
187
188 return pixel;
189}

199////////////////////////////////////////////////////////////
200void RenderTarget::draw(const Vertex* vertices, std::size_t vertexCount,
201 PrimitiveType type, const RenderStates& states)
202{
203 // Nothing to draw?
204 if (!vertices || (vertexCount == 0))
205 return;
206
207 // GL_QUADS is unavailable on OpenGL ES
208 #ifdef SFML_OPENGL_ES
209 if (type == Quads)
210 {
211 err() << "sf::Quads primitive type is not supported on OpenGL ES platforms, drawing skipped" << std::endl;
212 return;
213 }
214 #define GL_QUADS 0
215 #endif
216
217 if (setActive(true))
218 {
219 // First set the persistent OpenGL states if it's the very first call
220 if (!m_cache.glStatesSet)
221 resetGLStates();
222
223 // Check if the vertex count is low enough so that we can pre-transform them
224 bool useVertexCache = (vertexCount <= StatesCache::VertexCacheSize);
225 if (useVertexCache)
226 {
227 // Pre-transform the vertices and store them into the vertex cache
228 for (std::size_t i = 0; i < vertexCount; ++i)
229 {
230 Vertex& vertex = m_cache.vertexCache[i];
231 vertex.position = states.transform * vertices[i].position;
232 vertex.color = vertices[i].color;
233 vertex.texCoords = vertices[i].texCoords;
234 }
235
236 // Since vertices are transformed, we must use an identity transform to render them
237 if (!m_cache.useVertexCache)
238 applyTransform(Transform::Identity);
239 }
240 else
241 {
242 applyTransform(states.transform);
243 }
244
245 // Apply the view
246 if (m_cache.viewChanged)
247 applyCurrentView();
248
249 // Apply the blend mode
250 if (states.blendMode != m_cache.lastBlendMode)
251 applyBlendMode(states.blendMode);
252
253 // Apply the texture
254 Uint64 textureId = states.texture ? states.texture->m_cacheId : 0;
255 if (textureId != m_cache.lastTextureId)
256 applyTexture(states.texture);
257
258 // Apply the shader
259 if (states.shader)
260 applyShader(states.shader);
261
262 // If we pre-transform the vertices, we must use our internal vertex cache
263 if (useVertexCache)
264 {
265 // ... and if we already used it previously, we don't need to set the pointers again
266 if (!m_cache.useVertexCache)
267 vertices = m_cache.vertexCache;
268 else
269 vertices = NULL;
270 }
271
272 // Check if texture coordinates array is needed, and update client state accordingly
273 bool enableTexCoordsArray = (states.texture || states.shader);
274 if (enableTexCoordsArray != m_cache.texCoordsArrayEnabled)
275 {
276 if (enableTexCoordsArray)
277 glCheck(glEnableClientState(GL_TEXTURE_COORD_ARRAY));
278 else
279 glCheck(glDisableClientState(GL_TEXTURE_COORD_ARRAY));
280 m_cache.texCoordsArrayEnabled = enableTexCoordsArray;
281 }
282
283 // Setup the pointers to the vertices' components
284 if (vertices)
285 {
286 const char* data = reinterpret_cast<const char*>(vertices);
287 glCheck(glVertexPointer(2, GL_FLOAT, sizeof(Vertex), data + 0));
288 glCheck(glColorPointer(4, GL_UNSIGNED_BYTE, sizeof(Vertex), data + 8));
289 if (enableTexCoordsArray)
290 glCheck(glTexCoordPointer(2, GL_FLOAT, sizeof(Vertex), data + 12));
291 }
292
293 // Find the OpenGL primitive type
294 static const GLenum modes[] = {GL_POINTS, GL_LINES, GL_LINE_STRIP, GL_TRIANGLES,
295 GL_TRIANGLE_STRIP, GL_TRIANGLE_FAN, GL_QUADS};
296 GLenum mode = modes[type];
297
298 // Draw the primitives
299 glCheck(glDrawArrays(mode, 0, vertexCount));
300
301 // Unbind the shader, if any
302 if (states.shader)
303 applyShader(NULL);
304
305 // If the texture we used to draw belonged to a RenderTexture, then forcibly unbind that texture.
306 // This prevents a bug where some drivers do not clear RenderTextures properly.
307 if (states.texture && states.texture->m_fboAttachment)
308 applyTexture(NULL);
309
310 // Update the cache
311 m_cache.useVertexCache = useVertexCache;
312 }
313}

316////////////////////////////////////////////////////////////
317void RenderTarget::pushGLStates()
318{
319 if (setActive(true))
320 {
321 #ifdef SFML_DEBUG
322 // make sure that the user didn't leave an unchecked OpenGL error
323 GLenum error = glGetError();
324 if (error != GL_NO_ERROR)
325 {
326 err() << "OpenGL error (" << error << ") detected in user code, "
327 << "you should check for errors with glGetError()"
328 << std::endl;
329 }
330 #endif
331
332 #ifndef SFML_OPENGL_ES
333 glCheck(glPushClientAttrib(GL_CLIENT_ALL_ATTRIB_BITS));
334 glCheck(glPushAttrib(GL_ALL_ATTRIB_BITS));
335 #endif
336 glCheck(glMatrixMode(GL_MODELVIEW));
337 glCheck(glPushMatrix());
338 glCheck(glMatrixMode(GL_PROJECTION));
339 glCheck(glPushMatrix());
340 glCheck(glMatrixMode(GL_TEXTURE));
341 glCheck(glPushMatrix());
342 }
343
344 resetGLStates();
345}

348////////////////////////////////////////////////////////////
349void RenderTarget::popGLStates()
350{
351 if (setActive(true))
352 {
353 glCheck(glMatrixMode(GL_PROJECTION));
354 glCheck(glPopMatrix());
355 glCheck(glMatrixMode(GL_MODELVIEW));
356 glCheck(glPopMatrix());
357 glCheck(glMatrixMode(GL_TEXTURE));
358 glCheck(glPopMatrix());
359 #ifndef SFML_OPENGL_ES
360 glCheck(glPopClientAttrib());
361 glCheck(glPopAttrib());
362 #endif
363 }
364}

367////////////////////////////////////////////////////////////
368void RenderTarget::resetGLStates()
369{
370 // Check here to make sure a context change does not happen after activate(true)
371 bool shaderAvailable = Shader::isAvailable();
372
373 if (setActive(true))
374 {
375 // Make sure that extensions are initialized
376 priv::ensureExtensionsInit();
377
378 // Make sure that the texture unit which is active is the number 0
379 if (GLEXT_multitexture)
380 {
381 glCheck(GLEXT_glClientActiveTexture(GLEXT_GL_TEXTURE0));
382 glCheck(GLEXT_glActiveTexture(GLEXT_GL_TEXTURE0));
383 }
384
385 // Define the default OpenGL states
386 glCheck(glDisable(GL_CULL_FACE));
387 glCheck(glDisable(GL_LIGHTING));
388 glCheck(glDisable(GL_DEPTH_TEST));
389 glCheck(glDisable(GL_ALPHA_TEST));
390 glCheck(glEnable(GL_TEXTURE_2D));
391 glCheck(glEnable(GL_BLEND));
392 glCheck(glMatrixMode(GL_MODELVIEW));
393 glCheck(glEnableClientState(GL_VERTEX_ARRAY));
394 glCheck(glEnableClientState(GL_COLOR_ARRAY));
395 glCheck(glEnableClientState(GL_TEXTURE_COORD_ARRAY));
396 m_cache.glStatesSet = true;
397
398 // Apply the default SFML states
399 applyBlendMode(BlendAlpha);
400 applyTransform(Transform::Identity);
401 applyTexture(NULL);
402 if (shaderAvailable)
403 applyShader(NULL);
404
405 m_cache.texCoordsArrayEnabled = true;
406
407 m_cache.useVertexCache = false;
408
409 // Set the default view
410 setView(getView());
411 }
412}

415////////////////////////////////////////////////////////////
416void RenderTarget::initialize()
417{
418 // Setup the default and current views
419 m_defaultView.reset(FloatRect(0, 0, static_cast<float>(getSize().x), static_cast<float>(getSize().y)));
420 m_view = m_defaultView;
421
422 // Set GL states only on first draw, so that we don't pollute user's states
423 m_cache.glStatesSet = false;
424}

427////////////////////////////////////////////////////////////
428void RenderTarget::applyCurrentView()
429{
430 // Set the viewport
431 IntRect viewport = getViewport(m_view);
432 int top = getSize().y - (viewport.top + viewport.height);
433 glCheck(glViewport(viewport.left, top, viewport.width, viewport.height));
434
435 // Set the projection matrix
436 glCheck(glMatrixMode(GL_PROJECTION));
437 glCheck(glLoadMatrixf(m_view.getTransform().getMatrix()));
438
439 // Go back to model-view mode
440 glCheck(glMatrixMode(GL_MODELVIEW));
441
442 m_cache.viewChanged = false;
443}

446////////////////////////////////////////////////////////////
447void RenderTarget::applyBlendMode(const BlendMode& mode)
448{
449 // Apply the blend mode, falling back to the non-separate versions if necessary
450 if (GLEXT_blend_func_separate)
451 {
452 glCheck(GLEXT_glBlendFuncSeparate(
453 factorToGlConstant(mode.colorSrcFactor), factorToGlConstant(mode.colorDstFactor),
454 factorToGlConstant(mode.alphaSrcFactor), factorToGlConstant(mode.alphaDstFactor)));
455 }
456 else
457 {
458 glCheck(glBlendFunc(
459 factorToGlConstant(mode.colorSrcFactor),
460 factorToGlConstant(mode.colorDstFactor)));
461 }
462
463 if (GLEXT_blend_minmax && GLEXT_blend_subtract)
464 {
465 if (GLEXT_blend_equation_separate)
466 {
467 glCheck(GLEXT_glBlendEquationSeparate(
468 equationToGlConstant(mode.colorEquation),
469 equationToGlConstant(mode.alphaEquation)));
470 }
471 else
472 {
473 glCheck(GLEXT_glBlendEquation(equationToGlConstant(mode.colorEquation)));
474 }
475 }
476 else if ((mode.colorEquation != BlendMode::Add) || (mode.alphaEquation != BlendMode::Add))
477 {
478 static bool warned = false;
479
480 if (!warned)
481 {
482 err() << "OpenGL extension EXT_blend_minmax and/or EXT_blend_subtract unavailable" << std::endl;
483 err() << "Selecting a blend equation not possible" << std::endl;
484 err() << "Ensure that hardware acceleration is enabled if available" << std::endl;
485
486 warned = true;
487 }
488 }
489
490 m_cache.lastBlendMode = mode;
491}

494////////////////////////////////////////////////////////////
495void RenderTarget::applyTransform(const Transform& transform)
496{
497 // No need to call glMatrixMode(GL_MODELVIEW), it is always the
498 // current mode (for optimization purpose, since it's the most used)
499 glCheck(glLoadMatrixf(transform.getMatrix()));
500}

521////////////////////////////////////////////////////////////
522// Render states caching strategies
523//
524// * View
525// If SetView was called since last draw, the projection
526// matrix is updated. We don't need more, the view doesn't
527// change frequently.
528//
529// * Transform
530// The transform matrix is usually expensive because each
531// entity will most likely use a different transform. This can
532// lead, in worst case, to changing it every 4 vertices.
533// To avoid that, when the vertex count is low enough, we
534// pre-transform them and therefore use an identity transform
535// to render them.
536//
537// * Blending mode
538// Since it overloads the == operator, we can easily check
539// whether any of the 6 blending components changed and,
540// thus, whether we need to update the blend mode.
541//
542// * Texture
543// Storing the pointer or OpenGL ID of the last used texture
544// is not enough; if the sf::Texture instance is destroyed,
545// both the pointer and the OpenGL ID might be recycled in
546// a new texture instance. We need to use our own unique
547// identifier system to ensure consistent caching.
548//
549// * Shader
550// Shaders are very hard to optimize, because they have
551// parameters that can be hard (if not impossible) to track,
552// like matrices or textures. The only optimization that we
553// do is that we avoid setting a null shader if there was
554// already none for the previous draw.
555//
