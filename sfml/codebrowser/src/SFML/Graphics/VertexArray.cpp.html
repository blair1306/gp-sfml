107FloatRect VertexArray::getBounds() const
108{
109 if (!m_vertices.empty())
110 {
111 float left = m_vertices[0].position.x;
112 float top = m_vertices[0].position.y;
113 float right = m_vertices[0].position.x;
114 float bottom = m_vertices[0].position.y;
115
116 for (std::size_t i = 1; i < m_vertices.size(); ++i)
117 {
118 Vector2f position = m_vertices[i].position;
119
120 // Update left and right
121 if (position.x < left)
122 left = position.x;
123 else if (position.x > right)
124 right = position.x;
125
126 // Update top and bottom
127 if (position.y < top)
128 top = position.y;
129 else if (position.y > bottom)
130 bottom = position.y;
131 }
132
133 return FloatRect(left, top, right - left, bottom - top);
134 }
135 else
136 {
137 // Array is empty
138 return FloatRect();
139 }
140}
