51////////////////////////////////////////////////////////////
52bool RenderTexture::create(unsigned int width, unsigned int height, bool depthBuffer)
53{
54 // Create the texture
55 if (!m_texture.create(width, height))
56 {
57 err() << "Impossible to create render texture (failed to create the target texture)" << std::endl;
58 return false;
59 }
60
61 // We disable smoothing by default for render textures
62 setSmooth(false);
63
64 // Create the implementation
65 delete m_impl;
66 if (priv::RenderTextureImplFBO::isAvailable())
67 {
68 // Use frame-buffer object (FBO)
69 m_impl = new priv::RenderTextureImplFBO;
70
71 // Mark the texture as being a framebuffer object attachment
72 m_texture.m_fboAttachment = true;
73 }
74 else
75 {
76 // Use default implementation
77 m_impl = new priv::RenderTextureImplDefault;
78 }
79
80 // Initialize the render texture
81 if (!m_impl->create(width, height, m_texture.m_texture, depthBuffer))
82 return false;
83
84 // We can now initialize the render target part
85 RenderTarget::initialize();
86
87 return true;
88}

127bool RenderTexture::setActive(bool active)
128{
129 return m_impl && m_impl->activate(active);
130}

133////////////////////////////////////////////////////////////
134void RenderTexture::display()
135{
136 // Update the target texture
137 if (setActive(true))
138 {
139 m_impl->updateTexture(m_texture.m_texture);
140 m_texture.m_pixelsFlipped = true;
141 m_texture.invalidateMipmap();
142 }
143}
