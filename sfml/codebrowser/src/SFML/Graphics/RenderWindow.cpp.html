78////////////////////////////////////////////////////////////
79Image RenderWindow::capture() const
80{
81 Vector2u windowSize = getSize();
82
83 Texture texture;
84 texture.create(windowSize.x, windowSize.y);
85 texture.update(*this);
86
87 return texture.copyToImage();
88}
