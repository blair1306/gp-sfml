39 void copyMatrix(const Transform& source, Matrix<3, 3>& dest)
40 {
41 const float* from = source.getMatrix(); // 4x4
42 float* to = dest.array; // 3x3
43
44 // Use only left-upper 3x3 block (for a 2D transform)
45 to[0] = from[ 0]; to[1] = from[ 1]; to[2] = from[ 3];
46 to[3] = from[ 4]; to[4] = from[ 5]; to[5] = from[ 7];
47 to[6] = from[12]; to[7] = from[13]; to[8] = from[15];
48 }
49
50
51 ////////////////////////////////////////////////////////////
52 void copyMatrix(const Transform& source, Matrix<4, 4>& dest)
53 {
54 // Adopt 4x4 matrix as-is
55 copyMatrix(source.getMatrix(), 4 * 4, dest.array);
56 }
57
58
59 ////////////////////////////////////////////////////////////
60 void copyMatrix(const float* source, std::size_t elements, float* dest)
61 {
62 std::copy(source, source + elements, dest);
63 }

66 ////////////////////////////////////////////////////////////
67 void copyVector(const Color& source, Vector4<float>& dest)
68 {
69 dest.x = source.r / 255.f;
70 dest.y = source.g / 255.f;
71 dest.z = source.b / 255.f;
72 dest.w = source.a / 255.f;
73 }
74
75
76 ////////////////////////////////////////////////////////////
77 void copyVector(const Color& source, Vector4<int>& dest)
78 {
79 dest.x = static_cast<int>(source.r);
80 dest.y = static_cast<int>(source.g);
81 dest.z = static_cast<int>(source.b);
82 dest.w = static_cast<int>(source.a);
83 }
