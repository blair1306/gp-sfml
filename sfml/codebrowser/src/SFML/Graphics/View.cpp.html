109void View::setRotation(float angle)
110{
111 m_rotation = static_cast<float>(fmod(angle, 360));
112 if (m_rotation < 0)
113 m_rotation += 360.f;
114
115 m_transformUpdated = false;
116 m_invTransformUpdated = false;
117}

128void View::reset(const FloatRect& rectangle)
129{
130 m_center.x = rectangle.left + rectangle.width / 2.f;
131 m_center.y = rectangle.top + rectangle.height / 2.f;
132 m_size.x = rectangle.width;
133 m_size.y = rectangle.height;
134 m_rotation = 0;
135
136 m_transformUpdated = false;
137 m_invTransformUpdated = false;
138}

198const Transform& View::getTransform() const
199{
200 // Recompute the matrix if needed
201 if (!m_transformUpdated)
202 {
203 // Rotation components
204 float angle = m_rotation * 3.141592654f / 180.f;
205 float cosine = static_cast<float>(std::cos(angle));
206 float sine = static_cast<float>(std::sin(angle));
207 float tx = -m_center.x * cosine - m_center.y * sine + m_center.x;
208 float ty = m_center.x * sine - m_center.y * cosine + m_center.y;
209
210 // Projection components
211 float a = 2.f / m_size.x;
212 float b = -2.f / m_size.y;
213 float c = -a * m_center.x;
214 float d = -b * m_center.y;
215
216 // Rebuild the projection matrix
217 m_transform = Transform( a * cosine, a * sine, a * tx + c,
218 -b * sine, b * cosine, b * ty + d,
219 0.f, 0.f, 1.f);
220 m_transformUpdated = true;
221 }
222
223 return m_transform;
224}

227////////////////////////////////////////////////////////////
228const Transform& View::getInverseTransform() const
229{
230 // Recompute the matrix if needed
231 if (!m_invTransformUpdated)
232 {
233 m_inverseTransform = getTransform().getInverse();
234 m_invTransformUpdated = true;
235 }
236
237 return m_inverseTransform;
238}
