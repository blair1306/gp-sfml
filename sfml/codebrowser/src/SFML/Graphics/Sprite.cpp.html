54Sprite::Sprite(const Texture& texture, const IntRect& rectangle) :
55m_texture (NULL),
56m_textureRect()

58 setTexture(texture);
59 setTextureRect(rectangle);

63////////////////////////////////////////////////////////////
64void Sprite::setTexture(const Texture& texture, bool resetRect)
65{
66 // Recompute the texture area if requested, or if there was no valid texture & rect before
67 if (resetRect || (!m_texture && (m_textureRect == sf::IntRect())))
68 setTextureRect(IntRect(0, 0, texture.getSize().x, texture.getSize().y));
69
70 // Assign the new texture
71 m_texture = &texture;
72}
73
74
75////////////////////////////////////////////////////////////
76void Sprite::setTextureRect(const IntRect& rectangle)
77{
78 if (rectangle != m_textureRect)
79 {
80 m_textureRect = rectangle;
81 updatePositions();
82 updateTexCoords();
83 }
84}

88void Sprite::setColor(const Color& color)
89{
90 // Update the vertices' color
91 m_vertices[0].color = color;
92 m_vertices[1].color = color;
93 m_vertices[2].color = color;
94 m_vertices[3].color = color;
95}

119////////////////////////////////////////////////////////////
120FloatRect Sprite::getLocalBounds() const
121{
122 float width = static_cast<float>(std::abs(m_textureRect.width));
123 float height = static_cast<float>(std::abs(m_textureRect.height));
124
125 return FloatRect(0.f, 0.f, width, height);
126}
127
128
129////////////////////////////////////////////////////////////
130FloatRect Sprite::getGlobalBounds() const
131{
132 return getTransform().transformRect(getLocalBounds());
133}

135
136////////////////////////////////////////////////////////////
137void Sprite::draw(RenderTarget& target, RenderStates states) const
138{
139 if (m_texture)
140 {
141 states.transform *= getTransform();
142 states.texture = m_texture;
143 target.draw(m_vertices, 4, TriangleStrip, states);
144 }
145}

148////////////////////////////////////////////////////////////
149void Sprite::updatePositions()
150{
151 FloatRect bounds = getLocalBounds();
152
153 m_vertices[0].position = Vector2f(0, 0);
154 m_vertices[1].position = Vector2f(0, bounds.height);
155 m_vertices[2].position = Vector2f(bounds.width, 0);
156 m_vertices[3].position = Vector2f(bounds.width, bounds.height);
157}
158
159
160////////////////////////////////////////////////////////////
161void Sprite::updateTexCoords()
162{
163 float left = static_cast<float>(m_textureRect.left);
164 float right = left + m_textureRect.width;
165 float top = static_cast<float>(m_textureRect.top);
166 float bottom = top + m_textureRect.height;
167
168 m_vertices[0].texCoords = Vector2f(left, top);
169 m_vertices[1].texCoords = Vector2f(left, bottom);
170 m_vertices[2].texCoords = Vector2f(right, top);
171 m_vertices[3].texCoords = Vector2f(right, bottom);
172}
