25#ifndef SFML_RENDERTEXTUREIMPLFBO_HPP
26#define SFML_RENDERTEXTUREIMPLFBO_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Graphics/RenderTextureImpl.hpp>
32#include <SFML/Window/Context.hpp>
33#include <SFML/Window/GlResource.hpp>
34
35
36namespace sf
37{
38namespace priv
39{
40////////////////////////////////////////////////////////////
41/// \brief Specialization of RenderTextureImpl using the
42/// FrameBuffer Object OpenGL extension
43///
44////////////////////////////////////////////////////////////
45class RenderTextureImplFBO : public RenderTextureImpl, GlResource
46{
47public:
48
49 ////////////////////////////////////////////////////////////
50 /// \brief Default constructor
51 ///
52 ////////////////////////////////////////////////////////////
53 RenderTextureImplFBO();
54
55 ////////////////////////////////////////////////////////////
56 /// \brief Destructor
57 ///
58 ////////////////////////////////////////////////////////////
59 ~RenderTextureImplFBO();
60
61 ////////////////////////////////////////////////////////////
62 /// \brief Check whether the system supports FBOs or not
63 ///
64 /// \return True if FBO render textures are supported
65 ///
66 ////////////////////////////////////////////////////////////
67 static bool isAvailable();
68
69private:
70
71 ////////////////////////////////////////////////////////////
72 /// \brief Create the render texture implementation
73 ///
74 /// \param width Width of the texture to render to
75 /// \param height Height of the texture to render to
76 /// \param textureId OpenGL identifier of the target texture
77 /// \param depthBuffer Is a depth buffer requested?
78 ///
79 /// \return True if creation has been successful
80 ///
81 ////////////////////////////////////////////////////////////
82 virtual bool create(unsigned int width, unsigned int height, unsigned int textureId, bool depthBuffer);
83
84 ////////////////////////////////////////////////////////////
85 /// \brief Activate or deactivate the render texture for rendering
86 ///
87 /// \param active True to activate, false to deactivate
88 ///
89 /// \return True on success, false on failure
90 ///
91 ////////////////////////////////////////////////////////////
92 virtual bool activate(bool active);
93
94 ////////////////////////////////////////////////////////////
95 /// \brief Update the pixels of the target texture
96 ///
97 /// \param textureId OpenGL identifier of the target texture
98 ///
99 ////////////////////////////////////////////////////////////
100 virtual void updateTexture(unsigned textureId);
101
102 ////////////////////////////////////////////////////////////
103 // Member data
104 ////////////////////////////////////////////////////////////
105 Context* m_context; ///< Needs a separate OpenGL context for not messing up the other ones
106 unsigned int m_frameBuffer; ///< OpenGL frame buffer object
107 unsigned int m_depthBuffer; ///< Optional depth buffer attached to the frame buffer
108};
109
110} // namespace priv
111
112} // namespace sf
113
114
115#endif // SFML_RENDERTEXTUREIMPLFBO_HPP
