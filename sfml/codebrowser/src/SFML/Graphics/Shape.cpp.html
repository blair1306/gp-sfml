37 // Compute the normal of a segment
38 sf::Vector2f computeNormal(const sf::Vector2f& p1, const sf::Vector2f& p2)
39 {
40 sf::Vector2f normal(p1.y - p2.y, p2.x - p1.x);
41 float length = std::sqrt(normal.x * normal.x + normal.y * normal.y);
42 if (length != 0.f)
43 normal /= length;
44 return normal;
45 }
46
47 // Compute the dot product of two vectors
48 float dotProduct(const sf::Vector2f& p1, const sf::Vector2f& p2)
49 {
50 return p1.x * p2.x + p1.y * p2.y;
51 }
52}

64void Shape::setTexture(const Texture* texture, bool resetRect)
65{
66 if (texture)
67 {
68 // Recompute the texture area if requested, or if there was no texture & rect before
69 if (resetRect || (!m_texture && (m_textureRect == IntRect())))
70 setTextureRect(IntRect(0, 0, texture->getSize().x, texture->getSize().y));
71 }
72
73 // Assign the new texture
74 m_texture = texture;
75}

174////////////////////////////////////////////////////////////
175void Shape::update()
176{
177 // Get the total number of points of the shape
178 std::size_t count = getPointCount();
179 if (count < 3)
180 {
181 m_vertices.resize(0);
182 m_outlineVertices.resize(0);
183 return;
184 }
185
186 m_vertices.resize(count + 2); // + 2 for center and repeated first point
187
188 // Position
189 for (std::size_t i = 0; i < count; ++i)
190 m_vertices[i + 1].position = getPoint(i);
191 m_vertices[count + 1].position = m_vertices[1].position;
192
193 // Update the bounding rectangle
194 m_vertices[0] = m_vertices[1]; // so that the result of getBounds() is correct
195 m_insideBounds = m_vertices.getBounds();
196
197 // Compute the center and make it the first vertex
198 m_vertices[0].position.x = m_insideBounds.left + m_insideBounds.width / 2;
199 m_vertices[0].position.y = m_insideBounds.top + m_insideBounds.height / 2;
200
201 // Color
202 updateFillColors();
203
204 // Texture coordinates
205 updateTexCoords();
206
207 // Outline
208 updateOutline();
209}

212////////////////////////////////////////////////////////////
213void Shape::draw(RenderTarget& target, RenderStates states) const
214{
215 states.transform *= getTransform();
216
217 // Render the inside
218 states.texture = m_texture;
219 target.draw(m_vertices, states);
220
221 // Render the outline
222 if (m_outlineThickness != 0)
223 {
224 states.texture = NULL;
225 target.draw(m_outlineVertices, states);
226 }
227}

238////////////////////////////////////////////////////////////
239void Shape::updateTexCoords()
240{
241 for (std::size_t i = 0; i < m_vertices.getVertexCount(); ++i)
242 {
243 float xratio = m_insideBounds.width > 0 ? (m_vertices[i].position.x - m_insideBounds.left) / m_insideBounds.width : 0;
244 float yratio = m_insideBounds.height > 0 ? (m_vertices[i].position.y - m_insideBounds.top) / m_insideBounds.height : 0;
245 m_vertices[i].texCoords.x = m_textureRect.left + m_textureRect.width * xratio;
246 m_vertices[i].texCoords.y = m_textureRect.top + m_textureRect.height * yratio;
247 }
248}

251////////////////////////////////////////////////////////////
252void Shape::updateOutline()
253{
254 std::size_t count = m_vertices.getVertexCount() - 2;
255 m_outlineVertices.resize((count + 1) * 2);
256
257 for (std::size_t i = 0; i < count; ++i)
258 {
259 std::size_t index = i + 1;
260
261 // Get the two segments shared by the current point
262 Vector2f p0 = (i == 0) ? m_vertices[count].position : m_vertices[index - 1].position;
263 Vector2f p1 = m_vertices[index].position;
264 Vector2f p2 = m_vertices[index + 1].position;
265
266 // Compute their normal
267 Vector2f n1 = computeNormal(p0, p1);
268 Vector2f n2 = computeNormal(p1, p2);
269
270 // Make sure that the normals point towards the outside of the shape
271 // (this depends on the order in which the points were defined)
272 if (dotProduct(n1, m_vertices[0].position - p1) > 0)
273 n1 = -n1;
274 if (dotProduct(n2, m_vertices[0].position - p1) > 0)
275 n2 = -n2;
276
277 // Combine them to get the extrusion direction
278 float factor = 1.f + (n1.x * n2.x + n1.y * n2.y);
279 Vector2f normal = (n1 + n2) / factor;
280
281 // Update the outline points
282 m_outlineVertices[i * 2 + 0].position = p1;
283 m_outlineVertices[i * 2 + 1].position = p1 + normal * m_outlineThickness;
284 }
285
286 // Duplicate the first point at the end, to close the outline
287 m_outlineVertices[count * 2 + 0].position = m_outlineVertices[0].position;
288 m_outlineVertices[count * 2 + 1].position = m_outlineVertices[1].position;
289
290 // Update outline colors
291 updateOutlineColors();
292
293 // Update the shape's bounds
294 m_bounds = m_outlineVertices.getBounds();
295}
