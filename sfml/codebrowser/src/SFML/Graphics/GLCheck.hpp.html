25#ifndef SFML_GLCHECK_HPP
26#define SFML_GLCHECK_HPP
27
28////////////////////////////////////////////////////////////
29// Headers
30////////////////////////////////////////////////////////////
31#include <SFML/Config.hpp>
32#include <SFML/Graphics/GLExtensions.hpp>
33
34
35namespace sf
36{
37namespace priv
38{
39////////////////////////////////////////////////////////////
40/// Let's define a macro to quickly check every OpenGL API call
41////////////////////////////////////////////////////////////
42#ifdef SFML_DEBUG
43
44 // In debug mode, perform a test on every OpenGL call
45 // The do-while loop is needed so that glCheck can be used as a single statement in if/else branches
46 #define glCheck(expr) do { expr; sf::priv::glCheckError(__FILE__, __LINE__, #expr); } while (false)
47
48#else
49
50 // Else, we don't add any overhead
51 #define glCheck(expr) (expr)
52
53#endif
54
55////////////////////////////////////////////////////////////
56/// \brief Check the last OpenGL error
57///
58/// \param file Source file where the call is located
59/// \param line Line number of the source file where the call is located
60/// \param expression The evaluated expression as a string
61///
62////////////////////////////////////////////////////////////
63void glCheckError(const char* file, unsigned int line, const char* expression);
64
65} // namespace priv
66
67} // namespace sf
68
69
70#endif // SFML_GLCHECK_HPP
