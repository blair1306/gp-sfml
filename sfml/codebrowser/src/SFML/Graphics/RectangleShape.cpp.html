28#include <SFML/Graphics/RectangleShape.hpp>
29#include <cmath>
30
31
32namespace sf
33{
34////////////////////////////////////////////////////////////
35RectangleShape::RectangleShape(const Vector2f& size)
36{
37 setSize(size);
38}
39
40
41////////////////////////////////////////////////////////////
42void RectangleShape::setSize(const Vector2f& size)
43{
44 m_size = size;
45 update();
46}
47
48
49////////////////////////////////////////////////////////////
50const Vector2f& RectangleShape::getSize() const
51{
52 return m_size;
53}
54
55
56////////////////////////////////////////////////////////////
57std::size_t RectangleShape::getPointCount() const
58{
59 return 4;
60}
61
62
63////////////////////////////////////////////////////////////
64Vector2f RectangleShape::getPoint(std::size_t index) const
65{
66 switch (index)
67 {
68 default:
69 case 0: return Vector2f(0, 0);
70 case 1: return Vector2f(m_size.x, 0);
71 case 2: return Vector2f(m_size.x, m_size.y);
72 case 3: return Vector2f(0, m_size.y);
73 }
74}
