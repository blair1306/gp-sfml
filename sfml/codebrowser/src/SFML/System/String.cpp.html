67////////////////////////////////////////////////////////////
68String::String(const char* ansiString, const std::locale& locale)
69{
70 if (ansiString)
71 {
72 std::size_t length = strlen(ansiString);
73 if (length > 0)
74 {
75 m_string.reserve(length + 1);
76 Utf32::fromAnsi(ansiString, ansiString + length, std::back_inserter(m_string), locale);
77 }
78 }
79}

149////////////////////////////////////////////////////////////
150std::string String::toAnsiString(const std::locale& locale) const
151{
152 // Prepare the output string
153 std::string output;
154 output.reserve(m_string.length() + 1);
155
156 // Convert
157 Utf32::toAnsi(m_string.begin(), m_string.end(), std::back_inserter(output), 0, locale);
158
159 return output;
160}

291////////////////////////////////////////////////////////////
292void String::replace(const String& searchFor, const String& replaceWith)
293{
294 std::size_t step = replaceWith.getSize();
295 std::size_t len = searchFor.getSize();
296 std::size_t pos = find(searchFor);
297
298 // Replace each occurrence of search
299 while (pos != InvalidPos)
300 {
301 replace(pos, len, replaceWith);
302 pos = find(searchFor, pos + step);
303 }
304}
