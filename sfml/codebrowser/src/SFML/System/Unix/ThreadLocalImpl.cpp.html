31namespace sf
32{
33namespace priv
34{
35////////////////////////////////////////////////////////////
36ThreadLocalImpl::ThreadLocalImpl() :
37m_key(0)
38{
39 pthread_key_create(&m_key, NULL);
40}
41
42
43////////////////////////////////////////////////////////////
44ThreadLocalImpl::~ThreadLocalImpl()
45{
46 pthread_key_delete(m_key);
47}
48
49
50////////////////////////////////////////////////////////////
51void ThreadLocalImpl::setValue(void* value)
52{
53 pthread_setspecific(m_key, value);
54}
55
56
57////////////////////////////////////////////////////////////
58void* ThreadLocalImpl::getValue() const
59{
60 return pthread_getspecific(m_key);
61}
