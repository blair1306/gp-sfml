38////////////////////////////////////////////////////////////
39ThreadImpl::ThreadImpl(Thread* owner) :
40m_isActive(true)
41{
42 m_isActive = pthread_create(&m_thread, NULL, &ThreadImpl::entryPoint, owner) == 0;
43
44 if (!m_isActive)
45 std::cerr << "Failed to create thread" << std::endl;
46}
47
48
49////////////////////////////////////////////////////////////
50void ThreadImpl::wait()
51{
52 if (m_isActive)
53 {
54 assert(pthread_equal(pthread_self(), m_thread) == 0); // A thread cannot wait for itself!
55 pthread_join(m_thread, NULL);
56 }
57}
58
59
60////////////////////////////////////////////////////////////
61void ThreadImpl::terminate()
62{
63 if (m_isActive)
64 {
65 #ifndef SFML_SYSTEM_ANDROID
66 pthread_cancel(m_thread);
67 #else
68 // See http://stackoverflow.com/questions/4610086/pthread-cancel-al
69 pthread_kill(m_thread, SIGUSR1);
70 #endif
71 }
72}
73
74
75////////////////////////////////////////////////////////////
76void* ThreadImpl::entryPoint(void* userData)
77{
78 // The Thread instance is stored in the user data
79 Thread* owner = static_cast<Thread*>(userData);
80
81 #ifndef SFML_SYSTEM_ANDROID
82 // Tell the thread to handle cancel requests immediately
83 pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
84 #endif
85
86 // Forward to the owner
87 owner->run();
88
89 return NULL;
90}
