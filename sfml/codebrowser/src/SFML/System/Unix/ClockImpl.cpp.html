41Time ClockImpl::getCurrentTime()
42{
43#if defined(SFML_SYSTEM_MACOS) || defined(SFML_SYSTEM_IOS)
44
45 // Mac OS X specific implementation (it doesn't support clock_gettime)
46 static mach_timebase_info_data_t frequency = {0, 0};
47 if (frequency.denom == 0)
48 mach_timebase_info(&frequency);
49 Uint64 nanoseconds = mach_absolute_time() * frequency.numer / frequency.denom;
50 return sf::microseconds(nanoseconds / 1000);
51
52#else
53
54 // POSIX implementation
55 timespec time;
56 clock_gettime(CLOCK_MONOTONIC, &time);
57 return sf::microseconds(static_cast<Uint64>(time.tv_sec) * 1000000 + time.tv_nsec / 1000);
58
59#endif
60}
