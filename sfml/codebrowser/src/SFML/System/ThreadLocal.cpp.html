37namespace sf
38{
39////////////////////////////////////////////////////////////
40ThreadLocal::ThreadLocal(void* value)
41{
42 m_impl = new priv::ThreadLocalImpl;
43 setValue(value);
44}
45
46
47////////////////////////////////////////////////////////////
48ThreadLocal::~ThreadLocal()
49{
50 delete m_impl;
51}
52
53
54////////////////////////////////////////////////////////////
55void ThreadLocal::setValue(void* value)
56{
57 m_impl->setValue(value);
58}
59
60
61////////////////////////////////////////////////////////////
62void* ThreadLocal::getValue() const
63{
64 return m_impl->getValue();
65}
