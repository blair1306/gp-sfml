37FileInputStream::FileInputStream()
38: m_file(NULL)
39{
40
41}

44////////////////////////////////////////////////////////////
45FileInputStream::~FileInputStream()
46{
47#ifdef ANDROID
48 if (m_file)
49 delete m_file;
50#else
51 if (m_file)
52 std::fclose(m_file);
53#endif
54}
55
56
57////////////////////////////////////////////////////////////
58bool FileInputStream::open(const std::string& filename)
59{
60#ifdef ANDROID
61 if (m_file)
62 delete m_file;
63 m_file = new priv::ResourceStream(filename);
64 return m_file->tell() != -1;
65#else
66 if (m_file)
67 std::fclose(m_file);
68
69 m_file = std::fopen(filename.c_str(), "rb");
70
71 return m_file != NULL;
72#endif
73}

76////////////////////////////////////////////////////////////
77Int64 FileInputStream::read(void* data, Int64 size)
78{
79#ifdef ANDROID
80 return m_file->read(data, size);
81#else
82 if (m_file)
83 return std::fread(data, 1, static_cast<std::size_t>(size), m_file);
84 else
85 return -1;
86#endif
87}
88
89
90////////////////////////////////////////////////////////////
91Int64 FileInputStream::seek(Int64 position)
92{
93#ifdef ANDROID
94 return m_file->seek(position);
95#else
96 if (m_file)
97 {
98 if (std::fseek(m_file, static_cast<std::size_t>(position), SEEK_SET))
99 return -1;
100
101 return tell();
102 }
103 else
104 {
105 return -1;
106 }
107#endif
108}
109
110
111////////////////////////////////////////////////////////////
112Int64 FileInputStream::tell()
113{
114#ifdef ANDROID
115 return m_file->tell();
116#else
117 if (m_file)
118 return std::ftell(m_file);
119 else
120 return -1;
121#endif
122}
123
124
125////////////////////////////////////////////////////////////
126Int64 FileInputStream::getSize()
127{
128#ifdef ANDROID
129 return m_file->getSize();
130#else
131 if (m_file)
132 {
133 Int64 position = tell();
134 std::fseek(m_file, 0, SEEK_END);
135 Int64 size = tell();
136 seek(position);
137 return size;
138 }
139 else
140 {
141 return -1;
142 }
143#endif
144}
