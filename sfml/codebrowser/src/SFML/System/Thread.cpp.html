38namespace sf
39{
40////////////////////////////////////////////////////////////
41Thread::~Thread()
42{
43 wait();
44 delete m_entryPoint;
45}
46
47
48////////////////////////////////////////////////////////////
49void Thread::launch()
50{
51 wait();
52 m_impl = new priv::ThreadImpl(this);
53}
54
55
56////////////////////////////////////////////////////////////
57void Thread::wait()
58{
59 if (m_impl)
60 {
61 m_impl->wait();
62 delete m_impl;
63 m_impl = NULL;
64 }
65}
66
67
68////////////////////////////////////////////////////////////
69void Thread::terminate()
70{
71 if (m_impl)
72 {
73 m_impl->terminate();
74 delete m_impl;
75 m_impl = NULL;
76 }
77}
78
79
80////////////////////////////////////////////////////////////
81void Thread::run()
82{
83 m_entryPoint->run();
84}
